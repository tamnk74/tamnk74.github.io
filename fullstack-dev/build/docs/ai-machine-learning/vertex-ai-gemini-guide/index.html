Getting Started with Vertex AI and Gemini Models

This comprehensive guide covers everything you need to know about using Google's Vertex AI platform with Gemini models for enterprise AI applications. Learn how to integrate powerful generative AI capabilities into your applications using the Google Gen AI SDK.

Overview​

The Google Gen AI SDK provides a unified interface to Google's generative AI API services, including the powerful Gemini models. This SDK simplifies the process of integrating generative AI capabilities into applications and services, enabling developers to leverage Google's advanced AI models for various tasks.

Why Choose Vertex AI with Gemini?​

Enterprise-Ready Platform​

  • Scalability: Built for production workloads with enterprise-grade infrastructure
  • Security: Advanced security controls and compliance certifications
  • Integration: Seamless integration with Google Cloud services
  • Cost Efficiency: Flexible pricing models including batch processing options

Gemini Model Capabilities​

  • Multimodal: Process text, images, audio, video, and PDFs in a single request
  • Advanced Reasoning: State-of-the-art language understanding and generation
  • Function Calling: Integrate with external APIs and tools
  • Context Caching: Optimize costs for frequently used content
  • Controlled Generation: Enforce structured output formats

Getting Started​

Prerequisites​

Before you begin, ensure you have:

  1. Google Cloud Project: Create a project if you don't have one
  2. Vertex AI API: Enable the Vertex AI API
  3. Authentication: Set up Application Default Credentials
  4. Python Environment: Python 3.8 or higher

def create_document_cache(documents):

import time

def create_batch_job(input_data_uri, output_bucket):
    """Create a batch processing job for large-scale inference."""
    batch_job = client.batches.create(
        model=MODEL_ID,
        src=input_data_uri,  # JSONL file in Cloud Storage
        config=CreateBatchJobConfig(dest=output_bucket),
    )
    
    return batch_job

def wait_for_batch_completion(batch_job):
    """Wait for batch job to complete."""
    while batch_job.state == "JOB_STATE_RUNNING":
        print("Batch job still running...")
        time.sleep(30)
        batch_job = client.batches.get(name=batch_job.name)
    
    if batch_job.state == "JOB_STATE_SUCCEEDED":
        print("Batch job completed successfully!")
        return batch_job.dest.gcs_uri
    else:
        print(f"Batch job failed: {batch_job.error}")
        return None

# Example: Process thousands of customer reviews
input_data = "gs://your-bucket/customer-reviews.jsonl"

1. Error Handling and Resilience​

import time
from google.api_core import exceptions

def resilient_generate_content(prompt, max_retries=3):
    """Generate content with error handling and retries."""
    for attempt in range(max_retries):
        try:
            response = client.models.generate_content(
                model=MODEL_ID,
                contents=prompt
            )
            return response.text
        
        except exceptions.ResourceExhausted:
            # Rate limit exceeded
            wait_time = (2 ** attempt) + 1
            print(f"Rate limit hit. Waiting {wait_time}s before retry...")
            time.sleep(wait_time)
        
        except exceptions.InvalidArgument as e:
            print(f"Invalid request: {e}")
            break
        
        except Exception as e:
            print(f"Unexpected error on attempt {attempt + 1}: {e}")
            if attempt == max_retries - 1:
                raise
    
    return None

Performance Optimization​

1. Connection Pooling​

import threading

class GeminiClientPool:
    """Connection pool for Gemini clients."""
    
    def __init__(self, project_id, location, pool_size=5):
        self.clients = []
        self.lock = threading.Lock()
        
        for _ in range(pool_size):
            client = genai.Client(
                vertexai=True,
                project=project_id,
                location=location
            )
            self.clients.append(client)
    
    def get_client(self):
        """Get an available client from the pool."""
        with self.lock:
            if self.clients:
                return self.clients.pop()
            else:
                # Create new client if pool is empty
                return genai.Client(vertexai=True)
    
    def return_client(self, client):
        """Return client to the pool."""
        with self.lock:
            self.clients.append(client)

2. Caching Layer​

import redis
import json
import hashlib

class ResponseCache:
    """Redis-based cache for API responses."""
    
    def __init__(self, redis_host='localhost', redis_port=6379):
        self.redis_client = redis.Redis(host=redis_host, port=redis_port)
        self.ttl = 3600  # 1 hour
    
    def get_cache_key(self, prompt, model_id):
        """Generate cache key for prompt."""
        content = f"{prompt}:{model_id}"
        return hashlib.md5(content.encode()).hexdigest()
    
    def get(self, prompt, model_id):
        """Get cached response."""
        key = self.get_cache_key(prompt, model_id)
        cached = self.redis_client.get(key)
        return json.loads(cached) if cached else None
    
    def set(self, prompt, model_id, response):
        """Cache response."""
        key = self.get_cache_key(prompt, model_id)
        self.redis_client.setex(
            key, 
            self.ttl, 
            json.dumps(response)
        )

# Usage with cache
cache = ResponseCache()

def cached_generation(prompt):
    """Generate content with caching."""
    # Check cache first
    cached_response = cache.get(prompt, MODEL_ID)
    if cached_response:
        return cached_response['text']
    
    # Generate new response
    response = client.models.generate_content(
        model=MODEL_ID,
        contents=prompt
    )
    
    # Cache the response
    cache.set(prompt, MODEL_ID, {'text': response.text})
    
    return response.text

    series = monitoring_v3.TimeSeries()
    series.metric.type = "custom.googleapis.com/gemini/generation_duration"
    series.resource.type = "global"
    
    point = monitoring_v3.Point()
    point.value.double_value = duration
    point.interval.end_time.seconds = int(time.time())
    
    series.points = [point]
    
    # Send to monitoring
    project_name = f"projects/{PROJECT_ID}"
    monitoring_client.create_time_series(
        name=project_name, 
        time_series=[series]
    )

    """Comprehensive health check."""

        "vertex_ai": check_vertex_ai_connection(),
        "model_availability": check_model_availability(),

        "dependencies": check_dependencies()

    return {
