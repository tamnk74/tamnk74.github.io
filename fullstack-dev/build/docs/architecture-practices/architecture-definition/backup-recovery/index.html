Backup and Recovery Strategy

This guide provides comprehensive backup and recovery strategies for NestJS microservices and Next.js microfrontend applications, ensuring business continuity and data protection in production environments.

Overview​

A robust backup and recovery strategy is essential for maintaining data integrity, ensuring business continuity, and meeting compliance requirements. This guide covers database backups, application state recovery, disaster recovery planning, and automated recovery procedures.

Database Backup Strategy​

PostgreSQL Backup Implementation​

// src/backup/database-backup.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { Cron, CronExpression } from '@nestjs/schedule';
import { exec } from 'child_process';
import { promisify } from 'util';
import * as fs from 'fs/promises';
import * as path from 'path';
import * as AWS from 'aws-sdk';

const execAsync = promisify(exec);

export interface BackupMetadata {
  id: string;
  timestamp: Date;
  type: 'full' | 'incremental' | 'differential';
  size: number;
  duration: number;
  location: string;
  checksum: string;
  status: 'success' | 'failed' | 'partial';
}

@Injectable()
export class DatabaseBackupService {
  private readonly logger = new Logger(DatabaseBackupService.name);
  private readonly s3: AWS.S3;
  private readonly backupDir: string;

  constructor(private readonly configService: ConfigService) {
    this.s3 = new AWS.S3({
      region: this.configService.get('AWS_REGION'),
      accessKeyId: this.configService.get('AWS_ACCESS_KEY_ID'),
      secretAccessKey: this.configService.get('AWS_SECRET_ACCESS_KEY'),
    });

    this.backupDir = this.configService.get('BACKUP_DIRECTORY', '/tmp/backups');
  }

  // Full backup every day at 2 AM
  @Cron('0 2 * * *')
  async createFullBackup(): Promise<BackupMetadata> {
    const startTime = Date.now();
    const backupId = `full_${Date.now()}`;
    
    this.logger.log('Starting full database backup');

    try {
      // Create backup directory if it doesn't exist
      await fs.mkdir(this.backupDir, { recursive: true });

      const filename = `${backupId}.sql`;
      const filepath = path.join(this.backupDir, filename);

      // Create PostgreSQL dump
      const dumpCommand = this.buildPgDumpCommand(filepath, 'full');
      await execAsync(dumpCommand);

      // Compress the backup
      const compressedPath = `${filepath}.gz`;
      await execAsync(`gzip ${filepath}`);

      // Calculate file size and checksum
      const stats = await fs.stat(compressedPath);
      const checksum = await this.calculateChecksum(compressedPath);

      // Upload to S3
      await this.uploadToS3(compressedPath, `backups/full/${filename}.gz`);

      // Clean up local file
      await fs.unlink(compressedPath);

      const duration = Date.now() - startTime;
      const metadata: BackupMetadata = {
        id: backupId,
        timestamp: new Date(),
        type: 'full',
        size: stats.size,
        duration,
        location: `s3://backups/full/${filename}.gz`,
        checksum,
        status: 'success',
      };

      await this.saveBackupMetadata(metadata);
      this.logger.log(`Full backup completed successfully: ${backupId}`);

      return metadata;
    } catch (error) {
      this.logger.error(`Full backup failed: ${error.message}`);
      throw error;
    }
  }

  // Incremental backup every hour
  @Cron(CronExpression.EVERY_HOUR)
  async createIncrementalBackup(): Promise<BackupMetadata> {
    const startTime = Date.now();
    const backupId = `incremental_${Date.now()}`;
    
    this.logger.log('Starting incremental backup');

    try {
      // Get the last backup timestamp
      const lastBackup = await this.getLastBackupTimestamp();
      
      // Create WAL-E backup or transaction log backup
      const filename = `${backupId}.wal`;
      const filepath = path.join(this.backupDir, filename);

      // Archive WAL files since last backup
      await this.archiveWALFiles(lastBackup, filepath);

      // Compress and upload
      const compressedPath = `${filepath}.gz`;
      await execAsync(`gzip ${filepath}`);

      const stats = await fs.stat(compressedPath);
      const checksum = await this.calculateChecksum(compressedPath);

      await this.uploadToS3(compressedPath, `backups/incremental/${filename}.gz`);
      await fs.unlink(compressedPath);

      const duration = Date.now() - startTime;
      const metadata: BackupMetadata = {
        id: backupId,
        timestamp: new Date(),
        type: 'incremental',
        size: stats.size,
        duration,
        location: `s3://backups/incremental/${filename}.gz`,
        checksum,
        status: 'success',
      };

      await this.saveBackupMetadata(metadata);
      this.logger.log(`Incremental backup completed: ${backupId}`);

      return metadata;
    } catch (error) {
      this.logger.error(`Incremental backup failed: ${error.message}`);
      throw error;
    }
  }

  async restoreFromBackup(
    backupId: string,
    targetDatabase?: string,
  ): Promise<void> {
    this.logger.log(`Starting database restore from backup: ${backupId}`);

    try {
      const metadata = await this.getBackupMetadata(backupId);
      if (!metadata) {
        throw new Error(`Backup not found: ${backupId}`);
      }

      // Download backup from S3
      const localPath = await this.downloadFromS3(metadata.location);

      // Verify checksum
      const downloadedChecksum = await this.calculateChecksum(localPath);
      if (downloadedChecksum !== metadata.checksum) {
        throw new Error('Backup file corrupted - checksum mismatch');
      }

      // Decompress if needed
      const decompressedPath = localPath.replace('.gz', '');
      if (localPath.endsWith('.gz')) {
        await execAsync(`gunzip ${localPath}`);
      }

      // Create target database if specified
      const dbName = targetDatabase || this.configService.get('DB_NAME');
      if (targetDatabase) {
        await this.createDatabase(targetDatabase);
      }

      // Restore the backup
      const restoreCommand = this.buildPgRestoreCommand(decompressedPath, dbName);
      await execAsync(restoreCommand);

      // Clean up
      await fs.unlink(decompressedPath);

      this.logger.log(`Database restore completed successfully: ${backupId}`);
    } catch (error) {
      this.logger.error(`Database restore failed: ${error.message}`);
      throw error;
    }
  }

  private buildPgDumpCommand(filepath: string, type: 'full' | 'schema-only'): string {
    const host = this.configService.get('DB_HOST');
    const port = this.configService.get('DB_PORT');
    const username = this.configService.get('DB_USERNAME');
    const database = this.configService.get('DB_NAME');

    let command = `pg_dump -h ${host} -p ${port} -U ${username} -d ${database}`;
    
    if (type === 'schema-only') {
      command += ' --schema-only';
    } else {
      command += ' --verbose --no-owner --no-acl';
    }

    command += ` > ${filepath}`;
    
    return `PGPASSWORD=${this.configService.get('DB_PASSWORD')} ${command}`;
  }

  private buildPgRestoreCommand(filepath: string, database: string): string {
    const host = this.configService.get('DB_HOST');
    const port = this.configService.get('DB_PORT');
    const username = this.configService.get('DB_USERNAME');

    const command = `psql -h ${host} -p ${port} -U ${username} -d ${database} < ${filepath}`;
    
    return `PGPASSWORD=${this.configService.get('DB_PASSWORD')} ${command}`;
  }

  private async uploadToS3(localPath: string, s3Key: string): Promise<void> {
    const fileContent = await fs.readFile(localPath);
    
    await this.s3.upload({
      Bucket: this.configService.get('BACKUP_S3_BUCKET'),
      Key: s3Key,
      Body: fileContent,
      StorageClass: 'STANDARD_IA', // Infrequent Access for cost optimization
      ServerSideEncryption: 'AES256',
    }).promise();
  }

  private async downloadFromS3(s3Location: string): Promise<string> {
    const [, , bucket, ...keyParts] = s3Location.split('/');
    const key = keyParts.join('/');
    
    const response = await this.s3.getObject({
      Bucket: bucket,
      Key: key,
    }).promise();

    const localPath = path.join(this.backupDir, path.basename(key));
    await fs.writeFile(localPath, response.Body as Buffer);
    
    return localPath;
  }

  private async calculateChecksum(filepath: string): Promise<string> {
    const { stdout } = await execAsync(`sha256sum ${filepath}`);
    return stdout.split(' ')[0];
  }

  private async saveBackupMetadata(metadata: BackupMetadata): Promise<void> {
    // Save to database or metadata store
    // Implementation depends on your metadata storage strategy
  }

  private async getBackupMetadata(backupId: string): Promise<BackupMetadata | null> {
    // Retrieve from database or metadata store
    // Implementation depends on your metadata storage strategy
    return null;
  }

  private async getLastBackupTimestamp(): Promise<Date> {
    // Get timestamp of last backup for incremental backups
    return new Date(Date.now() - 3600000); // Default to 1 hour ago
  }

  private async archiveWALFiles(since: Date, outputPath: string): Promise<void> {
    // Archive PostgreSQL WAL files for incremental backup

  }

  private async createDatabase(name: string): Promise<void> {
    const command = this.buildCreateDatabaseCommand(name);
    await execAsync(command);

import * as fs from 'fs/promises';
import * as path from 'path';

@Injectable()
export class RedisBackupService {
  private readonly logger = new Logger(RedisBackupService.name);
  private readonly redis: Redis;
  private readonly backupDir: string;

  constructor(private readonly configService: ConfigService) {
    this.redis = new Redis({
      host: this.configService.get('REDIS_HOST'),
      port: this.configService.get('REDIS_PORT'),
      password: this.configService.get('REDIS_PASSWORD'),
    });

    this.backupDir = this.configService.get('REDIS_BACKUP_DIR', '/tmp/redis-backups');
  }

  // Redis backup every 6 hours
  @Cron('0 */6 * * *')
  async createRedisBackup(): Promise<void> {
    this.logger.log('Starting Redis backup');

    try {
      await fs.mkdir(this.backupDir, { recursive: true });

      const timestamp = new Date().toISOString().replace(/:/g, '-');
      const filename = `redis-backup-${timestamp}.rdb`;
      const filepath = path.join(this.backupDir, filename);

      // Trigger BGSAVE for point-in-time backup
      await this.redis.bgsave();

      // Wait for background save to complete
      await this.waitForBgsaveCompletion();

      // Copy the RDB file
      const rdbPath = await this.getRdbPath();
      await fs.copyFile(rdbPath, filepath);

      // Compress and upload to S3
      await this.compressAndUpload(filepath, `redis-backups/${filename}`);

      this.logger.log(`Redis backup completed: ${filename}`);
    } catch (error) {
      this.logger.error(`Redis backup failed: ${error.message}`);
      throw error;
    }
  }

  private async waitForBgsaveCompletion(): Promise<void> {
    let isRunning = true;
    while (isRunning) {
      const result = await this.redis.lastsave();
      await new Promise(resolve => setTimeout(resolve, 1000));
      const newResult = await this.redis.lastsave();
      isRunning = result === newResult;
    }
  }

  private async getRdbPath(): Promise<string> {
    const info = await this.redis.config('GET', 'dir');
    const dbfilename = await this.redis.config('GET', 'dbfilename');
    return path.join(info[1], dbfilename[1]);
  }

  }

// scripts/backup-static-assets.ts
import * as fs from 'fs/promises';
import * as path from 'path';
import { S3 } from 'aws-sdk';
import { createHash } from 'crypto';

interface AssetManifest {
  version: string;
  timestamp: string;
  assets: AssetInfo[];
}

interface AssetInfo {
  path: string;
  hash: string;
  size: number;
  mimeType: string;
}

class StaticAssetBackup {
  private s3: S3;
  private bucketName: string;

  constructor() {
    this.s3 = new S3({
      region: process.env.AWS_REGION,
      accessKeyId: process.env.AWS_ACCESS_KEY_ID,
      secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
    });
    this.bucketName = process.env.BACKUP_BUCKET!;
  }

  async backupStaticAssets(buildDir: string): Promise<void> {
    console.log('Starting static asset backup...');

    try {
      // Generate asset manifest
      const manifest = await this.generateAssetManifest(buildDir);
      
      // Upload assets to S3
      await this.uploadAssets(buildDir, manifest);
      
      // Upload manifest
      await this.uploadManifest(manifest);
      
      console.log('Static asset backup completed successfully');
    } catch (error) {
      console.error('Static asset backup failed:', error);
      throw error;
    }
  }

  private async generateAssetManifest(buildDir: string): Promise<AssetManifest> {
    const assets: AssetInfo[] = [];
    
    const traverseDirectory = async (dir: string, relativePath = ''): Promise<void> => {
      const items = await fs.readdir(dir);
      
      for (const item of items) {
        const fullPath = path.join(dir, item);
        const stats = await fs.stat(fullPath);
        
        if (stats.isDirectory()) {
          await traverseDirectory(fullPath, path.join(relativePath, item));
        } else {
          const content = await fs.readFile(fullPath);
          const hash = createHash('sha256').update(content).digest('hex');
          
          assets.push({
            path: path.join(relativePath, item),
            hash,
            size: stats.size,
            mimeType: this.getMimeType(item),
          });
        }
      }
    };

    await traverseDirectory(buildDir);

    return {
      version: process.env.npm_package_version || '1.0.0',
      timestamp: new Date().toISOString(),
      assets,
    };
  }

  private async uploadAssets(buildDir: string, manifest: AssetManifest): Promise<void> {
    for (const asset of manifest.assets) {
      const localPath = path.join(buildDir, asset.path);
      const s3Key = `static-assets/${manifest.version}/${asset.path}`;
      
      const content = await fs.readFile(localPath);
      
      await this.s3.upload({
        Bucket: this.bucketName,
        Key: s3Key,
        Body: content,
        ContentType: asset.mimeType,
        CacheControl: 'public, max-age=31536000', // 1 year
        Metadata: {
          originalHash: asset.hash,
          originalSize: asset.size.toString(),
        },
      }).promise();
      
      console.log(`Uploaded: ${asset.path}`);
    }
  }

  private async uploadManifest(manifest: AssetManifest): Promise<void> {
    const manifestKey = `static-assets/${manifest.version}/manifest.json`;
    
    await this.s3.upload({
      Bucket: this.bucketName,
      Key: manifestKey,
      Body: JSON.stringify(manifest, null, 2),
      ContentType: 'application/json',
    }).promise();
    
    console.log('Manifest uploaded successfully');
  }

  private getMimeType(filename: string): string {
    const ext = path.extname(filename).toLowerCase();
    const mimeTypes: Record<string, string> = {
      '.html': 'text/html',
      '.css': 'text/css',
      '.js': 'application/javascript',
      '.json': 'application/json',
      '.png': 'image/png',
      '.jpg': 'image/jpeg',
      '.jpeg': 'image/jpeg',
      '.gif': 'image/gif',
      '.svg': 'image/svg+xml',
      '.ico': 'image/x-icon',
      '.woff': 'font/woff',
      '.woff2': 'font/woff2',
      '.ttf': 'font/ttf',
      '.eot': 'application/vnd.ms-fontobject',
    };
    
    return mimeTypes[ext] || 'application/octet-stream';
  }
}

        return;

      const latestBackup = recentBackups[0];
      
      if (latestBackup.size < avgSize * 0.5) {
        await this.sendAlert(

          'warning'
        );

Best Practices​

Backup Strategy Guidelines​

  1. 3-2-1 Rule: 3 copies of data, 2 different media types, 1 offsite
  2. Regular Testing: Test backup restoration procedures monthly
  3. Incremental Backups: Use incremental backups to reduce storage costs
  4. Encryption: Encrypt backups both in transit and at rest
  5. Retention Policies: Define clear retention policies based on business requirements
