Security Scanning Guide

This guide provides comprehensive security scanning strategies and automated vulnerability detection for NestJS microservices and Next.js microfrontend applications, covering static analysis, dynamic testing, dependency scanning, and infrastructure security assessment.

Overview​

Security scanning is a critical component of the development lifecycle that helps identify vulnerabilities, security misconfigurations, and potential threats before they reach production. This guide covers automated security scanning pipelines, vulnerability management, and continuous security monitoring across the entire application stack.

    ...results: Partial<SASTScanResult>[]
  ): SASTScanResult {
    const allVulnerabilities: SASTVulnerability[] = [];
    
    for (const result of results) {
      if (result.vulnerabilities) {
        allVulnerabilities.push(...result.vulnerabilities);
      }
    }

    // Deduplicate vulnerabilities based on file, line, and CWE
    const uniqueVulnerabilities = this.deduplicateVulnerabilities(allVulnerabilities);

    return {
      scanId: Date.now().toString(),
      timestamp: new Date(),
      status: 'completed',
      projectId: 'example-project',
      branch: 'main',
      commit: 'abc123',
      vulnerabilities: uniqueVulnerabilities,

      metrics: {

        linesOfCode: 50000,
        filesScanned: 500,
      },

  private deduplicateVulnerabilities(vulnerabilities: SASTVulnerability[]): SASTVulnerability[] {
    const seen = new Set<string>();
    return vulnerabilities.filter(vuln => {
      const key = `${vuln.file}:${vuln.line}:${vuln.cwe}`;
      if (seen.has(key)) {
        return false;
      }
      seen.add(key);
      return true;
    });
  }

  private calculateSummary(vulnerabilities: SASTVulnerability[]) {
    const summary = {
      total: vulnerabilities.length,
      critical: 0,
      high: 0,
      medium: 0,
      low: 0,
      info: 0,
    };

    for (const vuln of vulnerabilities) {
      summary[vuln.severity]++;
    }

    return summary;
  }

  private mapSonarSeverity(severity: string): SASTVulnerability['severity'] {
    const mapping: Record<string, SASTVulnerability['severity']> = {
      BLOCKER: 'critical',
      CRITICAL: 'high',
      MAJOR: 'medium',
      MINOR: 'low',
      INFO: 'info',
    };
    return mapping[severity] || 'low';
  }

  private mapSonarConfidence(severity: string): SASTVulnerability['confidence'] {
    const mapping: Record<string, SASTVulnerability['confidence']> = {
      BLOCKER: 'high',
      CRITICAL: 'high',
      MAJOR: 'medium',
      MINOR: 'low',
      INFO: 'low',
    };
    return mapping[severity] || 'low';
  }

import { Injectable, Logger } from '@nestjs/common';

  scanId: string;
  timestamp: Date;
  projectPath: string;
  packageManager: 'npm' | 'yarn' | 'pnpm';
  totalDependencies: number;
  vulnerabilities: DependencyVulnerability[];
  summary: {
    total: number;
    critical: number;
    high: number;
    medium: number;
    low: number;
  };
  licensing: {
    totalLicenses: number;
    incompatibleLicenses: string[];
    unknownLicenses: string[];
  };

@Injectable()
export class DependencyScannerService {
  private readonly logger = new Logger(DependencyScannerService.name);

  constructor(private readonly httpService: HttpService) {}

  async scanDependencies(projectPath: string): Promise<DependencyScanResult> {
    this.logger.log(`Starting dependency scan for project: ${projectPath}`);

    const packageManager = this.detectPackageManager(projectPath);
    const scanId = `dep-scan-${Date.now()}`;

    try {
      // Run multiple dependency scanners
      const npmAuditResults = await this.runNpmAudit(projectPath);
      const snykResults = await this.runSnykScan(projectPath);
      const owaspResults = await this.runOwaspDependencyCheck(projectPath);

      // Aggregate results
      const vulnerabilities = this.aggregateDependencyVulnerabilities([
        ...npmAuditResults,
        ...snykResults,
        ...owaspResults,
      ]);

      const totalDependencies = await this.countDependencies(projectPath, packageManager);
      const licensing = await this.analyzeLicensing(projectPath);

      const result: DependencyScanResult = {
        scanId,
        timestamp: new Date(),
        projectPath,
        packageManager,
        totalDependencies,
        vulnerabilities,
        summary: this.calculateDependencySummary(vulnerabilities),
        licensing,
      };

      this.logger.log(`Dependency scan completed. Found ${vulnerabilities.length} vulnerabilities`);
      return result;
    } catch (error) {
      this.logger.error('Dependency scan failed', error);
      throw error;
    }
  }

  private detectPackageManager(projectPath: string): 'npm' | 'yarn' | 'pnpm' {
    if (fs.existsSync(path.join(projectPath, 'pnpm-lock.yaml'))) {
      return 'pnpm';
    } else if (fs.existsSync(path.join(projectPath, 'yarn.lock'))) {
      return 'yarn';
    } else {
      return 'npm';
    }
  }

  private async runNpmAudit(projectPath: string): Promise<DependencyVulnerability[]> {
    this.logger.debug('Running npm audit');

    try {
      const { spawn } = require('child_process');
      const auditProcess = spawn('npm', ['audit', '--json'], {
        cwd: projectPath,
        stdio: 'pipe',
      });

      let auditOutput = '';
      auditProcess.stdout.on('data', (data: Buffer) => {
        auditOutput += data.toString();
      });

      await new Promise((resolve, reject) => {
        auditProcess.on('close', (code: number) => {
          if (code === 0 || code === 1) { // npm audit returns 1 when vulnerabilities found
            resolve(code);
          } else {
            reject(new Error(`npm audit failed with code ${code}`));
          }
        });
      });

      const auditData = JSON.parse(auditOutput);
      return this.parseNpmAuditResults(auditData);
    } catch (error) {
      this.logger.warn('npm audit failed', error);
      return [];
    }
  }

  private parseNpmAuditResults(auditData: any): DependencyVulnerability[] {
    const vulnerabilities: DependencyVulnerability[] = [];

    if (auditData.vulnerabilities) {
      for (const [packageName, vuln] of Object.entries(auditData.vulnerabilities)) {
        const vulnData = vuln as any;
        
        for (const advisory of vulnData.via) {
          if (typeof advisory === 'object' && advisory.cwe) {
            vulnerabilities.push({
              id: advisory.id?.toString() || `npm-${Date.now()}`,
              cve: advisory.cve || '',
              packageName,
              packageVersion: vulnData.range || '',
              severity: this.mapNpmSeverity(advisory.severity),
              cvssScore: advisory.cvss?.score || 0,
              description: advisory.title || '',
              references: advisory.references || [],
              patchedVersions: vulnData.fixAvailable?.version ? [vulnData.fixAvailable.version] : [],
              vulnerableVersions: advisory.vulnerable_versions || '',
              publishedDate: new Date(advisory.created || Date.now()),
              updatedDate: new Date(advisory.updated || Date.now()),
              source: 'npm-audit',
            });
          }
        }
      }
    }

    return vulnerabilities;
  }

  ): DependencyVulnerability[] {
    // Deduplicate vulnerabilities based on CVE and package
    const seen = new Set<string>();
    return vulnerabilities.filter(vuln => {

      const dependencies = Object.keys(packageJson.dependencies || {});
      const devDependencies = Object.keys(packageJson.devDependencies || {});
      
      return dependencies.length + devDependencies.length;
    } catch (error) {

        // Test SQL injection on various endpoints
        const testUrls = [
          `${targetUrl}/api/users?id=${encodeURIComponent(payload)}`,
          `${targetUrl}/search?q=${encodeURIComponent(payload)}`,
        ];

        for (const testUrl of testUrls) {
          const startTime = Date.now();
          const response = await this.httpService.get(testUrl).toPromise();
          const responseTime = Date.now() - startTime;

          // Check for SQL injection indicators
          if (this.detectSQLInjection(response.data, responseTime)) {
            vulnerabilities.push({
              id: `sql-inj-${Date.now()}`,
              name: 'SQL Injection',
              description: 'Potential SQL injection vulnerability detected',
              severity: 'high',
              confidence: 'medium',
              url: testUrl,
              method: 'GET',
              parameter: 'query parameter',
              evidence: `Payload: ${payload}`,
              cwe: 'CWE-89',
              owasp: 'A03:2021',
              remediation: 'Use parameterized queries and input validation',
              references: ['https://owasp.org/www-community/attacks/SQL_Injection'],
            });
          }
        }
      } catch (error) {
        // Analyze error responses for SQL injection indicators
        if (this.detectSQLErrorMessages(error.message)) {
          vulnerabilities.push({
            id: `sql-err-${Date.now()}`,
            name: 'SQL Injection (Error-based)',
            description: 'SQL error messages revealed in response',
            severity: 'medium',
            confidence: 'high',
            url: targetUrl,
            method: 'GET',
            parameter: 'query parameter',
            evidence: error.message,
            cwe: 'CWE-89',
            owasp: 'A03:2021',
            remediation: 'Implement proper error handling and input validation',
            references: ['https://owasp.org/www-community/attacks/SQL_Injection'],
          });
        }
      }
    }

    return vulnerabilities;
  }

  private async checkSecurityHeaders(targetUrl: string): Promise<DASTVulnerability[]> {
    const vulnerabilities: DASTVulnerability[] = [];

    try {
      const response = await this.httpService.get(targetUrl).toPromise();
      const headers = response.headers;

      const securityHeaders = [
        'content-security-policy',
        'x-frame-options',
        'x-content-type-options',
        'strict-transport-security',
        'referrer-policy',
      ];

      for (const header of securityHeaders) {
        if (!headers[header]) {
          vulnerabilities.push({
            id: `header-${header}-${Date.now()}`,
            name: `Missing Security Header: ${header}`,
            description: `Security header ${header} is not set`,
            severity: this.getHeaderSeverity(header),
            confidence: 'high',
            url: targetUrl,
            method: 'GET',
            parameter: 'HTTP Header',
            evidence: `Missing header: ${header}`,
            cwe: 'CWE-693',
            owasp: 'A05:2021',
            remediation: `Implement ${header} security header`,
            references: ['https://owasp.org/www-project-secure-headers/'],
          });
        }
      }
    } catch (error) {
      this.logger.warn('Failed to check security headers', error);
    }

    return vulnerabilities;
  }

        }
      } catch (error) {

      }
    }

    return vulnerabilities;
  }

  // Helper methods
  private detectSQLInjection(responseData: any, responseTime: number): boolean {
    const sqlErrorPatterns = [
      /sql syntax/i,
      /mysql_fetch/i,
      /ora-\d{5}/i,
      /sqlstate/i,
    ];

    // Check for SQL error patterns
    const hasErrorPattern = sqlErrorPatterns.some(pattern => 
      pattern.test(JSON.stringify(responseData))
    );

    // Check for time-based injection (significant delay)
    const hasTimeDelay = responseTime > 5000;

    return hasErrorPattern || hasTimeDelay;
  }

  private detectSQLErrorMessages(errorMessage: string): boolean {
    const sqlErrorKeywords = [
      'syntax error',
      'mysql',
      'oracle',
      'postgresql',
      'sqlite',
      'sql server',
    ];

    return sqlErrorKeywords.some(keyword => 
      errorMessage.toLowerCase().includes(keyword)
    );
  }

  private getHeaderSeverity(header: string): DASTVulnerability['severity'] {
    const severityMap: Record<string, DASTVulnerability['severity']> = {
      'content-security-policy': 'high',
      'strict-transport-security': 'high',
      'x-frame-options': 'medium',
      'x-content-type-options': 'medium',
      'referrer-policy': 'low',
    };

    return severityMap[header] || 'medium';
  }

  private deduplicateDASTVulnerabilities(vulnerabilities: DASTVulnerability[]): DASTVulnerability[] {
    const seen = new Set<string>();
    return vulnerabilities.filter(vuln => {
      const key = `${vuln.url}:${vuln.method}:${vuln.cwe}`;
      if (seen.has(key)) {
        return false;
      }
      seen.add(key);
      return true;
    });
  }

  private calculateDASTSummary(vulnerabilities: DASTVulnerability[]) {
    const summary = {
      total: vulnerabilities.length,
      critical: 0,
      high: 0,
      medium: 0,
      low: 0,
      info: 0,
    };

    for (const vuln of vulnerabilities) {
      summary[vuln.severity]++;
    }

    return summary;
  }
}

    // Extract tag
    const parts = imageRef.split(':');
    return parts.length > 1 ? parts[1].split('@')[0] : 'latest';
  }

# .github/workflows/security-scanning.yml
name: Security Scanning

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 2 * * *' # Daily at 2 AM

env:
  REGISTRY: ghcr.io

jobs:
  sast-scan:
    name: Static Application Security Testing
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4
      with:
        fetch-depth: 0

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'

    - name: Install dependencies
      run: npm ci

    - name: Run ESLint Security Plugin
      run: npx eslint . --ext .js,.ts,.tsx --format json --output-file eslint-results.json
      continue-on-error: true

    - name: Run Semgrep
      uses: returntocorp/semgrep-action@v1
      with:
        config: >-
          p/security-audit
          p/secrets
          p/typescript
          p/react
          p/nodejs
        generateSarif: "1"

    - name: Run CodeQL Analysis
      uses: github/codeql-action/init@v3
      with:
        languages: javascript,typescript
        queries: security-and-quality,security-extended

    - name: Perform CodeQL Analysis
      uses: github/codeql-action/analyze@v3

    - name: Upload SAST Results
      uses: actions/upload-artifact@v4
      with:
        name: sast-results
        path: |
          eslint-results.json
          semgrep.sarif
        if-no-files-found: warn

  dependency-scan:
    name: Dependency Vulnerability Scanning
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'

    - name: Run npm audit
      run: |
        npm audit --audit-level=moderate --json > npm-audit.json || true
        
    - name: Run Snyk to check for vulnerabilities
      uses: snyk/actions/node@master
      env:
        SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
      with:
        args: --severity-threshold=high --json-file-output=snyk-results.json
      continue-on-error: true

    - name: Run OWASP Dependency Check
      uses: dependency-check/Dependency-Check_Action@main
      with:
        project: 'microservice-app'
        path: '.'
        format: 'JSON'
        out: 'dependency-check-report'

    - name: Upload Dependency Scan Results
      uses: actions/upload-artifact@v4
      with:
        name: dependency-scan-results
        path: |
          npm-audit.json
          snyk-results.json
          dependency-check-report/
        if-no-files-found: warn

  container-scan:
    name: Container Security Scanning
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Build Docker image
      run: |
        docker build -t test-image:${{ github.sha }} .

    - name: Run Trivy vulnerability scanner
      uses: aquasecurity/trivy-action@master
      with:
        image-ref: 'test-image:${{ github.sha }}'
        format: 'sarif'
        output: 'trivy-results.sarif'

    - name: Run Docker Bench Security
      run: |
        git clone https://github.com/docker/docker-bench-security.git
        cd docker-bench-security
        sudo sh docker-bench-security.sh -c container_images | tee ../docker-bench-results.txt

    - name: Upload Container Scan Results
      uses: actions/upload-artifact@v4
      with:
        name: container-scan-results
        path: |
          trivy-results.sarif
          docker-bench-results.txt
        if-no-files-found: warn

  dast-scan:
    name: Dynamic Application Security Testing
    runs-on: ubuntu-latest
    needs: [sast-scan]
    if: github.event_name == 'schedule' || github.ref == 'refs/heads/main'
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'

    - name: Install dependencies and build
      run: |
        npm ci
        npm run build

    - name: Start application
      run: |
        npm start &
        sleep 30
        curl -f http://localhost:3000/health || exit 1

    - name: Run OWASP ZAP Baseline Scan
      uses: zaproxy/action-baseline@v0.10.0
      with:
        target: 'http://localhost:3000'
        rules_file_name: '.zap/rules.tsv'
        cmd_options: '-a'

    - name: Upload DAST Results
      uses: actions/upload-artifact@v4
      with:
        name: dast-results
        path: report_html.html
        if-no-files-found: warn

  secrets-scan:
    name: Secrets Detection
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v4
      with:
        fetch-depth: 0

    - name: Run TruffleHog
      uses: trufflesecurity/trufflehog@main
      with:
        path: ./
        base: main
        head: HEAD
        extra_args: --debug --only-verified

    - name: Run GitLeaks
      uses: gitleaks/gitleaks-action@v2
      env:
        GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

    name: Generate Security Report
    runs-on: ubuntu-latest
    needs: [sast-scan, dependency-scan, container-scan, dast-scan, secrets-scan]
    if: always()

        echo "# Security Scan Report" > security-report.md

        echo "" >> security-report.md
        
        echo "## Summary" >> security-report.md
        # Add logic to parse and summarize all scan results
        
        echo "## SAST Results" >> security-report.md
        if [ -f sast-results/eslint-results.json ]; then
          echo "ESLint issues found:" >> security-report.md
          jq '.length' sast-results/eslint-results.json >> security-report.md

        name: security-report

      uses: actions/github-script@v7
      with:
        script: |

          const report = fs.readFileSync('security-report.md', 'utf8');
          
          github.rest.issues.createComment({
            issue_number: context.issue.number,
            owner: context.repo.owner,
            repo: context.repo.repo,
            body: report
          });
