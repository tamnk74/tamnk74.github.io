Resource Scaling Guide

This guide provides comprehensive resource scaling strategies and automated scaling mechanisms for NestJS microservices and Next.js microfrontend applications, covering horizontal scaling, vertical scaling, auto-scaling policies, and cloud-native scaling solutions.

Overview​

Resource scaling ensures applications can handle varying load levels efficiently while maintaining performance and cost-effectiveness. This guide covers both reactive scaling (responding to current load) and predictive scaling (anticipating future load) across infrastructure, application, and database layers.

Auto-Scaling Service Implementation​

NestJS Auto-Scaling Service​

// src/scaling/auto-scaling.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { EventEmitter2 } from '@nestjs/event-emitter';
import { Cron, CronExpression } from '@nestjs/schedule';
import { HttpService } from '@nestjs/axios';
import { firstValueFrom } from 'rxjs';

export interface ScalingMetrics {
  timestamp: Date;
  cpu: {
    current: number;
    average: number;
    target: number;
  };
  memory: {
    current: number;
    average: number;
    target: number;
  };
  requests: {
    perSecond: number;
    average: number;
    target: number;
  };
  responseTime: {
    current: number;
    p95: number;
    target: number;
  };
  queueDepth: {
    current: number;
    average: number;
    target: number;
  };
  errorRate: {
    current: number;
    average: number;
    threshold: number;
  };
  connections: {
    active: number;
    average: number;
    limit: number;
  };
}

export interface ScalingDecision {
  action: 'scale-up' | 'scale-down' | 'no-action';
  type: 'horizontal' | 'vertical';
  currentInstances: number;
  targetInstances: number;
  reason: string;
  confidence: number;
  estimatedDuration: number;
  costImpact: number;
}

export interface ScalingPolicy {
  name: string;
  service: string;
  enabled: boolean;
  minInstances: number;
  maxInstances: number;
  targetCpuUtilization: number;
  targetMemoryUtilization: number;
  targetRequestsPerSecond: number;
  targetResponseTime: number;
  scaleUpCooldown: number;
  scaleDownCooldown: number;
  scaleUpPolicy: {
    threshold: number;
    periodSeconds: number;
    step: number;
  };
  scaleDownPolicy: {
    threshold: number;
    periodSeconds: number;
    step: number;
  };
  predictiveScaling: {
    enabled: boolean;
    lookAheadMinutes: number;
    confidenceThreshold: number;
  };
}

@Injectable()
export class AutoScalingService {
  private readonly logger = new Logger(AutoScalingService.name);
  private scalingPolicies: Map<string, ScalingPolicy> = new Map();
  private lastScalingAction: Map<string, Date> = new Map();
  private metricsHistory: Map<string, ScalingMetrics[]> = new Map();

  constructor(
    private readonly eventEmitter: EventEmitter2,
    private readonly httpService: HttpService
  ) {
    this.initializeScalingPolicies();
  }

  private initializeScalingPolicies(): void {
    const policies: ScalingPolicy[] = [

        scaleDownCooldown: 600, // 10 minutes
        scaleUpPolicy: {
          threshold: 80,
          periodSeconds: 60,
          step: 2,
        },
        scaleDownPolicy: {
          threshold: 50,
          periodSeconds: 300,
          step: 1,
        },
        predictiveScaling: {
          enabled: true,
          lookAheadMinutes: 30,
          confidenceThreshold: 0.8,
        },
      },
      {
        name: 'user-service-scaling',
        service: 'user-service',
        enabled: true,
        minInstances: 1,
        maxInstances: 10,
        targetCpuUtilization: 65,
        targetMemoryUtilization: 75,
        targetRequestsPerSecond: 50,
        targetResponseTime: 150,
        scaleUpCooldown: 180,
        scaleDownCooldown: 480,
        scaleUpPolicy: {
          threshold: 75,

        },
        scaleDownPolicy: {

        },
        predictiveScaling: {
          enabled: false,
          lookAheadMinutes: 15,
          confidenceThreshold: 0.7,
        },
      },
      {
        name: 'product-service-scaling',
        service: 'product-service',
        enabled: true,
        minInstances: 2,
        maxInstances: 15,
        targetCpuUtilization: 60,
        targetMemoryUtilization: 70,
        targetRequestsPerSecond: 75,
        targetResponseTime: 180,
        scaleUpCooldown: 240,
        scaleDownCooldown: 540,
        scaleUpPolicy: {
          threshold: 70,
          periodSeconds: 90,
          step: 2,
        },
        scaleDownPolicy: {
          threshold: 35,
          periodSeconds: 360,
          step: 1,
        },
        predictiveScaling: {
          enabled: true,
          lookAheadMinutes: 45,
          confidenceThreshold: 0.75,
        },
      },
    ];

    policies.forEach(policy => {
      this.scalingPolicies.set(policy.service, policy);
    });

    this.logger.log(`Initialized ${policies.length} scaling policies`);
  }

  @Cron(CronExpression.EVERY_30_SECONDS)
  async evaluateScaling(): Promise<void> {
    this.logger.debug('Evaluating auto-scaling decisions');

    for (const [serviceName, policy] of this.scalingPolicies) {
      if (!policy.enabled) continue;

      try {
        const metrics = await this.collectServiceMetrics(serviceName);
        this.updateMetricsHistory(serviceName, metrics);

        const decision = await this.makeScalingDecision(serviceName, policy, metrics);
        
        if (decision.action !== 'no-action') {
          this.logger.log(
            `Scaling decision for ${serviceName}: ${decision.action} from ${decision.currentInstances} to ${decision.targetInstances} (${decision.reason})`
          );

          await this.executeScalingDecision(serviceName, decision);
        }
      } catch (error) {
        this.logger.error(`Failed to evaluate scaling for ${serviceName}`, error);
      }
    }
  }

  private async collectServiceMetrics(serviceName: string): Promise<ScalingMetrics> {
    try {
      // Collect metrics from Prometheus/monitoring system
      const [cpuMetrics, memoryMetrics, requestMetrics, responseTimeMetrics, queueMetrics] = await Promise.all([
        this.getCpuMetrics(serviceName),
        this.getMemoryMetrics(serviceName),
        this.getRequestMetrics(serviceName),
        this.getResponseTimeMetrics(serviceName),
        this.getQueueMetrics(serviceName),
      ]);

      const metrics: ScalingMetrics = {
        timestamp: new Date(),
        cpu: cpuMetrics,
        memory: memoryMetrics,
        requests: requestMetrics,
        responseTime: responseTimeMetrics,
        queueDepth: queueMetrics,
        errorRate: await this.getErrorRateMetrics(serviceName),
        connections: await this.getConnectionMetrics(serviceName),
      };

      return metrics;
    } catch (error) {
      this.logger.error(`Failed to collect metrics for ${serviceName}`, error);
      throw error;
    }
  }

  private async getCpuMetrics(serviceName: string): Promise<ScalingMetrics['cpu']> {
    // Query Prometheus for CPU metrics
    const query = `avg(rate(container_cpu_usage_seconds_total{container="${serviceName}"}[5m])) * 100`;
    const result = await this.queryPrometheus(query);
    
    return {
      current: parseFloat(result.data.result[0]?.value[1] || '0'),
      average: await this.getAverageMetric(serviceName, 'cpu', 300), // 5 minutes
      target: this.scalingPolicies.get(serviceName)?.targetCpuUtilization || 70,
    };
  }

  private async getMemoryMetrics(serviceName: string): Promise<ScalingMetrics['memory']> {
    const query = `avg(container_memory_usage_bytes{container="${serviceName}"} / container_spec_memory_limit_bytes{container="${serviceName}"}) * 100`;
    const result = await this.queryPrometheus(query);
    
    return {
      current: parseFloat(result.data.result[0]?.value[1] || '0'),
      average: await this.getAverageMetric(serviceName, 'memory', 300),
      target: this.scalingPolicies.get(serviceName)?.targetMemoryUtilization || 80,
    };
  }

  private async getRequestMetrics(serviceName: string): Promise<ScalingMetrics['requests']> {
    const query = `sum(rate(http_requests_total{service="${serviceName}"}[1m]))`;
    const result = await this.queryPrometheus(query);
    
    return {
      perSecond: parseFloat(result.data.result[0]?.value[1] || '0'),
      average: await this.getAverageMetric(serviceName, 'requests', 300),
      target: this.scalingPolicies.get(serviceName)?.targetRequestsPerSecond || 100,
    };
  }

  private async getResponseTimeMetrics(serviceName: string): Promise<ScalingMetrics['responseTime']> {
    const currentQuery = `avg(http_request_duration_seconds{service="${serviceName}"}) * 1000`;
    const p95Query = `histogram_quantile(0.95, rate(http_request_duration_seconds_bucket{service="${serviceName}"}[5m])) * 1000`;
    
    const [currentResult, p95Result] = await Promise.all([
      this.queryPrometheus(currentQuery),
      this.queryPrometheus(p95Query),
    ]);
    
    return {
      current: parseFloat(currentResult.data.result[0]?.value[1] || '0'),
      p95: parseFloat(p95Result.data.result[0]?.value[1] || '0'),
      target: this.scalingPolicies.get(serviceName)?.targetResponseTime || 200,
    };
  }

  private async getQueueMetrics(serviceName: string): Promise<ScalingMetrics['queueDepth']> {
    const query = `avg(queue_depth{service="${serviceName}"})`;
    const result = await this.queryPrometheus(query);
    
    return {
      current: parseFloat(result.data.result[0]?.value[1] || '0'),
      average: await this.getAverageMetric(serviceName, 'queue', 300),
      target: 30, // Default queue depth target
    };
  }

  private async getErrorRateMetrics(serviceName: string): Promise<ScalingMetrics['errorRate']> {
    const query = `(sum(rate(http_requests_total{service="${serviceName}",status=~"5.."}[5m])) / sum(rate(http_requests_total{service="${serviceName}"}[5m]))) * 100`;
    const result = await this.queryPrometheus(query);
    
    return {
      current: parseFloat(result.data.result[0]?.value[1] || '0'),
      average: await this.getAverageMetric(serviceName, 'error_rate', 300),
      threshold: 5, // 5% error rate threshold
    };
  }

  private async getConnectionMetrics(serviceName: string): Promise<ScalingMetrics['connections']> {
    const query = `avg(active_connections{service="${serviceName}"})`;
    const result = await this.queryPrometheus(query);
    
    return {
      active: parseFloat(result.data.result[0]?.value[1] || '0'),
      average: await this.getAverageMetric(serviceName, 'connections', 300),
      limit: 1000, // Default connection limit
    };
  }

  private async queryPrometheus(query: string): Promise<any> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/api/v1/query', {
          params: { query },
          baseURL: process.env.PROMETHEUS_URL || 'http://prometheus:9090',
        })
      );
      return response.data;
    } catch (error) {
      this.logger.error(`Prometheus query failed: ${query}`, error);
      throw error;
    }
  }

  private async getAverageMetric(serviceName: string, metricType: string, periodSeconds: number): Promise<number> {
    const history = this.metricsHistory.get(serviceName) || [];
    const cutoff = new Date(Date.now() - periodSeconds * 1000);
    const recentMetrics = history.filter(m => m.timestamp >= cutoff);
    
    if (recentMetrics.length === 0) return 0;
    
    const sum = recentMetrics.reduce((acc, m) => {
      switch (metricType) {
        case 'cpu': return acc + m.cpu.current;
        case 'memory': return acc + m.memory.current;
        case 'requests': return acc + m.requests.perSecond;
        case 'queue': return acc + m.queueDepth.current;
        case 'error_rate': return acc + m.errorRate.current;
        case 'connections': return acc + m.connections.active;
        default: return acc;
      }
    }, 0);
    
    return sum / recentMetrics.length;
  }

  private updateMetricsHistory(serviceName: string, metrics: ScalingMetrics): void {
    const history = this.metricsHistory.get(serviceName) || [];
    history.push(metrics);
    
    // Keep only last hour of metrics
    const cutoff = new Date(Date.now() - 3600 * 1000);
    const filteredHistory = history.filter(m => m.timestamp >= cutoff);
    
    this.metricsHistory.set(serviceName, filteredHistory);
  }

  private async makeScalingDecision(
    serviceName: string,
    policy: ScalingPolicy,
    metrics: ScalingMetrics
  ): Promise<ScalingDecision> {
    const currentInstances = await this.getCurrentInstanceCount(serviceName);
    
    // Check cooldown periods
    const lastAction = this.lastScalingAction.get(serviceName);
    if (lastAction) {
      const timeSinceLastAction = Date.now() - lastAction.getTime();
      const cooldownPeriod = policy.scaleUpCooldown * 1000; // Convert to milliseconds
      
      if (timeSinceLastAction < cooldownPeriod) {
        return {
          action: 'no-action',
          type: 'horizontal',
          currentInstances,
          targetInstances: currentInstances,
          reason: 'Cooldown period active',
          confidence: 1.0,
          estimatedDuration: 0,
          costImpact: 0,
        };
      }
    }

    // Evaluate scaling triggers
    const scaleUpTriggers = this.evaluateScaleUpTriggers(policy, metrics);
    const scaleDownTriggers = this.evaluateScaleDownTriggers(policy, metrics);

    // Predictive scaling
    let predictiveRecommendation: Partial<ScalingDecision> = {};
    if (policy.predictiveScaling.enabled) {
      predictiveRecommendation = await this.getPredictiveScalingRecommendation(serviceName, policy);
    }

    // Make decision based on triggers and prediction
    if (scaleUpTriggers.length > 0 && currentInstances < policy.maxInstances) {
      const step = Math.min(policy.scaleUpPolicy.step, policy.maxInstances - currentInstances);
      return {
        action: 'scale-up',
        type: 'horizontal',
        currentInstances,
        targetInstances: currentInstances + step,
        reason: `Scale up triggers: ${scaleUpTriggers.join(', ')}`,
        confidence: this.calculateConfidence(scaleUpTriggers, metrics),
        estimatedDuration: 120, // 2 minutes
        costImpact: this.calculateCostImpact(step, 'up'),
      };
    }

    if (scaleDownTriggers.length > 0 && currentInstances > policy.minInstances) {
      const step = Math.min(policy.scaleDownPolicy.step, currentInstances - policy.minInstances);
      return {
        action: 'scale-down',
        type: 'horizontal',
        currentInstances,
        targetInstances: currentInstances - step,
        reason: `Scale down triggers: ${scaleDownTriggers.join(', ')}`,
        confidence: this.calculateConfidence(scaleDownTriggers, metrics),
        estimatedDuration: 180, // 3 minutes
        costImpact: this.calculateCostImpact(step, 'down'),
      };
    }

    // Consider predictive scaling if no immediate triggers
    if (predictiveRecommendation.action && predictiveRecommendation.action !== 'no-action') {
      return {
        ...predictiveRecommendation,
        currentInstances,
        reason: `Predictive scaling: ${predictiveRecommendation.reason}`,
      } as ScalingDecision;
    }

    return {
      action: 'no-action',
      type: 'horizontal',
      currentInstances,
      targetInstances: currentInstances,
      reason: 'No scaling triggers met',
      confidence: 1.0,
      estimatedDuration: 0,
      costImpact: 0,
    };
  }

  private evaluateScaleUpTriggers(policy: ScalingPolicy, metrics: ScalingMetrics): string[] {
    const triggers: string[] = [];

    if (metrics.cpu.current > policy.scaleUpPolicy.threshold) {
      triggers.push(`CPU ${metrics.cpu.current}% > ${policy.scaleUpPolicy.threshold}%`);
    }

    if (metrics.memory.current > policy.scaleUpPolicy.threshold) {
      triggers.push(`Memory ${metrics.memory.current}% > ${policy.scaleUpPolicy.threshold}%`);
    }

    if (metrics.requests.perSecond > policy.targetRequestsPerSecond * 1.2) {
      triggers.push(`Requests ${metrics.requests.perSecond}/s > ${policy.targetRequestsPerSecond * 1.2}/s`);
    }

    if (metrics.responseTime.p95 > policy.targetResponseTime * 1.5) {
      triggers.push(`Response time ${metrics.responseTime.p95}ms > ${policy.targetResponseTime * 1.5}ms`);
    }

    if (metrics.queueDepth.current > 50) {
      triggers.push(`Queue depth ${metrics.queueDepth.current} > 50`);
    }

    if (metrics.errorRate.current > metrics.errorRate.threshold) {
      triggers.push(`Error rate ${metrics.errorRate.current}% > ${metrics.errorRate.threshold}%`);
    }

    return triggers;
  }

  private evaluateScaleDownTriggers(policy: ScalingPolicy, metrics: ScalingMetrics): string[] {
    const triggers: string[] = [];

    if (metrics.cpu.average < policy.scaleDownPolicy.threshold && 
        metrics.memory.average < policy.scaleDownPolicy.threshold) {
      triggers.push(`Low resource usage: CPU ${metrics.cpu.average}%, Memory ${metrics.memory.average}%`);
    }

    if (metrics.requests.perSecond < policy.targetRequestsPerSecond * 0.5) {
      triggers.push(`Low request rate: ${metrics.requests.perSecond}/s < ${policy.targetRequestsPerSecond * 0.5}/s`);
    }

    if (metrics.queueDepth.current < 5) {
      triggers.push(`Low queue depth: ${metrics.queueDepth.current} < 5`);
    }

    return triggers;
  }

  private async getPredictiveScalingRecommendation(
    serviceName: string,
    policy: ScalingPolicy
  ): Promise<Partial<ScalingDecision>> {
    // Implement predictive scaling based on historical patterns
    // This is a simplified implementation
    const history = this.metricsHistory.get(serviceName) || [];
    
    if (history.length < 10) {
      return { action: 'no-action', reason: 'Insufficient historical data' };
    }

    // Analyze trends in the last lookAheadMinutes
    const lookBackPeriod = policy.predictiveScaling.lookAheadMinutes * 60 * 1000;
    const cutoff = new Date(Date.now() - lookBackPeriod);
    const recentHistory = history.filter(m => m.timestamp >= cutoff);

    if (recentHistory.length < 3) {
      return { action: 'no-action', reason: 'Insufficient recent data' };
    }

    // Calculate trend
    const cpuTrend = this.calculateTrend(recentHistory.map(m => m.cpu.current));
    const requestTrend = this.calculateTrend(recentHistory.map(m => m.requests.perSecond));

    // Predict future load
    const currentCpu = recentHistory[recentHistory.length - 1].cpu.current;
    const predictedCpu = currentCpu + (cpuTrend * policy.predictiveScaling.lookAheadMinutes);

    const currentRequests = recentHistory[recentHistory.length - 1].requests.perSecond;
    const predictedRequests = currentRequests + (requestTrend * policy.predictiveScaling.lookAheadMinutes);

    // Make prediction-based recommendation
    if (predictedCpu > policy.targetCpuUtilization * 1.2 || 
        predictedRequests > policy.targetRequestsPerSecond * 1.2) {
      return {
        action: 'scale-up',
        type: 'horizontal',
        targetInstances: await this.getCurrentInstanceCount(serviceName) + 1,
        reason: `Predicted high load: CPU ${predictedCpu.toFixed(1)}%, Requests ${predictedRequests.toFixed(1)}/s`,
        confidence: policy.predictiveScaling.confidenceThreshold,
        estimatedDuration: 120,
        costImpact: this.calculateCostImpact(1, 'up'),
      };
    }

    return { action: 'no-action', reason: 'No scaling predicted' };
  }

  private calculateTrend(values: number[]): number {
    if (values.length < 2) return 0;

    // Simple linear regression slope
    const n = values.length;
    const sumX = (n * (n - 1)) / 2;
    const sumY = values.reduce((a, b) => a + b, 0);
    const sumXY = values.reduce((sum, y, x) => sum + x * y, 0);
    const sumXX = values.reduce((sum, _, x) => sum + x * x, 0);

    return (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
  }

  private calculateConfidence(triggers: string[], metrics: ScalingMetrics): number {
    // Calculate confidence based on number of triggers and severity
    const baseConfidence = Math.min(triggers.length / 3, 1.0); // More triggers = higher confidence
    
    // Adjust based on metric severity
    let severityMultiplier = 1.0;
    
    if (metrics.cpu.current > 90 || metrics.memory.current > 90) {
      severityMultiplier = 1.2;
    } else if (metrics.cpu.current < 30 && metrics.memory.current < 30) {
      severityMultiplier = 0.8;
    }

    return Math.min(baseConfidence * severityMultiplier, 1.0);
  }

  private calculateCostImpact(instanceDelta: number, direction: 'up' | 'down'): number {
    // Simplified cost calculation (would integrate with cloud pricing APIs)
    const costPerInstancePerHour = 0.10; // $0.10 per hour
    const multiplier = direction === 'up' ? 1 : -1;
    
    return instanceDelta * costPerInstancePerHour * multiplier;
  }

  private async getCurrentInstanceCount(serviceName: string): Promise<number> {
    try {
      // Query Kubernetes API for current replica count
      const query = `count(up{job="${serviceName}"})`;
      const result = await this.queryPrometheus(query);
      return parseInt(result.data.result[0]?.value[1] || '1');
    } catch (error) {
      this.logger.warn(`Failed to get instance count for ${serviceName}`, error);
      return 1; // Default to 1 instance
    }
  }

  private async executeScalingDecision(serviceName: string, decision: ScalingDecision): Promise<void> {
    try {
      this.logger.log(`Executing scaling decision for ${serviceName}: ${decision.action} to ${decision.targetInstances} instances`);

      // Update Kubernetes deployment
      await this.updateKubernetesDeployment(serviceName, decision.targetInstances);

      // Record scaling action
      this.lastScalingAction.set(serviceName, new Date());

      // Emit scaling event
      this.eventEmitter.emit('scaling.executed', {
        service: serviceName,
        decision,
        timestamp: new Date(),
      });

      // Wait for scaling to complete
      await this.waitForScalingCompletion(serviceName, decision.targetInstances, decision.estimatedDuration);

      this.logger.log(`Scaling completed for ${serviceName}`);
    } catch (error) {
      this.logger.error(`Failed to execute scaling decision for ${serviceName}`, error);
      throw error;
    }
  }

  private async updateKubernetesDeployment(serviceName: string, targetReplicas: number): Promise<void> {

    // This is a placeholder implementation

    // kubectl patch deployment ${serviceName} -p '{"spec":{"replicas":${targetReplicas}}}'
    // Or use Kubernetes JavaScript client
  }

  private async waitForScalingCompletion(
    serviceName: string,
    targetInstances: number,
    maxWaitTime: number
  ): Promise<void> {
    const startTime = Date.now();
    
    while (Date.now() - startTime < maxWaitTime * 1000) {
      const currentInstances = await this.getCurrentInstanceCount(serviceName);
      
      if (currentInstances === targetInstances) {
        return;
      }
      
      await new Promise(resolve => setTimeout(resolve, 5000)); // Wait 5 seconds
    }
    
    this.logger.warn(`Scaling did not complete within ${maxWaitTime} seconds for ${serviceName}`);
  }

  // Public API methods
  async getScalingPolicies(): Promise<ScalingPolicy[]> {
    return Array.from(this.scalingPolicies.values());
  }

  async updateScalingPolicy(serviceName: string, updates: Partial<ScalingPolicy>): Promise<void> {
    const policy = this.scalingPolicies.get(serviceName);
    if (!policy) {
      throw new Error(`Scaling policy not found for service: ${serviceName}`);
    }

    const updatedPolicy = { ...policy, ...updates };
    this.scalingPolicies.set(serviceName, updatedPolicy);

    this.logger.log(`Updated scaling policy for ${serviceName}`);
    this.eventEmitter.emit('scaling.policy.updated', {
      service: serviceName,
      policy: updatedPolicy,
    });
  }

  async getServiceMetrics(serviceName: string): Promise<ScalingMetrics | null> {
    try {
      return await this.collectServiceMetrics(serviceName);
    } catch (error) {
      this.logger.error(`Failed to get metrics for ${serviceName}`, error);
      return null;
    }
  }

  async getMetricsHistory(serviceName: string, hours: number = 1): Promise<ScalingMetrics[]> {
    const history = this.metricsHistory.get(serviceName) || [];
    const cutoff = new Date(Date.now() - hours * 3600 * 1000);
    return history.filter(m => m.timestamp >= cutoff);
  }

  async manualScale(serviceName: string, targetInstances: number, reason: string): Promise<void> {
    const policy = this.scalingPolicies.get(serviceName);
    if (!policy) {
      throw new Error(`Scaling policy not found for service: ${serviceName}`);
    }

    if (targetInstances < policy.minInstances || targetInstances > policy.maxInstances) {
      throw new Error(`Target instances ${targetInstances} outside allowed range [${policy.minInstances}, ${policy.maxInstances}]`);
    }

    const currentInstances = await this.getCurrentInstanceCount(serviceName);
    const decision: ScalingDecision = {
      action: targetInstances > currentInstances ? 'scale-up' : targetInstances < currentInstances ? 'scale-down' : 'no-action',
      type: 'horizontal',
      currentInstances,
      targetInstances,
      reason: `Manual scaling: ${reason}`,
      confidence: 1.0,
      estimatedDuration: Math.abs(targetInstances - currentInstances) * 30, // 30 seconds per instance
      costImpact: this.calculateCostImpact(Math.abs(targetInstances - currentInstances), targetInstances > currentInstances ? 'up' : 'down'),
    };

    if (decision.action !== 'no-action') {
      await this.executeScalingDecision(serviceName, decision);
    }
  }
}

  minConnections: number;
  maxConnections: number;
  targetUtilization: number;
  scaleUpThreshold: number;
  scaleDownThreshold: number;
  scaleStep: number;
  cooldownPeriod: number;
}

@Injectable()
export class ConnectionPoolScalingService {
  private readonly logger = new Logger(ConnectionPoolScalingService.name);
  private lastScalingAction: Date | null = null;
  
  constructor(private readonly dataSource: DataSource) {}

  async evaluatePoolScaling(config: PoolScalingConfig): Promise<void> {
    const metrics = await this.getConnectionPoolMetrics();
    
    // Check cooldown period
    if (this.lastScalingAction) {
      const timeSinceLastAction = Date.now() - this.lastScalingAction.getTime();
      if (timeSinceLastAction < config.cooldownPeriod * 1000) {
        this.logger.debug('Connection pool scaling in cooldown period');
        return;
      }
    }

    // Evaluate scaling decision
    if (metrics.utilizationRate > config.scaleUpThreshold && 
        metrics.maxConnections < config.maxConnections) {
      
      const newMaxConnections = Math.min(
        metrics.maxConnections + config.scaleStep,
        config.maxConnections
      );
      
      await this.scaleConnectionPool(newMaxConnections);
      this.logger.log(`Scaled up connection pool to ${newMaxConnections} connections`);
      
    } else if (metrics.utilizationRate < config.scaleDownThreshold && 
               metrics.maxConnections > config.minConnections) {
      
      const newMaxConnections = Math.max(
        metrics.maxConnections - config.scaleStep,
        config.minConnections
      );
      
      await this.scaleConnectionPool(newMaxConnections);
      this.logger.log(`Scaled down connection pool to ${newMaxConnections} connections`);
    }
  }

interface EdgeScalingDecision {
  action: 'scale-up' | 'scale-down' | 'no-action';
  regions: string[];
  cacheConfiguration: {
    ttl: number;
    strategies: string[];
  };
  reason: string;
}

class FrontendScalingService {
  private logger = console;

  async evaluateFrontendScaling(): Promise<EdgeScalingDecision> {
    const metrics = await this.getFrontendMetrics();
    
    // Evaluate if additional edge locations are needed
    if (metrics.responseTime > 500 || metrics.cacheHitRatio < 0.8) {
      const underservedRegions = await this.identifyUnderservedRegions(metrics);
      
      if (underservedRegions.length > 0) {
        return {
          action: 'scale-up',
          regions: underservedRegions,
          cacheConfiguration: {
            ttl: 3600,
            strategies: ['stale-while-revalidate', 'cache-first'],
          },
          reason: `High response time (${metrics.responseTime}ms) or low cache hit ratio (${metrics.cacheHitRatio})`,
        };
      }
    }

    // Evaluate if edge locations can be reduced
    if (metrics.requestsPerSecond < 10 && metrics.cacheHitRatio > 0.95) {
      const overProvisionedRegions = await this.identifyOverProvisionedRegions(metrics);
      
      if (overProvisionedRegions.length > 0) {
        return {
          action: 'scale-down',
          regions: overProvisionedRegions,
          cacheConfiguration: {
            ttl: 1800,
            strategies: ['cache-first'],
          },
          reason: `Low request rate (${metrics.requestsPerSecond}/s) with high cache efficiency`,
        };
      }
    }

    return {
      action: 'no-action',
      regions: [],
      cacheConfiguration: {
        ttl: 3600,
        strategies: ['stale-while-revalidate'],
      },
      reason: 'No scaling needed',
