Infrastructure Monitoring Guide

This guide provides comprehensive infrastructure monitoring strategies for NestJS microservices and Next.js microfrontend applications, covering system health, performance metrics, and automated alerting.

Overview​

Infrastructure monitoring ensures system reliability, performance optimization, and proactive issue detection. This guide covers monitoring setup for containers, Kubernetes clusters, databases, and cloud services using industry-standard tools and practices.

Monitoring Architecture​

Multi-Layer Monitoring Strategy​

NestJS Prometheus Integration​

// src/monitoring/prometheus.service.ts
import { Injectable } from '@nestjs/common';
import { register, Counter, Histogram, Gauge } from 'prom-client';

@Injectable()
export class PrometheusService {
  private readonly httpRequestsTotal = new Counter({
    name: 'http_requests_total',
    help: 'Total number of HTTP requests',
    labelNames: ['method', 'route', 'status_code'],
    registers: [register],
  });

  private readonly httpRequestDuration = new Histogram({
    name: 'http_request_duration_seconds',
    help: 'Duration of HTTP requests in seconds',
    labelNames: ['method', 'route'],
    buckets: [0.1, 0.3, 0.5, 0.7, 1, 3, 5, 7, 10],
    registers: [register],
  });

  private readonly databaseConnectionsActive = new Gauge({
    name: 'database_connections_active',
    help: 'Number of active database connections',
    registers: [register],
  });

  private readonly redisConnectionsActive = new Gauge({
    name: 'redis_connections_active',
    help: 'Number of active Redis connections',
    registers: [register],
  });

  private readonly businessMetrics = {
    usersRegistered: new Counter({
      name: 'users_registered_total',
      help: 'Total number of user registrations',
      labelNames: ['source'],
      registers: [register],
    }),
    ordersCreated: new Counter({
      name: 'orders_created_total',
      help: 'Total number of orders created',
      labelNames: ['status'],
      registers: [register],
    }),
    paymentProcessed: new Counter({
      name: 'payments_processed_total',
      help: 'Total number of payments processed',
      labelNames: ['status', 'provider'],
      registers: [register],
    }),
    activeUsers: new Gauge({
      name: 'active_users_current',
      help: 'Current number of active users',
      registers: [register],
    }),
  };

  recordHttpRequest(method: string, route: string, statusCode: number, duration: number): void {
    this.httpRequestsTotal.inc({ method, route, status_code: statusCode.toString() });
    this.httpRequestDuration.observe({ method, route }, duration);
  }

  updateDatabaseConnections(count: number): void {
    this.databaseConnectionsActive.set(count);
  }

  updateRedisConnections(count: number): void {
    this.redisConnectionsActive.set(count);
  }

  recordUserRegistration(source: string): void {
    this.businessMetrics.usersRegistered.inc({ source });
  }

  recordOrderCreation(status: string): void {
    this.businessMetrics.ordersCreated.inc({ status });
  }

  recordPayment(status: string, provider: string): void {
    this.businessMetrics.paymentProcessed.inc({ status, provider });
  }

  updateActiveUsers(count: number): void {
    this.businessMetrics.activeUsers.set(count);
  }

  getMetrics(): string {
    return register.metrics();
  }
}

Monitoring Middleware​

// src/middleware/monitoring.middleware.ts
import { Injectable, NestMiddleware } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';
import { PrometheusService } from '../monitoring/prometheus.service';

@Injectable()
export class MonitoringMiddleware implements NestMiddleware {
  constructor(private readonly prometheusService: PrometheusService) {}

  use(req: Request, res: Response, next: NextFunction): void {
    const startTime = Date.now();

    res.on('finish', () => {
      const duration = (Date.now() - startTime) / 1000;
      const route = req.route?.path || req.path;
      
      this.prometheusService.recordHttpRequest(
        req.method,
        route,
        res.statusCode,
        duration
      );
    });

    next();
  }
}

Health Check Implementation​

// src/health/health.controller.ts
import { Controller, Get } from '@nestjs/common';
import { HealthCheckService, HttpHealthIndicator, TypeOrmHealthIndicator } from '@nestjs/terminus';
import { PrometheusService } from '../monitoring/prometheus.service';

@Controller('health')
export class HealthController {
  constructor(
    private health: HealthCheckService,
    private http: HttpHealthIndicator,
    private db: TypeOrmHealthIndicator,
    private prometheus: PrometheusService,
  ) {}

  @Get()
  check() {
    return this.health.check([
      () => this.db.pingCheck('database'),
      () => this.http.pingCheck('redis', 'redis://localhost:6379'),
    ]);
  }

  @Get('ready')
  ready() {
    return this.health.check([
      () => this.db.pingCheck('database'),
      () => this.http.pingCheck('redis', 'redis://localhost:6379'),
      () => this.http.pingCheck('external-api', 'https://api.external-service.com/health'),
    ]);
  }

  @Get('metrics')
  metrics() {
    return this.prometheus.getMetrics();
  }
}

Client-Side Monitoring​

// lib/monitoring/client-metrics.ts
interface PerformanceMetric {
  name: string;
  value: number;
  labels?: Record<string, string>;
}

class ClientMetrics {
  private metrics: PerformanceMetric[] = [];

  // Core Web Vitals
  recordCLS(value: number, route: string): void {
    this.metrics.push({
      name: 'web_vitals_cls',
      value,
      labels: { route, metric: 'cls' },
    });
  }

  recordFID(value: number, route: string): void {
    this.metrics.push({
      name: 'web_vitals_fid',
      value,
      labels: { route, metric: 'fid' },
    });
  }

  recordLCP(value: number, route: string): void {
    this.metrics.push({
      name: 'web_vitals_lcp',
      value,
      labels: { route, metric: 'lcp' },
    });
  }

  recordFCP(value: number, route: string): void {
    this.metrics.push({
      name: 'web_vitals_fcp',
      value,
      labels: { route, metric: 'fcp' },
    });
  }

  recordTTFB(value: number, route: string): void {
    this.metrics.push({
      name: 'web_vitals_ttfb',
      value,
      labels: { route, metric: 'ttfb' },
    });
  }

  // Custom Business Metrics
  recordUserInteraction(action: string, component: string): void {
    this.metrics.push({
      name: 'user_interactions_total',
      value: 1,
      labels: { action, component },
    });
  }

  recordAPICall(endpoint: string, status: number, duration: number): void {
    this.metrics.push({
      name: 'api_calls_total',
      value: 1,
      labels: { endpoint, status: status.toString() },
    });
    
    this.metrics.push({
      name: 'api_call_duration_seconds',
      value: duration / 1000,
      labels: { endpoint },
    });
  }

  recordPageView(route: string, loadTime: number): void {
    this.metrics.push({
      name: 'page_views_total',
      value: 1,
      labels: { route },
    });
    
    this.metrics.push({
      name: 'page_load_time_seconds',
      value: loadTime / 1000,
      labels: { route },
    });
  }

  recordError(type: string, message: string, component?: string): void {
    this.metrics.push({
      name: 'client_errors_total',
      value: 1,
      labels: { type, component: component || 'unknown' },
    });
  }

  // Send metrics to backend
  async sendMetrics(): Promise<void> {
    if (this.metrics.length === 0) return;

    try {
      await fetch('/api/metrics', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ metrics: this.metrics }),
      });
      
      this.metrics = []; // Clear after sending
    } catch (error) {
      console.error('Failed to send metrics:', error);
    }
  }

  // Automatic metric collection
  collectWebVitals(): void {
    if (typeof window === 'undefined') return;

    // Collect Core Web Vitals
    import('web-vitals').then(({ getCLS, getFID, getFCP, getLCP, getTTFB }) => {
      const route = window.location.pathname;
      
      getCLS((metric) => this.recordCLS(metric.value, route));
      getFID((metric) => this.recordFID(metric.value, route));
      getFCP((metric) => this.recordFCP(metric.value, route));
      getLCP((metric) => this.recordLCP(metric.value, route));
      getTTFB((metric) => this.recordTTFB(metric.value, route));
    });

    // Page load metrics
    window.addEventListener('load', () => {
      const loadTime = performance.timing.loadEventEnd - performance.timing.navigationStart;
      this.recordPageView(window.location.pathname, loadTime);
    });

    // Unhandled errors
    window.addEventListener('error', (event) => {
      this.recordError('javascript', event.error?.message || 'Unknown error', event.filename);
    });

    // Unhandled promise rejections
    window.addEventListener('unhandledrejection', (event) => {
      this.recordError('promise', event.reason?.message || 'Unhandled promise rejection');
    });

    // Send metrics periodically
    setInterval(() => this.sendMetrics(), 30000); // Every 30 seconds
  }
}

export const clientMetrics = new ClientMetrics();

Next.js Monitoring Hook​

// hooks/use-monitoring.ts
import { useEffect } from 'react';
import { useRouter } from 'next/router';
import { clientMetrics } from '../lib/monitoring/client-metrics';

export const useMonitoring = () => {
  const router = useRouter();

  useEffect(() => {
    // Initialize monitoring
    clientMetrics.collectWebVitals();

    // Track route changes
    const handleRouteChange = (url: string) => {
      const startTime = Date.now();
      
      const handleRouteComplete = () => {
        const loadTime = Date.now() - startTime;
        clientMetrics.recordPageView(url, loadTime);
        router.events.off('routeChangeComplete', handleRouteComplete);
      };
      
      router.events.on('routeChangeComplete', handleRouteComplete);
    };

    router.events.on('routeChangeStart', handleRouteChange);

    return () => {
      router.events.off('routeChangeStart', handleRouteChange);
    };
  }, [router]);

  const trackUserInteraction = (action: string, component: string) => {
    clientMetrics.recordUserInteraction(action, component);
  };

  const trackAPICall = (endpoint: string, status: number, duration: number) => {
    clientMetrics.recordAPICall(endpoint, status, duration);
  };

  const trackError = (type: string, message: string, component?: string) => {
    clientMetrics.recordError(type, message, component);
  };

  return {
    trackUserInteraction,
    trackAPICall,
    trackError,
  };
};

# monitoring/rules/infrastructure.yml
groups:
- name: infrastructure
  rules:
  - alert: HighCPUUsage
    expr: 100 - (avg by(instance) (irate(node_cpu_seconds_total{mode="idle"}[5m])) * 100) > 80
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "High CPU usage detected"
      description: "CPU usage is above 80% for more than 5 minutes on {{ $labels.instance }}"

  - alert: HighMemoryUsage
    expr: (node_memory_MemTotal_bytes - node_memory_MemAvailable_bytes) / node_memory_MemTotal_bytes * 100 > 85
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "High memory usage detected"
      description: "Memory usage is above 85% for more than 5 minutes on {{ $labels.instance }}"

  - alert: DiskSpaceLow
    expr: 100 - ((node_filesystem_avail_bytes * 100) / node_filesystem_size_bytes) > 90
    for: 5m
    labels:
      severity: critical
    annotations:
      summary: "Disk space critically low"
      description: "Disk usage is above 90% on {{ $labels.instance }} filesystem {{ $labels.mountpoint }}"

  - alert: PodCrashLooping
    expr: rate(kube_pod_container_status_restarts_total[5m]) > 0
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "Pod is crash looping"
      description: "Pod {{ $labels.namespace }}/{{ $labels.pod }} is restarting frequently"

- name: applications
  rules:
  - alert: HighErrorRate
    expr: rate(http_requests_total{status_code=~"5.."}[5m]) / rate(http_requests_total[5m]) > 0.05
    for: 5m
    labels:
      severity: critical
    annotations:
      summary: "High application error rate"
      description: "Error rate is above 5% for service {{ $labels.kubernetes_name }}"

  - alert: HighResponseTime

    for: 5m

      severity: warning

      summary: "High response time detected"

    labels:
      severity: warning
    annotations:
      summary: "High database connection usage"

  - alert: ServiceDown
    expr: up == 0
    for: 1m

      severity: critical
    annotations:
      summary: "Service is down"
      description: "Service {{ $labels.job }} is down"

      severity: warning

Database Performance Monitor​

// src/monitoring/database-monitor.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import { DataSource } from 'typeorm';
import { PrometheusService } from './prometheus.service';

@Injectable()
export class DatabaseMonitorService {
  private readonly logger = new Logger(DatabaseMonitorService.name);

  constructor(
    private readonly dataSource: DataSource,
    private readonly prometheus: PrometheusService,
  ) {}

  @Cron(CronExpression.EVERY_30_SECONDS)
  async monitorDatabasePerformance(): Promise<void> {
    try {
      // Monitor active connections
      const connectionCount = await this.getActiveConnections();
      this.prometheus.updateDatabaseConnections(connectionCount);

      // Monitor slow queries
      const slowQueries = await this.getSlowQueries();
      this.logSlowQueries(slowQueries);

      // Monitor database size
      const dbSize = await this.getDatabaseSize();
      this.prometheus.updateDatabaseSize(dbSize);

      // Monitor table sizes
      const tableSizes = await this.getTableSizes();
      this.updateTableSizeMetrics(tableSizes);

    } catch (error) {
      this.logger.error('Database monitoring failed', error);
    }
  }

  private async getActiveConnections(): Promise<number> {
    const result = await this.dataSource.query(`
      SELECT count(*) as connections 
      FROM pg_stat_activity 
      WHERE state = 'active'
    `);
    return parseInt(result[0].connections);
  }

  private async getSlowQueries(): Promise<any[]> {
    return this.dataSource.query(`
      SELECT query, mean_exec_time, calls 
      FROM pg_stat_statements 
      WHERE mean_exec_time > 1000 
      ORDER BY mean_exec_time DESC 
      LIMIT 10
    `);
  }

  private async getDatabaseSize(): Promise<number> {
    const result = await this.dataSource.query(`
      SELECT pg_database_size(current_database()) as size
    `);
    return parseInt(result[0].size);
  }

  private async getTableSizes(): Promise<Array<{table: string, size: number}>> {
    return this.dataSource.query(`
      SELECT 
        schemaname,
        tablename as table,
        pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as size,
        pg_total_relation_size(schemaname||'.'||tablename) as size_bytes
      FROM pg_tables 
      WHERE schemaname = 'public'
      ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC
    `);
  }

  private logSlowQueries(queries: any[]): void {
    queries.forEach(query => {
      this.logger.warn(`Slow query detected: ${query.query.substring(0, 100)}... (${query.mean_exec_time}ms avg, ${query.calls} calls)`);
    });
  }

  }
}
