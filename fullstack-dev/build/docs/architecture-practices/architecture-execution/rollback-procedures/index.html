Rollback Procedures Guide

This guide provides comprehensive rollback strategies and automated procedures for NestJS microservices and Next.js microfrontend applications, covering database rollbacks, application version rollbacks, configuration rollbacks, and emergency recovery procedures.

Overview​

Rollback procedures are critical safety mechanisms that enable rapid recovery from failed deployments, problematic releases, or system issues. This guide covers automated rollback triggers, manual rollback procedures, and comprehensive recovery strategies across different failure scenarios.

Rollback Decision Engine​

// src/rollback/rollback-decision.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { EventEmitter2 } from '@nestjs/event-emitter';
import { Cron, CronExpression } from '@nestjs/schedule';

export interface RollbackCriteria {
  name: string;
  weight: number;
  threshold: number;
  metric: () => Promise<number>;
  enabled: boolean;
}

export interface RollbackDecision {
  shouldRollback: boolean;
  reason: string;
  severity: 'low' | 'medium' | 'high' | 'critical';
  triggeredCriteria: Array<{
    name: string;
    value: number;
    threshold: number;
    weight: number;
  }>;
  timestamp: Date;
}

export interface DeploymentInfo {
  version: string;
  deploymentId: string;
  startTime: Date;
  environment: string;
  services: string[];
}

@Injectable()
export class RollbackDecisionService {
  private readonly logger = new Logger(RollbackDecisionService.name);
  private currentDeployment: DeploymentInfo | null = null;
  private rollbackCriteria: RollbackCriteria[] = [];
  private isMonitoring = false;
  private stabilizationPeriod = 10 * 60 * 1000; // 10 minutes

  constructor(private readonly eventEmitter: EventEmitter2) {
    this.initializeRollbackCriteria();
  }

  private initializeRollbackCriteria(): void {
    this.rollbackCriteria = [
      {
        name: 'error-rate',
        weight: 30,
        threshold: 5, // 5% error rate
        metric: () => this.getErrorRate(),
        enabled: true,
      },
      {
        name: 'response-time',
        weight: 20,
        threshold: 2000, // 2 seconds
        metric: () => this.getAverageResponseTime(),
        enabled: true,
      },
      {
        name: 'throughput-drop',
        weight: 25,
        threshold: 50, // 50% throughput drop
        metric: () => this.getThroughputDrop(),
        enabled: true,
      },
      {
        name: 'memory-usage',
        weight: 15,
        threshold: 90, // 90% memory usage
        metric: () => this.getMemoryUsage(),
        enabled: true,
      },
      {
        name: 'cpu-usage',
        weight: 10,
        threshold: 85, // 85% CPU usage
        metric: () => this.getCpuUsage(),
        enabled: true,
      },
      {
        name: 'database-errors',
        weight: 35,
        threshold: 2, // 2% database error rate
        metric: () => this.getDatabaseErrorRate(),
        enabled: true,
      },
      {
        name: 'external-api-failures',
        weight: 20,
        threshold: 10, // 10% external API failure rate
        metric: () => this.getExternalApiFailureRate(),
        enabled: true,
      },
    ];
  }

  async startDeploymentMonitoring(deployment: DeploymentInfo): Promise<void> {
    this.logger.log(`Starting rollback monitoring for deployment ${deployment.deploymentId}`);
    
    this.currentDeployment = deployment;
    this.isMonitoring = true;

    // Set up automatic monitoring termination after stabilization period
    setTimeout(() => {
      if (this.isMonitoring && this.currentDeployment?.deploymentId === deployment.deploymentId) {
        this.logger.log(`Deployment ${deployment.deploymentId} stabilized. Reducing monitoring frequency.`);
        this.isMonitoring = false;
      }
    }, this.stabilizationPeriod);

    // Emit monitoring started event
    this.eventEmitter.emit('rollback.monitoring.started', deployment);
  }

  async stopDeploymentMonitoring(): Promise<void> {
    if (this.currentDeployment) {
      this.logger.log(`Stopping rollback monitoring for deployment ${this.currentDeployment.deploymentId}`);
      this.isMonitoring = false;
      this.currentDeployment = null;
      this.eventEmitter.emit('rollback.monitoring.stopped');
    }
  }

  @Cron(CronExpression.EVERY_30_SECONDS)
  async evaluateRollbackCriteria(): Promise<void> {
    if (!this.isMonitoring || !this.currentDeployment) {
      return;
    }

    try {
      const decision = await this.makeRollbackDecision();
      
      if (decision.shouldRollback) {
        this.logger.warn(
          `Rollback criteria met for deployment ${this.currentDeployment.deploymentId}: ${decision.reason}`
        );
        
        this.eventEmitter.emit('rollback.decision.made', {
          deployment: this.currentDeployment,
          decision,
        });

        // Stop monitoring to prevent multiple rollback triggers
        this.isMonitoring = false;
      } else {
        this.logger.debug(`Rollback criteria evaluation passed for deployment ${this.currentDeployment.deploymentId}`);
      }
    } catch (error) {
      this.logger.error('Error evaluating rollback criteria', error);
    }
  }

  private async makeRollbackDecision(): Promise<RollbackDecision> {
    const triggeredCriteria = [];
    let totalWeight = 0;
    let triggeredWeight = 0;

    for (const criteria of this.rollbackCriteria) {
      if (!criteria.enabled) continue;

      totalWeight += criteria.weight;

      try {
        const value = await criteria.metric();
        
        if (value > criteria.threshold) {
          triggeredCriteria.push({
            name: criteria.name,
            value,
            threshold: criteria.threshold,
            weight: criteria.weight,
          });
          triggeredWeight += criteria.weight;
        }
      } catch (error) {
        this.logger.error(`Error evaluating criteria ${criteria.name}`, error);
      }
    }

    const triggerPercentage = (triggeredWeight / totalWeight) * 100;
    const shouldRollback = triggerPercentage >= 30; // 30% weight threshold

    let severity: RollbackDecision['severity'] = 'low';
    if (triggerPercentage >= 70) severity = 'critical';
    else if (triggerPercentage >= 50) severity = 'high';
    else if (triggerPercentage >= 30) severity = 'medium';

    const reason = shouldRollback
      ? `${triggeredCriteria.length} criteria exceeded thresholds (${triggerPercentage.toFixed(1)}% weight)`
      : 'All criteria within acceptable ranges';

    return {
      shouldRollback,
      reason,
      severity,
      triggeredCriteria,
      timestamp: new Date(),
    };
  }

  // Metric collection methods
  private async getErrorRate(): Promise<number> {
    // Implementation would integrate with monitoring system
    // This is a placeholder returning mock data
    return Math.random() * 10; // 0-10% error rate
  }

  private async getAverageResponseTime(): Promise<number> {
    return Math.random() * 3000 + 500; // 500-3500ms response time
  }

  private async getThroughputDrop(): Promise<number> {
    return Math.random() * 100; // 0-100% throughput drop
  }

  private async getMemoryUsage(): Promise<number> {
    return Math.random() * 100; // 0-100% memory usage
  }

  private async getCpuUsage(): Promise<number> {
    return Math.random() * 100; // 0-100% CPU usage
  }

  private async getDatabaseErrorRate(): Promise<number> {
    return Math.random() * 5; // 0-5% database error rate
  }

  private async getExternalApiFailureRate(): Promise<number> {
    return Math.random() * 20; // 0-20% external API failure rate
  }

  getCurrentDeployment(): DeploymentInfo | null {
    return this.currentDeployment;
  }

  isCurrentlyMonitoring(): boolean {
    return this.isMonitoring;
  }

  getRollbackCriteria(): RollbackCriteria[] {
    return this.rollbackCriteria;
  }

  updateCriteria(name: string, updates: Partial<RollbackCriteria>): void {
    const criteria = this.rollbackCriteria.find(c => c.name === name);
    if (criteria) {
      Object.assign(criteria, updates);
      this.logger.log(`Updated rollback criteria ${name}`);
    }
  }
}

import { Injectable, Logger } from '@nestjs/common';
import { EventEmitter2, OnEvent } from '@nestjs/event-emitter';
import { HttpService } from '@nestjs/axios';
import { firstValueFrom } from 'rxjs';

export interface RollbackPlan {
  type: 'blue-green' | 'canary' | 'rolling' | 'emergency';
  targetVersion: string;
  estimatedDuration: number;
  steps: RollbackStep[];
  prerequisites: string[];
  rollbackOrder: string[];
}

export interface RollbackStep {
  id: string;
  name: string;
  type: 'application' | 'database' | 'configuration' | 'infrastructure';
  action: string;
  timeout: number;
  retries: number;
  rollbackAction?: string;
}

export interface RollbackExecution {
  id: string;
  plan: RollbackPlan;
  status: 'planned' | 'in-progress' | 'completed' | 'failed' | 'aborted';
  startTime: Date;
  endTime?: Date;
  currentStep?: string;
  executedSteps: Array<{
    stepId: string;
    status: 'pending' | 'in-progress' | 'completed' | 'failed';
    startTime: Date;
    endTime?: Date;
    error?: string;
  }>;
  error?: string;
}

@Injectable()
export class RollbackExecutionService {
  private readonly logger = new Logger(RollbackExecutionService.name);
  private currentExecution: RollbackExecution | null = null;

  constructor(
    private readonly eventEmitter: EventEmitter2,
    private readonly httpService: HttpService
  ) {}

  @OnEvent('rollback.decision.made')
  async handleAutomaticRollback(event: {
    deployment: any;
    decision: any;
  }): Promise<void> {
    this.logger.warn('Automatic rollback triggered', event.decision);
    
    try {
      const plan = await this.createRollbackPlan(event.deployment, 'emergency');
      await this.executeRollback(plan);
    } catch (error) {
      this.logger.error('Automatic rollback failed', error);
      this.eventEmitter.emit('rollback.failed', {
        deployment: event.deployment,
        error: error.message,
      });
    }
  }

  async executeManualRollback(
    deploymentId: string,
    targetVersion: string,
    type: RollbackPlan['type'] = 'blue-green'
  ): Promise<RollbackExecution> {
    this.logger.log(`Manual rollback requested for deployment ${deploymentId} to version ${targetVersion}`);

    if (this.currentExecution && this.currentExecution.status === 'in-progress') {
      throw new Error('Another rollback is currently in progress');
    }

    const deployment = await this.getDeploymentInfo(deploymentId);
    const plan = await this.createRollbackPlan(deployment, type, targetVersion);
    
    return this.executeRollback(plan);
  }

  private async createRollbackPlan(
    deployment: any,
    type: RollbackPlan['type'],
    targetVersion?: string
  ): Promise<RollbackPlan> {
    // Get the previous stable version if targetVersion not specified
    if (!targetVersion) {
      targetVersion = await this.getPreviousStableVersion(deployment.environment);
    }

    const basePlan: RollbackPlan = {
      type,
      targetVersion,
      estimatedDuration: this.getEstimatedDuration(type),
      steps: [],
      prerequisites: [],
      rollbackOrder: deployment.services.reverse(), // Reverse order of services
    };

    switch (type) {
      case 'emergency':
        return this.createEmergencyRollbackPlan(basePlan, deployment);
      case 'blue-green':
        return this.createBlueGreenRollbackPlan(basePlan, deployment);
      case 'canary':
        return this.createCanaryRollbackPlan(basePlan, deployment);
      case 'rolling':
        return this.createRollingRollbackPlan(basePlan, deployment);
      default:
        throw new Error(`Unsupported rollback type: ${type}`);
    }
  }

  private async executeRollback(plan: RollbackPlan): Promise<RollbackExecution> {
    const execution: RollbackExecution = {
      id: `rollback-${Date.now()}`,
      plan,
      status: 'planned',
      startTime: new Date(),
      executedSteps: [],
    };

    this.currentExecution = execution;

    try {
      this.logger.log(`Starting rollback execution ${execution.id} with plan type ${plan.type}`);
      
      // Verify prerequisites
      await this.verifyPrerequisites(plan.prerequisites);
      
      execution.status = 'in-progress';
      this.eventEmitter.emit('rollback.started', execution);

      // Execute each step
      for (const step of plan.steps) {
        execution.currentStep = step.id;
        
        const stepExecution = {
          stepId: step.id,
          status: 'in-progress' as const,
          startTime: new Date(),
        };
        
        execution.executedSteps.push(stepExecution);

        this.logger.log(`Executing rollback step: ${step.name}`);

        try {
          await this.executeStep(step);
          
          stepExecution.status = 'completed';
          stepExecution.endTime = new Date();
          
          this.logger.log(`Completed rollback step: ${step.name}`);
        } catch (error) {
          stepExecution.status = 'failed';
          stepExecution.endTime = new Date();
          stepExecution.error = error.message;
          
          this.logger.error(`Failed rollback step: ${step.name}`, error);
          
          // Attempt step rollback if available
          if (step.rollbackAction) {
            this.logger.log(`Attempting to rollback failed step: ${step.name}`);
            try {
              await this.executeStepRollback(step);
            } catch (rollbackError) {
              this.logger.error(`Step rollback also failed: ${step.name}`, rollbackError);
            }
          }
          
          throw error;
        }
      }

      execution.status = 'completed';
      execution.endTime = new Date();
      
      this.logger.log(`Rollback execution ${execution.id} completed successfully`);
      this.eventEmitter.emit('rollback.completed', execution);

    } catch (error) {
      execution.status = 'failed';
      execution.endTime = new Date();
      execution.error = error.message;
      
      this.logger.error(`Rollback execution ${execution.id} failed`, error);
      this.eventEmitter.emit('rollback.failed', execution);
      
      throw error;
    } finally {
      this.currentExecution = null;
    }

    return execution;
  }

  private async verifyPrerequisites(prerequisites: string[]): Promise<void> {
    for (const prerequisite of prerequisites) {
      this.logger.debug(`Verifying prerequisite: ${prerequisite}`);
      
      switch (prerequisite) {
        case 'validate-blue-environment-exists':
          await this.validateBlueEnvironmentExists();
          break;
        case 'verify-database-backup':
          await this.verifyDatabaseBackup();
          break;
        case 'verify-previous-version-availability':
          await this.verifyPreviousVersionAvailability();
          break;
        case 'verify-stable-version-exists':
          await this.verifyStableVersionExists();
          break;
        default:
          this.logger.warn(`Unknown prerequisite: ${prerequisite}`);
      }
    }
  }

  private async executeStep(step: RollbackStep): Promise<void> {
    const startTime = Date.now();
    let attempt = 0;

    while (attempt <= step.retries) {
      try {
        await this.performStepAction(step);
        return;
      } catch (error) {
        attempt++;
        
        if (attempt > step.retries) {
          throw error;
        }
        
        this.logger.warn(`Step ${step.name} failed (attempt ${attempt}/${step.retries + 1}), retrying...`);
        await new Promise(resolve => setTimeout(resolve, 1000 * attempt)); // Exponential backoff
      }
      
      if (Date.now() - startTime > step.timeout) {
        throw new Error(`Step ${step.name} timed out after ${step.timeout}ms`);
      }
    }
  }

  private async performStepAction(step: RollbackStep): Promise<void> {
    switch (step.action) {
      case 'redirect-traffic-to-previous-version':
        await this.redirectTrafficToPreviousVersion();
        break;
      case 'stop-current-version-services':
        await this.stopCurrentVersionServices();
        break;
      case 'start-previous-version-services':
        await this.startPreviousVersionServices();
        break;
      case 'verify-service-health':
        await this.verifyServiceHealth();
        break;
      case 'validate-blue-environment':
        await this.validateBlueEnvironment();
        break;
      case 'prepare-database-rollback':
        await this.prepareDatabaseRollback();
        break;
      case 'switch-to-blue-environment':
        await this.switchToBlueEnvironment();
        break;
      case 'execute-database-rollback':
        await this.executeDatabaseRollback();
        break;
      case 'verify-rollback-health':
        await this.verifyRollbackHealth();
        break;
      case 'cleanup-failed-deployment':
        await this.cleanupFailedDeployment();
        break;
      default:
        if (step.action.startsWith('rolling-rollback-service-')) {
          const serviceName = step.action.replace('rolling-rollback-service-', '');
          await this.performRollingServiceRollback(serviceName);
        } else if (step.action.startsWith('verify-service-')) {
          const serviceName = step.action.replace('verify-service-', '').replace('-health', '');
          await this.verifySpecificServiceHealth(serviceName);
        } else {
          throw new Error(`Unknown step action: ${step.action}`);
        }
    }
  }

import { Injectable, Logger } from '@nestjs/common';
import { DataSource } from 'typeorm';

export interface MigrationRollbackPlan {
  targetMigration: string;
  migrationsToRollback: string[];
  dataBackupRequired: boolean;
  estimatedDuration: number;
  risks: string[];
}

export interface RollbackValidation {
  isValid: boolean;
  errors: string[];
  warnings: string[];
  dataLossRisk: 'none' | 'low' | 'medium' | 'high';
}

@Injectable()
export class MigrationRollbackService {
  private readonly logger = new Logger(MigrationRollbackService.name);

  constructor(private readonly dataSource: DataSource) {}

  async planMigrationRollback(targetMigration: string): Promise<MigrationRollbackPlan> {
    this.logger.log(`Planning migration rollback to: ${targetMigration}`);

    const executedMigrations = await this.getExecutedMigrations();
    const targetIndex = executedMigrations.findIndex(m => m === targetMigration);

    if (targetIndex === -1) {
      throw new Error(`Target migration ${targetMigration} not found in executed migrations`);
    }

    const migrationsToRollback = executedMigrations.slice(0, targetIndex).reverse();

    const plan: MigrationRollbackPlan = {
      targetMigration,
      migrationsToRollback,
      dataBackupRequired: true,
      estimatedDuration: migrationsToRollback.length * 30000, // 30 seconds per migration
      risks: await this.assessRollbackRisks(migrationsToRollback),
    };

    return plan;
  }

  async validateRollbackPlan(plan: MigrationRollbackPlan): Promise<RollbackValidation> {
    const validation: RollbackValidation = {
      isValid: true,
      errors: [],
      warnings: [],
      dataLossRisk: 'none',
    };

    // Check if all migrations have down migrations
    for (const migration of plan.migrationsToRollback) {
      const hasDownMigration = await this.hasDownMigration(migration);
      if (!hasDownMigration) {
        validation.errors.push(`Migration ${migration} does not have a down migration`);
        validation.isValid = false;
      }
    }

    // Assess data loss risk
    const dataLossRisk = await this.assessDataLossRisk(plan.migrationsToRollback);
    validation.dataLossRisk = dataLossRisk;

    if (dataLossRisk === 'high') {
      validation.warnings.push('High risk of data loss detected');
    }

    // Check for foreign key constraints
    const constraintIssues = await this.checkConstraintIssues(plan.migrationsToRollback);
    validation.warnings.push(...constraintIssues);

    return validation;
  }

  async executeMigrationRollback(plan: MigrationRollbackPlan): Promise<void> {
    this.logger.log(`Executing migration rollback plan for ${plan.migrationsToRollback.length} migrations`);

    // Create backup before rollback
    if (plan.dataBackupRequired) {
      await this.createPreRollbackBackup();
    }

    // Execute rollback in transaction
    await this.dataSource.transaction(async (manager) => {
      for (const migration of plan.migrationsToRollback) {
        this.logger.log(`Rolling back migration: ${migration}`);
        
        try {
          await this.rollbackSingleMigration(migration, manager);
        } catch (error) {
          this.logger.error(`Failed to rollback migration ${migration}`, error);
          throw error;
        }
      }
    });

    this.logger.log('Migration rollback completed successfully');
  }

  private async getExecutedMigrations(): Promise<string[]> {
    const queryRunner = this.dataSource.createQueryRunner();
    try {
      const migrations = await queryRunner.query(
        'SELECT name FROM migrations ORDER BY timestamp DESC'
      );
      return migrations.map(m => m.name);
    } finally {
      await queryRunner.release();
    }
  }

  }

  private async assessRollbackRisks(migrations: string[]): Promise<string[]> {
    const risks: string[] = [];

    for (const migration of migrations) {
      // Check for destructive operations
      const migrationContent = await this.getMigrationContent(migration);
      
      if (migrationContent.includes('DROP TABLE')) {
        risks.push(`Migration ${migration} drops tables - high data loss risk`);
      }
      
      if (migrationContent.includes('DROP COLUMN')) {
        risks.push(`Migration ${migration} drops columns - data loss risk`);
      }
      
      if (migrationContent.includes('ALTER TABLE')) {
        risks.push(`Migration ${migration} alters table structure - compatibility risk`);
      }
    }

    return risks;
  }

  private async assessDataLossRisk(migrations: string[]): Promise<RollbackValidation['dataLossRisk']> {
    let highRiskCount = 0;
    let mediumRiskCount = 0;

    for (const migration of migrations) {
      const content = await this.getMigrationContent(migration);
      
      if (content.includes('DROP TABLE') || content.includes('DROP DATABASE')) {
        highRiskCount++;
      } else if (content.includes('DROP COLUMN') || content.includes('TRUNCATE')) {
        mediumRiskCount++;
      }
    }

    if (highRiskCount > 0) return 'high';
    if (mediumRiskCount > 2) return 'medium';
    if (mediumRiskCount > 0) return 'low';
    return 'none';
  }

      estimatedDuration: 0,

    };

    if (issues.includes('javascript-error') || issues.includes('render-error')) {
      plan.configs.push({
        type: 'version',
        target: targetVersion,
        immediate: true,
        userImpact: 'high',
      });
      plan.cacheInvalidation = true;
      plan.estimatedDuration += 5 * 60 * 1000; // 5 minutes
    }

    if (issues.includes('feature-malfunction')) {
      plan.configs.push({
        type: 'feature-flag',
        target: 'disable-problematic-features',
        immediate: true,
        userImpact: 'medium',
      });

    if (issues.includes('asset-loading-error')) {
      plan.configs.push({
        type: 'asset',
        target: 'revert-asset-versions',
        immediate: true,
        userImpact: 'low',
