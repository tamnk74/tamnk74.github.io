Environment Promotion Guide

This guide provides comprehensive environment promotion strategies for NestJS microservices and Next.js microfrontend applications, covering environment-specific configurations, automated promotion pipelines, and progressive delivery across development, staging, and production environments.

Overview​

Environment promotion ensures consistent and reliable deployment of applications across different environments while maintaining isolation, security, and proper testing at each stage. This guide covers automated promotion workflows, environment-specific configurations, and validation strategies.

Environment Configuration Service​

// src/config/environment.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { EnvironmentConfig } from '../config/environments/base.config';
import { developmentConfig } from '../config/environments/development.config';
import { stagingConfig } from '../config/environments/staging.config';
import { productionConfig } from '../config/environments/production.config';

@Injectable()
export class EnvironmentService {
  private readonly logger = new Logger(EnvironmentService.name);
  private readonly config: EnvironmentConfig;

  constructor(private readonly configService: ConfigService) {
    const environment = this.configService.get('NODE_ENV', 'development');
    this.config = this.loadEnvironmentConfig(environment);
    this.validateConfig();
  }

  private loadEnvironmentConfig(environment: string): EnvironmentConfig {
    const configs = {
      development: developmentConfig,
      staging: stagingConfig,
      production: productionConfig,
    };

    const config = configs[environment as keyof typeof configs];
    if (!config) {
      throw new Error(`Unknown environment: ${environment}`);
    }

    this.logger.log(`Loaded configuration for environment: ${environment}`);
    return config;
  }

  private validateConfig(): void {
    const required = [
      'database.host',
      'database.name',
      'redis.host',
      'api.baseUrl',
    ];

    for (const path of required) {
      const value = this.getNestedValue(this.config, path);
      if (!value) {
        throw new Error(`Missing required configuration: ${path}`);
      }
    }

    this.logger.log('Configuration validation passed');
  }

  private getNestedValue(obj: any, path: string): any {
    return path.split('.').reduce((current, key) => current?.[key], obj);
  }

  getConfig(): EnvironmentConfig {
    return this.config;
  }

  getDatabaseConfig() {
    return this.config.database;
  }

  getRedisConfig() {
    return this.config.redis;
  }

  getApiConfig() {
    return this.config.api;
  }

  getLoggingConfig() {
    return this.config.logging;
  }

  getMonitoringConfig() {
    return this.config.monitoring;
  }

  getSecurityConfig() {
    return this.config.security;
  }

  getFeatureFlag(flagName: string): boolean | object {
    return this.config.features[flagName] ?? false;
  }

  isFeatureEnabled(flagName: string): boolean {
    const flag = this.getFeatureFlag(flagName);
    return typeof flag === 'boolean' ? flag : false;
  }

  getEnvironmentName(): string {
    return this.config.name;
  }

  isProduction(): boolean {
    return this.config.name === 'production';
  }

  isStaging(): boolean {
    return this.config.name === 'staging';
  }

  isDevelopment(): boolean {
    return this.config.name === 'development';
  }

  // Environment-specific behavior helpers
  shouldEnableDebugMode(): boolean {
    return this.isFeatureEnabled('enableDebugMode') && !this.isProduction();
  }

  shouldUseMockData(): boolean {
    return this.isFeatureEnabled('enableMockData') && this.isDevelopment();
  }

  getLogLevel(): string {
    return this.config.logging.level;
  }

  getMonitoringSampleRate(): number {
    return this.config.monitoring.sampleRate;
  }
}

GitHub Actions Environment Promotion​

# .github/workflows/environment-promotion.yml
name: Environment Promotion

on:
  push:
    branches: [develop, main]
  pull_request:
    branches: [main]
  workflow_dispatch:
    inputs:
      target_environment:
        description: 'Target environment for promotion'
        required: true
        default: 'staging'
        type: choice
        options:
        - development
        - staging
        - production
      skip_tests:
        description: 'Skip automated tests'
        required: false
        default: false
        type: boolean

env:
  REGISTRY: ghcr.io
  IMAGE_NAME: ${{ github.repository }}

jobs:
  determine-environment:
    runs-on: ubuntu-latest
    outputs:
      target-env: ${{ steps.env.outputs.target-env }}
      should-deploy: ${{ steps.env.outputs.should-deploy }}
    steps:
    - name: Determine target environment
      id: env
      run: |
        if [ "${{ github.event_name }}" = "workflow_dispatch" ]; then
          echo "target-env=${{ github.event.inputs.target_environment }}" >> $GITHUB_OUTPUT
          echo "should-deploy=true" >> $GITHUB_OUTPUT
        elif [ "${{ github.ref }}" = "refs/heads/develop" ]; then
          echo "target-env=development" >> $GITHUB_OUTPUT
          echo "should-deploy=true" >> $GITHUB_OUTPUT
        elif [ "${{ github.ref }}" = "refs/heads/main" ]; then
          echo "target-env=staging" >> $GITHUB_OUTPUT
          echo "should-deploy=true" >> $GITHUB_OUTPUT
        else
          echo "target-env=none" >> $GITHUB_OUTPUT
          echo "should-deploy=false" >> $GITHUB_OUTPUT
        fi

  build-and-test:
    runs-on: ubuntu-latest
    needs: determine-environment
    if: needs.determine-environment.outputs.should-deploy == 'true'
    strategy:
      matrix:
        service: [api-gateway, user-service, product-service, frontend]
    
    steps:
    - name: Checkout
      uses: actions/checkout@v4

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'
        cache-dependency-path: ${{ matrix.service }}/package-lock.json

    - name: Install dependencies
      run: |
        cd ${{ matrix.service }}
        npm ci

    - name: Run tests
      if: ${{ github.event.inputs.skip_tests != 'true' }}
      run: |
        cd ${{ matrix.service }}
        npm run test
        npm run test:e2e

    - name: Build application
      run: |
        cd ${{ matrix.service }}
        npm run build

    - name: Build Docker image
      run: |
        cd ${{ matrix.service }}
        docker build -t ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}/${{ matrix.service }}:${{ github.sha }} .

    - name: Log in to Container Registry
      uses: docker/login-action@v3
      with:
        registry: ${{ env.REGISTRY }}
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}

    - name: Push Docker image
      run: |
        docker push ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}/${{ matrix.service }}:${{ github.sha }}

  deploy-development:
    runs-on: ubuntu-latest
    needs: [determine-environment, build-and-test]
    if: needs.determine-environment.outputs.target-env == 'development'
    environment: development
    
    steps:
    - name: Deploy to Development
      run: |
        echo "Deploying to development environment..."
        # Update development environment configurations
        # Trigger development deployment

  deploy-staging:
    runs-on: ubuntu-latest
    needs: [determine-environment, build-and-test]
    if: needs.determine-environment.outputs.target-env == 'staging'
    environment: staging
    
    steps:
    - name: Run integration tests
      run: |
        echo "Running integration tests..."
        # Run comprehensive integration tests

    - name: Deploy to Staging
      run: |
        echo "Deploying to staging environment..."
        # Update staging environment configurations
        # Trigger staging deployment

    - name: Run smoke tests
      run: |
        echo "Running smoke tests..."
        # Verify basic functionality in staging

  deploy-production:
    runs-on: ubuntu-latest
    needs: [determine-environment, build-and-test]
    if: needs.determine-environment.outputs.target-env == 'production'
    environment: production
    
    steps:
    - name: Pre-production validation
      run: |
        echo "Running pre-production validation..."
        # Validate staging environment
        # Check performance metrics
        # Verify security scans

    - name: Create production backup
      run: |
        echo "Creating production backup..."
        # Backup production database
        # Backup configuration

    - name: Deploy to Production
      run: |
        echo "Deploying to production environment..."
        # Blue-green deployment to production
        # Update production configurations

    - name: Post-deployment verification
      run: |
        echo "Running post-deployment verification..."
        # Health checks
        # Performance validation
        # User acceptance tests

  notification:
    runs-on: ubuntu-latest
    needs: [deploy-development, deploy-staging, deploy-production]
    if: always()
    
    steps:
    - name: Notify deployment status
      uses: 8398a7/action-slack@v3
      with:
        status: ${{ job.status }}

          Environment promotion completed for ${{ needs.determine-environment.outputs.target-env }}
          Status: ${{ job.status }}

        SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK }}

Promotion Gates and Validation​

// src/promotion/promotion-gate.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { HttpService } from '@nestjs/axios';
import { firstValueFrom } from 'rxjs';

export interface PromotionGate {
  name: string;
  required: boolean;
  timeout: number;
  validator: () => Promise<PromotionGateResult>;
}

export interface PromotionGateResult {
  passed: boolean;
  message: string;
  details?: any;
}

export interface PromotionRequest {
  sourceEnvironment: string;
  targetEnvironment: string;
  version: string;
  services: string[];
  skipOptionalGates?: boolean;
}

@Injectable()
export class PromotionGateService {
  private readonly logger = new Logger(PromotionGateService.name);

  constructor(private readonly httpService: HttpService) {}

  async validatePromotion(request: PromotionRequest): Promise<{
    approved: boolean;
    gates: Array<PromotionGateResult & { name: string; required: boolean }>;
    message: string;
  }> {
    this.logger.log(`Validating promotion from ${request.sourceEnvironment} to ${request.targetEnvironment}`);

    const gates = this.getPromotionGates(request.targetEnvironment);
    const results = [];
    let allRequiredPassed = true;

    for (const gate of gates) {
      try {
        this.logger.debug(`Running gate: ${gate.name}`);
        
        const result = await Promise.race([
          gate.validator(),
          this.createTimeoutPromise(gate.timeout, gate.name),
        ]);

        results.push({
          name: gate.name,
          required: gate.required,
          ...result,
        });

        if (gate.required && !result.passed) {
          allRequiredPassed = false;
        }

        this.logger.debug(`Gate ${gate.name}: ${result.passed ? 'PASSED' : 'FAILED'} - ${result.message}`);
      } catch (error) {
        const errorResult = {
          name: gate.name,
          required: gate.required,
          passed: false,
          message: `Gate execution failed: ${error.message}`,
          details: error,
        };

        results.push(errorResult);

        if (gate.required) {
          allRequiredPassed = false;
        }

        this.logger.error(`Gate ${gate.name} execution failed`, error);
      }
    }

    const approved = allRequiredPassed || request.skipOptionalGates === true;
    const message = approved 
      ? 'All required promotion gates passed'
      : 'One or more required promotion gates failed';

    return {
      approved,
      gates: results,
      message,
    };
  }

  private getPromotionGates(targetEnvironment: string): PromotionGate[] {
    const commonGates: PromotionGate[] = [
      {
        name: 'health-check',
        required: true,
        timeout: 30000,
        validator: () => this.validateHealthCheck(),
      },
      {
        name: 'database-migrations',
        required: true,
        timeout: 60000,
        validator: () => this.validateDatabaseMigrations(),
      },
      {
        name: 'security-scan',
        required: true,
        timeout: 120000,
        validator: () => this.validateSecurityScan(),
      },
    ];

    const environmentSpecificGates: Record<string, PromotionGate[]> = {
      staging: [
        {
          name: 'integration-tests',
          required: true,
          timeout: 300000,
          validator: () => this.validateIntegrationTests(),
        },
        {
          name: 'performance-baseline',
          required: false,
          timeout: 180000,
          validator: () => this.validatePerformanceBaseline(),
        },
      ],
      production: [
        {
          name: 'staging-validation',
          required: true,
          timeout: 60000,
          validator: () => this.validateStagingEnvironment(),
        },
        {
          name: 'performance-tests',
          required: true,
          timeout: 600000,
          validator: () => this.validatePerformanceTests(),
        },
        {
          name: 'backup-verification',
          required: true,
          timeout: 120000,
          validator: () => this.validateBackupStatus(),
        },
        {
          name: 'manual-approval',
          required: true,
          timeout: 3600000, // 1 hour
          validator: () => this.validateManualApproval(),
        },
      ],
    };

    return [
      ...commonGates,
      ...(environmentSpecificGates[targetEnvironment] || []),
    ];
  }

  private async createTimeoutPromise(timeout: number, gateName: string): Promise<PromotionGateResult> {
    return new Promise((_, reject) => {
      setTimeout(() => {
        reject(new Error(`Gate ${gateName} timed out after ${timeout}ms`));
      }, timeout);
    });
  }

  private async validateHealthCheck(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/health', { timeout: 5000 })
      );

      if (response.status === 200) {
        return {
          passed: true,
          message: 'Health check passed',
          details: response.data,
        };
      } else {
        return {
          passed: false,
          message: `Health check failed with status ${response.status}`,
          details: response.data,
        };
      }
    } catch (error) {
      return {
        passed: false,
        message: 'Health check endpoint unreachable',
        details: error.message,
      };
    }
  }

  private async validateDatabaseMigrations(): Promise<PromotionGateResult> {
    try {
      // Check if there are pending migrations
      const response = await firstValueFrom(
        this.httpService.get('/admin/migrations/status')
      );

      const { pending, errors } = response.data;

      if (errors && errors.length > 0) {
        return {
          passed: false,
          message: 'Database migration errors detected',
          details: errors,
        };
      }

      if (pending && pending.length > 0) {
        return {
          passed: false,
          message: `${pending.length} pending migrations found`,
          details: pending,
        };
      }

      return {
        passed: true,
        message: 'All database migrations are up to date',
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify database migration status',
        details: error.message,
      };
    }
  }

  private async validateSecurityScan(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/security/scan-status')
      );

      const { vulnerabilities, lastScan } = response.data;
      const highSeverityVulns = vulnerabilities.filter(v => v.severity === 'high' || v.severity === 'critical');

      if (highSeverityVulns.length > 0) {
        return {
          passed: false,
          message: `${highSeverityVulns.length} high/critical vulnerabilities found`,
          details: highSeverityVulns,
        };
      }

      // Check if scan is recent (within last 24 hours)
      const scanAge = Date.now() - new Date(lastScan).getTime();
      const maxAge = 24 * 60 * 60 * 1000; // 24 hours

      if (scanAge > maxAge) {
        return {
          passed: false,
          message: 'Security scan is outdated',
          details: { lastScan, ageHours: scanAge / (60 * 60 * 1000) },
        };
      }

      return {
        passed: true,
        message: 'Security scan passed with no critical vulnerabilities',
        details: { vulnerabilities: vulnerabilities.length, lastScan },
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify security scan status',
        details: error.message,
      };
    }
  }

  private async validateIntegrationTests(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/tests/integration/latest')
      );

      const { status, passRate, duration, failedTests } = response.data;

      if (status !== 'passed') {
        return {
          passed: false,
          message: `Integration tests failed with status: ${status}`,
          details: failedTests,
        };
      }

      if (passRate < 95) {
        return {
          passed: false,
          message: `Integration test pass rate below threshold: ${passRate}%`,
          details: { passRate, failedTests },
        };
      }

      return {
        passed: true,
        message: `Integration tests passed with ${passRate}% success rate`,
        details: { passRate, duration },
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify integration test status',
        details: error.message,
      };
    }
  }

  private async validatePerformanceBaseline(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/performance/baseline')
      );

      const { currentMetrics, baseline, deviation } = response.data;

      const thresholds = {
        responseTime: 20, // 20% increase allowed
        throughput: -10,  // 10% decrease allowed
        errorRate: 50,    // 50% increase allowed
      };

      const violations = [];

      for (const [metric, threshold] of Object.entries(thresholds)) {
        if (deviation[metric] > threshold) {
          violations.push({
            metric,
            deviation: deviation[metric],
            threshold,
            current: currentMetrics[metric],
            baseline: baseline[metric],
          });
        }
      }

      if (violations.length > 0) {
        return {
          passed: false,
          message: 'Performance metrics deviated beyond acceptable thresholds',
          details: violations,
        };
      }

      return {
        passed: true,
        message: 'Performance metrics within acceptable range',
        details: { currentMetrics, baseline, deviation },
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify performance baseline',
        details: error.message,
      };
    }
  }

  private async validateStagingEnvironment(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/staging/validation')
      );

      const { uptime, errorRate, responseTime, lastDeployment } = response.data;

      // Check if staging has been stable for at least 2 hours
      const deploymentAge = Date.now() - new Date(lastDeployment).getTime();
      const minStabilityTime = 2 * 60 * 60 * 1000; // 2 hours

      if (deploymentAge < minStabilityTime) {
        return {
          passed: false,
          message: 'Staging environment needs more stability time',
          details: { 
            deploymentAge: deploymentAge / (60 * 60 * 1000), 
            required: 2 
          },
        };
      }

      if (errorRate > 1) {
        return {
          passed: false,
          message: `Staging error rate too high: ${errorRate}%`,
          details: { errorRate, threshold: 1 },
        };
      }

      if (responseTime > 2000) {
        return {
          passed: false,
          message: `Staging response time too high: ${responseTime}ms`,
          details: { responseTime, threshold: 2000 },
        };
      }

      return {
        passed: true,
        message: 'Staging environment validation passed',
        details: { uptime, errorRate, responseTime, lastDeployment },
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to validate staging environment',
        details: error.message,
      };
    }
  }

  private async validatePerformanceTests(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/performance/tests/latest')
      );

      const { status, metrics, thresholds } = response.data;

      if (status !== 'passed') {
        return {
          passed: false,
          message: `Performance tests failed with status: ${status}`,
          details: metrics,
        };
      }

      const violations = [];
      for (const [metric, value] of Object.entries(metrics)) {
        const threshold = thresholds[metric];
        if (threshold && value > threshold) {
          violations.push({ metric, value, threshold });
        }
      }

      if (violations.length > 0) {
        return {
          passed: false,
          message: 'Performance tests exceeded thresholds',
          details: violations,
        };
      }

      return {
        passed: true,
        message: 'Performance tests passed all thresholds',
        details: metrics,
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify performance test results',
        details: error.message,
      };
    }
  }

  private async validateBackupStatus(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/backup/status')
      );

      const { lastBackup, status, size } = response.data;

      // Check if backup is recent (within last 24 hours)
      const backupAge = Date.now() - new Date(lastBackup).getTime();
      const maxAge = 24 * 60 * 60 * 1000; // 24 hours

      if (backupAge > maxAge) {
        return {
          passed: false,
          message: 'No recent backup found',
          details: { lastBackup, ageHours: backupAge / (60 * 60 * 1000) },
        };
      }

      if (status !== 'completed') {
        return {
          passed: false,
          message: `Backup status is ${status}, not completed`,
          details: { status, lastBackup },
        };
      }

      return {
        passed: true,
        message: 'Recent backup verified',
        details: { lastBackup, status, size },
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify backup status',
        details: error.message,
      };
    }
  }

  private async validateManualApproval(): Promise<PromotionGateResult> {
    try {
      const response = await firstValueFrom(
        this.httpService.get('/admin/approvals/production/latest')
      );

      const { approved, approver, timestamp, comments } = response.data;

      if (!approved) {
        return {
          passed: false,
          message: 'Manual approval required for production deployment',
          details: { status: 'pending' },
        };
      }

      // Check if approval is recent (within last 4 hours)
      const approvalAge = Date.now() - new Date(timestamp).getTime();
      const maxAge = 4 * 60 * 60 * 1000; // 4 hours

      if (approvalAge > maxAge) {
        return {
          passed: false,
          message: 'Manual approval is outdated',
          details: { timestamp, ageHours: approvalAge / (60 * 60 * 1000) },
        };
      }

      return {
        passed: true,
        message: 'Manual approval verified',
        details: { approver, timestamp, comments },
      };
    } catch (error) {
      return {
        passed: false,
        message: 'Unable to verify manual approval status',
        details: error.message,
      };
    }
  }
}

// next.config.js
const { PHASE_DEVELOPMENT_SERVER, PHASE_PRODUCTION_BUILD } = require('next/constants');

module.exports = (phase, { defaultConfig }) => {
  const isDev = phase === PHASE_DEVELOPMENT_SERVER;
  const isProd = phase === PHASE_PRODUCTION_BUILD;

  const env = {
    NEXT_PUBLIC_ENVIRONMENT: process.env.NODE_ENV,
    NEXT_PUBLIC_API_BASE_URL: (() => {
      switch (process.env.NODE_ENV) {
        case 'production':
          return 'https://api.example.com';
        case 'staging':
          return 'https://api-staging.example.com';
        default:
          return 'http://localhost:3001';
      }
    })(),
    NEXT_PUBLIC_APP_VERSION: process.env.npm_package_version || '1.0.0',
    NEXT_PUBLIC_BUILD_TIME: new Date().toISOString(),
  };

  const config = {
    ...defaultConfig,
    env,
    
    // Environment-specific optimizations
    compiler: {
      removeConsole: isProd ? { exclude: ['error'] } : false,
    },
    
    // Bundle analyzer for staging builds
    ...(process.env.ANALYZE === 'true' && {
      webpack: (config, { buildId, dev, isServer, defaultLoaders, webpack }) => {
        if (!isServer) {
          const { BundleAnalyzerPlugin } = require('webpack-bundle-analyzer');
          config.plugins.push(
            new BundleAnalyzerPlugin({
              analyzerMode: 'static',
              openAnalyzer: false,
            })
          );
        }
        return config;
      },
    }),

    // Environment-specific headers
    async headers() {
      const headers = [];
      
      if (isProd) {
        headers.push({
          source: '/(.*)',
          headers: [
            {
              key: 'X-Frame-Options',
              value: 'DENY',
            },
            {
              key: 'X-Content-Type-Options',
              value: 'nosniff',
            },
            {
              key: 'Referrer-Policy',
              value: 'origin-when-cross-origin',
            },
          ],
        });
      }

      return headers;
    },

    // Environment-specific redirects
    async redirects() {
      const redirects = [];

      // Staging-specific redirects
      if (process.env.NODE_ENV === 'staging') {
        redirects.push({
          source: '/admin',
          destination: '/admin/staging',
          permanent: false,
        });
      }

      return redirects;
    },

    // Environment-specific rewrites
    async rewrites() {
      return {
        beforeFiles: [
          // Environment-specific API rewrites
          {
            source: '/api/:path*',
            destination: `${env.NEXT_PUBLIC_API_BASE_URL}/api/:path*`,
          },
        ],
      };
    },
  };

  return config;
};

        },

          enableImageOptimization: true,
          enableCodeSplitting: true,
        },
      },
    };

  }

  getConfig(): FrontendEnvironmentConfig {
    return this.config;
  }

  getEnvironmentName(): string {
    return this.config.name;
  }

  getApiBaseUrl(): string {
    return this.config.apiBaseUrl;
  }

  isFeatureEnabled(feature: keyof FrontendEnvironmentConfig['features']): boolean {
    return this.config.features[feature];
  }

  getMonitoringConfig() {
    return this.config.monitoring;
  }

  getPerformanceConfig() {
    return this.config.performance;
  }

  isDevelopment(): boolean {
    return this.config.name === 'development';
  }

  isStaging(): boolean {
    return this.config.name === 'staging';
  }

  isProduction(): boolean {
    return this.config.name === 'production';
  }
}

import React, { useState, useEffect } from 'react';
import { Card, CardHeader, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';

interface EnvironmentHealth {
  name: string;
  status: 'healthy' | 'warning' | 'critical';
  services: Array<{
    name: string;
    status: 'up' | 'down' | 'degraded';
    responseTime: number;
    lastCheck: string;
  }>;
  metrics: {
    uptime: number;
    errorRate: number;
    responseTime: number;
    throughput: number;
  };
  lastDeployment: {
    version: string;
    timestamp: string;
    status: 'success' | 'failed' | 'in-progress';
  };
}

export const EnvironmentHealthDashboard: React.FC = () => {
  const [environments, setEnvironments] = useState<EnvironmentHealth[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchEnvironmentHealth();
    const interval = setInterval(fetchEnvironmentHealth, 30000); // Refresh every 30 seconds
    return () => clearInterval(interval);
  }, []);

  const fetchEnvironmentHealth = async () => {
    try {
      const response = await fetch('/api/admin/environment-health');
      const data = await response.json();
      setEnvironments(data);
    } catch (error) {
      console.error('Failed to fetch environment health:', error);
    } finally {
      setLoading(false);
    }
  };

  const getStatusColor = (status: string) => {
    switch (status) {
      case 'healthy':
      case 'up':
      case 'success':
        return 'green';
      case 'warning':
      case 'degraded':
      case 'in-progress':
        return 'yellow';
      case 'critical':
      case 'down':
      case 'failed':
        return 'red';
      default:
        return 'gray';
    }
  };

  if (loading) {
    return <div>Loading environment health...</div>;
  }

  return (
    <div className="space-y-6">
      <h1 className="text-3xl font-bold">Environment Health Dashboard</h1>
      
      <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
        {environments.map((env) => (
          <Card key={env.name} className="w-full">
            <CardHeader>
              <div className="flex justify-between items-center">
                <h3 className="text-xl font-semibold capitalize">{env.name}</h3>
                <Badge variant={getStatusColor(env.status)}>
                  {env.status}
                </Badge>
              </div>
            </CardHeader>
            
            <CardContent className="space-y-4">
              {/* Services Status */}
              <div>
                <h4 className="font-medium mb-2">Services</h4>
                <div className="space-y-1">

                    <div key={service.name} className="flex justify-between items-center text-sm">

                          {service.status}
                        </Badge>
                        <span className="text-gray-500">{service.responseTime}ms</span>
                      </div>
                    </div>
                  ))}
                </div>
              </div>

              {/* Metrics */}
              <div>
                <h4 className="font-medium mb-2">Metrics</h4>
                <div className="grid grid-cols-2 gap-2 text-sm">
                  <div>
                    <span className="text-gray-500">Uptime:</span>
                    <span className="ml-1 font-medium">{env.metrics.uptime.toFixed(2)}%</span>
