Performance Testing Guide

This guide provides comprehensive performance testing strategies for NestJS microservices and Next.js microfrontend applications, covering load testing, stress testing, performance benchmarking, and automated performance validation in CI/CD pipelines.

Overview​

Performance testing ensures applications meet performance requirements under various load conditions. This guide covers different types of performance testing, automation strategies, and performance optimization techniques for microservices and microfrontend architectures.

K6 Test Scripts​

// tests/performance/load-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate, Trend, Counter } from 'k6/metrics';

// Custom metrics
const errorRate = new Rate('error_rate');
const responseTimeP95 = new Trend('response_time_p95');
const successfulRequests = new Counter('successful_requests');
const failedRequests = new Counter('failed_requests');

// Test configuration
export const options = {
  stages: [
    { duration: '2m', target: 10 },   // Ramp up to 10 users
    { duration: '5m', target: 10 },   // Stay at 10 users
    { duration: '2m', target: 20 },   // Ramp up to 20 users
    { duration: '5m', target: 20 },   // Stay at 20 users
    { duration: '2m', target: 50 },   // Ramp up to 50 users
    { duration: '5m', target: 50 },   // Stay at 50 users
    { duration: '2m', target: 0 },    // Ramp down to 0 users
  ],
  thresholds: {
    http_req_duration: ['p(95)<2000'], // 95% of requests must be below 2s
    http_req_failed: ['rate<0.01'],    // Error rate must be below 1%
    error_rate: ['rate<0.01'],         // Custom error rate
    response_time_p95: ['p(95)<2000'], // Custom P95 response time
  },
};

// Test data
const testUsers = [
  { email: 'user1@example.com', password: 'password123' },
  { email: 'user2@example.com', password: 'password123' },
  { email: 'user3@example.com', password: 'password123' },
];

const baseUrl = __ENV.BASE_URL || 'http://localhost:3001';

// Authentication helper
function authenticate() {
  const user = testUsers[Math.floor(Math.random() * testUsers.length)];
  
  const loginResponse = http.post(`${baseUrl}/auth/login`, {
    email: user.email,
    password: user.password,
  }, {
    headers: {
      'Content-Type': 'application/json',
    },
  });

  const loginSuccess = check(loginResponse, {
    'login status is 200': (r) => r.status === 200,
    'login response has token': (r) => r.json('access_token') !== undefined,
  });

  if (loginSuccess) {
    successfulRequests.add(1);
    return loginResponse.json('access_token');
  } else {
    failedRequests.add(1);
    errorRate.add(1);
    return null;
  }
}

// Main test scenario
export default function () {
  // Authentication
  const token = authenticate();
  if (!token) {
    return;
  }

  const headers = {
    'Authorization': `Bearer ${token}`,
    'Content-Type': 'application/json',
  };

  // Test user profile endpoint
  const profileResponse = http.get(`${baseUrl}/users/profile`, { headers });
  
  const profileCheck = check(profileResponse, {
    'profile status is 200': (r) => r.status === 200,
    'profile response time < 500ms': (r) => r.timings.duration < 500,
    'profile has user data': (r) => r.json('id') !== undefined,
  });

  if (profileCheck) {
    successfulRequests.add(1);
    responseTimeP95.add(profileResponse.timings.duration);
  } else {
    failedRequests.add(1);
    errorRate.add(1);
  }

  // Test product listing endpoint
  const productsResponse = http.get(`${baseUrl}/products?page=1&limit=20`, { headers });
  
  check(productsResponse, {
    'products status is 200': (r) => r.status === 200,
    'products response time < 1000ms': (r) => r.timings.duration < 1000,
    'products has data': (r) => r.json('data') !== undefined,
  });

  // Test product creation
  const newProduct = {
    name: `Test Product ${Math.random()}`,
    description: 'Test product description',
    price: Math.floor(Math.random() * 1000) + 10,
    categoryId: '123e4567-e89b-12d3-a456-426614174000',
  };

  const createResponse = http.post(`${baseUrl}/products`, JSON.stringify(newProduct), { headers });
  
  const createCheck = check(createResponse, {
    'product creation status is 201': (r) => r.status === 201,
    'product creation response time < 1500ms': (r) => r.timings.duration < 1500,
    'created product has id': (r) => r.json('id') !== undefined,
  });

  if (createCheck) {
    const productId = createResponse.json('id');
    
    // Test product update
    const updateData = { name: `Updated ${newProduct.name}` };
    const updateResponse = http.put(`${baseUrl}/products/${productId}`, JSON.stringify(updateData), { headers });
    
    check(updateResponse, {
      'product update status is 200': (r) => r.status === 200,
      'product update response time < 1000ms': (r) => r.timings.duration < 1000,
    });

    // Test product deletion
    const deleteResponse = http.del(`${baseUrl}/products/${productId}`, null, { headers });
    
    check(deleteResponse, {
      'product deletion status is 204': (r) => r.status === 204,
      'product deletion response time < 500ms': (r) => r.timings.duration < 500,
    });
  }

  sleep(1); // Wait 1 second between iterations
}

// Setup function (runs once before the test)
export function setup() {
  console.log('Starting performance test setup...');
  
  // Health check
  const healthResponse = http.get(`${baseUrl}/health`);
  if (healthResponse.status !== 200) {
    throw new Error('Service is not healthy');
  }
  
  console.log('Service health check passed');
  return { timestamp: new Date().toISOString() };
}

// Teardown function (runs once after the test)
export function teardown(data) {
  console.log(`Performance test completed at ${new Date().toISOString()}`);
  console.log(`Test started at ${data.timestamp}`);
}

// tests/performance/stress-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Rate, Trend } from 'k6/metrics';

const errorRate = new Rate('error_rate');
const responseTime = new Trend('response_time');

export const options = {
  stages: [
    { duration: '1m', target: 50 },   // Ramp up to 50 users
    { duration: '3m', target: 100 },  // Ramp up to 100 users
    { duration: '3m', target: 200 },  // Ramp up to 200 users
    { duration: '3m', target: 300 },  // Ramp up to 300 users (stress point)
    { duration: '5m', target: 300 },  // Stay at 300 users
    { duration: '3m', target: 400 },  // Push to 400 users (breaking point)
    { duration: '5m', target: 400 },  // Stay at 400 users
    { duration: '2m', target: 0 },    // Ramp down
  ],
  thresholds: {
    http_req_duration: ['p(95)<5000'], // Allow higher response times during stress
    http_req_failed: ['rate<0.05'],    // Allow higher error rate during stress
    error_rate: ['rate<0.05'],
  },
};

const baseUrl = __ENV.BASE_URL || 'http://localhost:3001';

export default function () {
  const response = http.get(`${baseUrl}/products?page=1&limit=50`);
  
  const success = check(response, {
    'status is 200': (r) => r.status === 200,
    'response time < 5000ms': (r) => r.timings.duration < 5000,
  });

  if (!success) {
    errorRate.add(1);
  }
  
  responseTime.add(response.timings.duration);
  
  sleep(Math.random() * 2 + 1); // Random sleep between 1-3 seconds
}

// tests/performance/spike-test.js
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '2m', target: 10 },   // Normal load
    { duration: '30s', target: 100 }, // Spike to 100 users
    { duration: '1m', target: 100 },  // Stay at spike
    { duration: '30s', target: 10 },  // Return to normal
    { duration: '2m', target: 10 },   // Normal load
    { duration: '30s', target: 200 }, // Larger spike
    { duration: '1m', target: 200 },  // Stay at larger spike
    { duration: '30s', target: 10 },  // Return to normal
    { duration: '2m', target: 10 },   // Normal load
  ],
  thresholds: {
    http_req_duration: ['p(95)<3000'],
    http_req_failed: ['rate<0.02'],
  },
};

const baseUrl = __ENV.BASE_URL || 'http://localhost:3001';

export default function () {
  const responses = http.batch([
    ['GET', `${baseUrl}/products`],
    ['GET', `${baseUrl}/categories`],
    ['GET', `${baseUrl}/health`],
  ]);

  for (let i = 0; i < responses.length; i++) {
    check(responses[i], {
      'status is 200': (r) => r.status === 200,
      'response time < 3000ms': (r) => r.timings.duration < 3000,
    });
  }

  sleep(1);
}

Lighthouse Performance Testing​

// tests/performance/lighthouse-performance.ts
import lighthouse from 'lighthouse';
import * as chromeLauncher from 'chrome-launcher';
import * as fs from 'fs/promises';
import * as path from 'path';

interface PerformanceMetrics {
  url: string;
  timestamp: Date;
  scores: {
    performance: number;
    accessibility: number;
    bestPractices: number;
    seo: number;
  };
  metrics: {
    firstContentfulPaint: number;
    largestContentfulPaint: number;
    firstInputDelay: number;
    cumulativeLayoutShift: number;
    speedIndex: number;
    timeToInteractive: number;
  };
  opportunities: Array<{
    id: string;
    title: string;
    score: number;
    details: any;
  }>;
}

class LighthousePerformanceTester {
  private chrome?: chromeLauncher.LaunchedChrome;

  async initialize(): Promise<void> {
    this.chrome = await chromeLauncher.launch({
      chromeFlags: ['--headless', '--disable-gpu', '--no-sandbox'],
    });
  }

  async runPerformanceTest(url: string): Promise<PerformanceMetrics> {
    if (!this.chrome) {
      await this.initialize();
    }

    const options = {
      logLevel: 'info' as const,
      output: 'json' as const,
      onlyCategories: ['performance', 'accessibility', 'best-practices', 'seo'],
      port: this.chrome!.port,
      settings: {
        preset: 'desktop',
        throttlingMethod: 'simulate',
        throttling: {
          rttMs: 40,
          throughputKbps: 10240,
          cpuSlowdownMultiplier: 1,
        },
      },
    };

    const result = await lighthouse(url, options);
    const lhr = result!.lhr;

    return {
      url,
      timestamp: new Date(),
      scores: {
        performance: Math.round((lhr.categories.performance?.score || 0) * 100),
        accessibility: Math.round((lhr.categories.accessibility?.score || 0) * 100),
        bestPractices: Math.round((lhr.categories['best-practices']?.score || 0) * 100),
        seo: Math.round((lhr.categories.seo?.score || 0) * 100),
      },
      metrics: {
        firstContentfulPaint: lhr.audits['first-contentful-paint']?.numericValue || 0,
        largestContentfulPaint: lhr.audits['largest-contentful-paint']?.numericValue || 0,
        firstInputDelay: lhr.audits['max-potential-fid']?.numericValue || 0,
        cumulativeLayoutShift: lhr.audits['cumulative-layout-shift']?.numericValue || 0,
        speedIndex: lhr.audits['speed-index']?.numericValue || 0,
        timeToInteractive: lhr.audits['interactive']?.numericValue || 0,
      },
      opportunities: Object.values(lhr.audits)
        .filter(audit => audit.scoreDisplayMode === 'numeric' && audit.score !== null && audit.score < 0.9)
        .map(audit => ({
          id: audit.id,
          title: audit.title,
          score: audit.score || 0,
          details: audit.details,
        }))
        .sort((a, b) => a.score - b.score),
    };
  }

  async runPageSpeedTest(urls: string[]): Promise<PerformanceMetrics[]> {
    const results: PerformanceMetrics[] = [];

    for (const url of urls) {
      console.log(`Testing ${url}...`);
      try {
        const metrics = await this.runPerformanceTest(url);
        results.push(metrics);
        
        // Log key metrics
        console.log(`Performance Score: ${metrics.scores.performance}`);
        console.log(`LCP: ${metrics.metrics.largestContentfulPaint}ms`);
        console.log(`FID: ${metrics.metrics.firstInputDelay}ms`);
        console.log(`CLS: ${metrics.metrics.cumulativeLayoutShift}`);
        
        // Wait between tests
        await new Promise(resolve => setTimeout(resolve, 2000));
      } catch (error) {
        console.error(`Failed to test ${url}:`, error);
      }
    }

    return results;
  }

  async generateReport(results: PerformanceMetrics[], outputPath: string): Promise<void> {
    const report = {
      timestamp: new Date().toISOString(),
      summary: {
        totalPages: results.length,
        averagePerformanceScore: results.reduce((sum, r) => sum + r.scores.performance, 0) / results.length,
        averageAccessibilityScore: results.reduce((sum, r) => sum + r.scores.accessibility, 0) / results.length,
        pagesBelow90Performance: results.filter(r => r.scores.performance < 90).length,
        slowestPage: results.reduce((slowest, current) => 
          current.metrics.largestContentfulPaint > slowest.metrics.largestContentfulPaint ? current : slowest
        ),
      },
      details: results,
      recommendations: this.generateRecommendations(results),
    };

    await fs.writeFile(outputPath, JSON.stringify(report, null, 2));
    console.log(`Performance report saved to ${outputPath}`);
  }

  private generateRecommendations(results: PerformanceMetrics[]): string[] {
    const recommendations: string[] = [];
    
    // Analyze common issues
    const commonOpportunities = new Map<string, number>();
    
    results.forEach(result => {
      result.opportunities.forEach(opp => {
        commonOpportunities.set(opp.id, (commonOpportunities.get(opp.id) || 0) + 1);
      });
    });

    // Generate recommendations based on frequency
    const sortedOpportunities = Array.from(commonOpportunities.entries())
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5);

    sortedOpportunities.forEach(([opportunityId, count]) => {
      const percentage = Math.round((count / results.length) * 100);
      recommendations.push(`${opportunityId} affects ${percentage}% of tested pages`);
    });

    return recommendations;
  }

  async cleanup(): Promise<void> {
    if (this.chrome) {
      await this.chrome.kill();
    }
  }
}

// Usage example
export async function runLighthouseTests(): Promise<void> {
  const tester = new LighthousePerformanceTester();
  
  const urls = [
    'http://localhost:3000',
    'http://localhost:3000/products',
    'http://localhost:3000/profile',
    'http://localhost:3000/dashboard',
  ];

  try {
    const results = await tester.runPageSpeedTest(urls);
    await tester.generateReport(results, 'lighthouse-report.json');
    
    // Check performance thresholds
    const failedPages = results.filter(r => r.scores.performance < 90);
    if (failedPages.length > 0) {
      console.error(`${failedPages.length} pages failed performance threshold`);
      process.exit(1);
    }
  } finally {
    await tester.cleanup();
  }
}

Playwright Performance Testing​

// tests/performance/playwright-performance.ts
import { test, expect, Page } from '@playwright/test';

interface PerformanceMetrics {
  navigation: number;
  firstContentfulPaint: number;
  largestContentfulPaint: number;
  domContentLoaded: number;
  loadComplete: number;
  resources: Array<{
    name: string;
    duration: number;
    size: number;
  }>;
}

async function collectPerformanceMetrics(page: Page): Promise<PerformanceMetrics> {
  const performanceData = await page.evaluate(() => {
    const navigation = performance.getEntriesByType('navigation')[0] as PerformanceNavigationTiming;
    const paintEntries = performance.getEntriesByType('paint');
    const resourceEntries = performance.getEntriesByType('resource') as PerformanceResourceTiming[];

    const fcp = paintEntries.find(entry => entry.name === 'first-contentful-paint')?.startTime || 0;
    const lcp = paintEntries.find(entry => entry.name === 'largest-contentful-paint')?.startTime || 0;

    return {
      navigation: navigation.loadEventEnd - navigation.navigationStart,
      firstContentfulPaint: fcp,
      largestContentfulPaint: lcp,
      domContentLoaded: navigation.domContentLoadedEventEnd - navigation.navigationStart,
      loadComplete: navigation.loadEventEnd - navigation.navigationStart,
      resources: resourceEntries.map(resource => ({
        name: resource.name,
        duration: resource.responseEnd - resource.requestStart,
        size: resource.transferSize || 0,
      })),
    };
  });

  return performanceData;
}

test.describe('Frontend Performance Tests', () => {
  test('Home page should load within performance budget', async ({ page }) => {
    await page.goto('http://localhost:3000');
    
    const metrics = await collectPerformanceMetrics(page);
    
    // Performance assertions
    expect(metrics.navigation).toBeLessThan(3000); // 3 seconds max
    expect(metrics.firstContentfulPaint).toBeLessThan(1500); // 1.5 seconds max
    expect(metrics.domContentLoaded).toBeLessThan(2000); // 2 seconds max
    
    console.log('Performance Metrics:', metrics);
  });

  test('Product listing should handle large datasets efficiently', async ({ page }) => {
    // Navigate to products page
    await page.goto('http://localhost:3000/products');
    
    // Wait for products to load
    await page.waitForSelector('[data-testid="product-list"]');
    
    const metrics = await collectPerformanceMetrics(page);
    
    // Check that rendering large lists doesn't block UI
    expect(metrics.navigation).toBeLessThan(4000);
    
    // Test scrolling performance
    const scrollStart = Date.now();
    await page.evaluate(() => {
      window.scrollTo(0, document.body.scrollHeight);
    });
    const scrollDuration = Date.now() - scrollStart;
    
    expect(scrollDuration).toBeLessThan(100); // Smooth scrolling
  });

  test('Form submission should be responsive', async ({ page }) => {
    await page.goto('http://localhost:3000/login');
    
    // Measure form interaction performance
    const interactionStart = Date.now();
    
    await page.fill('[data-testid="email-input"]', 'test@example.com');
    await page.fill('[data-testid="password-input"]', 'password123');
    
    const fillDuration = Date.now() - interactionStart;
    expect(fillDuration).toBeLessThan(500); // Responsive input
    
    // Measure form submission
    const submitStart = Date.now();
    await page.click('[data-testid="submit-button"]');
    
    // Wait for navigation or response
    await page.waitForURL('**/dashboard');
    const submitDuration = Date.now() - submitStart;
    
    expect(submitDuration).toBeLessThan(2000); // Quick submission
  });

  test('Bundle size should be within limits', async ({ page }) => {
    await page.goto('http://localhost:3000');
    
    const metrics = await collectPerformanceMetrics(page);
    
    // Check JavaScript bundle size
    const jsResources = metrics.resources.filter(r => r.name.includes('.js'));
    const totalJSSize = jsResources.reduce((sum, r) => sum + r.size, 0);
    
    expect(totalJSSize).toBeLessThan(1024 * 1024); // 1MB max for JS
    
    // Check CSS bundle size
    const cssResources = metrics.resources.filter(r => r.name.includes('.css'));
    const totalCSSSize = cssResources.reduce((sum, r) => sum + r.size, 0);
    
    expect(totalCSSSize).toBeLessThan(256 * 1024); // 256KB max for CSS
  });
});

// tests/performance/database-performance.ts
import { Test } from '@nestjs/testing';
import { DataSource } from 'typeorm';
import { Performance } from 'perf_hooks';

interface DatabaseMetrics {
  operationType: string;
  duration: number;
  rowsAffected: number;
  connectionCount: number;
  timestamp: Date;
}

class DatabasePerformanceTester {
  private dataSource: DataSource;
  private metrics: DatabaseMetrics[] = [];

  constructor(dataSource: DataSource) {
    this.dataSource = dataSource;
  }

  async testConnectionPool(): Promise<void> {
    console.log('Testing database connection pool...');
    
    const promises: Promise<any>[] = [];
    const startTime = performance.now();
    
    // Simulate concurrent connections
    for (let i = 0; i < 50; i++) {
      promises.push(this.performDatabaseOperation('SELECT 1'));
    }
    
    await Promise.all(promises);
    const duration = performance.now() - startTime;
    
    console.log(`50 concurrent connections completed in ${duration}ms`);
    
    if (duration > 5000) {
      throw new Error('Connection pool performance degraded');
    }
  }

  async testQueryPerformance(): Promise<void> {
    console.log('Testing query performance...');
    
    // Test simple SELECT
    await this.performDatabaseOperation('SELECT COUNT(*) FROM users', 'count_users');
    
    // Test complex JOIN
    await this.performDatabaseOperation(`
      SELECT u.id, u.name, p.title, c.name as category 
      FROM users u 
      LEFT JOIN products p ON u.id = p.created_by 
      LEFT JOIN categories c ON p.category_id = c.id 
      WHERE u.created_at > NOW() - INTERVAL '30 days'
      ORDER BY u.created_at DESC 
      LIMIT 100
    `, 'complex_join');
    
    // Test aggregation
    await this.performDatabaseOperation(`
      SELECT DATE(created_at) as date, COUNT(*) as count 
      FROM products 
      WHERE created_at > NOW() - INTERVAL '30 days' 
      GROUP BY DATE(created_at) 
      ORDER BY date DESC
    `, 'aggregation');
    
    // Analyze results
    this.metrics.forEach(metric => {
      console.log(`${metric.operationType}: ${metric.duration}ms`);
      
      if (metric.duration > 1000) {
        console.warn(`Slow query detected: ${metric.operationType}`);
      }
    });
  }

  async testInsertPerformance(): Promise<void> {
    console.log('Testing bulk insert performance...');
    
    const batchSizes = [100, 500, 1000, 2000];
    
    for (const batchSize of batchSizes) {
      const startTime = performance.now();
      
      // Generate test data
      const testData = Array.from({ length: batchSize }, (_, i) => ({
        name: `Test Product ${i}`,
        description: `Description for test product ${i}`,
        price: Math.floor(Math.random() * 1000) + 10,
        category_id: '123e4567-e89b-12d3-a456-426614174000',
      }));
      
      // Bulk insert
      await this.dataSource.query(`
        INSERT INTO products (name, description, price, category_id) 
        VALUES ${testData.map(() => '(?, ?, ?, ?)').join(', ')}
      `, testData.flatMap(item => [item.name, item.description, item.price, item.category_id]));
      
      const duration = performance.now() - startTime;
      
      console.log(`Batch insert ${batchSize} records: ${duration}ms (${(batchSize / (duration / 1000)).toFixed(0)} records/sec)`);
      
      // Clean up test data
      await this.dataSource.query('DELETE FROM products WHERE name LIKE ?', ['Test Product %']);
    }
  }

  async testUpdatePerformance(): Promise<void> {
    console.log('Testing update performance...');
    
    // Create test data
    const testProductIds: string[] = [];
    for (let i = 0; i < 1000; i++) {
      const result = await this.dataSource.query(`
        INSERT INTO products (name, description, price, category_id) 
        VALUES (?, ?, ?, ?) 
        RETURNING id
      `, [`Test Product ${i}`, `Description ${i}`, 100, '123e4567-e89b-12d3-a456-426614174000']);
      
      testProductIds.push(result[0].id);
    }
    
    // Test single record updates
    const singleUpdateStart = performance.now();
    for (let i = 0; i < 100; i++) {
      await this.dataSource.query('UPDATE products SET price = ? WHERE id = ?', [200, testProductIds[i]]);
    }
    const singleUpdateDuration = performance.now() - singleUpdateStart;
    console.log(`100 single updates: ${singleUpdateDuration}ms`);
    
    // Test bulk update
    const bulkUpdateStart = performance.now();
    await this.dataSource.query(`
      UPDATE products 
      SET price = 300 
      WHERE id IN (${testProductIds.slice(100, 200).map(() => '?').join(', ')})
    `, testProductIds.slice(100, 200));
    const bulkUpdateDuration = performance.now() - bulkUpdateStart;
    console.log(`100 bulk update: ${bulkUpdateDuration}ms`);
    
    // Clean up
    await this.dataSource.query(`
      DELETE FROM products 
      WHERE id IN (${testProductIds.map(() => '?').join(', ')})
    `, testProductIds);
  }

  private async performDatabaseOperation(query: string, operationType = 'generic'): Promise<void> {
    const startTime = performance.now();
    const connectionsBefore = await this.getConnectionCount();
    
    try {
      const result = await this.dataSource.query(query);
      const duration = performance.now() - startTime;
      const connectionsAfter = await this.getConnectionCount();
      
      this.metrics.push({
        operationType,
        duration,
        rowsAffected: Array.isArray(result) ? result.length : 1,
        connectionCount: connectionsAfter,
        timestamp: new Date(),
      });
    } catch (error) {
      console.error(`Database operation failed: ${operationType}`, error);
      throw error;
    }
  }

  private async getConnectionCount(): Promise<number> {
    try {
      const result = await this.dataSource.query(`
        SELECT count(*) as count 
        FROM pg_stat_activity 
        WHERE state = 'active'
      `);
      return parseInt(result[0].count);
    } catch {
      return 0;
    }
  }

  generateReport(): any {
    const report = {
      timestamp: new Date().toISOString(),
      totalOperations: this.metrics.length,
      averageDuration: this.metrics.reduce((sum, m) => sum + m.duration, 0) / this.metrics.length,
      slowestOperation: this.metrics.reduce((slowest, current) => 
        current.duration > slowest.duration ? current : slowest
      ),
      operationBreakdown: this.metrics.reduce((breakdown, metric) => {
        if (!breakdown[metric.operationType]) {
          breakdown[metric.operationType] = {
            count: 0,
            totalDuration: 0,
            averageDuration: 0,
          };
        }
        breakdown[metric.operationType].count++;
        breakdown[metric.operationType].totalDuration += metric.duration;
        breakdown[metric.operationType].averageDuration = 
          breakdown[metric.operationType].totalDuration / breakdown[metric.operationType].count;
        return breakdown;
      }, {} as any),
    };

    return report;
  }
}

// src/monitoring/performance-monitoring.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Interval } from '@nestjs/schedule';
import { PrometheusService } from './prometheus.service';
import { Histogram, Counter, Gauge } from 'prom-client';

@Injectable()
export class PerformanceMonitoringService {
  private readonly logger = new Logger(PerformanceMonitoringService.name);
  
  private readonly responseTimeHistogram: Histogram<string>;
  private readonly throughputCounter: Counter<string>;
  private readonly activeUsersGauge: Gauge<string>;
  private readonly errorRateCounter: Counter<string>;

  constructor(private readonly prometheusService: PrometheusService) {
    this.responseTimeHistogram = new Histogram({
      name: 'http_request_duration_seconds',
      help: 'Duration of HTTP requests in seconds',
      labelNames: ['method', 'route', 'status_code'],
      buckets: [0.1, 0.3, 0.5, 0.7, 1, 3, 5, 7, 10],
      registers: [this.prometheusService.getRegistry()],
    });

    this.throughputCounter = new Counter({
      name: 'http_requests_total',
      help: 'Total number of HTTP requests',
      labelNames: ['method', 'route', 'status_code'],
      registers: [this.prometheusService.getRegistry()],
    });

    this.activeUsersGauge = new Gauge({
      name: 'active_users_total',
      help: 'Number of active users',
      registers: [this.prometheusService.getRegistry()],
    });

    this.errorRateCounter = new Counter({
      name: 'http_request_errors_total',
      help: 'Total number of HTTP request errors',
      labelNames: ['method', 'route', 'error_type'],
      registers: [this.prometheusService.getRegistry()],
    });
  }

  recordRequestMetrics(
    method: string,
    route: string,
    statusCode: number,
    duration: number
  ): void {
    this.responseTimeHistogram
      .labels(method, route, statusCode.toString())
      .observe(duration / 1000); // Convert to seconds

    this.throughputCounter
      .labels(method, route, statusCode.toString())
      .inc();

    if (statusCode >= 400) {
      this.errorRateCounter
        .labels(method, route, statusCode >= 500 ? 'server_error' : 'client_error')
        .inc();
    }
  }

  @Interval(60000) // Every minute
  async collectSystemMetrics(): Promise<void> {
    try {
      // Collect active users (example implementation)
      const activeUsers = await this.getActiveUsersCount();
      this.activeUsersGauge.set(activeUsers);

      // Additional metrics collection can be added here
    } catch (error) {
      this.logger.error('Failed to collect system metrics', error);
    }
  }

  private async getActiveUsersCount(): Promise<number> {
    // Implementation would depend on your authentication system
    // This is a placeholder
    return Math.floor(Math.random() * 100);
  }
}
