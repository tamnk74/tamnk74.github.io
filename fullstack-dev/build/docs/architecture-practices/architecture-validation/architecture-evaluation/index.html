Architecture Evaluation Process

This guide establishes comprehensive architecture evaluation processes for NestJS microservices and Next.js microfrontend applications, covering automated code quality monitoring, architectural compliance validation, and design decision tracking through systematic review frameworks.

Overview​

Architecture evaluation ensures that software systems maintain high quality standards, follow established patterns, and meet both functional and non-functional requirements. This process includes automated quality gates, peer review protocols, and continuous architectural health monitoring.

Automated Architecture Analysis​

Static Code Analysis Setup​

// tools/architecture-analysis/static-analyzer.ts
import { ESLint } from 'eslint';
import * as fs from 'fs';
import * as path from 'path';
import { SonarQubeAnalyzer } from './sonarqube-analyzer';
import { CodeClimateAnalyzer } from './codeclimate-analyzer';
import { ArchitecturalRulesEngine } from './architectural-rules';

export interface AnalysisResult {
  file: string;
  issues: Issue[];
  metrics: CodeMetrics;
  violations: ArchitecturalViolation[];
  score: number;
}

export interface Issue {
  type: 'error' | 'warning' | 'info';
  rule: string;
  message: string;
  line: number;
  column: number;
  severity: number;
}

export interface CodeMetrics {
  complexity: number;
  maintainability: number;
  testCoverage: number;
  technicalDebt: string;
  duplicatedLines: number;
  linesOfCode: number;
}

export interface ArchitecturalViolation {
  type: 'dependency' | 'layering' | 'pattern' | 'naming';
  severity: 'critical' | 'major' | 'minor';
  description: string;
  suggestion: string;
  file: string;
  line?: number;
}

export class ArchitectureAnalyzer {
  private eslint: ESLint;
  private sonarqube: SonarQubeAnalyzer;
  private codeclimate: CodeClimateAnalyzer;
  private rulesEngine: ArchitecturalRulesEngine;

  constructor() {
    this.eslint = new ESLint({
      configFile: '.eslintrc.js',
      extensions: ['.ts', '.tsx', '.js', '.jsx'],
    });
    
    this.sonarqube = new SonarQubeAnalyzer();
    this.codeclimate = new CodeClimateAnalyzer();
    this.rulesEngine = new ArchitecturalRulesEngine();
  }

  async analyzeProject(projectPath: string): Promise<AnalysisResult[]> {
    const files = await this.getSourceFiles(projectPath);
    const results: AnalysisResult[] = [];

    for (const file of files) {
      const result = await this.analyzeFile(file);
      results.push(result);
    }

    return results;
  }

  async analyzeFile(filePath: string): Promise<AnalysisResult> {
    const [eslintResults, metrics, violations] = await Promise.all([
      this.runESLintAnalysis(filePath),
      this.analyzeCodeMetrics(filePath),
      this.checkArchitecturalRules(filePath),
    ]);

    const issues = this.convertESLintResults(eslintResults);
    const score = this.calculateQualityScore(issues, metrics, violations);

    return {
      file: filePath,
      issues,
      metrics,
      violations,
      score,
    };
  }

  private async runESLintAnalysis(filePath: string) {
    const results = await this.eslint.lintFiles([filePath]);
    return results[0] || { messages: [] };
  }

  private async analyzeCodeMetrics(filePath: string): Promise<CodeMetrics> {
    const content = fs.readFileSync(filePath, 'utf8');
    
    // Use multiple analysis tools
    const [sonarMetrics, codeClimateMetrics] = await Promise.all([
      this.sonarqube.analyzeFile(filePath, content),
      this.codeclimate.analyzeFile(filePath, content),
    ]);

    return {
      complexity: sonarMetrics.complexity,
      maintainability: codeClimateMetrics.maintainability,
      testCoverage: sonarMetrics.testCoverage,
      technicalDebt: sonarMetrics.technicalDebt,
      duplicatedLines: sonarMetrics.duplicatedLines,
      linesOfCode: this.countLinesOfCode(content),
    };
  }

  private async checkArchitecturalRules(filePath: string): Promise<ArchitecturalViolation[]> {
    const content = fs.readFileSync(filePath, 'utf8');
    return this.rulesEngine.validateFile(filePath, content);
  }

  private convertESLintResults(eslintResult: any): Issue[] {
    return eslintResult.messages.map((message: any) => ({
      type: message.severity === 2 ? 'error' : 'warning',
      rule: message.ruleId,
      message: message.message,
      line: message.line,
      column: message.column,
      severity: message.severity,
    }));
  }

  private calculateQualityScore(
    issues: Issue[],
    metrics: CodeMetrics,
    violations: ArchitecturalViolation[]
  ): number {
    let score = 100;

    // Deduct for issues
    const errorCount = issues.filter(i => i.type === 'error').length;
    const warningCount = issues.filter(i => i.type === 'warning').length;
    score -= errorCount * 10 + warningCount * 5;

    // Deduct for violations
    const criticalViolations = violations.filter(v => v.severity === 'critical').length;
    const majorViolations = violations.filter(v => v.severity === 'major').length;
    const minorViolations = violations.filter(v => v.severity === 'minor').length;
    score -= criticalViolations * 15 + majorViolations * 10 + minorViolations * 5;

    // Factor in metrics
    if (metrics.complexity > 10) score -= 5;
    if (metrics.maintainability < 60) score -= 10;
    if (metrics.testCoverage < 80) score -= 15;

    return Math.max(0, score);
  }

  private async getSourceFiles(projectPath: string): Promise<string[]> {
    const files: string[] = [];
    const extensions = ['.ts', '.tsx', '.js', '.jsx'];

    function walkDir(dir: string) {
      const items = fs.readdirSync(dir);
      
      for (const item of items) {
        const fullPath = path.join(dir, item);
        const stat = fs.statSync(fullPath);
        
        if (stat.isDirectory() && !item.startsWith('.') && item !== 'node_modules') {
          walkDir(fullPath);
        } else if (stat.isFile() && extensions.some(ext => item.endsWith(ext))) {
          files.push(fullPath);
        }
      }
    }

    walkDir(projectPath);
    return files;
  }

  private countLinesOfCode(content: string): number {
    return content.split('\n').filter(line => line.trim().length > 0).length;
  }
}

Architectural Rules Engine​

// tools/architecture-analysis/architectural-rules.ts
import * as fs from 'fs';
import * as path from 'path';
import { ArchitecturalViolation } from './static-analyzer';

export interface ArchitecturalRule {
  name: string;
  description: string;
  severity: 'critical' | 'major' | 'minor';
  check: (filePath: string, content: string) => ArchitecturalViolation[];
}

export class ArchitecturalRulesEngine {
  private rules: ArchitecturalRule[] = [
    // NestJS specific rules
    {
      name: 'nestjs-service-naming',
      description: 'Service files should end with .service.ts',
      severity: 'minor',
      check: this.checkServiceNaming.bind(this),
    },
    {
      name: 'nestjs-controller-naming',
      description: 'Controller files should end with .controller.ts',
      severity: 'minor',
      check: this.checkControllerNaming.bind(this),
    },
    {
      name: 'nestjs-dependency-injection',
      description: 'Services should use proper dependency injection',
      severity: 'major',
      check: this.checkDependencyInjection.bind(this),
    },
    {
      name: 'nestjs-module-structure',
      description: 'Modules should follow proper structure',
      severity: 'major',
      check: this.checkModuleStructure.bind(this),
    },
    
    // Next.js specific rules
    {
      name: 'nextjs-page-structure',
      description: 'Pages should be in pages/ or app/ directory',
      severity: 'critical',
      check: this.checkPageStructure.bind(this),
    },
    {
      name: 'nextjs-api-routes',
      description: 'API routes should be in proper directory structure',
      severity: 'major',
      check: this.checkApiRoutes.bind(this),
    },
    {
      name: 'nextjs-component-naming',
      description: 'Components should use PascalCase naming',
      severity: 'minor',
      check: this.checkComponentNaming.bind(this),
    },
    
    // General architectural rules
    {
      name: 'circular-dependencies',
      description: 'Circular dependencies should be avoided',
      severity: 'critical',
      check: this.checkCircularDependencies.bind(this),
    },
    {
      name: 'layer-separation',
      description: 'Business logic should not depend on infrastructure',
      severity: 'major',
      check: this.checkLayerSeparation.bind(this),
    },
    {
      name: 'single-responsibility',
      description: 'Classes should have single responsibility',
      severity: 'major',
      check: this.checkSingleResponsibility.bind(this),
    },
  ];

  validateFile(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];

    for (const rule of this.rules) {
      try {
        const ruleViolations = rule.check(filePath, content);
        violations.push(...ruleViolations);
      } catch (error) {
        console.error(`Error checking rule ${rule.name}:`, error);
      }
    }

    return violations;
  }

  private checkServiceNaming(filePath: string, content: string): ArchitecturalViolation[] {
    const fileName = path.basename(filePath);
    const isService = content.includes('@Injectable()') && 
                     (content.includes('Service') || fileName.includes('service'));

    if (isService && !fileName.endsWith('.service.ts')) {
      return [{
        type: 'naming',
        severity: 'minor',
        description: `Service file ${fileName} should end with .service.ts`,
        suggestion: `Rename file to ${fileName.replace('.ts', '.service.ts')}`,
        file: filePath,
      }];
    }

    return [];
  }

  private checkControllerNaming(filePath: string, content: string): ArchitecturalViolation[] {
    const fileName = path.basename(filePath);
    const isController = content.includes('@Controller(') && 
                        (content.includes('Controller') || fileName.includes('controller'));

    if (isController && !fileName.endsWith('.controller.ts')) {
      return [{
        type: 'naming',
        severity: 'minor',
        description: `Controller file ${fileName} should end with .controller.ts`,
        suggestion: `Rename file to ${fileName.replace('.ts', '.controller.ts')}`,
        file: filePath,
      }];
    }

    return [];
  }

  private checkDependencyInjection(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];

    // Check for direct instantiation of services
    const directInstantiation = /new\s+[A-Z]\w*Service\s*\(/g;
    const matches = [...content.matchAll(directInstantiation)];

    for (const match of matches) {
      violations.push({
        type: 'dependency',
        severity: 'major',
        description: 'Direct service instantiation found, use dependency injection instead',
        suggestion: 'Inject the service through constructor parameters',
        file: filePath,
        line: this.getLineNumber(content, match.index!),
      });
    }

    return violations;
  }

  private checkModuleStructure(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];
    
    if (content.includes('@Module(') && !filePath.endsWith('.module.ts')) {
      violations.push({
        type: 'pattern',
        severity: 'major',
        description: 'Module files should end with .module.ts',
        suggestion: `Rename file to ${path.basename(filePath).replace('.ts', '.module.ts')}`,
        file: filePath,
      });
    }

    return violations;
  }

  private checkPageStructure(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];
    
    // Check if it's a Next.js page component
    const isPageComponent = content.includes('export default') && 
                           (filePath.includes('/pages/') || filePath.includes('/app/')) &&
                           (content.includes('NextPage') || content.includes('Page'));

    if (isPageComponent) {
      const properPageDir = filePath.includes('/pages/') || filePath.includes('/app/');
      
      if (!properPageDir) {
        violations.push({
          type: 'pattern',
          severity: 'critical',
          description: 'Page components should be in pages/ or app/ directory',
          suggestion: 'Move page component to appropriate directory structure',
          file: filePath,
        });
      }
    }

    return violations;
  }

  private checkApiRoutes(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];
    
    const isApiRoute = filePath.includes('/api/') && 
                      (content.includes('export default') || content.includes('export async function'));

    if (isApiRoute && !filePath.match(/\/api\/.*\.(ts|js)$/)) {
      violations.push({
        type: 'pattern',
        severity: 'major',
        description: 'API routes should follow Next.js API directory structure',
        suggestion: 'Place API routes in pages/api/ or app/api/ directory',
        file: filePath,
      });
    }

    return violations;
  }

  private checkComponentNaming(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];
    const fileName = path.basename(filePath, path.extname(filePath));
    
    const isComponent = content.includes('export default') && 
                       (content.includes('function') || content.includes('const')) &&
                       content.includes('return') && content.includes('<');

    if (isComponent && !fileName.match(/^[A-Z][a-zA-Z0-9]*$/)) {
      violations.push({
        type: 'naming',
        severity: 'minor',
        description: 'Component files should use PascalCase naming',
        suggestion: `Rename file to use PascalCase (e.g., ${this.toPascalCase(fileName)})`,
        file: filePath,
      });
    }

    return violations;
  }

  private checkCircularDependencies(filePath: string, content: string): ArchitecturalViolation[] {
    // This is a simplified check - full circular dependency detection requires graph analysis
    const violations: ArchitecturalViolation[] = [];
    const imports = this.extractImports(content);
    const fileName = path.basename(filePath, path.extname(filePath));

    for (const importPath of imports) {
      if (importPath.includes(fileName)) {
        violations.push({
          type: 'dependency',
          severity: 'critical',
          description: 'Potential circular dependency detected',
          suggestion: 'Refactor to remove circular dependency, consider using dependency inversion',
          file: filePath,
        });
      }
    }

    return violations;
  }

  private checkLayerSeparation(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];
    
    // Check if domain/business logic imports infrastructure concerns
    const isDomainLayer = filePath.includes('/domain/') || 
                         filePath.includes('/entities/') ||
                         filePath.includes('/use-cases/');

    if (isDomainLayer) {
      const infrastructureImports = [
        'typeorm',
        'mongoose',
        'redis',
        'axios',
        'fetch',
        '@nestjs/typeorm',
        'prisma',
      ];

      for (const infraImport of infrastructureImports) {
        if (content.includes(`from '${infraImport}'`) || content.includes(`from "${infraImport}"`)) {
          violations.push({
            type: 'layering',
            severity: 'major',
            description: `Domain layer should not depend on infrastructure layer (${infraImport})`,
            suggestion: 'Use dependency inversion principle and inject abstractions',
            file: filePath,
          });
        }
      }
    }

    return violations;
  }

  private checkSingleResponsibility(filePath: string, content: string): ArchitecturalViolation[] {
    const violations: ArchitecturalViolation[] = [];
    
    // Simple heuristic: if a class has too many public methods, it might violate SRP
    const publicMethods = content.match(/^\s*(public\s+)?(?:async\s+)?[a-zA-Z_]\w*\s*\(/gm);
    
    if (publicMethods && publicMethods.length > 10) {
      violations.push({
        type: 'pattern',
        severity: 'major',
        description: `Class has ${publicMethods.length} public methods, might violate Single Responsibility Principle`,
        suggestion: 'Consider splitting the class into smaller, more focused classes',
        file: filePath,
      });
    }

    return violations;
  }

  private extractImports(content: string): string[] {
    const importRegex = /import.*from\s+['"]([^'"]+)['"]/g;
    const imports: string[] = [];
    let match;

    while ((match = importRegex.exec(content)) !== null) {
      imports.push(match[1]);
    }

    return imports;
  }

  private getLineNumber(content: string, index: number): number {
    return content.substring(0, index).split('\n').length;
  }

  private toPascalCase(str: string): string {
    return str.replace(/(?:^|[-_])(\w)/g, (_, char) => char.toUpperCase());
  }
}

      uses: actions/download-artifact@v4

        name: architecture-analysis

      run: |
        # Parse analysis results
        ARCH_SCORE=$(jq -r '.overallScore' reports/architecture-analysis.json)
        CRITICAL_VIOLATIONS=$(jq -r '.violations.critical' reports/architecture-analysis.json)
        MAJOR_VIOLATIONS=$(jq -r '.violations.major' reports/architecture-analysis.json)
        
        echo "Architecture Score: $ARCH_SCORE"
        echo "Critical Violations: $CRITICAL_VIOLATIONS"
        echo "Major Violations: $MAJOR_VIOLATIONS"
        
        # Define quality gates
        MIN_ARCH_SCORE=75
        MAX_CRITICAL_VIOLATIONS=0
        MAX_MAJOR_VIOLATIONS=5
        
        # Check quality gates
        if [ "$ARCH_SCORE" -lt "$MIN_ARCH_SCORE" ]; then
          echo "❌ Architecture score ($ARCH_SCORE) below minimum ($MIN_ARCH_SCORE)"
          exit 1
        fi
        
        if [ "$CRITICAL_VIOLATIONS" -gt "$MAX_CRITICAL_VIOLATIONS" ]; then
          echo "❌ Too many critical violations ($CRITICAL_VIOLATIONS)"
          exit 1
        fi
        
        if [ "$MAJOR_VIOLATIONS" -gt "$MAX_MAJOR_VIOLATIONS" ]; then
          echo "❌ Too many major violations ($MAJOR_VIOLATIONS)"
          exit 1
        fi
        
        echo "✅ All quality gates passed!"

// tools/review-system/review-assignment.ts
import { Octokit } from '@octokit/rest';
import * as core from '@actions/core';

export interface ReviewRequirement {
  minReviewers: number;
  requiredExperts: string[];
  skipConditions: string[];
  autoAssign: boolean;
}

export interface Developer {
  username: string;
  expertise: string[];
  workload: number;
  timezone: string;
  availability: boolean;
}

export class ReviewAssignmentSystem {
  private octokit: Octokit;
  private developers: Developer[] = [
    {
      username: 'senior-backend-dev',
      expertise: ['nestjs', 'microservices', 'database', 'security'],
      workload: 3,
      timezone: 'UTC',
      availability: true,
    },
    {
      username: 'senior-frontend-dev',
      expertise: ['nextjs', 'react', 'typescript', 'ui-ux'],
      workload: 2,
      timezone: 'UTC',
      availability: true,
    },
    {
      username: 'architect',
      expertise: ['architecture', 'patterns', 'performance', 'scalability'],
      workload: 1,
      timezone: 'UTC',
      availability: true,
    },
    {
      username: 'security-expert',
      expertise: ['security', 'compliance', 'vulnerabilities'],
      workload: 2,
      timezone: 'UTC',
      availability: true,
    },
  ];

  constructor(token: string) {
    this.octokit = new Octokit({ auth: token });
  }

  async assignReviewers(
    owner: string,
    repo: string,
    pullNumber: number,
    changedFiles: string[]
  ): Promise<void> {
    const requirements = this.determineReviewRequirements(changedFiles);
    const assignees = await this.selectReviewers(requirements, changedFiles);

    if (assignees.length > 0) {
      await this.octokit.pulls.requestReviewers({
        owner,
        repo,
        pull_number: pullNumber,
        reviewers: assignees,
      });

      core.info(`Assigned reviewers: ${assignees.join(', ')}`);
    }
  }

  private determineReviewRequirements(changedFiles: string[]): ReviewRequirement {
    let minReviewers = 1;
    const requiredExperts: string[] = [];

    // Analyze changed files to determine requirements
    const hasBackendChanges = changedFiles.some(file => 
      file.includes('/src/') && file.endsWith('.ts') && 
      !file.includes('.spec.ts') && !file.includes('.test.ts')
    );

    const hasFrontendChanges = changedFiles.some(file => 
      file.includes('/components/') || file.includes('/pages/') ||
      file.includes('/hooks/') || file.endsWith('.tsx')
    );

    const hasArchitecturalChanges = changedFiles.some(file => 
      file.includes('module.ts') || file.includes('config') ||
      file.includes('infrastructure') || file.includes('architecture')
    );

    const hasSecurityChanges = changedFiles.some(file => 
      file.includes('auth') || file.includes('security') ||
      file.includes('guard') || file.includes('middleware')
    );

    const hasDatabaseChanges = changedFiles.some(file => 
      file.includes('migration') || file.includes('entity') ||
      file.includes('repository') || file.includes('schema')
    );

    // Determine required expertise
    if (hasBackendChanges) requiredExperts.push('nestjs', 'microservices');
    if (hasFrontendChanges) requiredExperts.push('nextjs', 'react');
    if (hasArchitecturalChanges) requiredExperts.push('architecture');
    if (hasSecurityChanges) requiredExperts.push('security');
    if (hasDatabaseChanges) requiredExperts.push('database');

    // Increase minimum reviewers for complex changes
    if (requiredExperts.length > 2) minReviewers = 2;
    if (hasArchitecturalChanges || hasSecurityChanges) minReviewers = Math.max(minReviewers, 2);

    return {
      minReviewers,
      requiredExperts,
      skipConditions: ['docs-only', 'test-only'],
      autoAssign: true,
    };
  }

  private async selectReviewers(
    requirements: ReviewRequirement,
    changedFiles: string[]
  ): Promise<string[]> {
    // Filter available developers
    const availableDevelopers = this.developers.filter(dev => 
      dev.availability && dev.workload < 5
    );

    // Find experts for required expertise
    const expertCandidates = availableDevelopers.filter(dev =>
      requirements.requiredExperts.some(expertise => 
        dev.expertise.includes(expertise)
      )
    );

    // Sort by expertise match and workload
    const sortedCandidates = expertCandidates.sort((a, b) => {
      const aExpertiseMatch = requirements.requiredExperts.filter(exp => 
        a.expertise.includes(exp)
      ).length;
      const bExpertiseMatch = requirements.requiredExperts.filter(exp => 
        b.expertise.includes(exp)
      ).length;

      if (aExpertiseMatch !== bExpertiseMatch) {
        return bExpertiseMatch - aExpertiseMatch;
      }

      return a.workload - b.workload;
    });

    // Select reviewers
    const selectedReviewers: string[] = [];
    const coverageMap = new Set<string>();

    for (const candidate of sortedCandidates) {
      if (selectedReviewers.length >= requirements.minReviewers) {
        break;
      }

      const candidateExpertise = candidate.expertise.filter(exp =>
        requirements.requiredExperts.includes(exp)
      );

      const addsCoverage = candidateExpertise.some(exp => !coverageMap.has(exp));

      if (addsCoverage || selectedReviewers.length < requirements.minReviewers) {
        selectedReviewers.push(candidate.username);
        candidateExpertise.forEach(exp => coverageMap.add(exp));
      }
    }

    return selectedReviewers;
  }
}

// tools/review-system/architecture-review-board.ts
export interface ArchitectureDecision {
  id: string;
  title: string;
  context: string;
  decision: string;
  consequences: string[];
  status: 'proposed' | 'accepted' | 'rejected' | 'superseded';
  author: string;
  reviewers: string[];
  dateCreated: Date;
  dateDecided?: Date;
  relatedDecisions: string[];
}

export interface ReviewCriteria {
  alignment: number; // 1-5 scale
  impact: number; // 1-5 scale
  complexity: number; // 1-5 scale
  risk: number; // 1-5 scale
  innovation: number; // 1-5 scale
}

export class ArchitectureReviewBoard {
  private decisions: Map<string, ArchitectureDecision> = new Map();
  private boardMembers: string[] = [
    'chief-architect',
    'senior-architect',
    'tech-lead-backend',
    'tech-lead-frontend',
    'security-architect',
  ];

  async submitDecision(decision: Omit<ArchitectureDecision, 'id' | 'status' | 'dateCreated'>): Promise<string> {
    const id = this.generateDecisionId();
    const fullDecision: ArchitectureDecision = {
      ...decision,
      id,
      status: 'proposed',
      dateCreated: new Date(),
    };

    this.decisions.set(id, fullDecision);
    await this.notifyReviewers(fullDecision);
    
    return id;
  }

  async reviewDecision(
    decisionId: string,
    reviewer: string,
    criteria: ReviewCriteria,
    comments: string,
    recommendation: 'approve' | 'reject' | 'request-changes'
  ): Promise<void> {
    const decision = this.decisions.get(decisionId);
    if (!decision) {
      throw new Error(`Decision ${decisionId} not found`);
    }

    // Store review
    await this.storeReview(decisionId, reviewer, criteria, comments, recommendation);

    // Check if all required reviews are complete
    const reviews = await this.getReviews(decisionId);
    const requiredReviewers = this.determineRequiredReviewers(decision);
    
    const completedReviews = reviews.filter(r => 
      requiredReviewers.includes(r.reviewer)
    );

    if (completedReviews.length >= requiredReviewers.length) {
      await this.makeDecision(decisionId);
    }
  }

  private async makeDecision(decisionId: string): Promise<void> {
    const decision = this.decisions.get(decisionId)!;
    const reviews = await this.getReviews(decisionId);
    
    const approvals = reviews.filter(r => r.recommendation === 'approve').length;
    const rejections = reviews.filter(r => r.recommendation === 'reject').length;
    const totalReviews = reviews.length;
    
    // Require 2/3 majority for approval
    const approvalThreshold = Math.ceil(totalReviews * 0.67);
    
    if (approvals >= approvalThreshold) {
      decision.status = 'accepted';
      decision.dateDecided = new Date();
      await this.notifyDecisionAccepted(decision);
      await this.updateArchitectureDocumentation(decision);
    } else if (rejections > totalReviews - approvalThreshold) {
      decision.status = 'rejected';
      decision.dateDecided = new Date();
      await this.notifyDecisionRejected(decision);
    }
    
    this.decisions.set(decisionId, decision);
  }

  private determineRequiredReviewers(decision: ArchitectureDecision): string[] {
    const requiredReviewers = ['chief-architect'];
    
    // Add specific expertise based on decision content
    if (decision.context.includes('security') || decision.decision.includes('security')) {
      requiredReviewers.push('security-architect');
    }
    
    if (decision.context.includes('backend') || decision.decision.includes('nestjs')) {
      requiredReviewers.push('tech-lead-backend');
    }
    
    if (decision.context.includes('frontend') || decision.decision.includes('nextjs')) {
      requiredReviewers.push('tech-lead-frontend');
    }
    
    // For high-impact decisions, require senior architect
    if (this.isHighImpactDecision(decision)) {
      requiredReviewers.push('senior-architect');
    }
    
    return [...new Set(requiredReviewers)];
  }

  private isHighImpactDecision(decision: ArchitectureDecision): boolean {
    const highImpactKeywords = [
      'migration', 'refactor', 'architecture change', 'breaking change',
      'new technology', 'framework change', 'security model', 'data model'
    ];
    
    const content = `${decision.title} ${decision.context} ${decision.decision}`.toLowerCase();
    return highImpactKeywords.some(keyword => content.includes(keyword));
  }

  private generateDecisionId(): string {
    return `ADR-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  private async notifyReviewers(decision: ArchitectureDecision): Promise<void> {
    const requiredReviewers = this.determineRequiredReviewers(decision);
    
    // Send notifications (email, Slack, etc.)
    console.log(`Notifying reviewers for decision ${decision.id}: ${requiredReviewers.join(', ')}`);
  }

  private async storeReview(
    decisionId: string,
    reviewer: string,
    criteria: ReviewCriteria,
    comments: string,
    recommendation: string
  ): Promise<void> {
    // Store in database or file system
    console.log(`Review stored for decision ${decisionId} by ${reviewer}: ${recommendation}`);
  }

  private async getReviews(decisionId: string): Promise<any[]> {
    // Retrieve reviews from storage
    return [];
  }

  private async notifyDecisionAccepted(decision: ArchitectureDecision): Promise<void> {
    console.log(`Decision ${decision.id} has been accepted`);
  }

  private async notifyDecisionRejected(decision: ArchitectureDecision): Promise<void> {
    console.log(`Decision ${decision.id} has been rejected`);
  }

  private async updateArchitectureDocumentation(decision: ArchitectureDecision): Promise<void> {
    // Update ADR documentation
    console.log(`Updating architecture documentation for decision ${decision.id}`);
  }
