OpenTelemetry Implementation Guide

Overview​

This guide provides comprehensive instructions for implementing OpenTelemetry in your ecommerce microservices architecture using NestJS, Next.js, and Google Cloud Platform.

Benefits of OpenTelemetry for Ecommerce​

Business Benefits​

Improved Customer Experience

  • Faster issue detection and resolution reduces customer-facing downtime
  • Performance insights help optimize checkout flows and reduce cart abandonment
  • Real-time monitoring of critical user journeys (search, add to cart, payment)
  • Proactive identification of performance bottlenecks before they impact customers

Revenue Protection

  • Rapid detection of payment processing issues prevents revenue loss
  • Monitoring of recommendation engines ensures optimal product discovery
  • Inventory synchronization tracking prevents overselling scenarios
  • Performance optimization leads to higher conversion rates

Operational Efficiency

  • Reduced mean time to resolution (MTTR) from hours to minutes
  • Automated root cause analysis across distributed services
  • Simplified debugging of complex microservices interactions
  • Unified observability reduces tool sprawl and training overhead

Technical Benefits​

Distributed Tracing

  • End-to-end request visibility across all microservices
  • Automatic correlation of logs, metrics, and traces
  • Performance bottleneck identification in complex service meshes
  • Context propagation maintains request flow visibility

Vendor Neutrality

  • Standardized telemetry data format prevents vendor lock-in
  • Easy migration between observability platforms (Datadog, New Relic, Grafana)
  • Future-proof instrumentation as the industry standard
  • Reduced integration complexity with multiple monitoring tools

Auto-Instrumentation

  • Zero-code instrumentation for popular frameworks and libraries
  • Automatic HTTP, database, and message queue tracing
  • Reduced development overhead and maintenance burden
  • Consistent telemetry data across all services

Custom Business Metrics

  • Track ecommerce-specific KPIs (conversion rates, cart abandonment, search performance)
  • Real-time business intelligence through operational data
  • Service Level Indicator (SLI) and Service Level Objective (SLO) monitoring
  • Custom dashboards for business stakeholders

Security & Compliance Benefits​

Data Privacy

  • Built-in PII sanitization and data masking capabilities
  • Configurable data retention policies
  • Audit trails for compliance requirements (GDPR, PCI-DSS)
  • Secure data transmission and storage

Security Monitoring

  • Anomaly detection in user behavior patterns
  • Monitoring of authentication and authorization flows
  • Detection of suspicious API usage patterns
  • Integration with security information and event management (SIEM) systems

Backend Implementation (NestJS)​

1. Base Telemetry Configuration​

Create src/telemetry/telemetry.service.ts:

import { Injectable, OnModuleInit } from '@nestjs/common';
import { NodeSDK } from '@opentelemetry/sdk-node';
import { getNodeAutoInstrumentations } from '@opentelemetry/auto-instrumentations-node';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes } from '@opentelemetry/semantic-conventions';
import { DatadogExporter } from '@opentelemetry/exporter-datadog';
import { OTLPTraceExporter } from '@opentelemetry/exporter-otlp-http';
import { PeriodicExportingMetricReader } from '@opentelemetry/sdk-metrics';

@Injectable()
export class TelemetryService implements OnModuleInit {
  private sdk: NodeSDK;

  constructor() {
    this.initializeSDK();
  }

  private initializeSDK() {
    const serviceName = process.env.SERVICE_NAME || 'ecommerce-api';
    const serviceVersion = process.env.SERVICE_VERSION || '1.0.0';
    const environment = process.env.NODE_ENV || 'development';

    // Configure resource attributes
    const resource = new Resource({
      [SemanticResourceAttributes.SERVICE_NAME]: serviceName,
      [SemanticResourceAttributes.SERVICE_VERSION]: serviceVersion,
      [SemanticResourceAttributes.DEPLOYMENT_ENVIRONMENT]: environment,
      [SemanticResourceAttributes.SERVICE_NAMESPACE]: 'ecommerce',
    });

    // Configure exporters
    const traceExporter = new DatadogExporter({
      url: process.env.DATADOG_TRACE_URL || 'https://trace.agent.datadoghq.com',
      headers: {
        'DD-API-KEY': process.env.DATADOG_API_KEY,
      },
    });

    // Initialize SDK
    this.sdk = new NodeSDK({
      resource,
      traceExporter,
      instrumentations: [
        getNodeAutoInstrumentations({
          // Disable file system instrumentation for performance
          '@opentelemetry/instrumentation-fs': { enabled: false },

          // Configure HTTP instrumentation
          '@opentelemetry/instrumentation-http': {
            requestHook: (span, request) => {
              span.setAttributes({
                'http.request.header.user-agent': request.headers['user-agent'],
                'http.request.header.x-forwarded-for': request.headers['x-forwarded-for'],
              });
            },
          },

          // Configure Express instrumentation
          '@opentelemetry/instrumentation-express': {
            requestHook: (span, info) => {
              span.setAttributes({
                'express.route': info.route,
                'express.method': info.req.method,
              });
            },
          },

          // Configure TypeORM instrumentation
          '@opentelemetry/instrumentation-typeorm': {
            enabled: true,
            logQueries: true,
          },

          // Configure Redis instrumentation
          '@opentelemetry/instrumentation-redis': {
            enabled: true,
            dbStatementSerializer: (cmdName, cmdArgs) => {
              return `${cmdName} ${cmdArgs.slice(0, 2).join(' ')}`;
            },
          },
        }),
      ],
    });
  }

  onModuleInit() {
    this.sdk.start();
    console.log('OpenTelemetry started successfully');
  }

  async shutdown() {
    await this.sdk.shutdown();
    console.log('OpenTelemetry terminated');
  }
}

2. Custom Instrumentation Decorator​

Create src/telemetry/trace.decorator.ts:

import { trace, context, SpanStatusCode } from '@opentelemetry/api';

export function Trace(operationName?: string) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    const tracer = trace.getTracer('ecommerce-service');

    descriptor.value = async function (...args: any[]) {
      const spanName = operationName || `${target.constructor.name}.${propertyKey}`;

      return tracer.startActiveSpan(spanName, async (span) => {
        try {
          // Add method metadata
          span.setAttributes({
            'code.function': propertyKey,
            'code.namespace': target.constructor.name,
            'operation.type': 'business_logic',
          });

          // Execute original method
          const result = await originalMethod.apply(this, args);

          // Mark span as successful
          span.setStatus({ code: SpanStatusCode.OK });

          return result;
        } catch (error) {
          // Record error
          span.recordException(error);
          span.setStatus({
            code: SpanStatusCode.ERROR,
            message: error.message,
          });

          throw error;
        } finally {
          span.end();
        }
      });
    };

    return descriptor;
  };
}

import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { trace, metrics } from '@opentelemetry/api';
import { Trace } from '../telemetry/trace.decorator';
import { Product } from './entities/product.entity';

@Injectable()
export class ProductService {
  private readonly tracer = trace.getTracer('product-service');
  private readonly meter = metrics.getMeter('product-service');
  private readonly productSearchCounter = this.meter.createCounter('product_searches_total');
  private readonly productSearchDuration = this.meter.createHistogram('product_search_duration_ms');

  constructor(
    @InjectRepository(Product)
    private productRepository: Repository<Product>,
  ) {}

  @Trace('search-products')
  async searchProducts(query: string, filters: any): Promise<Product[]> {
    const startTime = Date.now();

    try {
      // Add custom attributes to current span
      const span = trace.getActiveSpan();
      span?.setAttributes({
        'search.query': query,
        'search.filters': JSON.stringify(filters),
        'operation.type': 'database_query',
      });

      // Perform search
      const products = await this.productRepository
        .createQueryBuilder('product')
        .where('product.name ILIKE :query', { query: `%${query}%` })
        .andWhere(filters)
        .getMany();

      // Record metrics
      this.productSearchCounter.add(1, {
        'search.type': 'text',
        'results.count': products.length.toString(),
      });

      // Add result metadata to span
      span?.setAttributes({
        'search.results.count': products.length,
        'search.results.found': products.length > 0,
      });

      return products;
    } finally {
      const duration = Date.now() - startTime;
      this.productSearchDuration.record(duration, {
        'search.type': 'text',
      });
    }
  }

For Kafka/RabbitMQ integration:

import { Injectable } from '@nestjs/common';
import { trace, context, SpanKind } from '@opentelemetry/api';
import { KafkaService } from './kafka.service';

@Injectable()
export class OrderEventService {
  private readonly tracer = trace.getTracer('order-events');

  constructor(private kafkaService: KafkaService) {}

  async publishOrderCreated(order: any) {
    return this.tracer.startActiveSpan(
      'order.created.publish',
      {
        kind: SpanKind.PRODUCER,
        attributes: {
          'messaging.system': 'kafka',
          'messaging.destination': 'order-events',
          'messaging.operation': 'publish',
          'order.id': order.id,
          'order.total': order.total,
        },
      },
      async (span) => {
        try {
          // Inject trace context into message headers
          const headers = {};
          trace.setSpanContext(context.active(), span.spanContext());

          await this.kafkaService.publish('order-events', {
            ...order,
            traceContext: headers,
          });

          span.setStatus({ code: SpanStatusCode.OK });
        } catch (error) {
          span.recordException(error);
          span.setStatus({ code: SpanStatusCode.ERROR, message: error.message });
          throw error;
        } finally {
          span.end();
        }
      },
    );
  }

          'messaging.system': 'kafka',
          'messaging.source': 'order-events',

          'order.id': message.orderId,
        },
      },

1. Client-Side Telemetry Setup​

Create lib/telemetry.ts:

import { WebSDK } from '@opentelemetry/sdk-web';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes } from '@opentelemetry/semantic-conventions';
import { getWebAutoInstrumentations } from '@opentelemetry/auto-instrumentations-web';
import { OTLPTraceExporter } from '@opentelemetry/exporter-otlp-http';

export function initializeTelemetry() {
  // Only initialize in browser
  if (typeof window === 'undefined') return;

  const sdk = new WebSDK({
    resource: new Resource({
      [SemanticResourceAttributes.SERVICE_NAME]: 'ecommerce-frontend',
      [SemanticResourceAttributes.SERVICE_VERSION]: process.env.NEXT_PUBLIC_APP_VERSION || '1.0.0',
      [SemanticResourceAttributes.DEPLOYMENT_ENVIRONMENT]: process.env.NODE_ENV,
    }),

    traceExporter: new OTLPTraceExporter({
      url: process.env.NEXT_PUBLIC_OTEL_EXPORTER_OTLP_TRACES_ENDPOINT,
    }),

    instrumentations: [
      getWebAutoInstrumentations({
        // Configure fetch instrumentation
        '@opentelemetry/instrumentation-fetch': {
          propagateTraceHeaderCorsUrls: [/^https:\/\/api\.yourdomain\.com\/.*/],
          clearTimingResources: true,
        },

        // Configure user interaction instrumentation
        '@opentelemetry/instrumentation-user-interaction': {
          enabled: true,
          eventNames: ['click', 'submit', 'keydown'],
        },

        // Configure XMLHttpRequest instrumentation
        '@opentelemetry/instrumentation-xml-http-request': {
          propagateTraceHeaderCorsUrls: [/^https:\/\/api\.yourdomain\.com\/.*/],
        },
      }),
    ],
  });

  sdk.start();
  console.log('OpenTelemetry Web SDK started');
}

3. Custom Frontend Spans​

Create hooks/useTracing.ts:

import { trace, context } from '@opentelemetry/api';
import { useCallback } from 'react';

export function useTracing() {
  const tracer = trace.getTracer('ecommerce-frontend-user-actions');

  const traceUserAction = useCallback(
    async (actionName: string, actionData: any, callback: () => Promise<any>) => {
      return tracer.startActiveSpan(
        `user.${actionName}`,
        {
          attributes: {
            'user.action': actionName,
            'user.action.data': JSON.stringify(actionData),
            'page.url': window.location.href,
            'page.title': document.title,
          },
        },
        async (span) => {
          try {
            const result = await callback();
            span.setStatus({ code: SpanStatusCode.OK });
            return result;
          } catch (error) {
            span.recordException(error);
            span.setStatus({ code: SpanStatusCode.ERROR, message: error.message });
            throw error;
          } finally {
            span.end();
          }
        },
      );
    },
    [tracer],
  );

  return { traceUserAction };
}

4. Component-Level Tracing​

Example usage in components:

'use client';

import { useState } from 'react';
import { useTracing } from '../hooks/useTracing';

export default function ProductSearch() {
  const [query, setQuery] = useState('');
  const [products, setProducts] = useState([]);
  const { traceUserAction } = useTracing();

  const handleSearch = async () => {
    await traceUserAction('product_search', { query, timestamp: Date.now() }, async () => {
      const response = await fetch(`/api/products/search?q=${query}`);
      const data = await response.json();
      setProducts(data);
      return data;
    });
  };

  const handleAddToCart = async (productId: string) => {
    await traceUserAction('add_to_cart', { productId, timestamp: Date.now() }, async () => {
      const response = await fetch('/api/cart/add', {
        method: 'POST',
        body: JSON.stringify({ productId }),
      });
      return response.json();
    });
  };

  return (
    <div>
      <input value={query} onChange={(e) => setQuery(e.target.value)} placeholder="Search products..." />
      <button onClick={handleSearch}>Search</button>

      {products.map((product) => (
        <div key={product.id}>
          <h3>{product.name}</h3>
          <button onClick={() => handleAddToCart(product.id)}>Add to Cart</button>
        </div>
      ))}
    </div>
  );
}

2. Advanced Sampling Strategies​

import { TraceIdRatioBasedSampler, ParentBasedSampler } from '@opentelemetry/sdk-trace-base';

// Custom sampling based on operation importance
class EcommerceSampler implements Sampler {
  private highPrioritySampler = new TraceIdRatioBasedSampler(1.0); // 100% sampling
  private normalSampler = new TraceIdRatioBasedSampler(0.1); // 10% sampling
  private lowPrioritySampler = new TraceIdRatioBasedSampler(0.01); // 1% sampling

  shouldSample(context: Context, traceId: string, spanName: string, spanKind: SpanKind): SamplingResult {
    // Always sample critical ecommerce operations
    if (spanName.includes('checkout') || spanName.includes('payment') || spanName.includes('order.create')) {
      return this.highPrioritySampler.shouldSample(context, traceId, spanName, spanKind);
    }

    // Medium sampling for user interactions
    if (spanName.includes('user.') || spanName.includes('search') || spanName.includes('cart')) {
      return this.normalSampler.shouldSample(context, traceId, spanName, spanKind);
    }

    // Low sampling for background operations
    return this.lowPrioritySampler.shouldSample(context, traceId, spanName, spanKind);
  }
}

// Request context management
export class RequestContext {
  private static storage = new AsyncLocalStorage<Map<string, any>>();

  static run<T>(fn: () => T): T {
    return this.storage.run(new Map(), fn);
  }

  static set(key: string, value: any): void {
    const store = this.storage.getStore();
    if (store) {
      store.set(key, value);
    }
  }

  static get(key: string): any {
    const store = this.storage.getStore();
    return store?.get(key);
  }
}

// Middleware for request correlation
@Injectable()
export class CorrelationMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction) {
    const correlationId = req.headers['x-correlation-id'] || randomUUID();

    RequestContext.run(() => {
      RequestContext.set('correlationId', correlationId);
      RequestContext.set('userId', req.user?.id);
      RequestContext.set('sessionId', req.sessionID);

      // Add correlation ID to response headers
      res.setHeader('x-correlation-id', correlationId);

      // Add to current span
      const span = trace.getActiveSpan();
      span?.setAttributes({
        'correlation.id': correlationId,
        'user.id': req.user?.id,
        'session.id': req.sessionID,
      });

      next();
    });
  }
}

4. Health Check and Readiness Probes​

@Controller('health')
export class HealthController {
  private readonly healthCheckSpan = trace.getTracer('health-checks');

  @Get('live')
  async liveness() {
    return this.healthCheckSpan.startActiveSpan('health.liveness', async (span) => {
      try {
        span.setAttributes({
          'health.check.type': 'liveness',
          'service.version': process.env.SERVICE_VERSION,
        });

        // Basic health check
        const isHealthy = await this.performBasicHealthCheck();

        span.setAttributes({
          'health.status': isHealthy ? 'healthy' : 'unhealthy',
        });

        if (!isHealthy) {
          span.setStatus({ code: SpanStatusCode.ERROR, message: 'Service unhealthy' });
          throw new ServiceUnavailableException('Service is not healthy');
        }

        return { status: 'healthy', timestamp: new Date().toISOString() };
      } finally {
        span.end();
      }
    });
  }

  @Get('ready')
  async readiness() {
    return this.healthCheckSpan.startActiveSpan('health.readiness', async (span) => {
      try {
        span.setAttributes({
          'health.check.type': 'readiness',
        });

        // Check dependencies
        const checks = await Promise.allSettled([
          this.checkDatabase(),
          this.checkRedis(),
          this.checkKafka(),
          this.checkExternalServices(),
        ]);

        const results = checks.map((check, index) => ({
          service: ['database', 'redis', 'kafka', 'external'][index],
          status: check.status === 'fulfilled' ? 'healthy' : 'unhealthy',
          error: check.status === 'rejected' ? check.reason.message : null,
        }));

        const allHealthy = results.every((r) => r.status === 'healthy');

        span.setAttributes({
          'health.overall.status': allHealthy ? 'ready' : 'not_ready',
          'health.dependencies.count': results.length,
          'health.dependencies.healthy': results.filter((r) => r.status === 'healthy').length,
        });

        if (!allHealthy) {
          span.setStatus({ code: SpanStatusCode.ERROR, message: 'Dependencies not ready' });
        }

        return {
          status: allHealthy ? 'ready' : 'not_ready',
          checks: results,
          timestamp: new Date().toISOString(),
        };
      } finally {
        span.end();
      }
    });
  }
}

Security Considerations​

1. PII and Sensitive Data Protection​

// Data sanitization utilities
export class DataSanitizer {
  private static readonly PII_PATTERNS = {
    email: /\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b/g,
    phone: /\b\d{3}-?\d{3}-?\d{4}\b/g,
    ssn: /\b\d{3}-?\d{2}-?\d{4}\b/g,
    creditCard: /\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b/g,
  };

  static sanitizeForTracing(data: any): any {
    if (typeof data === 'string') {
      let sanitized = data;
      Object.entries(this.PII_PATTERNS).forEach(([type, pattern]) => {
        sanitized = sanitized.replace(pattern, `[REDACTED_${type.toUpperCase()}]`);
      });
      return sanitized;
    }

    if (typeof data === 'object' && data !== null) {
      const sanitized = {};
      Object.entries(data).forEach(([key, value]) => {
        if (this.isSensitiveField(key)) {
          sanitized[key] = '[REDACTED]';
        } else {
          sanitized[key] = this.sanitizeForTracing(value);
        }
      });
      return sanitized;
    }

    return data;
  }

  private static isSensitiveField(fieldName: string): boolean {
    const sensitiveFields = [
      'password',
      'token',
      'secret',
      'key',
      'ssn',
      'credit_card',
      'cvv',
      'pin',
      'bank_account',
      'routing_number',
    ];
    return sensitiveFields.some((field) => fieldName.toLowerCase().includes(field));
  }
}

// Secure span decorator
export function SecureTrace(operationName?: string) {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    const tracer = trace.getTracer('secure-service');

    descriptor.value = async function (...args: any[]) {
      const spanName = operationName || `${target.constructor.name}.${propertyKey}`;

      return tracer.startActiveSpan(spanName, async (span) => {
        try {
          // Sanitize arguments before logging
          const sanitizedArgs = args.map((arg) => DataSanitizer.sanitizeForTracing(arg));

          span.setAttributes({
            'code.function': propertyKey,
            'code.namespace': target.constructor.name,
            'operation.args.count': args.length,
            // Don't log actual argument values for security
          });

          const result = await originalMethod.apply(this, args);
          span.setStatus({ code: SpanStatusCode.OK });
          return result;
        } catch (error) {
          // Sanitize error messages
          const sanitizedError = DataSanitizer.sanitizeForTracing(error.message);
          span.recordException(new Error(sanitizedError));
          span.setStatus({ code: SpanStatusCode.ERROR, message: sanitizedError });
          throw error;
        } finally {
          span.end();
        }
      });
    };

    return descriptor;
  };
}

Performance Optimization​

1. Intelligent Span Creation​

// Conditional instrumentation based on load
export class AdaptiveInstrumentation {
  private static currentLoad = 0;
  private static readonly LOAD_THRESHOLD = 0.8;

  static shouldCreateSpan(spanType: 'critical' | 'normal' | 'debug'): boolean {
    switch (spanType) {
      case 'critical':
        return true; // Always create critical spans
      case 'normal':
        return this.currentLoad < this.LOAD_THRESHOLD;
      case 'debug':
        return this.currentLoad < 0.5 && process.env.NODE_ENV !== 'production';
      default:
        return false;
    }
  }

  static updateLoad(cpuUsage: number, memoryUsage: number) {
    this.currentLoad = Math.max(cpuUsage, memoryUsage);
  }
}

// Adaptive tracing decorator
export function AdaptiveTrace(spanType: 'critical' | 'normal' | 'debug' = 'normal') {
  return function (target: any, propertyKey: string, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    const tracer = trace.getTracer('adaptive-service');

    descriptor.value = async function (...args: any[]) {
      if (!AdaptiveInstrumentation.shouldCreateSpan(spanType)) {
        return originalMethod.apply(this, args);
      }

      const spanName = `${target.constructor.name}.${propertyKey}`;
      return tracer.startActiveSpan(spanName, async (span) => {
        try {
          const result = await originalMethod.apply(this, args);
          span.setStatus({ code: SpanStatusCode.OK });
          return result;
        } catch (error) {
          span.recordException(error);
          span.setStatus({ code: SpanStatusCode.ERROR, message: error.message });
          throw error;
        } finally {
          span.end();
        }
      });
    };

    return descriptor;
  };
}

2. Resource-Aware Sampling​

export class ResourceAwareSampler implements Sampler {
  private readonly systemMetrics = new SystemMetrics();

  shouldSample(context: Context, traceId: string, spanName: string): SamplingResult {
    const { cpuUsage, memoryUsage, diskIO } = this.systemMetrics.getCurrentMetrics();

    // Reduce sampling under high load
    let samplingRate = 0.1; // Default 10%

    if (cpuUsage > 0.8 || memoryUsage > 0.8) {
      samplingRate = 0.01; // Reduce to 1%
    } else if (cpuUsage > 0.6 || memoryUsage > 0.6) {
      samplingRate = 0.05; // Reduce to 5%
    }

    // Always sample errors and critical operations
    if (spanName.includes('error') || spanName.includes('checkout')) {
      samplingRate = 1.0;
    }

    return {
      decision: Math.random() < samplingRate ? SamplingDecision.RECORD_AND_SAMPLED : SamplingDecision.NOT_RECORD,
    };
  }
}
