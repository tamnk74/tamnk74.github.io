Cookie Consent Management for E-commerce Website

Overview​

This document provides a comprehensive guide for implementing a robust cookie consent management system for an e-commerce website, ensuring compliance with GDPR, CCPA, and other privacy regulations while maintaining optimal user experience and business functionality.

Legal Requirements & Compliance​

GDPR (General Data Protection Regulation)​

  • Explicit Consent: Users must actively opt-in for non-essential cookies
  • Granular Control: Users must be able to choose specific cookie categories
  • Withdrawal: Users must be able to withdraw consent at any time
  • Clear Information: Cookie purposes must be clearly explained
  • Legitimate Interest: Must be properly documented and justified

CCPA (California Consumer Privacy Act)​

  • Right to Opt-Out: Users must be able to opt-out of data selling
  • Do Not Sell My Personal Information: Clear opt-out mechanism required
  • Notice at Collection: Users must be informed about data collection

ePrivacy Directive (Cookie Law)​

  • Prior Consent: Consent must be obtained before placing cookies
  • Essential Cookies Exception: Strictly necessary cookies don't require consent
  • Clear Information: Purpose and duration of cookies must be explained

Cookie Categories & Classification​

1. Strictly Necessary Cookies​

Purpose: Essential for website functionality Consent Required: No Examples:

  • Session management
  • Authentication tokens
  • Shopping cart persistence
  • Load balancing
  • Security features
interface StrictlyNecessaryCookies {
  sessionId: string;
  authToken: string;
  cartItems: string;
  csrfToken: string;
  loadBalancer: string;
}

2. Performance/Analytics Cookies​

Purpose: Website performance analysis and optimization Consent Required: Yes Examples:

  • Google Analytics
  • Page load time tracking
  • Error monitoring
  • A/B testing
interface AnalyticsCookies {
  googleAnalytics: '_ga' | '_gid' | '_gat';
  hotjar: '_hjSessionUser_*' | '_hjSession_*';
  mixpanel: 'mp_*';
  amplitude: 'amplitude_*';
}

3. Functional Cookies​

Purpose: Enhanced user experience and personalization Consent Required: Yes (recommended) Examples:

  • Language preferences
  • Currency selection
  • Recently viewed products
  • Wishlist items
  • Chat widgets
interface FunctionalCookies {
  language: string;
  currency: string;
  recentlyViewed: string;
  wishlist: string;
  chatPreferences: string;
}

4. Marketing/Advertising Cookies​

Purpose: Targeted advertising and marketing campaigns Consent Required: Yes Examples:

  • Facebook Pixel
  • Google Ads
  • Retargeting pixels
  • Social media plugins
  • Email marketing tracking
interface MarketingCookies {
  facebookPixel: '_fbp' | '_fbc';
  googleAds: '_gcl_au' | '_gcl_aw';
  linkedIn: 'li_*';
  twitter: 'twtr_*';
  pinterest: '_pin_*';
}

1. Cookie Consent Provider​

// lib/cookie-consent.tsx
'use client';

import React, { createContext, useContext, useState, useEffect } from 'react';

export interface CookieConsent {
  necessary: boolean;
  analytics: boolean;
  functional: boolean;
  marketing: boolean;
}

interface CookieConsentContextType {
  consent: CookieConsent;
  updateConsent: (newConsent: Partial<CookieConsent>) => void;
  showBanner: boolean;
  acceptAll: () => void;
  rejectAll: () => void;
  isLoaded: boolean;
}

const CookieConsentContext = createContext<CookieConsentContextType | undefined>(undefined);

const CONSENT_COOKIE_NAME = 'cookie-consent';
const CONSENT_VERSION = '1.0';

export function CookieConsentProvider({ children }: { children: React.ReactNode }) {
  const [consent, setConsent] = useState<CookieConsent>({
    necessary: true, // Always true
    analytics: false,
    functional: false,
    marketing: false,
  });
  const [showBanner, setShowBanner] = useState(false);
  const [isLoaded, setIsLoaded] = useState(false);

  useEffect(() => {
    // Load consent from localStorage or cookie
    const savedConsent = loadConsentFromStorage();
    if (savedConsent) {
      setConsent(savedConsent);
      setShowBanner(false);
    } else {
      setShowBanner(true);
    }
    setIsLoaded(true);
  }, []);

  const updateConsent = (newConsent: Partial<CookieConsent>) => {
    const updatedConsent = { ...consent, ...newConsent, necessary: true };
    setConsent(updatedConsent);
    saveConsentToStorage(updatedConsent);
    setShowBanner(false);
    
    // Trigger consent change event
    window.dispatchEvent(new CustomEvent('cookieConsentChange', {
      detail: updatedConsent
    }));
  };

  const acceptAll = () => {
    updateConsent({
      analytics: true,
      functional: true,
      marketing: true,
    });
  };

  const rejectAll = () => {
    updateConsent({
      analytics: false,
      functional: false,
      marketing: false,
    });
  };

  return (
    <CookieConsentContext.Provider 
      value={{ 
        consent, 
        updateConsent, 
        showBanner, 
        acceptAll, 
        rejectAll, 
        isLoaded 
      }}
    >
      {children}
    </CookieConsentContext.Provider>
  );
}

export const useCookieConsent = () => {
  const context = useContext(CookieConsentContext);
  if (!context) {
    throw new Error('useCookieConsent must be used within CookieConsentProvider');
  }
  return context;
};

function loadConsentFromStorage(): CookieConsent | null {
  try {
    const stored = localStorage.getItem(CONSENT_COOKIE_NAME);
    if (stored) {
      const parsed = JSON.parse(stored);
      if (parsed.version === CONSENT_VERSION) {
        return parsed.consent;
      }
    }
  } catch (error) {
    console.error('Error loading consent from storage:', error);
  }
  return null;
}

function saveConsentToStorage(consent: CookieConsent) {
  try {
    const toStore = {
      consent,
      version: CONSENT_VERSION,
      timestamp: new Date().toISOString(),
    };
    localStorage.setItem(CONSENT_COOKIE_NAME, JSON.stringify(toStore));
    
    // Also set a cookie for server-side access
    document.cookie = `${CONSENT_COOKIE_NAME}=${JSON.stringify(toStore)}; path=/; max-age=31536000; SameSite=Lax`;
  } catch (error) {
    console.error('Error saving consent to storage:', error);
  }
}

3. Cookie Management Hook​

// hooks/useCookieManager.ts
import { useEffect } from 'react';
import { useCookieConsent } from '@/lib/cookie-consent';

export function useCookieManager() {
  const { consent, isLoaded } = useCookieConsent();

  useEffect(() => {
    if (!isLoaded) return;

    // Google Analytics
    if (consent.analytics) {
      loadGoogleAnalytics();
    } else {
      removeGoogleAnalytics();
    }

    // Facebook Pixel
    if (consent.marketing) {
      loadFacebookPixel();
    } else {
      removeFacebookPixel();
    }

    // Hotjar
    if (consent.analytics) {
      loadHotjar();
    } else {
      removeHotjar();
    }

  }, [consent, isLoaded]);

  return { consent, isLoaded };
}

function loadGoogleAnalytics() {
  if (typeof window === 'undefined') return;
  
  // Load Google Analytics script
  const script = document.createElement('script');
  script.src = `https://www.googletagmanager.com/gtag/js?id=${process.env.NEXT_PUBLIC_GA_ID}`;
  script.async = true;
  document.head.appendChild(script);

  // Initialize gtag
  window.dataLayer = window.dataLayer || [];
  function gtag(...args: any[]) {
    window.dataLayer.push(arguments);
  }
  window.gtag = gtag;
  
  gtag('js', new Date());
  gtag('config', process.env.NEXT_PUBLIC_GA_ID, {
    anonymize_ip: true,
    cookie_flags: 'SameSite=Lax;Secure'
  });
}

function removeGoogleAnalytics() {
  // Remove GA cookies
  const gaCookies = ['_ga', '_gid', '_gat_gtag_' + process.env.NEXT_PUBLIC_GA_ID];
  gaCookies.forEach(cookie => {
    document.cookie = `${cookie}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
  });
}

function loadFacebookPixel() {
  if (typeof window === 'undefined') return;
  
  // Facebook Pixel code
  window.fbq = window.fbq || function() {
    window.fbq.callMethod ? window.fbq.callMethod.apply(window.fbq, arguments) : 
    window.fbq.queue.push(arguments);
  };
  
  if (!window._fbq) window._fbq = window.fbq;
  window.fbq.push = window.fbq;
  window.fbq.loaded = true;
  window.fbq.version = '2.0';
  window.fbq.queue = [];
  
  const script = document.createElement('script');
  script.src = 'https://connect.facebook.net/en_US/fbevents.js';
  script.async = true;
  document.head.appendChild(script);
  
  window.fbq('init', process.env.NEXT_PUBLIC_FACEBOOK_PIXEL_ID);
  window.fbq('track', 'PageView');
}

function removeFacebookPixel() {
  // Remove Facebook Pixel cookies
  const fbCookies = ['_fbp', '_fbc'];
  fbCookies.forEach(cookie => {
    document.cookie = `${cookie}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
  });
}

function loadHotjar() {
  if (typeof window === 'undefined') return;
  
  window.hj = window.hj || function() {
    (window.hj.q = window.hj.q || []).push(arguments);
  };
  window._hjSettings = { hjid: process.env.NEXT_PUBLIC_HOTJAR_ID, hjsv: 6 };
  
  const script = document.createElement('script');
  script.src = `https://static.hotjar.com/c/hotjar-${process.env.NEXT_PUBLIC_HOTJAR_ID}.js?sv=6`;
  script.async = true;
  document.head.appendChild(script);
}

function removeHotjar() {
  // Remove Hotjar cookies
  const hjCookies = Object.keys(document.cookie).filter(key => key.startsWith('_hj'));
  hjCookies.forEach(cookie => {
    document.cookie = `${cookie}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
  });
}

4. Server-Side Cookie Handling (NestJS)​

// src/middleware/cookie-consent.middleware.ts
import { Injectable, NestMiddleware } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';

@Injectable()
export class CookieConsentMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction) {
    const consentCookie = req.cookies['cookie-consent'];
    
    if (consentCookie) {
      try {
        const consent = JSON.parse(consentCookie);
        req.cookieConsent = consent;
      } catch (error) {
        console.error('Error parsing cookie consent:', error);
      }
    }
    
    next();
  }
}

// src/types/express.d.ts
declare global {
  namespace Express {
    interface Request {
      cookieConsent?: {
        consent: {
          necessary: boolean;
          analytics: boolean;
          functional: boolean;
          marketing: boolean;
        };
        version: string;
        timestamp: string;
      };
    }
  }
}

UI/UX Design Guidelines​

Design Principles​

  1. Transparency: Clear and honest communication about cookie usage
  2. User Control: Easy-to-use granular controls
  3. Accessibility: WCAG 2.1 AA compliant
  4. Performance: Minimal impact on page load time
  5. Mobile-First: Responsive design for all devices

Banner Placement Options​

  • Bottom of screen: Most common, less intrusive
  • Top of screen: More prominent, harder to ignore
  • Modal overlay: Maximum attention, can be disruptive
  • Slide-in from corner: Balanced approach

// utils/analytics.ts
import { useCookieConsent } from '@/lib/cookie-consent';

export function trackEvent(eventName: string, parameters?: any) {
  const { consent } = useCookieConsent();
  
  if (consent.analytics && typeof window !== 'undefined' && window.gtag) {
    window.gtag('event', eventName, {
      ...parameters,
      cookie_flags: 'SameSite=Lax;Secure'
    });
  }
}

export function trackPurchase(transactionId: string, value: number, currency: string, items: any[]) {
  const { consent } = useCookieConsent();
  
  if (consent.analytics && typeof window !== 'undefined' && window.gtag) {
    window.gtag('event', 'purchase', {
      transaction_id: transactionId,
      value: value,
      currency: currency,
      items: items
    });
  }
}

  • Cookie banner doesn't appear after consent is given
  • "Accept All" button works correctly
  • "Reject All" button works correctly
  • Granular cookie settings work
  • Cookie preferences are persisted
  • Analytics scripts load only with consent
  • Marketing scripts load only with consent
  • Cookie banner is accessible via keyboard
  • Cookie banner works on mobile devices
  • Cookie policy page is accessible
  • Cookie settings can be changed later

GDPR Compliance Testing​

  1. Consent Granularity: Verify users can choose specific cookie categories
  2. Withdrawal: Test that users can withdraw consent easily
  3. Clear Information: Ensure cookie purposes are clearly explained
  4. No Pre-ticked Boxes: Verify non-essential cookies aren't pre-selected
  5. Legitimate Interest: Test opt-out mechanisms for legitimate interest

Key Metrics to Track​

// utils/consent-analytics.ts
export function trackConsentMetrics() {
  const { consent } = useCookieConsent();
  
  // Track consent rates
  trackEvent('cookie_consent_given', {
    analytics_consent: consent.analytics,
    functional_consent: consent.functional,
    marketing_consent: consent.marketing
  });
}

export function trackConsentChanges(oldConsent: any, newConsent: any) {
  trackEvent('cookie_consent_changed', {
    analytics_changed: oldConsent.analytics !== newConsent.analytics,
    functional_changed: oldConsent.functional !== newConsent.functional,
    marketing_changed: oldConsent.marketing !== newConsent.marketing
  });
}

Troubleshooting Common Issues​

Issue: Cookies not being removed after consent withdrawal​

Solution: Implement proper cookie cleanup functions

function cleanupCookies(domain: string, cookieNames: string[]) {
  cookieNames.forEach(name => {
    // Remove from current domain
    document.cookie = `${name}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
    // Remove from domain with dot prefix
    document.cookie = `${name}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/; domain=${domain};`;
    document.cookie = `${name}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/; domain=.${domain};`;
  });
}

Issue: Scripts loading before consent​

Solution: Use dynamic script loading based on consent

function loadScriptWithConsent(src: string, consentType: keyof CookieConsent) {
  const { consent } = useCookieConsent();
  
  if (consent[consentType]) {
    const script = document.createElement('script');
    script.src = src;
    script.async = true;
    document.head.appendChild(script);
  }
}

Issue: Banner not showing on Safari​

Solution: Check for localStorage availability and fallback to cookies

function isLocalStorageAvailable(): boolean {
  try {
    const testKey = '__localStorage_test__';
    localStorage.setItem(testKey, 'test');
    localStorage.removeItem(testKey);
    return true;
  } catch {
    return false;
  }
}
