Real-time Notification Application with Node.js, Next.js, FCM & GKE

This comprehensive guide walks you through building a production-ready real-time notification system using modern technologies and deploying it to Google Kubernetes Engine (GKE).

Overview​

We'll build a complete notification system that includes:

  • Backend: Node.js/Express API server with Socket.IO for real-time connections
  • Frontend: Next.js application with push notifications
  • Push Notifications: Firebase Cloud Messaging (FCM) integration
  • Real-time Communication: WebSocket connections via Socket.IO
  • Deployment: Container orchestration on Google Kubernetes Engine

Architecture​

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│   Next.js App   │    │   Node.js API   │    │   Firebase FCM  │
│   (Frontend)    │◄──►│   (Backend)     │◄──►│   (Push Notify) │
└─────────────────┘    └─────────────────┘    └─────────────────┘
         │                       │                       │
         │              ┌─────────────────┐              │

                        │   (WebSocket)   │
                        └─────────────────┘
                                 │
                        ┌─────────────────┐
                        │   Google GKE    │

1.3 Firebase FCM Service​

Create src/services/firebase.ts:

import admin from 'firebase-admin';

export const initializeFirebase = () => {
  if (!admin.apps.length) {
    admin.initializeApp({
      credential: admin.credential.cert({
        projectId: process.env.FIREBASE_PROJECT_ID,
        clientEmail: process.env.FIREBASE_CLIENT_EMAIL,
        privateKey: process.env.FIREBASE_PRIVATE_KEY?.replace(/\\n/g, '\n'),
      }),
    });
  }
};

export interface NotificationPayload {
  title: string;
  body: string;
  data?: Record<string, string>;
  token?: string;
  tokens?: string[];
  topic?: string;
}

export const sendPushNotification = async (payload: NotificationPayload) => {
  try {
    const message: admin.messaging.Message = {
      notification: {
        title: payload.title,
        body: payload.body,
      },
      data: payload.data || {},
    };

    let result;
    
    if (payload.token) {
      message.token = payload.token;
      result = await admin.messaging().send(message);
    } else if (payload.tokens && payload.tokens.length > 0) {
      const multicastMessage: admin.messaging.MulticastMessage = {
        ...message,
        tokens: payload.tokens,
      };
      result = await admin.messaging().sendMulticast(multicastMessage);
    } else if (payload.topic) {
      message.topic = payload.topic;
      result = await admin.messaging().send(message);
    }

    console.log('Successfully sent message:', result);
    return result;
  } catch (error) {
    console.error('Error sending message:', error);
    throw error;
  }
};

  return redis;
};

export const storeUserSession = async (redis: Redis, userId: string, socketId: string) => {
  await redis.setex(`user:${userId}`, 3600, socketId);
};

export const getUserSession = async (redis: Redis, userId: string): Promise<string | null> => {
  return await redis.get(`user:${userId}`);
};

export const removeUserSession = async (redis: Redis, userId: string) => {
  await redis.del(`user:${userId}`);
};

1.5 Socket.IO Handlers​

Create src/socket/handlers.ts:

import { Server, Socket } from 'socket.io';
import Redis from 'ioredis';
import { storeUserSession, removeUserSession } from '../services/redis';
import { sendPushNotification } from '../services/firebase';

interface AuthenticatedSocket extends Socket {
  userId?: string;
}

export const setupSocketHandlers = (io: Server, redis: Redis) => {
  io.on('connection', (socket: AuthenticatedSocket) => {
    console.log('User connected:', socket.id);

    // Handle user authentication
    socket.on('authenticate', async (data: { userId: string; fcmToken?: string }) => {
      socket.userId = data.userId;
      await storeUserSession(redis, data.userId, socket.id);
      
      // Store FCM token for push notifications
      if (data.fcmToken) {
        await redis.setex(`fcm:${data.userId}`, 86400, data.fcmToken);
      }
      
      socket.join(`user:${data.userId}`);
      socket.emit('authenticated', { userId: data.userId });
      console.log(`User ${data.userId} authenticated`);
    });

    // Handle sending notifications
    socket.on('sendNotification', async (data: {
      recipientId: string;
      title: string;
      body: string;
      type: 'realtime' | 'push' | 'both';
      data?: Record<string, any>;
    }) => {
      const { recipientId, title, body, type, data: notificationData } = data;

      try {
        // Send real-time notification via Socket.IO
        if (type === 'realtime' || type === 'both') {
          io.to(`user:${recipientId}`).emit('notification', {
            id: Date.now().toString(),
            title,
            body,
            data: notificationData,
            timestamp: new Date().toISOString(),
          });
        }

        // Send push notification via FCM
        if (type === 'push' || type === 'both') {
          const fcmToken = await redis.get(`fcm:${recipientId}`);
          if (fcmToken) {
            await sendPushNotification({
              title,
              body,
              token: fcmToken,
              data: notificationData ? JSON.stringify(notificationData) : undefined,
            });
          }
        }

        socket.emit('notificationSent', { success: true });
      } catch (error) {
        console.error('Error sending notification:', error);
        socket.emit('notificationSent', { success: false, error: error.message });
      }
    });

    // Handle disconnect
    socket.on('disconnect', async () => {
      if (socket.userId) {
        await removeUserSession(redis, socket.userId);
        console.log(`User ${socket.userId} disconnected`);
      }
    });
  });
};

1.6 API Routes​

Create src/routes/notifications.ts:

import { Router } from 'express';
import { sendPushNotification } from '../services/firebase';

const router = Router();

router.post('/send', async (req, res) => {
  try {
    const { title, body, token, tokens, topic, data } = req.body;

    if (!title || !body) {
      return res.status(400).json({ error: 'Title and body are required' });
    }

    if (!token && !tokens && !topic) {
      return res.status(400).json({ error: 'At least one of token, tokens, or topic is required' });
    }

    const result = await sendPushNotification({
      title,
      body,
      token,
      tokens,
      topic,
      data,
    });

    res.json({ success: true, result });
  } catch (error) {
    console.error('Error sending notification:', error);
    res.status(500).json({ error: 'Failed to send notification' });
  }
});

router.post('/subscribe-topic', async (req, res) => {
  try {
    const { tokens, topic } = req.body;

    if (!tokens || !topic) {
      return res.status(400).json({ error: 'Tokens and topic are required' });
    }

    const result = await admin.messaging().subscribeToTopic(tokens, topic);
    res.json({ success: true, result });
  } catch (error) {
    console.error('Error subscribing to topic:', error);
    res.status(500).json({ error: 'Failed to subscribe to topic' });
  }
});

export { router as notificationRoutes };

    });
  });

2.3 Socket.IO Hook​

Create src/hooks/useSocket.ts:

import { useEffect, useRef, useState } from 'react';
import { io, Socket } from 'socket.io-client';

interface Notification {
  id: string;
  title: string;
  body: string;
  data?: any;
  timestamp: string;
}

export const useSocket = (serverUrl: string, userId?: string) => {
  const [socket, setSocket] = useState<Socket | null>(null);
  const [connected, setConnected] = useState(false);
  const [notifications, setNotifications] = useState<Notification[]>([]);
  const socketRef = useRef<Socket | null>(null);

  useEffect(() => {
    if (!userId) return;

    const socketInstance = io(serverUrl);
    socketRef.current = socketInstance;
    setSocket(socketInstance);

    socketInstance.on('connect', () => {
      setConnected(true);
      console.log('Connected to server');
    });

    socketInstance.on('disconnect', () => {
      setConnected(false);
      console.log('Disconnected from server');
    });

    socketInstance.on('authenticated', (data) => {
      console.log('Authenticated:', data);
    });

    socketInstance.on('notification', (notification: Notification) => {
      setNotifications((prev) => [notification, ...prev]);
      
      // Show browser notification if permission granted
      if ('Notification' in window && Notification.permission === 'granted') {
        new Notification(notification.title, {
          body: notification.body,
          icon: '/icon-192x192.png',
        });
      }
    });

    return () => {
      socketInstance.disconnect();
      socketRef.current = null;
    };
  }, [serverUrl, userId]);

  const authenticate = async (fcmToken?: string) => {
    if (socket && userId) {
      socket.emit('authenticate', { userId, fcmToken });
    }
  };

  const sendNotification = (data: {
    recipientId: string;
    title: string;
    body: string;
    type: 'realtime' | 'push' | 'both';
    data?: any;
  }) => {
    if (socket) {
      socket.emit('sendNotification', data);
    }
  };

  return {
    socket,
    connected,
    notifications,
    authenticate,
    sendNotification,
    clearNotifications: () => setNotifications([]),
  };
};

2.4 Main App Component​

Create src/app/page.tsx:

'use client';

import { useEffect, useState } from 'react';
import { useSocket } from '@/hooks/useSocket';
import { getFCMToken } from '@/lib/firebase';

export default function Home() {
  const [userId, setUserId] = useState('');
  const [fcmToken, setFcmToken] = useState<string | null>(null);
  const [recipientId, setRecipientId] = useState('');
  const [notificationTitle, setNotificationTitle] = useState('');
  const [notificationBody, setNotificationBody] = useState('');
  const [notificationType, setNotificationType] = useState<'realtime' | 'push' | 'both'>('both');

  const { connected, notifications, authenticate, sendNotification, clearNotifications } = useSocket(
    process.env.NEXT_PUBLIC_BACKEND_URL || 'http://localhost:5000',
    userId
  );

  useEffect(() => {
    // Request notification permission
    if ('Notification' in window && Notification.permission === 'default') {
      Notification.requestPermission();
    }

    // Get FCM token
    getFCMToken().then(setFcmToken);
  }, []);

  const handleLogin = () => {
    if (userId.trim()) {
      authenticate(fcmToken || undefined);
    }
  };

  const handleSendNotification = () => {
    if (recipientId && notificationTitle && notificationBody) {
      sendNotification({
        recipientId,
        title: notificationTitle,
        body: notificationBody,
        type: notificationType,
      });
      
      // Clear form
      setRecipientId('');
      setNotificationTitle('');
      setNotificationBody('');
    }
  };

  return (
    <div className="min-h-screen bg-gray-100 p-8">
      <div className="max-w-4xl mx-auto">
        <h1 className="text-3xl font-bold text-center mb-8">Real-time Notifications Demo</h1>
        
        {/* Connection Status */}
        <div className="bg-white rounded-lg shadow p-6 mb-6">
          <h2 className="text-xl font-semibold mb-4">Connection Status</h2>
          <div className="flex items-center space-x-4">
            <div className={`w-4 h-4 rounded-full ${connected ? 'bg-green-500' : 'bg-red-500'}`}></div>
            <span>{connected ? 'Connected' : 'Disconnected'}</span>
          </div>
          {fcmToken && (
            <div className="mt-4">
              <p className="text-sm text-gray-600">FCM Token: {fcmToken.substring(0, 50)}...</p>
            </div>
          )}
        </div>

        {/* Login */}
        {!connected && (
          <div className="bg-white rounded-lg shadow p-6 mb-6">
            <h2 className="text-xl font-semibold mb-4">Login</h2>
            <div className="flex space-x-4">
              <input
                type="text"
                placeholder="Enter your user ID"
                value={userId}
                onChange={(e) => setUserId(e.target.value)}
                className="flex-1 px-4 py-2 border rounded-lg"
              />
              <button
                onClick={handleLogin}
                className="px-6 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600"
              >
                Connect
              </button>
            </div>
          </div>
        )}

        {/* Send Notification */}
        {connected && (
          <div className="bg-white rounded-lg shadow p-6 mb-6">
            <h2 className="text-xl font-semibold mb-4">Send Notification</h2>
            <div className="space-y-4">
              <input
                type="text"
                placeholder="Recipient User ID"
                value={recipientId}
                onChange={(e) => setRecipientId(e.target.value)}
                className="w-full px-4 py-2 border rounded-lg"
              />
              <input
                type="text"
                placeholder="Notification Title"
                value={notificationTitle}
                onChange={(e) => setNotificationTitle(e.target.value)}
                className="w-full px-4 py-2 border rounded-lg"
              />
              <textarea
                placeholder="Notification Body"
                value={notificationBody}
                onChange={(e) => setNotificationBody(e.target.value)}
                className="w-full px-4 py-2 border rounded-lg h-24"
              />
              <select
                value={notificationType}
                onChange={(e) => setNotificationType(e.target.value as any)}
                className="w-full px-4 py-2 border rounded-lg"
              >
                <option value="realtime">Real-time Only</option>
                <option value="push">Push Only</option>
                <option value="both">Both</option>
              </select>
              <button
                onClick={handleSendNotification}
                className="w-full px-6 py-2 bg-green-500 text-white rounded-lg hover:bg-green-600"
              >
                Send Notification
              </button>
            </div>
          </div>
        )}

        {/* Notifications */}
        <div className="bg-white rounded-lg shadow p-6">
          <div className="flex justify-between items-center mb-4">
            <h2 className="text-xl font-semibold">Notifications ({notifications.length})</h2>
            {notifications.length > 0 && (
              <button
                onClick={clearNotifications}
                className="px-4 py-2 bg-gray-500 text-white rounded-lg hover:bg-gray-600"
              >
                Clear All
              </button>
            )}
          </div>
          <div className="space-y-4 max-h-96 overflow-y-auto">
            {notifications.length === 0 ? (
              <p className="text-gray-500 text-center py-8">No notifications yet</p>
            ) : (
              notifications.map((notification) => (
                <div key={notification.id} className="border rounded-lg p-4 bg-gray-50">
                  <h3 className="font-semibold">{notification.title}</h3>
                  <p className="text-gray-700">{notification.body}</p>
                  <p className="text-sm text-gray-500 mt-2">
                    {new Date(notification.timestamp).toLocaleString()}
                  </p>
                </div>
              ))
            )}
          </div>
        </div>
      </div>
    </div>
  );
}

Part 7: Testing and Validation​

7.1 Load Testing Script​

Create test/load-test.js:

const io = require('socket.io-client');

const BACKEND_URL = process.env.BACKEND_URL || 'http://localhost:5000';
const NUM_CLIENTS = 100;
const TEST_DURATION = 60000; // 1 minute

async function runLoadTest() {
  console.log(`Starting load test with ${NUM_CLIENTS} clients...`);
  
  const clients = [];
  const startTime = Date.now();
  
  // Create clients
  for (let i = 0; i < NUM_CLIENTS; i++) {
    const client = io(BACKEND_URL);
    clients.push(client);
    
    client.on('connect', () => {
      client.emit('authenticate', { userId: `user-${i}` });
    });
    
    client.on('notification', (data) => {
      console.log(`Client ${i} received notification: ${data.title}`);
    });
  }
  
  // Send notifications periodically
  const interval = setInterval(() => {
    const sender = Math.floor(Math.random() * NUM_CLIENTS);
    const recipient = Math.floor(Math.random() * NUM_CLIENTS);
    
    clients[sender].emit('sendNotification', {
      recipientId: `user-${recipient}`,
      title: 'Load Test Notification',
      body: `Message from user-${sender} to user-${recipient}`,
      type: 'realtime'
    });
  }, 100);
  
  // Stop after test duration
  setTimeout(() => {
    clearInterval(interval);
    clients.forEach(client => client.disconnect());
    console.log(`Load test completed in ${Date.now() - startTime}ms`);
    process.exit(0);
  }, TEST_DURATION);
}

runLoadTest();

7.2 Health Check Script​

Create scripts/health-check.sh:

#!/bin/bash

BACKEND_URL="http://localhost:5000"
FRONTEND_URL="http://localhost:3000"

# Check backend health
echo "Checking backend health..."
BACKEND_STATUS=$(curl -s -o /dev/null -w "%{http_code}" $BACKEND_URL/health)
if [ $BACKEND_STATUS -eq 200 ]; then
  echo "✅ Backend is healthy"
else
  echo "❌ Backend is unhealthy (HTTP $BACKEND_STATUS)"
  exit 1
fi

# Check frontend health
echo "Checking frontend health..."
FRONTEND_STATUS=$(curl -s -o /dev/null -w "%{http_code}" $FRONTEND_URL)
if [ $FRONTEND_STATUS -eq 200 ]; then
  echo "✅ Frontend is healthy"
else
  echo "❌ Frontend is unhealthy (HTTP $FRONTEND_STATUS)"
  exit 1
fi

echo "🎉 All services are healthy!"
