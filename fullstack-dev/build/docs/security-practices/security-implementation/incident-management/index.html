// src/security/incident-detection.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { Cron, CronExpression } from '@nestjs/schedule';

interface SecurityIncident {
  id: string;
  type: 'BREACH' | 'ATTACK' | 'VULNERABILITY' | 'POLICY_VIOLATION' | 'SYSTEM_COMPROMISE';
  severity: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  title: string;
  description: string;
  source: string;
  timestamp: Date;
  status: 'OPEN' | 'INVESTIGATING' | 'CONTAINED' | 'RESOLVED' | 'CLOSED';
  assignee?: string;
  metadata: Record<string, any>;
  affectedSystems: string[];
  evidence: string[];
}

@Injectable()
export class IncidentDetectionService {
  private readonly logger = new Logger(IncidentDetectionService.name);
  private readonly incidents = new Map<string, SecurityIncident>();

  constructor(private configService: ConfigService) {}

  @Cron(CronExpression.EVERY_MINUTE)
  async detectSecurityIncidents(): Promise<void> {
    await Promise.all([
      this.checkUnauthorizedAccess(),
      this.checkSuspiciousNetworkActivity(),
      this.checkDataExfiltration(),
      this.checkMalwareSignatures(),
      this.checkConfigurationChanges(),
    ]);
  }

  async createIncident(incident: Omit<SecurityIncident, 'id' | 'timestamp' | 'status'>): Promise<string> {
    const incidentId = this.generateIncidentId();
    const fullIncident: SecurityIncident = {
      ...incident,
      id: incidentId,
      timestamp: new Date(),
      status: 'OPEN',
    };

    this.incidents.set(incidentId, fullIncident);
    
    await this.triggerIncidentResponse(fullIncident);
    
    this.logger.error(`Security incident created: ${incidentId}`, fullIncident);
    
    return incidentId;
  }

  async updateIncidentStatus(incidentId: string, status: SecurityIncident['status']): Promise<void> {
    const incident = this.incidents.get(incidentId);
    if (incident) {
      incident.status = status;
      this.logger.log(`Incident ${incidentId} status updated to: ${status}`);
    }
  }

  async getIncident(incidentId: string): Promise<SecurityIncident | undefined> {
    return this.incidents.get(incidentId);
  }

  async getAllIncidents(): Promise<SecurityIncident[]> {
    return Array.from(this.incidents.values());
  }

  private async checkUnauthorizedAccess(): Promise<void> {
    // Check for failed login attempts, privilege escalation, etc.
    const failedLogins = await this.getFailedLoginAttempts();
    
    if (failedLogins.length > 10) {
      await this.createIncident({
        type: 'ATTACK',
        severity: 'HIGH',
        title: 'Multiple Failed Login Attempts Detected',
        description: `${failedLogins.length} failed login attempts detected in the last minute`,
        source: 'auth-monitor',
        metadata: { failedLogins },
        affectedSystems: ['authentication-service'],
        evidence: ['auth-logs'],
      });
    }
  }

  private async checkSuspiciousNetworkActivity(): Promise<void> {
    // Check for unusual network patterns, port scans, etc.
    const networkMetrics = await this.getNetworkMetrics();
    
    if (networkMetrics.anomalousConnections > 100) {
      await this.createIncident({
        type: 'ATTACK',
        severity: 'MEDIUM',
        title: 'Suspicious Network Activity Detected',
        description: 'Unusual network connection patterns detected',
        source: 'network-monitor',
        metadata: { networkMetrics },
        affectedSystems: ['network-infrastructure'],
        evidence: ['network-logs'],
      });
    }
  }

  private async checkDataExfiltration(): Promise<void> {
    // Check for unusual data access patterns or large data transfers
    const dataAccessMetrics = await this.getDataAccessMetrics();
    
    if (dataAccessMetrics.unusualAccess) {
      await this.createIncident({
        type: 'BREACH',
        severity: 'CRITICAL',
        title: 'Potential Data Exfiltration Detected',
        description: 'Unusual data access patterns suggest potential data breach',
        source: 'data-monitor',
        metadata: { dataAccessMetrics },
        affectedSystems: ['database', 'file-storage'],
        evidence: ['audit-logs', 'database-logs'],
      });
    }
  }

  private async checkMalwareSignatures(): Promise<void> {
    // Check for malware signatures in uploads, memory, etc.
    // This would integrate with antivirus/malware detection systems
  }

  private async checkConfigurationChanges(): Promise<void> {
    // Check for unauthorized configuration changes
    const configChanges = await this.getRecentConfigChanges();
    
    const unauthorizedChanges = configChanges.filter(change => !change.authorized);
    
    if (unauthorizedChanges.length > 0) {
      await this.createIncident({
        type: 'POLICY_VIOLATION',
        severity: 'HIGH',
        title: 'Unauthorized Configuration Changes',
        description: 'Unauthorized changes detected to system configuration',
        source: 'config-monitor',
        metadata: { unauthorizedChanges },
        affectedSystems: ['kubernetes', 'database'],
        evidence: ['audit-logs', 'config-history'],
      });
    }
  }

  private async triggerIncidentResponse(incident: SecurityIncident): Promise<void> {
    // Trigger automated incident response based on severity
    switch (incident.severity) {
      case 'CRITICAL':
        await this.triggerEmergencyResponse(incident);
        break;
      case 'HIGH':
        await this.triggerHighPriorityResponse(incident);
        break;
      case 'MEDIUM':
        await this.triggerMediumPriorityResponse(incident);
        break;
      case 'LOW':
        await this.triggerLowPriorityResponse(incident);
        break;
    }
  }

  private async triggerEmergencyResponse(incident: SecurityIncident): Promise<void> {
    // Immediate response for critical incidents
    await this.notifySecurityTeam(incident, 'IMMEDIATE');
    await this.activateIncidentResponseTeam(incident);
    await this.initializeContainmentProcedures(incident);
  }

  private async triggerHighPriorityResponse(incident: SecurityIncident): Promise<void> {
    // Response for high priority incidents
    await this.notifySecurityTeam(incident, 'HIGH');
    await this.escalateToSecurityLead(incident);
  }

  private async triggerMediumPriorityResponse(incident: SecurityIncident): Promise<void> {
    // Response for medium priority incidents
    await this.notifySecurityTeam(incident, 'MEDIUM');
    await this.assignToSecurityAnalyst(incident);
  }

  private async triggerLowPriorityResponse(incident: SecurityIncident): Promise<void> {
    // Response for low priority incidents
    await this.logIncidentForReview(incident);
  }

  private generateIncidentId(): string {
    const timestamp = Date.now();
    const random = Math.random().toString(36).substring(2, 8);
    return `INC-${timestamp}-${random}`.toUpperCase();
  }

  }

  private async getRecentConfigChanges(): Promise<any[]> {

  }

    this.logger.error(`[${priority}] Security team notified about incident: ${incident.id}`);
  }

Google Cloud Alerting Integration​

// src/monitoring/cloud-alerting.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { Monitoring } from '@google-cloud/monitoring';

@Injectable()
export class CloudAlertingService {
  private readonly logger = new Logger(CloudAlertingService.name);
  private readonly client: Monitoring.AlertPolicyServiceClient;

  constructor(private configService: ConfigService) {
    this.client = new Monitoring.AlertPolicyServiceClient();
  }

  async createSecurityAlertPolicies(): Promise<void> {
    const projectId = this.configService.get('GCP_PROJECT_ID');
    const projectName = this.client.projectPath(projectId);

    const alertPolicies = [
      {
        displayName: 'High Failed Login Rate',
        conditions: [{
          displayName: 'Failed login rate above threshold',
          conditionThreshold: {
            filter: 'resource.type="k8s_container" AND metric.type="custom.googleapis.com/auth/failed_logins"',
            comparison: 'COMPARISON_GREATER_THAN',
            thresholdValue: { doubleValue: 10 },
            duration: { seconds: 300 },
          },
        }],
        alertStrategy: {
          autoClose: { seconds: 1800 },
        },
        enabled: { value: true },
      },
      {
        displayName: 'Suspicious Network Activity',
        conditions: [{
          displayName: 'Unusual network connections',
          conditionThreshold: {
            filter: 'resource.type="gce_instance" AND metric.type="compute.googleapis.com/instance/network/received_bytes_count"',
            comparison: 'COMPARISON_GREATER_THAN',
            thresholdValue: { doubleValue: 1000000000 }, // 1GB
            duration: { seconds: 600 },
          },
        }],
        enabled: { value: true },
      },
    ];

    for (const policy of alertPolicies) {
      try {
        const [createdPolicy] = await this.client.createAlertPolicy({
          name: projectName,
          alertPolicy: policy,
        });
        this.logger.log(`Created alert policy: ${createdPolicy.displayName}`);
      } catch (error) {
        this.logger.error(`Failed to create alert policy ${policy.displayName}:`, error);
      }
    }
  }

  async sendIncidentNotification(incident: any): Promise<void> {
    // Integration with notification channels (Slack, PagerDuty, email)
    const notificationChannels = await this.getNotificationChannels();
    
    for (const channel of notificationChannels) {
      await this.sendNotification(channel, incident);
    }
  }

  private async getNotificationChannels(): Promise<any[]> {
    const projectId = this.configService.get('GCP_PROJECT_ID');
    const projectName = this.client.projectPath(projectId);

    try {
      const [channels] = await this.client.listNotificationChannels({
        name: projectName,
      });
      return channels;
    } catch (error) {
      this.logger.error('Failed to get notification channels:', error);
      return [];
    }
  }

  private async sendNotification(channel: any, incident: any): Promise<void> {
    // Implementation depends on notification channel type
    this.logger.log(`Sending notification via ${channel.type} for incident: ${incident.id}`);
  }
}

import { Injectable, Logger } from '@nestjs/common';

interface IncidentResponsePlan {
  incidentType: string;
  severity: string;
  steps: IncidentResponseStep[];
  timeline: Record<string, number>; // minutes
  roles: string[];
  approvals: string[];
}

interface IncidentResponseStep {
  id: string;
  title: string;
  description: string;
  automated: boolean;
  requiredRole: string;
  estimatedTime: number;
  dependencies: string[];
  actions: string[];
}

@Injectable()
export class IncidentResponseService {
  private readonly logger = new Logger(IncidentResponseService.name);
  private readonly responsePlans = new Map<string, IncidentResponsePlan>();

  constructor(private configService: ConfigService) {
    this.initializeResponsePlans();
  }

  async executeIncidentResponse(incidentId: string, incidentType: string, severity: string): Promise<void> {
    const planKey = `${incidentType}-${severity}`;
    const plan = this.responsePlans.get(planKey);

    if (!plan) {
      this.logger.error(`No response plan found for ${planKey}`);
      return;
    }

    this.logger.log(`Executing incident response plan for incident: ${incidentId}`);

    for (const step of plan.steps) {
      try {
        await this.executeResponseStep(incidentId, step);
      } catch (error) {
        this.logger.error(`Failed to execute step ${step.id} for incident ${incidentId}:`, error);
        
        if (step.id === 'containment' || step.id === 'isolation') {
          // Critical steps - escalate immediately
          await this.escalateIncident(incidentId, `Critical step failed: ${step.title}`);
        }
      }
    }
  }

  private async executeResponseStep(incidentId: string, step: IncidentResponseStep): Promise<void> {
    this.logger.log(`Executing step: ${step.title} for incident: ${incidentId}`);

    if (step.automated) {
      await this.executeAutomatedActions(incidentId, step);
    } else {
      await this.executeManualActions(incidentId, step);
    }
  }

  private async executeAutomatedActions(incidentId: string, step: IncidentResponseStep): Promise<void> {
    for (const action of step.actions) {
      switch (action) {
        case 'isolate-affected-systems':
          await this.isolateAffectedSystems(incidentId);
          break;
        case 'block-suspicious-ips':
          await this.blockSuspiciousIPs(incidentId);
          break;
        case 'rotate-secrets':
          await this.rotateSecrets(incidentId);
          break;
        case 'backup-evidence':
          await this.backupEvidence(incidentId);
          break;
        case 'notify-stakeholders':
          await this.notifyStakeholders(incidentId);
          break;
        default:
          this.logger.warn(`Unknown automated action: ${action}`);
      }
    }
  }

  private async executeManualActions(incidentId: string, step: IncidentResponseStep): Promise<void> {

// src/security/digital-forensics.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

interface ForensicEvidence {
  id: string;
  incidentId: string;
  type: 'LOG' | 'DISK_IMAGE' | 'MEMORY_DUMP' | 'NETWORK_CAPTURE' | 'FILE_SYSTEM';
  source: string;
  timestamp: Date;
  hash: string;
  location: string;
  metadata: Record<string, any>;
  chainOfCustody: ChainOfCustodyEntry[];
}

interface ChainOfCustodyEntry {
  timestamp: Date;
  action: 'COLLECTED' | 'TRANSFERRED' | 'ANALYZED' | 'STORED';
  person: string;
  location: string;
  notes: string;
}

@Injectable()
export class DigitalForensicsService {
  private readonly logger = new Logger(DigitalForensicsService.name);
  private readonly evidence = new Map<string, ForensicEvidence>();

  constructor(private configService: ConfigService) {}

  async collectEvidence(incidentId: string, evidenceType: string, source: string): Promise<string> {
    const evidenceId = this.generateEvidenceId();
    
    let evidence: ForensicEvidence;
    
    switch (evidenceType) {
      case 'LOG':
        evidence = await this.collectLogEvidence(incidentId, source);
        break;
      case 'DISK_IMAGE':
        evidence = await this.collectDiskImage(incidentId, source);
        break;
      case 'MEMORY_DUMP':
        evidence = await this.collectMemoryDump(incidentId, source);
        break;
      case 'NETWORK_CAPTURE':
        evidence = await this.collectNetworkCapture(incidentId, source);
        break;
      default:
        throw new Error(`Unsupported evidence type: ${evidenceType}`);
    }

    evidence.id = evidenceId;
    this.evidence.set(evidenceId, evidence);
    
    this.logger.log(`Evidence collected: ${evidenceId} for incident: ${incidentId}`);
    
    return evidenceId;
  }

  private async collectLogEvidence(incidentId: string, source: string): Promise<ForensicEvidence> {
    const timestamp = new Date();
    
    // Collect logs from various sources
    const logs = await this.exportLogs(source, timestamp);
    const hash = await this.calculateHash(logs);
    const location = await this.storeSecurely(logs, `logs-${incidentId}-${Date.now()}`);

    return {
      id: '',
      incidentId,
      type: 'LOG',
      source,
      timestamp,
      hash,
      location,
      metadata: {
        logSize: logs.length,
        timeRange: {
          start: timestamp,
          end: timestamp,
        },
      },
      chainOfCustody: [{
        timestamp,
        action: 'COLLECTED',
        person: 'automated-system',
        location: 'kubernetes-cluster',
        notes: 'Automated log collection during incident response',
      }],
    };
  }

  private async collectDiskImage(incidentId: string, source: string): Promise<ForensicEvidence> {
    const timestamp = new Date();
    
    // Create disk image of affected systems
    const imageFile = await this.createDiskImage(source);
    const hash = await this.calculateFileHash(imageFile);
    const location = await this.storeSecurely(imageFile, `disk-image-${incidentId}-${Date.now()}`);

    return {
      id: '',
      incidentId,
      type: 'DISK_IMAGE',
      source,
      timestamp,
      hash,
      location,
      metadata: {
        imageSize: await this.getFileSize(imageFile),
        compressionUsed: true,
        mountPoints: await this.getMountPoints(source),
      },
      chainOfCustody: [{
        timestamp,
        action: 'COLLECTED',
        person: 'automated-system',
        location: source,
        notes: 'Forensic disk image created',
      }],
    };
  }

  private async collectMemoryDump(incidentId: string, source: string): Promise<ForensicEvidence> {
    const timestamp = new Date();
    
    // Create memory dump of affected systems
    const memoryDump = await this.createMemoryDump(source);
    const hash = await this.calculateFileHash(memoryDump);
    const location = await this.storeSecurely(memoryDump, `memory-dump-${incidentId}-${Date.now()}`);

    return {
      id: '',
      incidentId,
      type: 'MEMORY_DUMP',
      source,
      timestamp,
      hash,
      location,
      metadata: {
        memorySize: await this.getFileSize(memoryDump),
        processCount: await this.getProcessCount(source),
      },
      chainOfCustody: [{
        timestamp,
        action: 'COLLECTED',
        person: 'automated-system',
        location: source,
        notes: 'Memory dump captured for analysis',
      }],
    };
  }

  private async collectNetworkCapture(incidentId: string, source: string): Promise<ForensicEvidence> {
    const timestamp = new Date();
    
    // Capture network traffic
    const networkCapture = await this.captureNetworkTraffic(source);
    const hash = await this.calculateFileHash(networkCapture);
    const location = await this.storeSecurely(networkCapture, `network-capture-${incidentId}-${Date.now()}`);

    return {
      id: '',
      incidentId,
      type: 'NETWORK_CAPTURE',
      source,

    this.logger.log(`Automating incident response for: ${incidentId}`);

    const automationPlaybook = this.getAutomationPlaybook(incidentType);
    
    for (const step of automationPlaybook.steps) {
      try {
        await this.executeAutomationStep(incidentId, step);

        }
      }
    }
  }
