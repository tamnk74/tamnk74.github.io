Sensitive System Data Protection Implementation

This document provides implementation guidelines for protecting sensitive system data including connection strings, cryptographic keys, API tokens, and other critical infrastructure secrets in NestJS/Next.js applications on GCP Kubernetes.

Google Secret Manager Integration​

1. Secret Manager Service​

// src/config/secret-manager.service.ts
import { Injectable, Logger, OnModuleInit } from '@nestjs/common';
import { SecretManagerServiceClient } from '@google-cloud/secret-manager';

@Injectable()
export class SecretManagerService implements OnModuleInit {
  private readonly logger = new Logger(SecretManagerService.name);
  private client: SecretManagerServiceClient;
  private projectId: string;

  constructor() {
    this.projectId = process.env.GCP_PROJECT_ID || '';
    this.client = new SecretManagerServiceClient();
  }

  async onModuleInit() {
    this.logger.log('Initializing Secret Manager Service');
    await this.validateConnection();
  }

  async getSecret(secretName: string, version: string = 'latest'): Promise<string> {
    try {
      const secretPath = `projects/${this.projectId}/secrets/${secretName}/versions/${version}`;
      
      const [accessResponse] = await this.client.accessSecretVersion({
        name: secretPath,
      });

      const secretValue = accessResponse.payload?.data?.toString();
      if (!secretValue) {
        throw new Error(`Secret ${secretName} is empty or undefined`);
      }

      this.logger.debug(`Successfully retrieved secret: ${secretName}`);
      return secretValue;
    } catch (error) {
      this.logger.error(`Failed to retrieve secret ${secretName}:`, error);
      throw new Error(`Failed to retrieve secret: ${secretName}`);
    }
  }

  async createSecret(secretName: string, secretValue: string): Promise<void> {
    try {
      // Create the secret
      await this.client.createSecret({
        parent: `projects/${this.projectId}`,
        secretId: secretName,
        secret: {
          replication: {
            automatic: {},
          },
        },
      });

      // Add the secret version
      await this.client.addSecretVersion({
        parent: `projects/${this.projectId}/secrets/${secretName}`,
        payload: {
          data: Buffer.from(secretValue, 'utf8'),
        },
      });

      this.logger.log(`Secret ${secretName} created successfully`);
    } catch (error) {
      this.logger.error(`Failed to create secret ${secretName}:`, error);
      throw error;
    }
  }

  async updateSecret(secretName: string, secretValue: string): Promise<void> {
    try {
      await this.client.addSecretVersion({
        parent: `projects/${this.projectId}/secrets/${secretName}`,
        payload: {
          data: Buffer.from(secretValue, 'utf8'),
        },
      });

      this.logger.log(`Secret ${secretName} updated successfully`);
    } catch (error) {
      this.logger.error(`Failed to update secret ${secretName}:`, error);
      throw error;
    }
  }

  async listSecrets(): Promise<string[]> {
    try {
      const [secrets] = await this.client.listSecrets({
        parent: `projects/${this.projectId}`,
      });

      return secrets.map(secret => secret.name?.split('/').pop() || '');
    } catch (error) {
      this.logger.error('Failed to list secrets:', error);
      throw error;
    }
  }

  private async validateConnection(): Promise<void> {
    try {
      await this.client.listSecrets({
        parent: `projects/${this.projectId}`,
        pageSize: 1,
      });
      this.logger.log('Secret Manager connection validated');
    } catch (error) {
      this.logger.error('Failed to validate Secret Manager connection:', error);
      throw error;
    }
  }
}

2. Configuration Service with Secret Manager​

// src/config/config.service.ts
import { Injectable } from '@nestjs/common';
import { SecretManagerService } from './secret-manager.service';

@Injectable()
export class ConfigService {
  private secrets: Map<string, string> = new Map();

  constructor(private secretManager: SecretManagerService) {}

  async loadSecrets(): Promise<void> {
    const secretNames = [
      'database-password',
      'jwt-secret',
      'api-key-stripe',
      'api-key-sendgrid',
      'encryption-key',
      'session-secret'
    ];

    for (const secretName of secretNames) {
      try {
        const secretValue = await this.secretManager.getSecret(secretName);
        this.secrets.set(secretName, secretValue);
      } catch (error) {
        console.error(`Failed to load secret ${secretName}:`, error);
        // Use environment variable as fallback
        const envValue = process.env[secretName.toUpperCase().replace(/-/g, '_')];
        if (envValue) {
          this.secrets.set(secretName, envValue);
        }
      }
    }
  }

  get(key: string): string {
    const value = this.secrets.get(key) || process.env[key];
    if (!value) {
      throw new Error(`Configuration value for ${key} not found`);
    }
    return value;
  }

  getDatabaseConfig() {
    return {
      host: this.get('DB_HOST'),
      port: parseInt(this.get('DB_PORT')),
      username: this.get('DB_USERNAME'),
      password: this.get('database-password'),
      database: this.get('DB_NAME'),
    };
  }

  getJwtConfig() {
    return {
      secret: this.get('jwt-secret'),
      expiresIn: this.get('JWT_EXPIRES_IN') || '1h',
    };
  }
}

Key Rotation Strategies​

1. Automated Key Rotation Service​

// src/security/key-rotation.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import { SecretManagerService } from '../config/secret-manager.service';
import { randomBytes } from 'crypto';

@Injectable()
export class KeyRotationService {
  private readonly logger = new Logger(KeyRotationService.name);

  constructor(private secretManager: SecretManagerService) {}

  @Cron(CronExpression.EVERY_1ST_DAY_OF_MONTH_AT_MIDNIGHT)
  async rotateKeys(): Promise<void> {
    this.logger.log('Starting monthly key rotation');

    try {
      await this.rotateJwtSecret();
      await this.rotateEncryptionKey();
      await this.rotateSessionSecret();
      
      this.logger.log('Key rotation completed successfully');
    } catch (error) {
      this.logger.error('Key rotation failed:', error);
      // Send alert to monitoring system
      await this.sendRotationAlert('failed', error.message);
    }
  }

  async rotateJwtSecret(): Promise<void> {
    const newSecret = this.generateSecureSecret(64);
    await this.secretManager.updateSecret('jwt-secret', newSecret);
    this.logger.log('JWT secret rotated');
  }

  async rotateEncryptionKey(): Promise<void> {
    const newKey = this.generateSecureSecret(32);
    await this.secretManager.updateSecret('encryption-key', newKey);
    this.logger.log('Encryption key rotated');
  }

  async rotateSessionSecret(): Promise<void> {
    const newSecret = this.generateSecureSecret(64);
    await this.secretManager.updateSecret('session-secret', newSecret);
    this.logger.log('Session secret rotated');
  }

  private generateSecureSecret(length: number): string {
    return randomBytes(length).toString('hex');
  }

  private async sendRotationAlert(status: 'success' | 'failed', message: string): Promise<void> {
    // Implementation for sending alerts (Slack, email, etc.)
    this.logger.log(`Key rotation ${status}: ${message}`);
  }

  // Manual rotation endpoint for emergency situations
  async emergencyRotation(secretName: string): Promise<void> {
    this.logger.warn(`Emergency rotation initiated for ${secretName}`);
    
    const newSecret = this.generateSecureSecret(64);
    await this.secretManager.updateSecret(secretName, newSecret);
    
    await this.sendRotationAlert('success', `Emergency rotation completed for ${secretName}`);
  }
}

2. Database Password Rotation​

// src/security/db-rotation.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { SecretManagerService } from '../config/secret-manager.service';
import { DataSource } from 'typeorm';

@Injectable()
export class DatabaseRotationService {
  private readonly logger = new Logger(DatabaseRotationService.name);

  constructor(
    private secretManager: SecretManagerService,
    private dataSource: DataSource
  ) {}

  async rotateDatabasePassword(): Promise<void> {
    const newPassword = this.generatePassword(16);
    const username = process.env.DB_USERNAME;

    try {
      // Update password in database
      await this.dataSource.query(
        `ALTER USER ${username} WITH PASSWORD $1`,
        [newPassword]
      );

      // Update secret in Secret Manager
      await this.secretManager.updateSecret('database-password', newPassword);

      this.logger.log('Database password rotated successfully');
    } catch (error) {
      this.logger.error('Failed to rotate database password:', error);
      throw error;
    }
  }

  private generatePassword(length: number): string {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*';
    let password = '';
    for (let i = 0; i < length; i++) {
      password += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return password;
  }
}

import { ConfigModule } from '@nestjs/config';

      console.error('Failed to load secrets from Secret Manager:', error);
      throw error;
    }
  }

  return baseConfig;
};

2. Environment Validation​

// src/config/env.validation.ts
import { plainToClass } from 'class-transformer';
import { IsEnum, IsNumber, IsString, validateSync, IsOptional } from 'class-validator';

enum Environment {
  Development = 'development',
  Production = 'production',
  Test = 'test',
}

class EnvironmentVariables {
  @IsEnum(Environment)
  NODE_ENV: Environment;

  @IsNumber()
  PORT: number;

  @IsString()
  DB_HOST: string;

  @IsNumber()
  DB_PORT: number;

  @IsString()
  DB_USERNAME: string;

  @IsString()
  DB_NAME: string;

  @IsString()
  @IsOptional()
  DB_PASSWORD: string; // Optional as it comes from Secret Manager

  @IsString()
  GCP_PROJECT_ID: string;

  @IsString()
  @IsOptional()
  CORS_ORIGIN: string;
}

export function validate(config: Record<string, unknown>) {
  const validatedConfig = plainToClass(EnvironmentVariables, config, {
    enableImplicitConversion: true,
  });

  const errors = validateSync(validatedConfig, { skipMissingProperties: false });

  if (errors.length > 0) {
    throw new Error(errors.toString());
  }
  return validatedConfig;
}

Secret Scanning and Detection​

1. Pre-commit Hook for Secret Scanning​

#!/bin/bash
# .git/hooks/pre-commit

echo "Running secret scan..."

# Install truffleHog if not present
if ! command -v trufflehog &> /dev/null; then
    echo "Installing truffleHog..."
    go install github.com/trufflesecurity/trufflehog/v3@latest
fi

# Scan staged files for secrets
trufflehog git file://. --since-commit HEAD --only-verified --fail

if [ $? -ne 0 ]; then
    echo "❌ Secret scan failed! Please remove secrets before committing."
    exit 1
fi

echo "✅ Secret scan passed!"

3. Runtime Secret Detection​

// src/security/secret-detector.service.ts
import { Injectable, Logger } from '@nestjs/common';

@Injectable()
export class SecretDetectorService {
  private readonly logger = new Logger(SecretDetectorService.name);
  
  private readonly patterns = {
    awsAccessKey: /AKIA[0-9A-Z]{16}/g,
    awsSecretKey: /[0-9a-zA-Z/+]{40}/g,
    googleApiKey: /AIza[0-9A-Za-z-_]{35}/g,
    githubToken: /gh[ospru]_[A-Za-z0-9_]{36}/g,
    slackToken: /xox[baprs]-([0-9a-zA-Z]{10,48})/g,
    jwt: /eyJ[A-Za-z0-9-_=]+\.[A-Za-z0-9-_=]+\.?[A-Za-z0-9-_.+/=]*/g,
    genericSecret: /(?i)(secret|password|key|token|pwd|pass)[\s]*[:=][\s]*['"]?[a-zA-Z0-9!@#$%^&*]{8,}/g,
  };

  scanText(text: string): { type: string; match: string; position: number }[] {
    const findings: { type: string; match: string; position: number }[] = [];

    Object.entries(this.patterns).forEach(([type, pattern]) => {
      let match;
      while ((match = pattern.exec(text)) !== null) {
        findings.push({
          type,
          match: match[0],
          position: match.index,
        });
      }
    });

    if (findings.length > 0) {
      this.logger.warn(`Potential secrets detected:`, findings);
    }

    return findings;
  }

  scanObject(obj: any, path: string = ''): { type: string; path: string; value: string }[] {
    const findings: { type: string; path: string; value: string }[] = [];

    if (typeof obj === 'string') {
      const textFindings = this.scanText(obj);
      textFindings.forEach(finding => {
        findings.push({
          type: finding.type,
          path,
          value: finding.match,
        });
      });
    } else if (typeof obj === 'object' && obj !== null) {
      Object.entries(obj).forEach(([key, value]) => {
        const newPath = path ? `${path}.${key}` : key;
        findings.push(...this.scanObject(value, newPath));
      });
    }

    return findings;
  }
}

Audit and Monitoring​

1. Secret Access Logging​

// src/audit/secret-access.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { SecretAccessLog } from './entities/secret-access-log.entity';

@Injectable()
export class SecretAccessService {
  private readonly logger = new Logger(SecretAccessService.name);

  constructor(
    @InjectRepository(SecretAccessLog)
    private secretAccessRepository: Repository<SecretAccessLog>
  ) {}

  async logSecretAccess(data: {
    secretName: string;
    userId?: string;
    serviceAccount?: string;
    action: 'READ' | 'CREATE' | 'UPDATE' | 'DELETE';
    source: string;
    success: boolean;
    reason?: string;
  }): Promise<void> {
    const log = this.secretAccessRepository.create({
      ...data,
      timestamp: new Date(),
      ipAddress: this.getClientIP(),
    });

    await this.secretAccessRepository.save(log);
    
    this.logger.log(`Secret access logged: ${data.secretName} - ${data.action}`);
  }

  private getClientIP(): string {
    // Implementation to get client IP
    return 'unknown';
  }

  async getAuditLogs(filters: {
    secretName?: string;
    startDate?: Date;
    endDate?: Date;
    action?: string;
  }): Promise<SecretAccessLog[]> {
    const query = this.secretAccessRepository.createQueryBuilder('log');

    if (filters.secretName) {
      query.andWhere('log.secretName = :secretName', { secretName: filters.secretName });
    }

    if (filters.startDate) {
      query.andWhere('log.timestamp >= :startDate', { startDate: filters.startDate });
    }

    if (filters.endDate) {
      query.andWhere('log.timestamp <= :endDate', { endDate: filters.endDate });
    }

    if (filters.action) {
      query.andWhere('log.action = :action', { action: filters.action });
    }

    return query.orderBy('log.timestamp', 'DESC').getMany();
  }
}

// src/monitoring/secret-monitoring.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import { SecretManagerService } from '../config/secret-manager.service';

@Injectable()
export class SecretMonitoringService {
  private readonly logger = new Logger(SecretMonitoringService.name);

  constructor(private secretManager: SecretManagerService) {}

  @Cron(CronExpression.EVERY_HOUR)
  async checkSecretHealth(): Promise<void> {
    const criticalSecrets = [
      'database-password',
      'jwt-secret',
      'encryption-key',
    ];

    for (const secretName of criticalSecrets) {
      try {
        const secret = await this.secretManager.getSecret(secretName);
        if (!secret) {
          await this.alertSecretIssue(secretName, 'Secret is empty or null');
        }
      } catch (error) {
        await this.alertSecretIssue(secretName, error.message);
      }
    }
  }

  @Cron(CronExpression.EVERY_DAY_AT_MIDNIGHT)
  async checkSecretRotation(): Promise<void> {
    // Check if secrets are due for rotation
    const secrets = await this.secretManager.listSecrets();
    
    for (const secretName of secrets) {
      const lastRotation = await this.getLastRotationDate(secretName);
      const daysSinceRotation = this.getDaysSince(lastRotation);
      
      if (daysSinceRotation > 90) { // 90 days rotation policy
        await this.alertRotationDue(secretName, daysSinceRotation);
      }
    }
  }

  private async alertSecretIssue(secretName: string, issue: string): Promise<void> {
    this.logger.error(`Secret issue detected: ${secretName} - ${issue}`);
    // Send alert to monitoring system
  }

  private async alertRotationDue(secretName: string, days: number): Promise<void> {
    this.logger.warn(`Secret rotation overdue: ${secretName} (${days} days)`);
    // Send rotation reminder
  }

  private async getLastRotationDate(secretName: string): Promise<Date> {
    // Implementation to get last rotation date
    return new Date();
  }

  private getDaysSince(date: Date): number {
    const now = new Date();
    const diffTime = Math.abs(now.getTime() - date.getTime());
    return Math.ceil(diffTime / (1000 * 60 * 60 * 24));
  }
}

Deployment Checklist​

  • GCP Secret Manager API enabled
  • Service account with appropriate permissions created
  • Workload Identity configured for GKE
  • Secret Store CSI driver installed
  • All secrets migrated to Secret Manager
  • Environment variables updated to reference secrets
  • Key rotation schedule configured
  • Monitoring and alerting set up
  • Secret scanning tools integrated in CI/CD
  • Team trained on secret management procedures

This implementation ensures that sensitive system data is properly protected, rotated, and monitored throughout the application lifecycle.
