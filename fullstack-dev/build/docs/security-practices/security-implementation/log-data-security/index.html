Secure Logging Architecture​

Centralized Logging Service​

// src/logging/secure-logger.service.ts
import { Injectable, LoggerService } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import * as winston from 'winston';
import { LoggingWinston } from '@google-cloud/logging-winston';

interface LogContext {
  userId?: string;
  sessionId?: string;
  requestId?: string;
  ipAddress?: string;
  userAgent?: string;
  action?: string;
  resource?: string;
  timestamp?: Date;
  severity?: 'DEBUG' | 'INFO' | 'WARN' | 'ERROR' | 'CRITICAL';
}

interface SecurityEvent {
  eventType: 'AUTHENTICATION' | 'AUTHORIZATION' | 'DATA_ACCESS' | 'CONFIGURATION_CHANGE' | 'SECURITY_VIOLATION';
  eventCategory: 'SUCCESS' | 'FAILURE' | 'ATTEMPT';
  description: string;
  riskLevel: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  context: LogContext;
  sensitiveData?: boolean;
  complianceRelevant?: boolean;
}

@Injectable()
export class SecureLoggerService implements LoggerService {
  private readonly logger: winston.Logger;
  private readonly securityLogger: winston.Logger;
  private readonly auditLogger: winston.Logger;
  private readonly sensitiveDataPatterns: RegExp[];

  constructor(private configService: ConfigService) {
    this.sensitiveDataPatterns = this.initializeSensitivePatterns();
    this.logger = this.createApplicationLogger();
    this.securityLogger = this.createSecurityLogger();
    this.auditLogger = this.createAuditLogger();
  }

  log(message: string, context?: LogContext): void {
    const sanitizedMessage = this.sanitizeMessage(message);
    const logEntry = this.createLogEntry('INFO', sanitizedMessage, context);
    this.logger.info(logEntry);
  }

  error(message: string, trace?: string, context?: LogContext): void {
    const sanitizedMessage = this.sanitizeMessage(message);
    const sanitizedTrace = trace ? this.sanitizeMessage(trace) : undefined;
    const logEntry = this.createLogEntry('ERROR', sanitizedMessage, context, sanitizedTrace);
    this.logger.error(logEntry);
  }

  warn(message: string, context?: LogContext): void {
    const sanitizedMessage = this.sanitizeMessage(message);
    const logEntry = this.createLogEntry('WARN', sanitizedMessage, context);
    this.logger.warn(logEntry);
  }

  debug(message: string, context?: LogContext): void {
    if (this.configService.get('NODE_ENV') !== 'production') {
      const sanitizedMessage = this.sanitizeMessage(message);
      const logEntry = this.createLogEntry('DEBUG', sanitizedMessage, context);
      this.logger.debug(logEntry);
    }
  }

  verbose(message: string, context?: LogContext): void {
    this.debug(message, context);
  }

  async logSecurityEvent(event: SecurityEvent): Promise<void> {
    const sanitizedEvent = this.sanitizeSecurityEvent(event);
    const logEntry = {
      timestamp: new Date().toISOString(),
      eventType: sanitizedEvent.eventType,
      eventCategory: sanitizedEvent.eventCategory,
      description: sanitizedEvent.description,
      riskLevel: sanitizedEvent.riskLevel,
      context: this.sanitizeContext(sanitizedEvent.context),
      environment: this.configService.get('NODE_ENV'),
      application: this.configService.get('APP_NAME'),
      version: this.configService.get('APP_VERSION'),
    };

    this.securityLogger.info(logEntry);

    // Alert on high-risk security events
    if (event.riskLevel === 'HIGH' || event.riskLevel === 'CRITICAL') {
      await this.alertSecurityIncident(logEntry);
    }
  }

  async logAuditEvent(
    action: string,
    resource: string,
    result: 'SUCCESS' | 'FAILURE',
    context: LogContext,
    details?: any
  ): Promise<void> {
    const auditEntry = {
      timestamp: new Date().toISOString(),
      action,
      resource,
      result,
      userId: context.userId,
      sessionId: context.sessionId,
      ipAddress: context.ipAddress,
      userAgent: this.sanitizeUserAgent(context.userAgent),
      details: details ? this.sanitizeObject(details) : undefined,
      environment: this.configService.get('NODE_ENV'),
    };

    this.auditLogger.info(auditEntry);
  }

  async logDataAccess(
    dataType: string,
    operation: 'READ' | 'WRITE' | 'DELETE',
    recordIds: string[],
    context: LogContext
  ): Promise<void> {
    const dataAccessEvent: SecurityEvent = {
      eventType: 'DATA_ACCESS',
      eventCategory: 'SUCCESS',
      description: `${operation} operation on ${dataType}`,
      riskLevel: this.calculateDataAccessRisk(dataType, operation),
      context,
      sensitiveData: this.isSensitiveDataType(dataType),
      complianceRelevant: this.isComplianceRelevantData(dataType),
    };

    await this.logSecurityEvent(dataAccessEvent);

    // Separate audit log for data access
    await this.logAuditEvent(
      `DATA_${operation}`,
      dataType,
      'SUCCESS',
      context,
      { recordCount: recordIds.length }
    );
  }

  async logAuthenticationEvent(
    eventType: 'LOGIN' | 'LOGOUT' | 'FAILED_LOGIN' | 'PASSWORD_CHANGE' | 'MFA_CHALLENGE',
    userId: string,
    result: 'SUCCESS' | 'FAILURE',
    context: LogContext,
    additionalInfo?: any
  ): Promise<void> {
    const authEvent: SecurityEvent = {
      eventType: 'AUTHENTICATION',
      eventCategory: result,
      description: `${eventType} ${result.toLowerCase()}`,
      riskLevel: this.calculateAuthRisk(eventType, result),
      context: { ...context, userId },
    };

    await this.logSecurityEvent(authEvent);
  }

  async logAuthorizationEvent(
    action: string,
    resource: string,
    decision: 'ALLOW' | 'DENY',
    context: LogContext
  ): Promise<void> {
    const authzEvent: SecurityEvent = {
      eventType: 'AUTHORIZATION',
      eventCategory: decision === 'ALLOW' ? 'SUCCESS' : 'FAILURE',
      description: `Authorization ${decision.toLowerCase()} for ${action} on ${resource}`,
      riskLevel: decision === 'DENY' ? 'MEDIUM' : 'LOW',
      context,
    };

    await this.logSecurityEvent(authzEvent);
  }

  async logConfigurationChange(
    configType: string,
    changeType: 'CREATE' | 'UPDATE' | 'DELETE',
    context: LogContext,
    oldValue?: any,
    newValue?: any
  ): Promise<void> {
    const configEvent: SecurityEvent = {
      eventType: 'CONFIGURATION_CHANGE',
      eventCategory: 'SUCCESS',
      description: `${changeType} configuration: ${configType}`,
      riskLevel: 'MEDIUM',
      context,
      complianceRelevant: true,
    };

    await this.logSecurityEvent(configEvent);

    // Detailed audit log for configuration changes
    await this.logAuditEvent(
      `CONFIG_${changeType}`,
      configType,
      'SUCCESS',
      context,
      {
        oldValue: oldValue ? this.sanitizeObject(oldValue) : undefined,
        newValue: newValue ? this.sanitizeObject(newValue) : undefined,
      }
    );
  }

  private createApplicationLogger(): winston.Logger {
    const transports: winston.transport[] = [];

    // Console transport for development
    if (this.configService.get('NODE_ENV') !== 'production') {
      transports.push(
        new winston.transports.Console({
          format: winston.format.combine(
            winston.format.colorize(),
            winston.format.timestamp(),
            winston.format.printf(({ timestamp, level, message, ...meta }) => {
              return `${timestamp} [${level}]: ${message} ${
                Object.keys(meta).length ? JSON.stringify(meta, null, 2) : ''
              }`;
            })
          ),
        })
      );
    }

    // Google Cloud Logging transport for production
    if (this.configService.get('NODE_ENV') === 'production') {
      transports.push(
        new LoggingWinston({
          projectId: this.configService.get('GCP_PROJECT_ID'),
          logName: 'application-logs',
          resource: {
            type: 'k8s_container',
            labels: {
              cluster_name: this.configService.get('GKE_CLUSTER_NAME'),
              namespace_name: this.configService.get('K8S_NAMESPACE'),
            },
          },
        })
      );
    }

    return winston.createLogger({
      level: this.configService.get('LOG_LEVEL', 'info'),
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.errors({ stack: true }),
        winston.format.json()
      ),
      transports,
    });
  }

  private createSecurityLogger(): winston.Logger {
    const transports: winston.transport[] = [];

    if (this.configService.get('NODE_ENV') === 'production') {
      transports.push(
        new LoggingWinston({
          projectId: this.configService.get('GCP_PROJECT_ID'),
          logName: 'security-events',
          resource: {
            type: 'k8s_container',
            labels: {
              cluster_name: this.configService.get('GKE_CLUSTER_NAME'),
              namespace_name: this.configService.get('K8S_NAMESPACE'),
            },
          },
        })
      );
    } else {
      transports.push(new winston.transports.Console());
    }

    return winston.createLogger({
      level: 'info',
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.json()
      ),
      transports,
    });
  }

  private createAuditLogger(): winston.Logger {
    const transports: winston.transport[] = [];

    if (this.configService.get('NODE_ENV') === 'production') {
      transports.push(
        new LoggingWinston({
          projectId: this.configService.get('GCP_PROJECT_ID'),
          logName: 'audit-logs',
          resource: {
            type: 'k8s_container',
            labels: {
              cluster_name: this.configService.get('GKE_CLUSTER_NAME'),
              namespace_name: this.configService.get('K8S_NAMESPACE'),
            },
          },
        })
      );
    } else {
      transports.push(new winston.transports.Console());
    }

    return winston.createLogger({
      level: 'info',
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.json()
      ),
      transports,
    });
  }

  private createLogEntry(
    level: string,
    message: string,
    context?: LogContext,
    trace?: string
  ): any {
    return {
      level,
      message,
      timestamp: new Date().toISOString(),
      context: context ? this.sanitizeContext(context) : undefined,
      trace,
      environment: this.configService.get('NODE_ENV'),
      application: this.configService.get('APP_NAME'),
      version: this.configService.get('APP_VERSION'),
    };
  }

  private sanitizeMessage(message: string): string {
    let sanitized = message;
    
    for (const pattern of this.sensitiveDataPatterns) {
      sanitized = sanitized.replace(pattern, '[REDACTED]');
    }
    
    return sanitized;
  }

  private sanitizeContext(context: LogContext): LogContext {
    const sanitized = { ...context };
    
    // Remove or hash sensitive context data
    if (sanitized.userId) {
      sanitized.userId = this.hashUserId(sanitized.userId);
    }
    
    return sanitized;
  }

  private sanitizeSecurityEvent(event: SecurityEvent): SecurityEvent {
    return {
      ...event,
      description: this.sanitizeMessage(event.description),
      context: this.sanitizeContext(event.context),
    };
  }

  private sanitizeObject(obj: any): any {
    if (typeof obj !== 'object' || obj === null) {
      return obj;
    }

    const sanitized = { ...obj };
    const sensitiveKeys = ['password', 'token', 'secret', 'key', 'credential'];
    
    for (const key in sanitized) {
      if (sensitiveKeys.some(sensitive => key.toLowerCase().includes(sensitive))) {
        sanitized[key] = '[REDACTED]';
      } else if (typeof sanitized[key] === 'object') {
        sanitized[key] = this.sanitizeObject(sanitized[key]);
      }
    }
    
    return sanitized;
  }

  private sanitizeUserAgent(userAgent?: string): string {
    if (!userAgent) return '';
    
    // Remove potential sensitive information from user agent
    return userAgent.substring(0, 200); // Limit length
  }

  private hashUserId(userId: string): string {
    const crypto = require('crypto');
    return crypto.createHash('sha256').update(userId).digest('hex').substring(0, 16);
  }

  private calculateDataAccessRisk(dataType: string, operation: string): 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL' {
    const sensitiveDataTypes = ['user', 'payment', 'pii', 'credential'];
    const highRiskOperations = ['DELETE'];
    
    if (sensitiveDataTypes.some(type => dataType.toLowerCase().includes(type))) {
      if (highRiskOperations.includes(operation)) {
        return 'CRITICAL';
      }
      return 'HIGH';
    }
    
    return operation === 'DELETE' ? 'MEDIUM' : 'LOW';
  }

  private calculateAuthRisk(eventType: string, result: string): 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL' {
    if (result === 'FAILURE') {
      return eventType === 'FAILED_LOGIN' ? 'MEDIUM' : 'LOW';
    }
    
    const highRiskEvents = ['PASSWORD_CHANGE', 'MFA_CHALLENGE'];
    return highRiskEvents.includes(eventType) ? 'MEDIUM' : 'LOW';
  }

  private isSensitiveDataType(dataType: string): boolean {
    const sensitiveTypes = ['user', 'payment', 'pii', 'credential', 'personal'];
    return sensitiveTypes.some(type => dataType.toLowerCase().includes(type));
  }

  private isComplianceRelevantData(dataType: string): boolean {
    const complianceTypes = ['payment', 'pii', 'health', 'financial'];
    return complianceTypes.some(type => dataType.toLowerCase().includes(type));
  }

  private async alertSecurityIncident(logEntry: any): Promise<void> {
    // Send alert to security team
    this.logger.error('SECURITY INCIDENT ALERT', logEntry);
    // Integration with alerting system would go here
  }

  private initializeSensitivePatterns(): RegExp[] {
    return [
      // Credit card numbers
      /\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b/g,
      // Social Security Numbers
      /\b\d{3}-\d{2}-\d{4}\b/g,
      // Email addresses (partial masking)
      /\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b/g,
      // Phone numbers
      /\b\d{3}[-.]?\d{3}[-.]?\d{4}\b/g,
      // Passwords in logs
      /password["\s]*[:=]["\s]*[^"\s,}]+/gi,
      // API keys
      /["\s](sk|pk)_[a-zA-Z0-9]{20,}["\s]/g,
      // JWT tokens
      /eyJ[A-Za-z0-9-_=]+\.[A-Za-z0-9-_=]+\.?[A-Za-z0-9-_.+/=]*/g,
    ];
  }
}

// src/logging/log-classification.service.ts
import { Injectable, Logger } from '@nestjs/common';

interface LogClassification {
  category: 'APPLICATION' | 'SECURITY' | 'AUDIT' | 'PERFORMANCE' | 'DEBUG';
  sensitivity: 'PUBLIC' | 'INTERNAL' | 'CONFIDENTIAL' | 'RESTRICTED';
  retention: {
    period: number; // days
    archiveAfter: number; // days
    purgeAfter: number; // days
  };
  encryption: boolean;
  accessControl: string[];
  complianceRelevant: boolean;
}

interface LogPolicy {
  pattern: RegExp;
  classification: LogClassification;
  processingRules: string[];
  monitoringRules: string[];
}

@Injectable()
export class LogClassificationService {
  private readonly logger = new Logger(LogClassificationService.name);
  private readonly logPolicies = new Map<string, LogPolicy>();

  constructor() {
    this.initializeLogPolicies();
  }

  classifyLog(logMessage: string, logLevel: string, source: string): LogClassification {
    // Default classification
    let classification: LogClassification = {
      category: 'APPLICATION',
      sensitivity: 'INTERNAL',
      retention: { period: 30, archiveAfter: 90, purgeAfter: 365 },
      encryption: false,
      accessControl: ['developer', 'devops'],
      complianceRelevant: false,
    };

    // Apply policy-based classification
    for (const [policyId, policy] of this.logPolicies) {
      if (policy.pattern.test(logMessage)) {
        classification = { ...policy.classification };
        break;
      }
    }

    // Override based on log level
    if (logLevel === 'ERROR' || logLevel === 'CRITICAL') {
      classification.retention.period = Math.max(classification.retention.period, 90);
    }

    // Override based on source
    if (source.includes('security') || source.includes('auth')) {
      classification.category = 'SECURITY';
      classification.retention.period = Math.max(classification.retention.period, 180);
      classification.encryption = true;
    }

    return classification;
  }

  getRetentionPolicy(classification: LogClassification): {
    retentionDays: number;
    archiveDays: number;
    purgeDays: number;
  } {
    return {
      retentionDays: classification.retention.period,
      archiveDays: classification.retention.archiveAfter,
      purgeDays: classification.retention.purgeAfter,
    };
  }

  shouldEncrypt(classification: LogClassification): boolean {
    return classification.encryption || 
           classification.sensitivity === 'CONFIDENTIAL' || 
           classification.sensitivity === 'RESTRICTED';
  }

  getAccessControlList(classification: LogClassification): string[] {
    return classification.accessControl;
  }

  isComplianceRelevant(classification: LogClassification): boolean {
    return classification.complianceRelevant;
  }

  private initializeLogPolicies(): void {
    const policies: Array<{ id: string; policy: LogPolicy }> = [
      {
        id: 'authentication_logs',
        policy: {
          pattern: /login|logout|authentication|password|credential/i,
          classification: {
            category: 'SECURITY',
            sensitivity: 'CONFIDENTIAL',
            retention: { period: 180, archiveAfter: 365, purgeAfter: 2555 }, // 7 years
            encryption: true,
            accessControl: ['security-team', 'compliance-officer'],
            complianceRelevant: true,
          },
          processingRules: ['sanitize_credentials', 'hash_user_ids'],
          monitoringRules: ['alert_failed_attempts', 'detect_anomalies'],
        },
      },
      {
        id: 'payment_logs',
        policy: {
          pattern: /payment|transaction|card|billing|purchase/i,
          classification: {
            category: 'AUDIT',
            sensitivity: 'RESTRICTED',
            retention: { period: 365, archiveAfter: 1095, purgeAfter: 3650 }, // 10 years
            encryption: true,
            accessControl: ['finance-team', 'compliance-officer', 'auditor'],
            complianceRelevant: true,
          },
          processingRules: ['mask_payment_data', 'encrypt_sensitive_fields'],
          monitoringRules: ['fraud_detection', 'compliance_monitoring'],
        },
      },
      {
        id: 'pii_logs',
        policy: {
          pattern: /email|phone|address|social|ssn|personal/i,
          classification: {
            category: 'AUDIT',
            sensitivity: 'CONFIDENTIAL',
            retention: { period: 90, archiveAfter: 365, purgeAfter: 2555 }, // 7 years
            encryption: true,
            accessControl: ['privacy-team', 'legal-team', 'compliance-officer'],
            complianceRelevant: true,
          },
          processingRules: ['mask_pii', 'anonymize_data'],
          monitoringRules: ['gdpr_compliance', 'data_breach_detection'],
        },
      },
      {
        id: 'application_errors',
        policy: {
          pattern: /error|exception|stack trace|failure/i,
          classification: {
            category: 'APPLICATION',
            sensitivity: 'INTERNAL',
            retention: { period: 90, archiveAfter: 180, purgeAfter: 365 },
            encryption: false,
            accessControl: ['developer', 'devops', 'support'],
            complianceRelevant: false,
          },
          processingRules: ['sanitize_stack_traces', 'remove_sensitive_data'],
          monitoringRules: ['error_rate_monitoring', 'alert_critical_errors'],
        },
      },
      {
        id: 'performance_logs',
        policy: {
          pattern: /performance|latency|response time|throughput/i,
          classification: {
            category: 'PERFORMANCE',
            sensitivity: 'INTERNAL',
            retention: { period: 30, archiveAfter: 90, purgeAfter: 180 },
            encryption: false,
            accessControl: ['developer', 'devops', 'sre'],
            complianceRelevant: false,
          },
          processingRules: ['aggregate_metrics', 'compress_data'],
          monitoringRules: ['performance_monitoring', 'capacity_planning'],
        },
      },
    ];

    policies.forEach(({ id, policy }) => {
      this.logPolicies.set(id, policy);
    });
  }
}

Log Encryption Service​

// src/logging/log-encryption.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import * as crypto from 'crypto';

interface EncryptionConfig {
  algorithm: string;
  keyDerivation: string;
  keyLength: number;
  ivLength: number;
  tagLength: number;
}

interface EncryptedLog {
  encryptedData: string;
  iv: string;
  tag: string;
  keyId: string;
  algorithm: string;
  timestamp: string;
}

@Injectable()
export class LogEncryptionService {
  private readonly logger = new Logger(LogEncryptionService.name);
  private readonly encryptionConfig: EncryptionConfig;
  private readonly encryptionKeys = new Map<string, Buffer>();

  constructor(private configService: ConfigService) {
    this.encryptionConfig = {
      algorithm: 'aes-256-gcm',
      keyDerivation: 'pbkdf2',
      keyLength: 32,
      ivLength: 12,
      tagLength: 16,
    };
    this.initializeEncryptionKeys();
  }

  async encryptLog(logData: any, keyId: string = 'default'): Promise<EncryptedLog> {
    const key = this.getEncryptionKey(keyId);
    const iv = crypto.randomBytes(this.encryptionConfig.ivLength);
    
    const cipher = crypto.createCipher(this.encryptionConfig.algorithm, key);
    cipher.setAAD(Buffer.from(keyId)); // Additional authenticated data
    
    const serializedData = JSON.stringify(logData);
    const encrypted = Buffer.concat([
      cipher.update(serializedData, 'utf8'),
      cipher.final(),
    ]);
    
    const tag = cipher.getAuthTag();
    
    return {
      encryptedData: encrypted.toString('base64'),
      iv: iv.toString('base64'),
      tag: tag.toString('base64'),
      keyId,
      algorithm: this.encryptionConfig.algorithm,
      timestamp: new Date().toISOString(),
    };
  }

  async decryptLog(encryptedLog: EncryptedLog): Promise<any> {
    const key = this.getEncryptionKey(encryptedLog.keyId);
    const iv = Buffer.from(encryptedLog.iv, 'base64');
    const tag = Buffer.from(encryptedLog.tag, 'base64');
    const encryptedData = Buffer.from(encryptedLog.encryptedData, 'base64');
    
    const decipher = crypto.createDecipher(encryptedLog.algorithm, key);
    decipher.setAAD(Buffer.from(encryptedLog.keyId));
    decipher.setAuthTag(tag);
    
    const decrypted = Buffer.concat([
      decipher.update(encryptedData),
      decipher.final(),
    ]);
    
    return JSON.parse(decrypted.toString('utf8'));
  }

  async encryptLogField(value: string, fieldType: 'PII' | 'CREDENTIAL' | 'SENSITIVE'): Promise<string> {
    const keyId = this.getKeyIdForFieldType(fieldType);
    const encrypted = await this.encryptLog({ value }, keyId);
    return `ENC:${encrypted.keyId}:${encrypted.encryptedData}:${encrypted.iv}:${encrypted.tag}`;
  }

  async decryptLogField(encryptedValue: string): Promise<string> {
    if (!encryptedValue.startsWith('ENC:')) {
      return encryptedValue; // Not encrypted
    }
    
    const parts = encryptedValue.substring(4).split(':');
    if (parts.length !== 4) {
      throw new Error('Invalid encrypted field format');
    }
    
    const [keyId, encryptedData, iv, tag] = parts;
    const encryptedLog: EncryptedLog = {
      encryptedData,
      iv,
      tag,
      keyId,
      algorithm: this.encryptionConfig.algorithm,
      timestamp: new Date().toISOString(),
    };
    
    const decrypted = await this.decryptLog(encryptedLog);
    return decrypted.value;
  }

  async rotateEncryptionKey(keyId: string): Promise<void> {
    const newKey = this.generateEncryptionKey();
    const oldKey = this.encryptionKeys.get(keyId);
    
    if (oldKey) {
      // Store old key for decryption of existing logs
      this.encryptionKeys.set(`${keyId}_old_${Date.now()}`, oldKey);
    }
    
    this.encryptionKeys.set(keyId, newKey);
    
    this.logger.log(`Encryption key rotated for keyId: ${keyId}`);
  }

// src/logging/log-access-control.service.ts
import { Injectable, Logger, UnauthorizedException } from '@nestjs/common';

interface LogAccessRequest {
  userId: string;
  logCategory: string;
  logLevel: string;
  timeRange: {
    start: Date;
    end: Date;
  };
  purpose: string;
  requiredApproval?: boolean;
}

interface LogAccessPolicy {
  logCategory: string;
  allowedRoles: string[];
  restrictions: {
    maxTimeRange: number; // days
    requiresApproval: boolean;
    approvalRoles: string[];
    allowedOperations: ('READ' | 'SEARCH' | 'EXPORT')[];
  };
  auditRequired: boolean;
}

@Injectable()
export class LogAccessControlService {
  private readonly logger = new Logger(LogAccessControlService.name);
  private readonly accessPolicies = new Map<string, LogAccessPolicy>();

  constructor() {
    this.initializeAccessPolicies();
  }

  async requestLogAccess(request: LogAccessRequest): Promise<string> {
    // Validate user permissions
    await this.validateLogAccess(request);
    
    // Check if approval is required
    const policy = this.getAccessPolicy(request.logCategory);
    if (policy?.restrictions.requiresApproval) {
      return await this.requestApproval(request);
    }
    
    // Grant immediate access
    const accessToken = this.generateAccessToken(request);
    await this.auditLogAccess(request, 'GRANTED', accessToken);
    
    return accessToken;
  }

  async validateLogAccess(request: LogAccessRequest): Promise<void> {
    const userRoles = await this.getUserRoles(request.userId);
    const policy = this.getAccessPolicy(request.logCategory);
    
    if (!policy) {
      throw new UnauthorizedException(`No access policy for log category: ${request.logCategory}`);
    }
    
    // Check role permissions
    const hasRequiredRole = policy.allowedRoles.some(role => userRoles.includes(role));
    if (!hasRequiredRole) {
      throw new UnauthorizedException(`Insufficient permissions for log category: ${request.logCategory}`);
    }
    
    // Check time range restrictions
    const requestedDays = this.calculateDaysBetween(request.timeRange.start, request.timeRange.end);
    if (requestedDays > policy.restrictions.maxTimeRange) {
      throw new UnauthorizedException(
        `Requested time range (${requestedDays} days) exceeds maximum allowed (${policy.restrictions.maxTimeRange} days)`
      );
    }
    
    // Additional security checks
    await this.performSecurityChecks(request);
  }

  async auditLogAccess(
    request: LogAccessRequest,
    result: 'GRANTED' | 'DENIED' | 'REVOKED',
    accessToken?: string
  ): Promise<void> {
    const auditEntry = {
      timestamp: new Date().toISOString(),
      userId: request.userId,
      action: 'LOG_ACCESS_REQUEST',
      logCategory: request.logCategory,
      logLevel: request.logLevel,
      timeRange: request.timeRange,
      purpose: request.purpose,
      result,
      accessToken: accessToken ? this.hashToken(accessToken) : undefined,
    };
    
    this.logger.log(`Log access audit: ${JSON.stringify(auditEntry)}`);
  }

  async revokeLogAccess(accessToken: string, reason: string): Promise<void> {
    // Revoke the access token
    await this.invalidateAccessToken(accessToken);
    
    // Audit the revocation
    const auditEntry = {
      timestamp: new Date().toISOString(),
      action: 'LOG_ACCESS_REVOKED',
      accessToken: this.hashToken(accessToken),
      reason,
    };
    
    this.logger.log(`Log access revoked: ${JSON.stringify(auditEntry)}`);
  }

  async monitorLogAccess(): Promise<void> {
    // Monitor for suspicious log access patterns
    await this.detectAnomalousAccess();
    await this.checkComplianceViolations();
    await this.generateAccessReport();
  }

  }

  }

  private generateAccessToken(request: LogAccessRequest): string {
    const crypto = require('crypto');
    const tokenData = {
      userId: request.userId,
      logCategory: request.logCategory,
      timeRange: request.timeRange,
      issuedAt: new Date().toISOString(),
      expiresAt: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(), // 24 hours
    };
    
    return crypto.createHash('sha256').update(JSON.stringify(tokenData)).digest('hex');
  }

  }

#!/bin/bash
# scripts/log-retention-management.sh

set -e

PROJECT_ID=${1:-"your-project-id"}
ENVIRONMENT=${2:-"production"}

echo "Setting up log retention management for project: $PROJECT_ID"

# Create Cloud Storage buckets for log archival
gsutil mb -p $PROJECT_ID -l us-central1 gs://$PROJECT_ID-log-archive-$ENVIRONMENT
gsutil mb -p $PROJECT_ID -l us-central1 gs://$PROJECT_ID-log-cold-storage-$ENVIRONMENT

# Set lifecycle policies for log retention
cat > log-lifecycle-policy.json << EOF
{
  "lifecycle": {
    "rule": [
      {
        "action": {"type": "SetStorageClass", "storageClass": "NEARLINE"},
        "condition": {"age": 30}
      },
      {
        "action": {"type": "SetStorageClass", "storageClass": "COLDLINE"},
        "condition": {"age": 90}
      },
      {
        "action": {"type": "SetStorageClass", "storageClass": "ARCHIVE"},
        "condition": {"age": 365}
      },
      {
        "action": {"type": "Delete"},
        "condition": {"age": 2555}
      }
    ]
  }
}
EOF

# Apply lifecycle policies
gsutil lifecycle set log-lifecycle-policy.json gs://$PROJECT_ID-log-archive-$ENVIRONMENT
gsutil lifecycle set log-lifecycle-policy.json gs://$PROJECT_ID-log-cold-storage-$ENVIRONMENT

# Set up log sinks for different log categories
gcloud logging sinks create security-logs-sink \
    storage.googleapis.com/$PROJECT_ID-log-archive-$ENVIRONMENT/security-logs \
    --log-filter='resource.type="k8s_container" AND logName="projects/'$PROJECT_ID'/logs/security-events"'

gcloud logging sinks create audit-logs-sink \
    storage.googleapis.com/$PROJECT_ID-log-archive-$ENVIRONMENT/audit-logs \
    --log-filter='resource.type="k8s_container" AND logName="projects/'$PROJECT_ID'/logs/audit-logs"'

gcloud logging sinks create application-logs-sink \
    storage.googleapis.com/$PROJECT_ID-log-archive-$ENVIRONMENT/application-logs \
    --log-filter='resource.type="k8s_container" AND logName="projects/'$PROJECT_ID'/logs/application-logs"'

# Set up BigQuery datasets for log analytics
bq mk --dataset --location=US $PROJECT_ID:log_analytics_$ENVIRONMENT

# Create log export jobs
cat > log-export-config.json << EOF
{
  "destinationTable": {
    "projectId": "$PROJECT_ID",
    "datasetId": "log_analytics_$ENVIRONMENT",
    "tableId": "security_logs"
  },
  "filter": "resource.type=\"k8s_container\" AND logName=\"projects/$PROJECT_ID/logs/security-events\"",
  "description": "Export security logs to BigQuery for analysis"
}
EOF

echo "Log retention management setup completed for: $PROJECT_ID"
