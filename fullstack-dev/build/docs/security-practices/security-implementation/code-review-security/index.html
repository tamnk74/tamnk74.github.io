Code Review & Solution Design Security Guide

This guide provides comprehensive secure code review and solution design practices for NestJS/Next.js applications on GCP Kubernetes, ensuring security is integrated throughout the development process.

Code Review Security Service​

// src/code-review/security-review.service.ts
import { Injectable, Logger } from '@nestjs/common';

interface SecurityReviewRule {
  id: string;
  category: 'AUTHENTICATION' | 'AUTHORIZATION' | 'INPUT_VALIDATION' | 'OUTPUT_ENCODING' | 'CRYPTOGRAPHY' | 'LOGGING' | 'ERROR_HANDLING';
  pattern: RegExp;
  description: string;
  severity: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  guidance: string;
  examples: {
    vulnerable: string;
    secure: string;
  };
}

interface ReviewFinding {
  ruleId: string;
  file: string;
  line: number;
  code: string;
  severity: string;
  description: string;
  guidance: string;
  suggestion?: string;
}

@Injectable()
export class SecurityReviewService {
  private readonly logger = new Logger(SecurityReviewService.name);
  private readonly securityRules = new Map<string, SecurityReviewRule>();

  constructor() {
    this.initializeSecurityRules();
  }

  async reviewCode(
    files: Array<{ path: string; content: string }>,
    reviewType: 'PULL_REQUEST' | 'COMMIT' | 'MANUAL'
  ): Promise<ReviewFinding[]> {
    const findings: ReviewFinding[] = [];

    for (const file of files) {
      const fileFindings = await this.reviewFile(file.path, file.content);
      findings.push(...fileFindings);
    }

    // Generate review report
    await this.generateSecurityReviewReport(findings, reviewType);

    return findings;
  }

  async reviewFile(filePath: string, content: string): Promise<ReviewFinding[]> {
    const findings: ReviewFinding[] = [];
    const lines = content.split('\n');

    for (const [ruleId, rule] of this.securityRules) {
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];
        const match = rule.pattern.exec(line);

        if (match) {
          findings.push({
            ruleId,
            file: filePath,
            line: i + 1,
            code: line.trim(),
            severity: rule.severity,
            description: rule.description,
            guidance: rule.guidance,
            suggestion: this.generateSuggestion(rule, line),
          });
        }
      }
    }

    return findings;
  }

  async reviewPullRequest(prId: string, files: string[]): Promise<void> {
    this.logger.log(`Starting security review for PR: ${prId}`);

    const reviewFiles = await this.getChangedFiles(files);
    const findings = await this.reviewCode(reviewFiles, 'PULL_REQUEST');

    // Block PR if critical security issues found
    const criticalFindings = findings.filter(f => f.severity === 'CRITICAL');
    if (criticalFindings.length > 0) {
      await this.blockPullRequest(prId, criticalFindings);
    }

    // Add review comments
    await this.addReviewComments(prId, findings);
  }

  async generateSecurityReviewGuide(): Promise<string> {
    const categories = Array.from(new Set(
      Array.from(this.securityRules.values()).map(rule => rule.category)
    ));

    let guide = '# Security Code Review Guide\n\n';
    
    for (const category of categories) {
      guide += `## ${category.replace(/_/g, ' ')}\n\n`;
      
      const categoryRules = Array.from(this.securityRules.values())
        .filter(rule => rule.category === category);

      for (const rule of categoryRules) {
        guide += `### ${rule.description}\n\n`;
        guide += `**Severity:** ${rule.severity}\n\n`;
        guide += `${rule.guidance}\n\n`;
        
        guide += '**Vulnerable Example:**\n```typescript\n';
        guide += rule.examples.vulnerable;
        guide += '\n```\n\n';
        
        guide += '**Secure Example:**\n```typescript\n';
        guide += rule.examples.secure;
        guide += '\n```\n\n';
      }
    }

    return guide;
  }

  private async getChangedFiles(filePaths: string[]): Promise<Array<{ path: string; content: string }>> {
    // Mock implementation - would integrate with Git/GitHub API
    return filePaths.map(path => ({
      path,
      content: '// File content would be loaded here',
    }));
  }

  private async blockPullRequest(prId: string, findings: ReviewFinding[]): Promise<void> {
    this.logger.error(`Blocking PR ${prId} due to critical security issues: ${findings.length}`);
    // Would integrate with GitHub/GitLab API to block PR
  }

  private async addReviewComments(prId: string, findings: ReviewFinding[]): Promise<void> {
    for (const finding of findings) {
      await this.addInlineComment(prId, finding);
    }
  }

  private async addInlineComment(prId: string, finding: ReviewFinding): Promise<void> {
    const comment = `
🔒 **Security Review Finding**

**Severity:** ${finding.severity}
**Description:** ${finding.description}

**Guidance:** ${finding.guidance}

${finding.suggestion ? `**Suggestion:** ${finding.suggestion}` : ''}
    `;

    // Would post comment to PR at specific line
    this.logger.log(`Would add comment to PR ${prId} at ${finding.file}:${finding.line}`);
  }

  private generateSuggestion(rule: SecurityReviewRule, code: string): string {
    // Generate specific suggestions based on the rule and code
    switch (rule.category) {
      case 'INPUT_VALIDATION':
        return 'Consider using class-validator decorators or Joi schemas for input validation';
      case 'OUTPUT_ENCODING':
        return 'Use proper HTML encoding or sanitization libraries';
      case 'AUTHENTICATION':
        return 'Implement proper JWT validation with refresh tokens';
      default:
        return 'Review the security implications of this code';
    }
  }

  private async generateSecurityReviewReport(
    findings: ReviewFinding[],
    reviewType: string
  ): Promise<void> {
    const report = {
      timestamp: new Date(),
      reviewType,
      totalFindings: findings.length,
      bySeverity: {
        critical: findings.filter(f => f.severity === 'CRITICAL').length,
        high: findings.filter(f => f.severity === 'HIGH').length,
        medium: findings.filter(f => f.severity === 'MEDIUM').length,
        low: findings.filter(f => f.severity === 'LOW').length,
      },
      byCategory: this.groupFindingsByCategory(findings),
      findings,
    };

    this.logger.log(`Security review report: ${JSON.stringify(report)}`);
  }

  private groupFindingsByCategory(findings: ReviewFinding[]): Record<string, number> {
    const grouped: Record<string, number> = {};
    
    for (const finding of findings) {
      const rule = this.securityRules.get(finding.ruleId);
      if (rule) {
        grouped[rule.category] = (grouped[rule.category] || 0) + 1;
      }
    }

    return grouped;
  }

  private initializeSecurityRules(): void {
    const rules: SecurityReviewRule[] = [
      {
        id: 'SQL_INJECTION',
        category: 'INPUT_VALIDATION',
        pattern: /\$\{.*\}.*query|query.*\$\{.*\}/i,
        description: 'Potential SQL injection vulnerability',
        severity: 'CRITICAL',
        guidance: 'Use parameterized queries or ORM methods to prevent SQL injection',
        examples: {
          vulnerable: 'const query = `SELECT * FROM users WHERE id = ${userId}`;',
          secure: 'const user = await userRepository.findOne({ where: { id: userId } });',
        },
      },
      {
        id: 'XSS_RISK',
        category: 'OUTPUT_ENCODING',
        pattern: /innerHTML\s*=|dangerouslySetInnerHTML/i,
        description: 'Potential XSS vulnerability',
        severity: 'HIGH',
        guidance: 'Use proper HTML encoding or sanitization for dynamic content',
        examples: {
          vulnerable: 'element.innerHTML = userInput;',
          secure: 'element.textContent = userInput; // or use DOMPurify.sanitize()',
        },
      },
      {
        id: 'HARDCODED_SECRET',
        category: 'CRYPTOGRAPHY',
        pattern: /(password|secret|key|token)\s*[:=]\s*["'][^"']*["']/i,
        description: 'Hardcoded secret detected',
        severity: 'CRITICAL',
        guidance: 'Use environment variables or secure secret management for sensitive data',
        examples: {
          vulnerable: 'const apiKey = "sk-1234567890abcdef";',
          secure: 'const apiKey = process.env.API_KEY;',
        },
      },
      {
        id: 'WEAK_CRYPTO',
        category: 'CRYPTOGRAPHY',
        pattern: /\b(md5|sha1)\b/i,
        description: 'Weak cryptographic algorithm',
        severity: 'HIGH',
        guidance: 'Use strong cryptographic algorithms like SHA-256 or bcrypt',
        examples: {
          vulnerable: 'crypto.createHash("md5").update(password).digest("hex");',
          secure: 'await bcrypt.hash(password, 12);',
        },
      },
      {
        id: 'INSUFFICIENT_LOGGING',
        category: 'LOGGING',
        pattern: /console\.(log|error|warn)/i,
        description: 'Using console for logging',
        severity: 'MEDIUM',
        guidance: 'Use proper logging framework with appropriate log levels',
        examples: {
          vulnerable: 'console.log("User logged in:", userId);',
          secure: 'this.logger.log("User logged in", { userId });',
        },
      },
      {
        id: 'SENSITIVE_DATA_LOG',
        category: 'LOGGING',
        pattern: /(password|token|secret|key).*log|log.*(password|token|secret|key)/i,
        description: 'Potential sensitive data in logs',
        severity: 'HIGH',
        guidance: 'Never log sensitive information like passwords or tokens',
        examples: {
          vulnerable: 'logger.info("Login attempt", { password, username });',
          secure: 'logger.info("Login attempt", { username });',
        },
      },
      {
        id: 'ERROR_INFORMATION_LEAK',
        category: 'ERROR_HANDLING',
        pattern: /throw\s+new\s+Error\(\s*error\s*\)|res\.status\(500\)\.send\(error\)/i,
        description: 'Potential information leak in error messages',
        severity: 'MEDIUM',
        guidance: 'Sanitize error messages before sending to client',
        examples: {
          vulnerable: 'throw new Error(error);',
          secure: 'throw new Error("Invalid input provided");',
        },
      },
      {
        id: 'MISSING_AUTH_CHECK',
        category: 'AUTHORIZATION',
        pattern: /@Controller|@Get|@Post|@Put|@Delete/,
        description: 'Check for missing authentication/authorization',
        severity: 'HIGH',
        guidance: 'Ensure all endpoints have appropriate guards or decorators',
        examples: {
          vulnerable: '@Get("/admin") getAdminData() { ... }',
          secure: '@UseGuards(JwtAuthGuard, RolesGuard) @Roles("admin") @Get("/admin") getAdminData() { ... }',
        },
      },
      {
        id: 'CORS_MISCONFIGURATION',
        category: 'AUTHENTICATION',
        pattern: /origin:\s*\*|credentials:\s*true.*origin:\s*\*/s,
        description: 'Potential CORS misconfiguration',
        severity: 'HIGH',
        guidance: 'Avoid using wildcard origin with credentials enabled',
        examples: {
          vulnerable: 'app.enableCors({ origin: "*", credentials: true });',
          secure: 'app.enableCors({ origin: ["https://yourdomain.com"], credentials: true });',
        },
      },
    ];

    rules.forEach(rule => {
      this.securityRules.set(rule.id, rule);
    });
  }
}

Security Architecture Review Service​

// src/design/security-architecture.service.ts
import { Injectable, Logger } from '@nestjs/common';

interface SecurityRequirement {
  id: string;
  category: 'CONFIDENTIALITY' | 'INTEGRITY' | 'AVAILABILITY' | 'AUTHENTICATION' | 'AUTHORIZATION' | 'AUDITABILITY';
  description: string;
  implementation: string;
  verification: string;
  riskLevel: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
}

interface ArchitectureComponent {
  name: string;
  type: 'SERVICE' | 'DATABASE' | 'CACHE' | 'QUEUE' | 'GATEWAY' | 'STORAGE';
  securityRequirements: string[];
  dataClassification: 'PUBLIC' | 'INTERNAL' | 'CONFIDENTIAL' | 'RESTRICTED';
  trustBoundary: boolean;
  communicationProtocols: string[];
  authenticationMethods: string[];
}

interface ThreatModel {
  component: string;
  threats: Threat[];
  mitigations: Mitigation[];
  residualRisk: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
}

interface Threat {
  id: string;
  category: 'SPOOFING' | 'TAMPERING' | 'REPUDIATION' | 'INFORMATION_DISCLOSURE' | 'DENIAL_OF_SERVICE' | 'ELEVATION_OF_PRIVILEGE';
  description: string;
  likelihood: 'LOW' | 'MEDIUM' | 'HIGH';
  impact: 'LOW' | 'MEDIUM' | 'HIGH';
  riskScore: number;
}

interface Mitigation {
  threatId: string;
  description: string;
  implementation: string;
  effectiveness: 'LOW' | 'MEDIUM' | 'HIGH';
  cost: 'LOW' | 'MEDIUM' | 'HIGH';
}

@Injectable()
export class SecurityArchitectureService {
  private readonly logger = new Logger(SecurityArchitectureService.name);
  private readonly securityRequirements = new Map<string, SecurityRequirement>();
  private readonly threatModels = new Map<string, ThreatModel>();

  constructor() {
    this.initializeSecurityRequirements();
  }

  async reviewArchitecture(
    components: ArchitectureComponent[],
    dataFlows: any[],
    trustBoundaries: any[]
  ): Promise<{
    securityAssessment: any;
    threatModels: ThreatModel[];
    recommendations: string[];
  }> {
    this.logger.log('Starting security architecture review');

    const securityAssessment = await this.assessSecurityPosture(components);
    const threatModels = await this.generateThreatModels(components, dataFlows);
    const recommendations = await this.generateSecurityRecommendations(
      securityAssessment,
      threatModels
    );

    return {
      securityAssessment,
      threatModels,
      recommendations,
    };
  }

  async generateThreatModel(component: ArchitectureComponent): Promise<ThreatModel> {
    const threats = await this.identifyThreats(component);
    const mitigations = await this.identifyMitigations(threats, component);
    const residualRisk = this.calculateResidualRisk(threats, mitigations);

    const threatModel: ThreatModel = {
      component: component.name,
      threats,
      mitigations,
      residualRisk,
    };

    this.threatModels.set(component.name, threatModel);
    return threatModel;
  }

  async validateSecurityRequirements(
    component: ArchitectureComponent
  ): Promise<{
    compliant: boolean;
    missingRequirements: string[];
    recommendations: string[];
  }> {
    const requiredSecurityControls = this.getRequiredSecurityControls(component);
    const implementedControls = component.securityRequirements;
    
    const missingRequirements = requiredSecurityControls.filter(
      req => !implementedControls.includes(req)
    );

    const recommendations = this.generateImplementationRecommendations(
      component,
      missingRequirements
    );

    return {
      compliant: missingRequirements.length === 0,
      missingRequirements,
      recommendations,
    };
  }

  async generateSecurityDesignDocument(
    projectName: string,
    components: ArchitectureComponent[],
    threatModels: ThreatModel[]
  ): Promise<string> {
    let document = `# Security Design Document: ${projectName}\n\n`;
    
    document += '## Executive Summary\n\n';
    document += this.generateExecutiveSummary(components, threatModels);
    
    document += '\n## Security Architecture Overview\n\n';
    document += this.generateArchitectureOverview(components);
    
    document += '\n## Threat Models\n\n';
    document += this.generateThreatModelSection(threatModels);
    
    document += '\n## Security Controls\n\n';
    document += this.generateSecurityControlsSection(components);
    
    document += '\n## Risk Assessment\n\n';
    document += this.generateRiskAssessmentSection(threatModels);
    
    document += '\n## Implementation Recommendations\n\n';
    document += this.generateImplementationSection(components, threatModels);

    return document;
  }

  private async assessSecurityPosture(components: ArchitectureComponent[]): Promise<any> {
    const assessment = {
      overallRisk: 'MEDIUM',
      componentAssessments: new Map(),
      securityGaps: [],
      strengths: [],
    };

    for (const component of components) {
      const componentAssessment = await this.assessComponent(component);
      assessment.componentAssessments.set(component.name, componentAssessment);
      
      if (componentAssessment.risk === 'HIGH' || componentAssessment.risk === 'CRITICAL') {
        assessment.securityGaps.push({
          component: component.name,
          issues: componentAssessment.issues,
        });
      }
    }

    return assessment;
  }

  private async assessComponent(component: ArchitectureComponent): Promise<any> {
    const requiredControls = this.getRequiredSecurityControls(component);
    const missingControls = requiredControls.filter(
      control => !component.securityRequirements.includes(control)
    );

    const risk = this.calculateComponentRisk(component, missingControls);
    
    return {
      risk,
      implementedControls: component.securityRequirements.length,
      missingControls: missingControls.length,
      issues: missingControls,
    };
  }

  private async generateThreatModels(
    components: ArchitectureComponent[],
    dataFlows: any[]
  ): Promise<ThreatModel[]> {
    const threatModels: ThreatModel[] = [];

    for (const component of components) {
      const threatModel = await this.generateThreatModel(component);
      threatModels.push(threatModel);
    }

    return threatModels;
  }

  private async identifyThreats(component: ArchitectureComponent): Promise<Threat[]> {
    const threats: Threat[] = [];

    // STRIDE threat modeling
    const strideThreats = {
      SPOOFING: this.getSpoofingThreats(component),
      TAMPERING: this.getTamperingThreats(component),
      REPUDIATION: this.getRepudiationThreats(component),
      INFORMATION_DISCLOSURE: this.getInformationDisclosureThreats(component),
      DENIAL_OF_SERVICE: this.getDenialOfServiceThreats(component),
      ELEVATION_OF_PRIVILEGE: this.getElevationOfPrivilegeThreats(component),
    };

    for (const [category, categoryThreats] of Object.entries(strideThreats)) {
      threats.push(...categoryThreats.map(threat => ({
        ...threat,
        category: category as any,
        riskScore: this.calculateRiskScore(threat.likelihood, threat.impact),
      })));
    }

    return threats;
  }

  private getSpoofingThreats(component: ArchitectureComponent): Partial<Threat>[] {
    const threats: Partial<Threat>[] = [];

    if (!component.authenticationMethods.includes('MUTUAL_TLS')) {
      threats.push({
        id: `SPOOF_${component.name}_001`,
        description: 'Attacker could impersonate legitimate service',
        likelihood: 'MEDIUM',
        impact: 'HIGH',
      });
    }

    if (component.type === 'SERVICE' && !component.authenticationMethods.includes('JWT')) {
      threats.push({
        id: `SPOOF_${component.name}_002`,
        description: 'Unauthorized access to service endpoints',
        likelihood: 'HIGH',
        impact: 'HIGH',
      });
    }

    return threats;
  }

  private getTamperingThreats(component: ArchitectureComponent): Partial<Threat>[] {
    const threats: Partial<Threat>[] = [];

    if (!component.communicationProtocols.includes('TLS')) {
      threats.push({
        id: `TAMPER_${component.name}_001`,
        description: 'Data could be modified in transit',
        likelihood: 'HIGH',
        impact: 'HIGH',
      });
    }

    if (component.type === 'DATABASE' && !component.securityRequirements.includes('ENCRYPTION_AT_REST')) {
      threats.push({
        id: `TAMPER_${component.name}_002`,
        description: 'Stored data could be modified',
        likelihood: 'MEDIUM',
        impact: 'HIGH',
      });
    }

    return threats;
  }

  private getRepudiationThreats(component: ArchitectureComponent): Partial<Threat>[] {
    const threats: Partial<Threat>[] = [];

    if (!component.securityRequirements.includes('AUDIT_LOGGING')) {
      threats.push({
        id: `REPUDIATE_${component.name}_001`,
        description: 'Actions cannot be traced to specific users',
        likelihood: 'MEDIUM',
        impact: 'MEDIUM',
      });
    }

    return threats;
  }

  private getInformationDisclosureThreats(component: ArchitectureComponent): Partial<Threat>[] {
    const threats: Partial<Threat>[] = [];

    if (component.dataClassification === 'CONFIDENTIAL' || component.dataClassification === 'RESTRICTED') {
      if (!component.securityRequirements.includes('ENCRYPTION_AT_REST')) {
        threats.push({
          id: `DISCLOSURE_${component.name}_001`,
          description: 'Sensitive data could be exposed at rest',
          likelihood: 'MEDIUM',
          impact: 'HIGH',
        });
      }

      if (!component.communicationProtocols.includes('TLS')) {
        threats.push({
          id: `DISCLOSURE_${component.name}_002`,
          description: 'Sensitive data could be exposed in transit',
          likelihood: 'HIGH',
          impact: 'HIGH',
        });
      }
    }

    return threats;
  }

  private getDenialOfServiceThreats(component: ArchitectureComponent): Partial<Threat>[] {
    const threats: Partial<Threat>[] = [];

    if (!component.securityRequirements.includes('RATE_LIMITING')) {
      threats.push({
        id: `DOS_${component.name}_001`,
        description: 'Service could be overwhelmed by excessive requests',
        likelihood: 'HIGH',
        impact: 'MEDIUM',
      });
    }

    return threats;
  }

  private getElevationOfPrivilegeThreats(component: ArchitectureComponent): Partial<Threat>[] {
    const threats: Partial<Threat>[] = [];

    if (!component.securityRequirements.includes('RBAC')) {
      threats.push({
        id: `PRIVILEGE_${component.name}_001`,
        description: 'Users could gain unauthorized access levels',
        likelihood: 'MEDIUM',
        impact: 'HIGH',
      });
    }

    return threats;
  }

  private async identifyMitigations(threats: Threat[], component: ArchitectureComponent): Promise<Mitigation[]> {
    const mitigations: Mitigation[] = [];

    for (const threat of threats) {
      const threatMitigations = this.getMitigationsForThreat(threat, component);
      mitigations.push(...threatMitigations);
    }

    return mitigations;
  }

  private getMitigationsForThreat(threat: Threat, component: ArchitectureComponent): Mitigation[] {
    const mitigations: Mitigation[] = [];

    switch (threat.category) {
      case 'SPOOFING':
        mitigations.push({
          threatId: threat.id,
          description: 'Implement mutual TLS authentication',
          implementation: 'Configure mTLS certificates and validation',
          effectiveness: 'HIGH',
          cost: 'MEDIUM',
        });
        break;

      case 'TAMPERING':
        mitigations.push({
          threatId: threat.id,
          description: 'Use TLS for all communications',
          implementation: 'Enable TLS 1.3 with strong cipher suites',
          effectiveness: 'HIGH',
          cost: 'LOW',
        });
        break;

      case 'INFORMATION_DISCLOSURE':
        mitigations.push({
          threatId: threat.id,
          description: 'Implement encryption at rest',
          implementation: 'Use AES-256 encryption for stored data',
          effectiveness: 'HIGH',
          cost: 'MEDIUM',
        });
        break;

      case 'DENIAL_OF_SERVICE':
        mitigations.push({
          threatId: threat.id,
          description: 'Implement rate limiting',
          implementation: 'Configure API gateway rate limits and circuit breakers',
          effectiveness: 'MEDIUM',
          cost: 'LOW',
        });
        break;
    }

    return mitigations;
  }

  private calculateRiskScore(likelihood: string, impact: string): number {
    const likelihoodValues = { LOW: 1, MEDIUM: 2, HIGH: 3 };
    const impactValues = { LOW: 1, MEDIUM: 2, HIGH: 3 };
    
    return likelihoodValues[likelihood] * impactValues[impact];
  }

  private calculateResidualRisk(threats: Threat[], mitigations: Mitigation[]): 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL' {
    // Calculate residual risk after mitigations
    const highRiskThreats = threats.filter(t => t.riskScore >= 6);
    const mitigatedThreats = highRiskThreats.filter(t => 
      mitigations.some(m => m.threatId === t.id && m.effectiveness === 'HIGH')
    );

    const unmitigatedHighRisk = highRiskThreats.length - mitigatedThreats.length;

    if (unmitigatedHighRisk > 2) return 'CRITICAL';
    if (unmitigatedHighRisk > 1) return 'HIGH';
    if (unmitigatedHighRisk > 0) return 'MEDIUM';
    return 'LOW';
  }

  private calculateComponentRisk(component: ArchitectureComponent, missingControls: string[]): string {
    const criticalControls = ['ENCRYPTION_AT_REST', 'TLS', 'AUTHENTICATION'];
    const criticalMissing = missingControls.filter(control => criticalControls.includes(control));

    if (criticalMissing.length > 1) return 'CRITICAL';
    if (criticalMissing.length > 0) return 'HIGH';
    if (missingControls.length > 3) return 'MEDIUM';
    return 'LOW';
  }

  private getRequiredSecurityControls(component: ArchitectureComponent): string[] {
    const baseControls = ['AUTHENTICATION', 'AUTHORIZATION', 'AUDIT_LOGGING'];
    
    const typeSpecificControls = {
      SERVICE: ['RATE_LIMITING', 'INPUT_VALIDATION', 'OUTPUT_ENCODING'],
      DATABASE: ['ENCRYPTION_AT_REST', 'ACCESS_CONTROL', 'BACKUP_ENCRYPTION'],
      CACHE: ['ACCESS_CONTROL', 'DATA_EXPIRATION'],
      STORAGE: ['ENCRYPTION_AT_REST', 'ACCESS_CONTROL', 'VERSIONING'],
    };

    const dataClassificationControls = {
      CONFIDENTIAL: ['ENCRYPTION_AT_REST', 'ENCRYPTION_IN_TRANSIT', 'ACCESS_LOGGING'],
      RESTRICTED: ['ENCRYPTION_AT_REST', 'ENCRYPTION_IN_TRANSIT', 'ACCESS_LOGGING', 'DLP'],
    };

    return [
      ...baseControls,
      ...(typeSpecificControls[component.type] || []),
      ...(dataClassificationControls[component.dataClassification] || []),
    ];
  }

  private generateImplementationRecommendations(
    component: ArchitectureComponent,
    missingRequirements: string[]
  ): string[] {
    return missingRequirements.map(requirement => {
      switch (requirement) {
        case 'ENCRYPTION_AT_REST':
          return `Implement AES-256 encryption for ${component.name} data storage`;
        case 'RATE_LIMITING':
          return `Configure rate limiting for ${component.name} API endpoints`;
        case 'AUDIT_LOGGING':
          return `Enable comprehensive audit logging for ${component.name}`;
        default:
          return `Implement ${requirement} for ${component.name}`;
      }
    });
  }

  private async generateSecurityRecommendations(
    securityAssessment: any,
    threatModels: ThreatModel[]
  ): Promise<string[]> {
    const recommendations: string[] = [];

    // High-level recommendations based on assessment
    if (securityAssessment.overallRisk === 'HIGH' || securityAssessment.overallRisk === 'CRITICAL') {
      recommendations.push('Conduct immediate security remediation for critical components');
    }

    // Specific recommendations from threat models
    for (const threatModel of threatModels) {
      if (threatModel.residualRisk === 'HIGH' || threatModel.residualRisk === 'CRITICAL') {
        recommendations.push(`Address high-risk threats in ${threatModel.component}`);
      }
    }

    return recommendations;
  }

  // Document generation methods
  private generateExecutiveSummary(components: ArchitectureComponent[], threatModels: ThreatModel[]): string {
    const highRiskComponents = threatModels.filter(tm => 
      tm.residualRisk === 'HIGH' || tm.residualRisk === 'CRITICAL'
    ).length;

    return `This document presents the security design for a system with ${components.length} components. 
Security assessment identified ${highRiskComponents} components requiring immediate attention.`;
  }

  private generateArchitectureOverview(components: ArchitectureComponent[]): string {
    let overview = 'The system architecture consists of the following components:\n\n';
    
    for (const component of components) {
      overview += `- **${component.name}** (${component.type}): ${component.dataClassification} data\n`;
    }

    return overview;
  }

  private generateThreatModelSection(threatModels: ThreatModel[]): string {
    let section = '';
    
    for (const threatModel of threatModels) {
      section += `### ${threatModel.component}\n\n`;
      section += `**Residual Risk**: ${threatModel.residualRisk}\n\n`;
      section += `**Threats Identified**: ${threatModel.threats.length}\n\n`;
      section += `**Mitigations Planned**: ${threatModel.mitigations.length}\n\n`;
    }

    return section;
  }

  private generateSecurityControlsSection(components: ArchitectureComponent[]): string {
    let section = 'Security controls by component:\n\n';
    
    for (const component of components) {
      section += `### ${component.name}\n\n`;
      section += 'Implemented Controls:\n';
      for (const control of component.securityRequirements) {
        section += `- ${control}\n`;
      }
      section += '\n';
    }

    return section;
  }

  private generateRiskAssessmentSection(threatModels: ThreatModel[]): string {
    const riskDistribution = {
      LOW: threatModels.filter(tm => tm.residualRisk === 'LOW').length,
      MEDIUM: threatModels.filter(tm => tm.residualRisk === 'MEDIUM').length,
      HIGH: threatModels.filter(tm => tm.residualRisk === 'HIGH').length,
      CRITICAL: threatModels.filter(tm => tm.residualRisk === 'CRITICAL').length,
    };

    return `Risk assessment results:
- Critical Risk: ${riskDistribution.CRITICAL} components
- High Risk: ${riskDistribution.HIGH} components  
- Medium Risk: ${riskDistribution.MEDIUM} components
- Low Risk: ${riskDistribution.LOW} components`;
  }

  private generateImplementationSection(
    components: ArchitectureComponent[],

  ): string {
    let section = 'Implementation priorities:\n\n';

    // Priority 1: Critical risk components
    const criticalComponents = threatModels.filter(tm => tm.residualRisk === 'CRITICAL');
    if (criticalComponents.length > 0) {
      section += '**Priority 1 (Critical)**:\n';
      for (const component of criticalComponents) {
        section += `- Remediate ${component.component}\n`;
      }
      section += '\n';
    }

    return section;
  }

// src/checklists/security-checklist.service.ts
import { Injectable, Logger } from '@nestjs/common';

interface ChecklistItem {
  id: string;
  category: string;
  description: string;
  checkFunction: string;
  severity: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  automatable: boolean;
  documentation?: string;
}

interface ChecklistResult {
  itemId: string;
  status: 'PASS' | 'FAIL' | 'WARNING' | 'NOT_APPLICABLE';
  details: string;
  evidence?: string[];
  recommendations?: string[];
}

@Injectable()
export class SecurityChecklistService {
  private readonly logger = new Logger(SecurityChecklistService.name);
  private readonly checklistItems = new Map<string, ChecklistItem>();

  constructor() {
    this.initializeChecklist();
  }

  async runSecurityChecklist(
    codebase: string,
    architecture: any,
    deployment: any
  ): Promise<{
    overall: 'PASS' | 'FAIL' | 'WARNING';
    results: ChecklistResult[];
    summary: any;
  }> {
    const results: ChecklistResult[] = [];

    for (const [itemId, item] of this.checklistItems) {
      try {
        const result = await this.executeChecklistItem(item, {
          codebase,
          architecture,
          deployment,
        });
        results.push(result);
      } catch (error) {
        results.push({
          itemId,
          status: 'FAIL',
          details: `Checklist execution failed: ${error.message}`,
        });
      }
    }

    const summary = this.generateSummary(results);
    const overall = this.determineOverallStatus(results);

    return { overall, results, summary };
  }

  async executeChecklistItem(
    item: ChecklistItem,
    context: any
  ): Promise<ChecklistResult> {
    switch (item.checkFunction) {
      case 'checkAuthentication':
        return await this.checkAuthentication(item, context);
      case 'checkAuthorization':
        return await this.checkAuthorization(item, context);
      case 'checkInputValidation':
        return await this.checkInputValidation(item, context);
      case 'checkOutputEncoding':
        return await this.checkOutputEncoding(item, context);
      case 'checkCryptography':
        return await this.checkCryptography(item, context);
      case 'checkLogging':
        return await this.checkLogging(item, context);
      case 'checkErrorHandling':
        return await this.checkErrorHandling(item, context);
      case 'checkSecurityHeaders':
        return await this.checkSecurityHeaders(item, context);
      case 'checkDependencies':
        return await this.checkDependencies(item, context);
      case 'checkConfiguration':
        return await this.checkConfiguration(item, context);
      default:
        throw new Error(`Unknown check function: ${item.checkFunction}`);
    }
  }

  private async checkAuthentication(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    // Check for proper authentication implementation
    const authPatterns = [
      /UseGuards\(.*AuthGuard/,
      /jwt\.verify/,
      /@IsAuthenticated/,
    ];

    const hasAuthentication = authPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    return {
      itemId: item.id,
      status: hasAuthentication ? 'PASS' : 'FAIL',
      details: hasAuthentication 
        ? 'Authentication mechanisms found in codebase'
        : 'No authentication mechanisms detected',
      recommendations: hasAuthentication ? [] : [
        'Implement JWT authentication guards',
        'Add authentication decorators to protected endpoints',
        'Ensure all sensitive endpoints require authentication',
      ],
    };
  }

  private async checkAuthorization(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    // Check for proper authorization implementation
    const authzPatterns = [
      /@Roles\(/,
      /RolesGuard/,
      /canActivate/,
    ];

    const hasAuthorization = authzPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    return {
      itemId: item.id,
      status: hasAuthorization ? 'PASS' : 'FAIL',
      details: hasAuthorization
        ? 'Authorization controls found in codebase'
        : 'No authorization controls detected',
      recommendations: hasAuthorization ? [] : [
        'Implement role-based access control',
        'Add authorization guards to endpoints',
        'Define and enforce user permissions',
      ],
    };
  }

  private async checkInputValidation(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    const validationPatterns = [
      /@IsString\(/,
      /@IsEmail\(/,
      /@Length\(/,
      /ValidationPipe/,
      /class-validator/,
    ];

    const hasValidation = validationPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    return {
      itemId: item.id,
      status: hasValidation ? 'PASS' : 'WARNING',
      details: hasValidation
        ? 'Input validation found in codebase'
        : 'Limited input validation detected',
      recommendations: hasValidation ? [] : [
        'Implement comprehensive input validation using class-validator',
        'Add validation pipes to all endpoints',
        'Validate all user inputs at the boundary',
      ],
    };
  }

  private async checkOutputEncoding(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    const dangerousPatterns = [
      /innerHTML\s*=/,
      /dangerouslySetInnerHTML/,
      /eval\(/,
      /Function\(/,
    ];

    const hasDangerousOutput = dangerousPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    return {
      itemId: item.id,
      status: hasDangerousOutput ? 'FAIL' : 'PASS',
      details: hasDangerousOutput
        ? 'Potentially dangerous output methods detected'
        : 'No dangerous output methods found',
      recommendations: hasDangerousOutput ? [
        'Replace innerHTML with textContent or use sanitization',
        'Avoid dangerouslySetInnerHTML without proper sanitization',
        'Never use eval() or Function() with user input',
      ] : [],
    };
  }

  private async checkCryptography(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    const weakCryptoPatterns = [
      /md5|sha1/i,
      /DES|RC4/i,
      /Math\.random\(\)/,
    ];

    const strongCryptoPatterns = [
      /bcrypt/,
      /crypto\.randomBytes/,
      /sha256|sha512/i,
    ];

    const hasWeakCrypto = weakCryptoPatterns.some(pattern => 
      pattern.test(context.codebase)
    );
    
    const hasStrongCrypto = strongCryptoPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    let status: 'PASS' | 'FAIL' | 'WARNING' = 'PASS';
    let details = 'Cryptography usage appears secure';
    const recommendations: string[] = [];

    if (hasWeakCrypto) {
      status = 'FAIL';
      details = 'Weak cryptographic algorithms detected';
      recommendations.push(
        'Replace MD5/SHA1 with SHA-256 or better',
        'Use bcrypt for password hashing',
        'Replace weak encryption algorithms'
      );
    } else if (!hasStrongCrypto) {
      status = 'WARNING';
      details = 'No cryptographic usage detected - verify if needed';
      recommendations.push('Ensure proper cryptography for sensitive data');
    }

    return {
      itemId: item.id,
      status,
      details,
      recommendations,
    };
  }

  private async checkLogging(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    const properLoggingPatterns = [
      /Logger/,
      /winston/,
      /pino/,
      /this\.logger\./,
    ];

    const improperLoggingPatterns = [
      /console\.(log|error|warn)/,
    ];

    const hasProperLogging = properLoggingPatterns.some(pattern => 
      pattern.test(context.codebase)
    );
    
    const hasImproperLogging = improperLoggingPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    let status: 'PASS' | 'FAIL' | 'WARNING' = 'PASS';
    let details = 'Proper logging framework detected';
    const recommendations: string[] = [];

    if (hasImproperLogging) {
      status = 'WARNING';
      details = 'Console logging detected - should use proper logger';
      recommendations.push('Replace console.log with proper logging framework');
    }

    if (!hasProperLogging) {
      status = 'FAIL';
      details = 'No proper logging framework detected';
      recommendations.push(
        'Implement proper logging with Winston or similar',
        'Add security event logging',
        'Ensure logs do not contain sensitive data'
      );
    }

    return {
      itemId: item.id,
      status,
      details,
      recommendations,
    };
  }

  private async checkErrorHandling(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    const errorHandlingPatterns = [
      /try\s*\{[\s\S]*catch/,
      /@Catch\(/,
      /ExceptionFilter/,
    ];

    const informationLeakPatterns = [
      /throw new Error\(error\)/,
      /res\.status\(500\)\.send\(error\)/,
      /console\.error\(error\)/,
    ];

    const hasErrorHandling = errorHandlingPatterns.some(pattern => 
      pattern.test(context.codebase)
    );
    
    const hasInformationLeak = informationLeakPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    let status: 'PASS' | 'FAIL' | 'WARNING' = 'PASS';
    let details = 'Error handling appears secure';
    const recommendations: string[] = [];

    if (hasInformationLeak) {
      status = 'FAIL';
      details = 'Potential information leakage in error handling';
      recommendations.push(
        'Sanitize error messages before sending to client',
        'Log detailed errors server-side only',
        'Use generic error messages for client responses'
      );
    } else if (!hasErrorHandling) {
      status = 'WARNING';
      details = 'Limited error handling detected';
      recommendations.push('Implement comprehensive error handling');
    }

    return {
      itemId: item.id,
      status,
      details,
      recommendations,
    };
  }

  private async checkSecurityHeaders(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    // This would typically check deployment configuration

      details: hasSecurityHeaders

        'Implement Content Security Policy',
        'Enable HSTS for HTTPS enforcement',
        'Configure proper CORS settings',
      ],
    };
  }

    return {

      ],
    };
  }

  private async checkConfiguration(item: ChecklistItem, context: any): Promise<ChecklistResult> {
    const configPatterns = [
      /ConfigService/,
      /process\.env\./,
      /\.env/,
    ];

    const hardcodedSecretPatterns = [
      /(password|secret|key|token)\s*[:=]\s*["'][^"']*["']/i,
    ];

    const hasConfigManagement = configPatterns.some(pattern => 
      pattern.test(context.codebase)
    );
    
    const hasHardcodedSecrets = hardcodedSecretPatterns.some(pattern => 
      pattern.test(context.codebase)
    );

    let status: 'PASS' | 'FAIL' | 'WARNING' = 'PASS';
    let details = 'Configuration management appears secure';
    const recommendations: string[] = [];

    if (hasHardcodedSecrets) {
      status = 'FAIL';
      details = 'Hardcoded secrets detected in configuration';
      recommendations.push(
        'Move secrets to environment variables',
        'Use secret management service',
        'Never commit secrets to version control'
      );
    } else if (!hasConfigManagement) {
      status = 'WARNING';

  private generateSummary(results: ChecklistResult[]): any {
    return {

  private determineOverallStatus(results: ChecklistResult[]): 'PASS' | 'FAIL' | 'WARNING' {
    const failed = results.filter(r => r.status === 'FAIL');
    const warnings = results.filter(r => r.status === 'WARNING');

    if (failed.length > 0) return 'FAIL';
    if (warnings.length > 0) return 'WARNING';
    return 'PASS';
