Access Control Implementation​

1. Role-Based Access Control (RBAC)​

// src/auth/guards/pii-access.guard.ts
import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { GqlExecutionContext } from '@nestjs/graphql';

export enum PIIAccessLevel {
  NONE = 'none',
  READ = 'read',
  WRITE = 'write',
  ADMIN = 'admin'
}

@Injectable()
export class PIIAccessGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredLevel = this.reflector.get<PIIAccessLevel>('piiAccess', context.getHandler());
    if (!requiredLevel) return true;

    const request = context.switchToHttp().getRequest();
    const user = request.user;

    if (!user) {
      throw new ForbiddenException('Authentication required for PII access');
    }

    // Check if user has required PII access level
    const userPIILevel = this.getUserPIIAccessLevel(user);
    
    if (!this.hasRequiredAccess(userPIILevel, requiredLevel)) {
      throw new ForbiddenException(`Insufficient PII access level. Required: ${requiredLevel}`);
    }

    // Log PII access attempt
    this.auditPIIAccess(user, requiredLevel, context);

    return true;
  }

  private getUserPIIAccessLevel(user: any): PIIAccessLevel {
    // Check user roles and permissions
    if (user.roles.includes('pii-admin')) return PIIAccessLevel.ADMIN;
    if (user.roles.includes('pii-write')) return PIIAccessLevel.WRITE;
    if (user.roles.includes('pii-read')) return PIIAccessLevel.READ;
    return PIIAccessLevel.NONE;
  }

  private hasRequiredAccess(userLevel: PIIAccessLevel, requiredLevel: PIIAccessLevel): boolean {
    const levels = [PIIAccessLevel.NONE, PIIAccessLevel.READ, PIIAccessLevel.WRITE, PIIAccessLevel.ADMIN];
    return levels.indexOf(userLevel) >= levels.indexOf(requiredLevel);
  }

  private auditPIIAccess(user: any, accessLevel: PIIAccessLevel, context: ExecutionContext) {
    // Implementation in audit logging section
  }
}

// Decorator for PII access control
export const RequirePIIAccess = (level: PIIAccessLevel) => 
  SetMetadata('piiAccess', level);

2. Controller Implementation​

// src/users/users.controller.ts
import { Controller, Get, Param, UseGuards } from '@nestjs/common';
import { PIIAccessGuard, RequirePIIAccess, PIIAccessLevel } from '../auth/guards/pii-access.guard';

@Controller('users')
@UseGuards(PIIAccessGuard)
export class UsersController {
  
  @Get(':id')
  @RequirePIIAccess(PIIAccessLevel.READ)
  async getUser(@Param('id') id: string) {
    // This endpoint requires PII read access
    return this.usersService.findOne(id);
  }

  @Get(':id/pii')
  @RequirePIIAccess(PIIAccessLevel.ADMIN)
  async getUserPII(@Param('id') id: string) {
    // This endpoint requires PII admin access
    return this.usersService.findPII(id);
  }
}

import { MigrationInterface, QueryRunner } from 'typeorm';
import { faker } from '@faker-js/faker';

export class MaskPIIData1699000000000 implements MigrationInterface {
  public async up(queryRunner: QueryRunner): Promise<void> {

      console.log('Skipping PII masking in production');

    }

      WHERE email IS NOT NULL;
    `);

    // Mask phone numbers
    await queryRunner.query(`
      UPDATE users 
      SET phone = CONCAT('+1555', LPAD(id, 7, '0'))
      WHERE phone IS NOT NULL;
    `);

    // Mask names
    await queryRunner.query(`
      UPDATE users 
      SET 
        first_name = CONCAT('FirstName', id),
        last_name = CONCAT('LastName', id)
      WHERE first_name IS NOT NULL OR last_name IS NOT NULL;
    `);

    // Mask addresses
    await queryRunner.query(`
      UPDATE addresses 
      SET 
        street = CONCAT(id, ' Main Street'),
        city = 'TestCity',
        postal_code = '12345'
      WHERE street IS NOT NULL;
    `);
  }

  public async down(queryRunner: QueryRunner): Promise<void> {
    // This migration cannot be reversed as original data is lost
    throw new Error('Cannot reverse PII masking migration');
  }
}

2. Service for Data Masking​

// src/common/services/data-masking.service.ts
import { Injectable } from '@nestjs/common';
import { faker } from '@faker-js/faker';

@Injectable()
export class DataMaskingService {
  
  maskEmail(email: string): string {
    if (!email) return email;
    const [, domain] = email.split('@');
    return `user${faker.number.int()}@${domain || 'example.com'}`;
  }

  maskPhone(phone: string): string {
    if (!phone) return phone;
    return faker.phone.number('+1-555-###-####');
  }

  maskName(name: string): string {
    if (!name) return name;
    return faker.person.firstName();
  }

  maskAddress(address: any): any {
    if (!address) return address;
    return {
      ...address,
      street: faker.location.streetAddress(),
      city: faker.location.city(),
      postalCode: faker.location.zipCode(),
    };
  }

  maskCreditCard(cardNumber: string): string {
    if (!cardNumber) return cardNumber;
    return '**** **** **** ' + cardNumber.slice(-4);
  }

  maskSSN(ssn: string): string {
    if (!ssn) return ssn;
    return '***-**-' + ssn.slice(-4);
  }
}

PII Detection and Classification​

1. PII Detection Utility​

// src/common/utils/pii-detector.ts
export interface PIIField {
  field: string;
  type: PIIType;
  confidence: number;
  value?: string;
}

export enum PIIType {
  EMAIL = 'email',
  PHONE = 'phone',
  SSN = 'ssn',
  CREDIT_CARD = 'credit_card',
  NAME = 'name',
  ADDRESS = 'address',
  DOB = 'date_of_birth',
  PASSPORT = 'passport',
  DRIVER_LICENSE = 'driver_license'
}

export class PIIDetector {
  private static patterns = {
    [PIIType.EMAIL]: /\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b/g,
    [PIIType.PHONE]: /(\+?1?[-.\s]?)?\(?([0-9]{3})\)?[-.\s]?([0-9]{3})[-.\s]?([0-9]{4})/g,
    [PIIType.SSN]: /\b\d{3}-?\d{2}-?\d{4}\b/g,
    [PIIType.CREDIT_CARD]: /\b(?:\d{4}[-\s]?){3}\d{4}\b/g,
  };

  static detectPII(data: any, path: string = ''): PIIField[] {
    const results: PIIField[] = [];

    if (typeof data === 'string') {
      Object.entries(this.patterns).forEach(([type, pattern]) => {
        const matches = data.match(pattern);
        if (matches) {
          matches.forEach(match => {
            results.push({
              field: path,
              type: type as PIIType,
              confidence: this.calculateConfidence(match, type as PIIType),
              value: match
            });
          });
        }
      });
    } else if (typeof data === 'object' && data !== null) {
      Object.entries(data).forEach(([key, value]) => {
        const newPath = path ? `${path}.${key}` : key;
        results.push(...this.detectPII(value, newPath));
        
        // Check field names for PII indicators
        if (this.isPIIFieldName(key)) {
          results.push({
            field: newPath,
            type: this.inferTypeFromFieldName(key),
            confidence: 0.8,
            value: typeof value === 'string' ? value : JSON.stringify(value)
          });
        }
      });
    }

    return results;
  }

  private static isPIIFieldName(fieldName: string): boolean {
    const piiFieldNames = [
      'email', 'phone', 'ssn', 'social_security',
      'first_name', 'last_name', 'full_name',
      'address', 'street', 'city', 'zip',
      'birthday', 'birth_date', 'dob',
      'passport', 'license'
    ];
    
    return piiFieldNames.some(pii => 
      fieldName.toLowerCase().includes(pii)
    );
  }

  private static inferTypeFromFieldName(fieldName: string): PIIType {
    const fieldLower = fieldName.toLowerCase();
    if (fieldLower.includes('email')) return PIIType.EMAIL;
    if (fieldLower.includes('phone')) return PIIType.PHONE;
    if (fieldLower.includes('ssn') || fieldLower.includes('social')) return PIIType.SSN;
    if (fieldLower.includes('name')) return PIIType.NAME;
    if (fieldLower.includes('address') || fieldLower.includes('street')) return PIIType.ADDRESS;
    if (fieldLower.includes('birth') || fieldLower.includes('dob')) return PIIType.DOB;
    return PIIType.NAME; // Default
  }

  private static calculateConfidence(value: string, type: PIIType): number {
    // Implement confidence calculation logic
    return 0.9; // Simplified
  }
}

2. PII Scanning Interceptor​

// src/common/interceptors/pii-scanning.interceptor.ts
import { Injectable, NestInterceptor, ExecutionContext, CallHandler, Logger } from '@nestjs/common';
import { Observable } from 'rxjs';
import { tap } from 'rxjs/operators';
import { PIIDetector, PIIField } from '../utils/pii-detector';

@Injectable()
export class PIIScanningInterceptor implements NestInterceptor {
  private readonly logger = new Logger(PIIScanningInterceptor.name);

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const request = context.switchToHttp().getRequest();
    
    // Scan request body for PII
    if (request.body) {
      const detectedPII = PIIDetector.detectPII(request.body);
      if (detectedPII.length > 0) {
        this.logPIIDetection('request', detectedPII, request);
      }
    }

    return next.handle().pipe(
      tap(response => {
        // Scan response for PII
        if (response) {
          const detectedPII = PIIDetector.detectPII(response);
          if (detectedPII.length > 0) {
            this.logPIIDetection('response', detectedPII, request);
          }
        }
      })
    );
  }

  private logPIIDetection(type: 'request' | 'response', piiFields: PIIField[], request: any) {
    this.logger.warn(`PII detected in ${type}`, {
      endpoint: request.url,
      method: request.method,
      userId: request.user?.id,
      piiFields: piiFields.map(field => ({
        field: field.field,
        type: field.type,
        confidence: field.confidence
      }))
    });
  }
}

Encryption Implementation​

1. Field-Level Encryption Service​

// src/common/services/encryption.service.ts
import { Injectable } from '@nestjs/common';
import { createCipher, createDecipher, randomBytes, scrypt } from 'crypto';
import { promisify } from 'util';

@Injectable()
export class EncryptionService {
  private readonly algorithm = 'aes-256-gcm';
  private readonly keyLength = 32;

  async encrypt(text: string, key?: string): Promise<string> {
    if (!text) return text;

    const derivedKey = key ? await this.deriveKey(key) : await this.getDefaultKey();
    const iv = randomBytes(16);
    const cipher = createCipher(this.algorithm, derivedKey);
    
    let encrypted = cipher.update(text, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    return iv.toString('hex') + ':' + encrypted;
  }

  async decrypt(encryptedText: string, key?: string): Promise<string> {
    if (!encryptedText) return encryptedText;

    const [ivHex, encrypted] = encryptedText.split(':');
    if (!ivHex || !encrypted) return encryptedText;

    const derivedKey = key ? await this.deriveKey(key) : await this.getDefaultKey();
    const iv = Buffer.from(ivHex, 'hex');
    const decipher = createDecipher(this.algorithm, derivedKey);
    
    let decrypted = decipher.update(encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }

  private async deriveKey(password: string): Promise<Buffer> {
    const salt = Buffer.from(process.env.ENCRYPTION_SALT || 'default-salt', 'utf8');
    const scryptAsync = promisify(scrypt);
    return scryptAsync(password, salt, this.keyLength) as Promise<Buffer>;
  }

  private async getDefaultKey(): Promise<Buffer> {
    return this.deriveKey(process.env.ENCRYPTION_KEY || 'default-key');
  }
}

2. Database Column Encryption​

// src/common/transformers/encryption.transformer.ts
import { ValueTransformer } from 'typeorm';
import { EncryptionService } from '../services/encryption.service';

export class EncryptionTransformer implements ValueTransformer {
  constructor(private encryptionService: EncryptionService) {}

  to(value: string): Promise<string> {
    return this.encryptionService.encrypt(value);
  }

  from(value: string): Promise<string> {
    return this.encryptionService.decrypt(value);
  }
}

// Usage in entity
@Entity('users')
export class User {
  @Column({
    transformer: new EncryptionTransformer(new EncryptionService())
  })
  email: string;

  @Column({
    transformer: new EncryptionTransformer(new EncryptionService())
  })
  phone: string;
}

Audit Logging​

1. PII Access Audit Service​

// src/audit/services/pii-audit.service.ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { PIIAccessLog } from '../entities/pii-access-log.entity';

@Injectable()
export class PIIAuditService {
  constructor(
    @InjectRepository(PIIAccessLog)
    private piiAccessLogRepository: Repository<PIIAccessLog>
  ) {}

  async logPIIAccess(data: {
    userId: string;
    action: string;
    resource: string;
    piiFields: string[];
    ipAddress: string;
    userAgent: string;
    success: boolean;
    reason?: string;
  }): Promise<void> {
    const log = this.piiAccessLogRepository.create({
      ...data,
      timestamp: new Date(),
    });

    await this.piiAccessLogRepository.save(log);
  }

  async getPIIAccessLogs(filters: {
    userId?: string;
    startDate?: Date;
    endDate?: Date;
    action?: string;
  }): Promise<PIIAccessLog[]> {
    const query = this.piiAccessLogRepository.createQueryBuilder('log');

    if (filters.userId) {
      query.andWhere('log.userId = :userId', { userId: filters.userId });
    }

    if (filters.startDate) {
      query.andWhere('log.timestamp >= :startDate', { startDate: filters.startDate });
    }

    if (filters.endDate) {
      query.andWhere('log.timestamp <= :endDate', { endDate: filters.endDate });
    }

    if (filters.action) {
      query.andWhere('log.action = :action', { action: filters.action });
    }

    return query.orderBy('log.timestamp', 'DESC').getMany();
  }
}

// src/common/services/data-retention.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, LessThan } from 'typeorm';

@Injectable()
export class DataRetentionService {
  private readonly logger = new Logger(DataRetentionService.name);

  constructor(
    @InjectRepository(PIIAccessLog)
    private piiAccessLogRepository: Repository<PIIAccessLog>,
    // Add other repositories as needed
  ) {}

  @Cron(CronExpression.EVERY_DAY_AT_MIDNIGHT)
  async executeRetentionPolicy(): Promise<void> {
    this.logger.log('Starting data retention policy execution');

    try {
      // Delete PII access logs older than 2 years
      await this.deletePIIAccessLogs(2);
      
      // Delete inactive user data older than 5 years
      await this.deleteInactiveUserData(5);
      
      // Archive old order data
      await this.archiveOldOrders(7);

      this.logger.log('Data retention policy executed successfully');
    } catch (error) {
      this.logger.error('Error executing data retention policy', error);
    }
  }

  private async deletePIIAccessLogs(yearsOld: number): Promise<void> {
    const cutoffDate = new Date();
    cutoffDate.setFullYear(cutoffDate.getFullYear() - yearsOld);

    const result = await this.piiAccessLogRepository.delete({
      timestamp: LessThan(cutoffDate)
    });

    this.logger.log(`Deleted ${result.affected} PII access logs older than ${yearsOld} years`);
  }

  private async deleteInactiveUserData(yearsOld: number): Promise<void> {
    // Implementation for deleting inactive user data

  }

  private async archiveOldOrders(yearsOld: number): Promise<void> {

2. Cloud KMS Integration​

// src/common/services/kms-encryption.service.ts
import { Injectable } from '@nestjs/common';
import { KeyManagementServiceClient } from '@google-cloud/kms';

@Injectable()
export class KMSEncryptionService {
  private client = new KeyManagementServiceClient();
  private keyName = `projects/${process.env.GCP_PROJECT_ID}/locations/global/keyRings/pii-ring/cryptoKeys/pii-key`;

  async encrypt(data: string): Promise<string> {
    const [result] = await this.client.encrypt({
      name: this.keyName,
      plaintext: Buffer.from(data),
    });

    return Buffer.from(result.ciphertext!).toString('base64');
  }

  async decrypt(encryptedData: string): Promise<string> {
    const [result] = await this.client.decrypt({
      name: this.keyName,
      ciphertext: Buffer.from(encryptedData, 'base64'),
    });

    return result.plaintext!.toString();
  }
}
