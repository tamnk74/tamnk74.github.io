Session Storage Configuration​

NestJS Session Configuration​

// src/config/session.config.ts
import { ConfigService } from '@nestjs/config';
import { SessionOptions } from 'express-session';
import * as connectRedis from 'connect-redis';
import { Redis } from 'ioredis';

export interface SessionConfig {
  secret: string;
  name: string;
  maxAge: number;
  secure: boolean;
  httpOnly: boolean;
  sameSite: 'strict' | 'lax' | 'none';
  domain?: string;
  rolling: boolean;
  saveUninitialized: boolean;
  resave: boolean;
}

export class SessionConfigService {
  constructor(private configService: ConfigService) {}

  getSessionConfig(): SessionOptions {
    const RedisStore = connectRedis.default;
    const redisClient = new Redis({
      host: this.configService.get('REDIS_HOST'),
      port: this.configService.get('REDIS_PORT'),
      password: this.configService.get('REDIS_PASSWORD'),
      retryDelayOnFailover: 100,
      maxRetriesPerRequest: 3,
      db: this.configService.get('REDIS_SESSION_DB', 1),
    });

    return {
      store: new RedisStore({ client: redisClient }),
      secret: this.configService.get('SESSION_SECRET'),
      name: 'ecom.session.id',
      cookie: {
        maxAge: this.getSessionMaxAge(),
        secure: this.configService.get('NODE_ENV') === 'production',
        httpOnly: true,
        sameSite: 'strict',
        domain: this.configService.get('COOKIE_DOMAIN'),
      },
      rolling: true, // Reset expiration on activity
      saveUninitialized: false,
      resave: false,
      genid: () => this.generateSecureSessionId(),
    };
  }

  private getSessionMaxAge(): number {
    const environment = this.configService.get('NODE_ENV');
    const businessRequirement = this.configService.get('SESSION_TIMEOUT_MINUTES', 30);
    
    // Different timeouts based on environment and business rules
    switch (environment) {
      case 'production':
        return businessRequirement * 60 * 1000; // Convert to milliseconds
      case 'staging':
        return 60 * 60 * 1000; // 1 hour for testing
      default:
        return 24 * 60 * 60 * 1000; // 24 hours for development
    }
  }

  private generateSecureSessionId(): string {
    const crypto = require('crypto');
    return crypto.randomBytes(32).toString('hex');
  }
}

  exports: [SessionService],
})

    private sessionConfigService: SessionConfigService,
    private configService: ConfigService

  }
}

Secure Cookie Management​

Cookie Security Service​

// src/session/cookie-security.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { Request, Response } from 'express';
import * as crypto from 'crypto';

interface SecureCookieOptions {
  name: string;
  value: string;
  maxAge?: number;
  secure?: boolean;
  httpOnly?: boolean;
  sameSite?: 'strict' | 'lax' | 'none';
  domain?: string;
  path?: string;
  signed?: boolean;
}

@Injectable()
export class CookieSecurityService {
  private readonly logger = new Logger(CookieSecurityService.name);
  private readonly cookieSecret: string;

  constructor(private configService: ConfigService) {
    this.cookieSecret = this.configService.get('COOKIE_SECRET');
  }

  setSecureCookie(res: Response, options: SecureCookieOptions): void {
    const cookieOptions = {
      maxAge: options.maxAge || 30 * 60 * 1000, // 30 minutes default
      secure: this.configService.get('NODE_ENV') === 'production',
      httpOnly: options.httpOnly !== false, // Default to true
      sameSite: options.sameSite || 'strict' as const,
      domain: options.domain || this.configService.get('COOKIE_DOMAIN'),
      path: options.path || '/',
      signed: options.signed !== false, // Default to true
    };

    if (cookieOptions.signed) {
      res.cookie(options.name, options.value, cookieOptions);
    } else {
      res.cookie(options.name, options.value, { ...cookieOptions, signed: false });
    }

    this.logger.debug(`Set secure cookie: ${options.name}`);
  }

  clearSecureCookie(res: Response, cookieName: string): void {
    res.clearCookie(cookieName, {
      secure: this.configService.get('NODE_ENV') === 'production',
      httpOnly: true,
      sameSite: 'strict',
      domain: this.configService.get('COOKIE_DOMAIN'),
      path: '/',
    });

    this.logger.debug(`Cleared cookie: ${cookieName}`);
  }

  validateCookieIntegrity(req: Request, cookieName: string): boolean {
    const cookieValue = req.signedCookies[cookieName] || req.cookies[cookieName];
    
    if (!cookieValue) {
      return false;
    }

    // Additional integrity checks can be added here
    return true;
  }

  createAntiCSRFToken(): string {
    return crypto.randomBytes(32).toString('hex');
  }

  setCSRFToken(res: Response, token: string): void {
    this.setSecureCookie(res, {
      name: 'csrf-token',
      value: token,
      httpOnly: false, // Needs to be accessible by JavaScript
      sameSite: 'strict',
      maxAge: 60 * 60 * 1000, // 1 hour
    });
  }

  validateCSRFToken(req: Request, providedToken: string): boolean {
    const cookieToken = req.cookies['csrf-token'];
    return cookieToken && cookieToken === providedToken;
  }

  // Implement cookie encryption for sensitive data
  encryptCookieValue(value: string): string {
    const algorithm = 'aes-256-gcm';
    const key = crypto.scryptSync(this.cookieSecret, 'salt', 32);
    const iv = crypto.randomBytes(16);
    
    const cipher = crypto.createCipher(algorithm, key);
    cipher.setAAD(Buffer.from('cookie-encryption'));
    
    let encrypted = cipher.update(value, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return `${iv.toString('hex')}:${authTag.toString('hex')}:${encrypted}`;
  }

  decryptCookieValue(encryptedValue: string): string | null {
    try {
      const [ivHex, authTagHex, encrypted] = encryptedValue.split(':');
      const algorithm = 'aes-256-gcm';
      const key = crypto.scryptSync(this.cookieSecret, 'salt', 32);
      const iv = Buffer.from(ivHex, 'hex');
      const authTag = Buffer.from(authTagHex, 'hex');
      
      const decipher = crypto.createDecipher(algorithm, key);
      decipher.setAuthTag(authTag);
      decipher.setAAD(Buffer.from('cookie-encryption'));
      
      let decrypted = decipher.update(encrypted, 'hex', 'utf8');
      decrypted += decipher.final('utf8');
      
      return decrypted;
    } catch (error) {
      this.logger.error('Cookie decryption failed:', error);
      return null;
    }
  }
}

Cookie Validation Middleware​

// src/session/middleware/cookie-validation.middleware.ts
import { Injectable, NestMiddleware, UnauthorizedException } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';
import { CookieSecurityService } from '../cookie-security.service';

@Injectable()
export class CookieValidationMiddleware implements NestMiddleware {
  constructor(private cookieSecurityService: CookieSecurityService) {}

  use(req: Request, res: Response, next: NextFunction): void {
    // Validate session cookie integrity
    const sessionCookieValid = this.cookieSecurityService.validateCookieIntegrity(
      req,
      'ecom.session.id'
    );

    if (!sessionCookieValid && req.path !== '/auth/login') {
      throw new UnauthorizedException('Invalid session cookie');
    }

    // Validate CSRF token for state-changing operations
    if (['POST', 'PUT', 'DELETE', 'PATCH'].includes(req.method)) {
      const csrfToken = req.headers['x-csrf-token'] as string;
      const csrfValid = this.cookieSecurityService.validateCSRFToken(req, csrfToken);
      
      if (!csrfValid) {
        throw new UnauthorizedException('Invalid CSRF token');
      }
    }

    next();
  }
}

Session Timeout & Auto-logout​

Session Timeout Middleware​

// src/session/middleware/session-timeout.middleware.ts
import { Injectable, NestMiddleware, Logger } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';
import { ConfigService } from '@nestjs/config';

interface SessionData {
  userId?: string;
  lastActivity?: number;
  loginTime?: number;
  userAgent?: string;
  ipAddress?: string;
}

@Injectable()
export class SessionTimeoutMiddleware implements NestMiddleware {
  private readonly logger = new Logger(SessionTimeoutMiddleware.name);

  constructor(private configService: ConfigService) {}

  use(req: Request & { session: SessionData }, res: Response, next: NextFunction): void {
    if (!req.session) {
      return next();
    }

    const now = Date.now();
    const sessionTimeout = this.configService.get('SESSION_TIMEOUT_MINUTES', 30) * 60 * 1000;
    const absoluteTimeout = this.configService.get('ABSOLUTE_SESSION_TIMEOUT_HOURS', 8) * 60 * 60 * 1000;

    // Check for session timeout due to inactivity
    if (req.session.lastActivity) {
      const timeSinceLastActivity = now - req.session.lastActivity;
      
      if (timeSinceLastActivity > sessionTimeout) {
        this.logger.warn(`Session timeout for user ${req.session.userId} due to inactivity`);
        return this.destroySession(req, res, 'Session expired due to inactivity');
      }
    }

    // Check for absolute session timeout
    if (req.session.loginTime) {
      const sessionAge = now - req.session.loginTime;
      
      if (sessionAge > absoluteTimeout) {
        this.logger.warn(`Absolute session timeout for user ${req.session.userId}`);
        return this.destroySession(req, res, 'Session expired due to maximum duration');
      }
    }

    // Check for session hijacking (different user agent or IP)
    if (this.detectSessionHijacking(req)) {
      this.logger.error(`Potential session hijacking detected for user ${req.session.userId}`);
      return this.destroySession(req, res, 'Session terminated for security reasons');
    }

    // Update last activity
    req.session.lastActivity = now;

    // Set session timeout warning header (5 minutes before expiry)
    const timeUntilTimeout = sessionTimeout - (now - req.session.lastActivity);
    if (timeUntilTimeout <= 5 * 60 * 1000) {
      res.set('X-Session-Warning', 'Session will expire soon');
    }

    next();
  }

  private detectSessionHijacking(req: Request & { session: SessionData }): boolean {
    const currentUserAgent = req.headers['user-agent'];
    const currentIP = req.ip || req.connection.remoteAddress;

    // Compare with stored session data
    if (req.session.userAgent && req.session.userAgent !== currentUserAgent) {
      return true;
    }

    if (req.session.ipAddress && req.session.ipAddress !== currentIP) {
      return true;
    }

    return false;
  }

  private destroySession(
    req: Request & { session: SessionData },
    res: Response,
    reason: string
  ): void {
    req.session.destroy((err) => {
      if (err) {
        this.logger.error('Failed to destroy session:', err);
      }
    });

    res.status(401).json({
      error: 'Session expired',
      reason,
      redirectTo: '/auth/login',
    });
  }
}

Next.js Client-Side Session Management​

// src/hooks/useSessionTimeout.ts
import { useEffect, useCallback, useRef } from 'react';
import { useRouter } from 'next/router';
import { useAuth } from '../contexts/AuthContext';

interface UseSessionTimeoutOptions {
  timeoutMinutes?: number;
  warningMinutes?: number;
  onWarning?: () => void;
  onTimeout?: () => void;
}

export const useSessionTimeout = (options: UseSessionTimeoutOptions = {}) => {
  const {
    timeoutMinutes = 30,
    warningMinutes = 5,
    onWarning,
    onTimeout,
  } = options;

  const router = useRouter();
  const { logout } = useAuth();
  const timeoutRef = useRef<NodeJS.Timeout>();
  const warningRef = useRef<NodeJS.Timeout>();
  const lastActivityRef = useRef<number>(Date.now());

  const handleTimeout = useCallback(() => {
    logout();
    onTimeout?.();
    router.push('/auth/login?reason=timeout');
  }, [logout, onTimeout, router]);

  const handleWarning = useCallback(() => {
    onWarning?.();
  }, [onWarning]);

  const resetTimeout = useCallback(() => {
    lastActivityRef.current = Date.now();

    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }
    if (warningRef.current) {
      clearTimeout(warningRef.current);
    }

    // Set warning timeout
    warningRef.current = setTimeout(
      handleWarning,
      (timeoutMinutes - warningMinutes) * 60 * 1000
    );

    // Set session timeout
    timeoutRef.current = setTimeout(
      handleTimeout,
      timeoutMinutes * 60 * 1000
    );
  }, [timeoutMinutes, warningMinutes, handleTimeout, handleWarning]);

  const extendSession = useCallback(async () => {
    try {
      const response = await fetch('/api/auth/extend-session', {
        method: 'POST',
        credentials: 'include',
      });

      if (response.ok) {
        resetTimeout();
        return true;
      }
      return false;
    } catch (error) {
      console.error('Failed to extend session:', error);
      return false;
    }
  }, [resetTimeout]);

  useEffect(() => {
    const events = ['mousedown', 'mousemove', 'keypress', 'scroll', 'touchstart'];
    
    const updateActivity = () => {
      const now = Date.now();
      // Only reset if more than 30 seconds since last activity
      if (now - lastActivityRef.current > 30000) {
        resetTimeout();
      }
    };

    events.forEach(event => {
      document.addEventListener(event, updateActivity, true);
    });

    // Check for session warning from server
    const checkSessionWarning = () => {
      const xhr = new XMLHttpRequest();
      xhr.open('HEAD', '/api/auth/session-status');
      xhr.onreadystatechange = () => {
        if (xhr.readyState === 4) {
          const warning = xhr.getResponseHeader('X-Session-Warning');
          if (warning) {
            handleWarning();
          }
        }
      };
      xhr.send();
    };

    const warningCheckInterval = setInterval(checkSessionWarning, 60000); // Check every minute

    resetTimeout();

    return () => {
      events.forEach(event => {
        document.removeEventListener(event, updateActivity, true);
      });
      
      if (timeoutRef.current) {
        clearTimeout(timeoutRef.current);
      }
      if (warningRef.current) {
        clearTimeout(warningRef.current);
      }
      clearInterval(warningCheckInterval);
    };
  }, [resetTimeout, handleWarning]);

  return {
    extendSession,
    resetTimeout,
  };
};

Session Timeout Warning Component​

// src/components/SessionTimeoutWarning.tsx
import React, { useState, useEffect } from 'react';
import { Dialog } from '@headlessui/react';
import { useSessionTimeout } from '../hooks/useSessionTimeout';

interface SessionTimeoutWarningProps {
  show: boolean;
  onExtend: () => Promise<boolean>;
  onLogout: () => void;
}

export const SessionTimeoutWarning: React.FC<SessionTimeoutWarningProps> = ({
  show,
  onExtend,
  onLogout,
}) => {
  const [countdown, setCountdown] = useState(300); // 5 minutes
  const [extending, setExtending] = useState(false);

  useEffect(() => {
    if (!show) {
      setCountdown(300);
      return;
    }

    const interval = setInterval(() => {
      setCountdown(prev => {
        if (prev <= 1) {
          onLogout();
          return 0;
        }
        return prev - 1;
      });
    }, 1000);

    return () => clearInterval(interval);
  }, [show, onLogout]);

  const handleExtend = async () => {
    setExtending(true);
    const success = await onExtend();
    setExtending(false);
    
    if (!success) {
      onLogout();
    }
  };

  const formatTime = (seconds: number) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins}:${secs.toString().padStart(2, '0')}`;
  };

  return (
    <Dialog open={show} onClose={() => {}} className="relative z-50">
      <div className="fixed inset-0 bg-black/30" aria-hidden="true" />
      
      <div className="fixed inset-0 flex items-center justify-center p-4">
        <Dialog.Panel className="mx-auto max-w-sm rounded bg-white p-6 shadow-lg">
          <Dialog.Title className="text-lg font-medium text-gray-900 mb-4">
            Session Expiring Soon
          </Dialog.Title>
          
          <p className="text-sm text-gray-500 mb-4">
            Your session will expire in{' '}
            <span className="font-mono font-semibold text-red-600">
              {formatTime(countdown)}
            </span>
          </p>
          
          <div className="flex gap-3">
            <button
              onClick={handleExtend}
              disabled={extending}
              className="flex-1 bg-blue-600 text-white px-4 py-2 rounded text-sm font-medium hover:bg-blue-700 disabled:opacity-50"
            >
              {extending ? 'Extending...' : 'Stay Signed In'}
            </button>
            
            <button
              onClick={onLogout}
              className="flex-1 bg-gray-600 text-white px-4 py-2 rounded text-sm font-medium hover:bg-gray-700"
            >
              Sign Out
            </button>
          </div>
        </Dialog.Panel>
      </div>
    </Dialog>
  );
};

JWT Session Management​

JWT Session Service​

// src/auth/jwt-session.service.ts
import { Injectable, UnauthorizedException, Logger } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { ConfigService } from '@nestjs/config';
import { Redis } from 'ioredis';

interface JWTPayload {
  sub: string;
  email: string;
  role: string;
  sessionId: string;
  iat: number;
  exp: number;
}

interface RefreshTokenPayload {
  sub: string;
  sessionId: string;
  tokenVersion: number;
}

@Injectable()
export class JWTSessionService {
  private readonly logger = new Logger(JWTSessionService.name);
  private redis: Redis;

  constructor(
    private jwtService: JwtService,
    private configService: ConfigService
  ) {
    this.redis = new Redis({
      host: this.configService.get('REDIS_HOST'),
      port: this.configService.get('REDIS_PORT'),
      password: this.configService.get('REDIS_PASSWORD'),
      db: this.configService.get('REDIS_JWT_DB', 2),
    });
  }

  async createSession(userId: string, email: string, role: string): Promise<{
    accessToken: string;
    refreshToken: string;
    sessionId: string;
  }> {
    const sessionId = this.generateSessionId();
    const tokenVersion = await this.getTokenVersion(userId);

    // Store session metadata in Redis
    await this.redis.hset(`session:${sessionId}`, {
      userId,
      email,
      role,
      createdAt: Date.now(),
      lastActivity: Date.now(),
      active: 'true',
    });

    // Set session expiration
    await this.redis.expire(`session:${sessionId}`, 24 * 60 * 60); // 24 hours

    const accessToken = this.jwtService.sign(
      { sub: userId, email, role, sessionId },
      { expiresIn: '15m' }
    );

    const refreshToken = this.jwtService.sign(
      { sub: userId, sessionId, tokenVersion },
      { 
        secret: this.configService.get('JWT_REFRESH_SECRET'),
        expiresIn: '7d'
      }
    );

    return { accessToken, refreshToken, sessionId };
  }

  async refreshSession(refreshToken: string): Promise<{
    accessToken: string;
    refreshToken?: string;
  }> {
    try {
      const payload = this.jwtService.verify(refreshToken, {
        secret: this.configService.get('JWT_REFRESH_SECRET'),
      }) as RefreshTokenPayload;

      // Verify session exists and is active
      const sessionData = await this.redis.hgetall(`session:${payload.sessionId}`);
      if (!sessionData.userId || sessionData.active !== 'true') {
        throw new UnauthorizedException('Session not found or inactive');
      }

      // Verify token version
      const currentTokenVersion = await this.getTokenVersion(payload.sub);
      if (payload.tokenVersion !== currentTokenVersion) {
        throw new UnauthorizedException('Token version mismatch');
      }

      // Update last activity
      await this.redis.hset(`session:${payload.sessionId}`, 'lastActivity', Date.now());

      // Generate new access token
      const accessToken = this.jwtService.sign({
        sub: payload.sub,
        email: sessionData.email,
        role: sessionData.role,
        sessionId: payload.sessionId,
      }, { expiresIn: '15m' });

      // Optionally rotate refresh token
      let newRefreshToken: string | undefined;
      if (this.shouldRotateRefreshToken(refreshToken)) {
        newRefreshToken = this.jwtService.sign(
          { sub: payload.sub, sessionId: payload.sessionId, tokenVersion: currentTokenVersion },
          {
            secret: this.configService.get('JWT_REFRESH_SECRET'),
            expiresIn: '7d'
          }
        );
      }

      return { accessToken, refreshToken: newRefreshToken };
    } catch (error) {
      this.logger.error('Token refresh failed:', error);
      throw new UnauthorizedException('Invalid refresh token');
    }
  }

  async validateSession(sessionId: string): Promise<boolean> {
    const sessionData = await this.redis.hgetall(`session:${sessionId}`);
    return sessionData.userId && sessionData.active === 'true';
  }

  async destroySession(sessionId: string): Promise<void> {
    await this.redis.del(`session:${sessionId}`);
    this.logger.log(`Session destroyed: ${sessionId}`);
  }

  async destroyAllUserSessions(userId: string): Promise<void> {
    const sessionKeys = await this.redis.keys(`session:*`);
    
    for (const key of sessionKeys) {
      const sessionData = await this.redis.hgetall(key);
      if (sessionData.userId === userId) {
        await this.redis.del(key);
      }
    }

    // Increment token version to invalidate all tokens
    await this.incrementTokenVersion(userId);
    
    this.logger.log(`All sessions destroyed for user: ${userId}`);
  }

  async getActiveSessions(userId: string): Promise<any[]> {
    const sessionKeys = await this.redis.keys(`session:*`);
    const activeSessions = [];

    for (const key of sessionKeys) {
      const sessionData = await this.redis.hgetall(key);
      if (sessionData.userId === userId && sessionData.active === 'true') {
        activeSessions.push({
          sessionId: key.split(':')[1],
          createdAt: new Date(parseInt(sessionData.createdAt)),
          lastActivity: new Date(parseInt(sessionData.lastActivity)),
        });
      }
    }

    return activeSessions;
  }

  private generateSessionId(): string {
    const crypto = require('crypto');
    return crypto.randomBytes(32).toString('hex');
  }

  private async getTokenVersion(userId: string): Promise<number> {
    const version = await this.redis.get(`token_version:${userId}`);
    return version ? parseInt(version) : 0;
  }

  private async incrementTokenVersion(userId: string): Promise<number> {
    return await this.redis.incr(`token_version:${userId}`);
  }

  private shouldRotateRefreshToken(refreshToken: string): boolean {
    try {
      const decoded = this.jwtService.decode(refreshToken) as any;
      const now = Date.now() / 1000;
      const tokenAge = now - decoded.iat;
      
      // Rotate if token is older than 1 day
      return tokenAge > 24 * 60 * 60;
    } catch {
      return true;
    }
  }
}

// src/session/redis-session.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { Redis, Cluster } from 'ioredis';

interface SessionMetadata {
  userId: string;
  email: string;
  role: string;
  createdAt: number;
  lastActivity: number;
  ipAddress: string;
  userAgent: string;
  loginMethod: string;
  metadata?: Record<string, any>;
}

@Injectable()
export class RedisSessionService {
  private readonly logger = new Logger(RedisSessionService.name);
  private redis: Redis | Cluster;

  constructor(private configService: ConfigService) {
    this.redis = this.createRedisConnection();
  }

  private createRedisConnection(): Redis | Cluster {
    const isCluster = this.configService.get('REDIS_CLUSTER_MODE', false);
    
    if (isCluster) {
      const clusterNodes = this.configService
        .get('REDIS_CLUSTER_NODES', 'localhost:7000')
        .split(',')
        .map((node: string) => {
          const [host, port] = node.split(':');
          return { host, port: parseInt(port) };
        });

      return new Cluster(clusterNodes, {
        redisOptions: {
          password: this.configService.get('REDIS_PASSWORD'),
        },
        clusterRetryDelayOnFailover: 100,
        maxRetriesPerRequest: 3,
      });
    }

    return new Redis({
      host: this.configService.get('REDIS_HOST'),
      port: this.configService.get('REDIS_PORT'),
      password: this.configService.get('REDIS_PASSWORD'),
      retryDelayOnFailover: 100,
      maxRetriesPerRequest: 3,
      db: this.configService.get('REDIS_SESSION_DB', 1),
    });
  }

  async createSession(sessionId: string, metadata: SessionMetadata): Promise<void> {
    const sessionKey = `session:${sessionId}`;
    const sessionData = {
      ...metadata,
      active: 'true',
    };

    try {
      await this.redis.hmset(sessionKey, sessionData);
      await this.redis.expire(sessionKey, this.getSessionTTL());
      
      // Add to user's active sessions list
      await this.redis.sadd(`user_sessions:${metadata.userId}`, sessionId);
      
      this.logger.log(`Session created: ${sessionId} for user: ${metadata.userId}`);
    } catch (error) {
      this.logger.error('Failed to create session:', error);
      throw error;
    }
  }

  async getSession(sessionId: string): Promise<SessionMetadata | null> {
    try {
      const sessionData = await this.redis.hgetall(`session:${sessionId}`);
      
      if (!sessionData.userId || sessionData.active !== 'true') {
        return null;
      }

      return {
        userId: sessionData.userId,
        email: sessionData.email,
        role: sessionData.role,
        createdAt: parseInt(sessionData.createdAt),
        lastActivity: parseInt(sessionData.lastActivity),
        ipAddress: sessionData.ipAddress,
        userAgent: sessionData.userAgent,
        loginMethod: sessionData.loginMethod,
        metadata: sessionData.metadata ? JSON.parse(sessionData.metadata) : {},
      };
    } catch (error) {
      this.logger.error(`Failed to get session ${sessionId}:`, error);
      return null;
    }
  }

  async updateSessionActivity(sessionId: string): Promise<void> {
    try {
      const sessionKey = `session:${sessionId}`;
      await this.redis.hset(sessionKey, 'lastActivity', Date.now());
      await this.redis.expire(sessionKey, this.getSessionTTL());
    } catch (error) {
      this.logger.error(`Failed to update session activity ${sessionId}:`, error);
    }
  }

  async destroySession(sessionId: string): Promise<void> {
    try {
      const sessionData = await this.getSession(sessionId);
      if (sessionData) {
        await this.redis.srem(`user_sessions:${sessionData.userId}`, sessionId);
      }
      
      await this.redis.del(`session:${sessionId}`);
      this.logger.log(`Session destroyed: ${sessionId}`);
    } catch (error) {
      this.logger.error(`Failed to destroy session ${sessionId}:`, error);
    }
  }

  async destroyAllUserSessions(userId: string): Promise<void> {
    try {
      const sessionIds = await this.redis.smembers(`user_sessions:${userId}`);
      
      if (sessionIds.length > 0) {
        const sessionKeys = sessionIds.map(id => `session:${id}`);
        await this.redis.del(...sessionKeys);
        await this.redis.del(`user_sessions:${userId}`);
      }
      
      this.logger.log(`All sessions destroyed for user: ${userId}`);
    } catch (error) {
      this.logger.error(`Failed to destroy all sessions for user ${userId}:`, error);
    }
  }

  async getActiveSessionsForUser(userId: string): Promise<SessionMetadata[]> {
    try {
      const sessionIds = await this.redis.smembers(`user_sessions:${userId}`);
      const sessions: SessionMetadata[] = [];

      for (const sessionId of sessionIds) {
        const session = await this.getSession(sessionId);
        if (session) {
          sessions.push(session);
        }
      }

      return sessions;
    } catch (error) {
      this.logger.error(`Failed to get active sessions for user ${userId}:`, error);
      return [];
    }
  }

  async cleanupExpiredSessions(): Promise<void> {
    try {
      const cursor = '0';
      const sessionKeys = await this.redis.keys('session:*');
      let cleanedCount = 0;

      for (const key of sessionKeys) {
        const ttl = await this.redis.ttl(key);
        if (ttl === -1 || ttl === -2) { // No expiry or expired
          const sessionId = key.split(':')[1];
          await this.destroySession(sessionId);
          cleanedCount++;
        }
      }

      this.logger.log(`Cleaned up ${cleanedCount} expired sessions`);
    } catch (error) {
      this.logger.error('Failed to cleanup expired sessions:', error);
    }
  }

  private getSessionTTL(): number {
    return this.configService.get('SESSION_TIMEOUT_MINUTES', 30) * 60; // Convert to seconds
  }
}
