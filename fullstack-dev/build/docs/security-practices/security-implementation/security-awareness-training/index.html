Training Management System​

// src/training/training-management.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';

interface TrainingModule {
  id: string;
  title: string;
  category: 'OWASP' | 'CWE' | 'COMPLIANCE' | 'GENERAL';
  level: 'BEGINNER' | 'INTERMEDIATE' | 'ADVANCED';
  duration: number; // minutes
  prerequisites: string[];
  objectives: string[];
  content: TrainingContent[];
  assessment: Assessment;
  certification: boolean;
}

interface TrainingContent {
  type: 'VIDEO' | 'DOCUMENT' | 'LAB' | 'QUIZ' | 'DEMO';
  title: string;
  url: string;
  duration: number;
  description: string;
}

interface Assessment {
  type: 'QUIZ' | 'PRACTICAL' | 'PROJECT';
  passingScore: number;
  questions: Question[];
  timeLimit: number;
}

interface Question {
  id: string;
  type: 'MULTIPLE_CHOICE' | 'TRUE_FALSE' | 'CODE_REVIEW' | 'SCENARIO';
  question: string;
  options?: string[];
  correctAnswer: string | string[];
  explanation: string;
}

interface UserProgress {
  userId: string;
  moduleId: string;
  status: 'NOT_STARTED' | 'IN_PROGRESS' | 'COMPLETED' | 'CERTIFIED';
  score?: number;
  completedDate?: Date;
  certificateId?: string;
  timeSpent: number;
}

@Injectable()
export class TrainingManagementService {
  private readonly logger = new Logger(TrainingManagementService.name);
  private readonly modules = new Map<string, TrainingModule>();
  private readonly userProgress = new Map<string, UserProgress[]>();

  constructor(private configService: ConfigService) {
    this.initializeTrainingModules();
  }

  async enrollUser(userId: string, moduleId: string): Promise<void> {
    const module = this.modules.get(moduleId);
    if (!module) {
      throw new Error(`Training module not found: ${moduleId}`);
    }

    const userProgressList = this.userProgress.get(userId) || [];
    const existingProgress = userProgressList.find(p => p.moduleId === moduleId);

    if (existingProgress) {
      this.logger.warn(`User ${userId} already enrolled in module ${moduleId}`);
      return;
    }

    const progress: UserProgress = {
      userId,
      moduleId,
      status: 'NOT_STARTED',
      timeSpent: 0,
    };

    userProgressList.push(progress);
    this.userProgress.set(userId, userProgressList);

    this.logger.log(`User ${userId} enrolled in training module: ${module.title}`);
  }

  async startModule(userId: string, moduleId: string): Promise<void> {
    const progress = this.getUserProgress(userId, moduleId);
    progress.status = 'IN_PROGRESS';
    
    this.logger.log(`User ${userId} started training module: ${moduleId}`);
  }

  async completeModule(userId: string, moduleId: string, score: number): Promise<void> {
    const progress = this.getUserProgress(userId, moduleId);
    const module = this.modules.get(moduleId);

    if (!module) {
      throw new Error(`Module not found: ${moduleId}`);
    }

    progress.score = score;
    progress.completedDate = new Date();

    if (score >= module.assessment.passingScore) {
      progress.status = 'COMPLETED';
      
      if (module.certification) {
        const certificateId = await this.issueCertificate(userId, moduleId);
        progress.certificateId = certificateId;
        progress.status = 'CERTIFIED';
      }
    }

    this.logger.log(`User ${userId} completed module ${moduleId} with score: ${score}`);
  }

  async getTrainingPlan(userId: string, role: string): Promise<TrainingModule[]> {
    const allModules = Array.from(this.modules.values());
    
    // Customize training plan based on user role
    switch (role) {
      case 'developer':
        return allModules.filter(m => 
          m.category === 'OWASP' || 
          m.category === 'CWE' || 
          (m.category === 'COMPLIANCE' && m.level !== 'ADVANCED')
        );
      
      case 'security-engineer':
        return allModules; // All modules
      
      case 'devops':
        return allModules.filter(m => 
          m.category === 'OWASP' || 
          m.title.includes('Infrastructure')
        );
      
      default:
        return allModules.filter(m => m.level === 'BEGINNER');
    }
  }

  async generateTrainingReport(teamId: string): Promise<any> {
    // Generate comprehensive training report for the team
    const teamMembers = await this.getTeamMembers(teamId);
    const report = {
      teamId,
      totalMembers: teamMembers.length,
      completionRates: {},
      averageScores: {},
      certifications: 0,
      overallProgress: 0,
    };

    // Calculate metrics
    for (const member of teamMembers) {
      const userProgressList = this.userProgress.get(member.id) || [];
      // Calculate individual progress and aggregate
    }

    return report;
  }

  private getUserProgress(userId: string, moduleId: string): UserProgress {
    const userProgressList = this.userProgress.get(userId) || [];
    const progress = userProgressList.find(p => p.moduleId === moduleId);
    
    if (!progress) {
      throw new Error(`User ${userId} not enrolled in module ${moduleId}`);
    }
    
    return progress;
  }

  private async issueCertificate(userId: string, moduleId: string): Promise<string> {
    const certificateId = `CERT-${userId}-${moduleId}-${Date.now()}`;

  }

  private readonly logger = new Logger(TrainingSchedulerService.name);

    // Schedule OWASP Top 10 monthly sessions

// src/training/security-champions.service.ts
import { Injectable, Logger } from '@nestjs/common';

interface SecurityChampion {
  id: string;
  name: string;
  email: string;
  team: string;
  level: 'BRONZE' | 'SILVER' | 'GOLD' | 'PLATINUM';
  certifications: string[];
  achievements: Achievement[];
  menteeCount: number;
  lastActivityDate: Date;
  skillAreas: string[];
}

interface Achievement {
  id: string;
  title: string;
  description: string;
  category: 'TRAINING' | 'MENTORING' | 'CONTRIBUTION' | 'LEADERSHIP';
  dateEarned: Date;
  points: number;
}

interface ChampionProgram {
  requirements: Record<string, any>;
  benefits: string[];
  responsibilities: string[];
  progression: ProgressionLevel[];
}

interface ProgressionLevel {
  level: string;
  requirements: {
    trainingsCompleted: number;
    certificationsRequired: string[];
    mentoringHours: number;
    contributionsRequired: number;
  };
  benefits: string[];
}

@Injectable()
export class SecurityChampionsService {
  private readonly logger = new Logger(SecurityChampionsService.name);
  private readonly champions = new Map<string, SecurityChampion>();

  constructor() {
    this.initializeChampionProgram();
  }

  async nominateChampion(userId: string, nominatorId: string, reason: string): Promise<void> {
    this.logger.log(`User ${userId} nominated as security champion by ${nominatorId}`);
    
    // Verify nomination criteria
    const userProgress = await this.getUserTrainingProgress(userId);
    if (!this.meetsChampionCriteria(userProgress)) {
      throw new Error('User does not meet champion criteria');
    }

    // Create champion profile
    const champion: SecurityChampion = {
      id: userId,
      name: await this.getUserName(userId),
      email: await this.getUserEmail(userId),
      team: await this.getUserTeam(userId),
      level: 'BRONZE',
      certifications: await this.getUserCertifications(userId),
      achievements: [],
      menteeCount: 0,
      lastActivityDate: new Date(),
      skillAreas: this.identifySkillAreas(userProgress),
    };

    this.champions.set(userId, champion);
    
    await this.awardAchievement(userId, {
      id: 'champion-nomination',
      title: 'Security Champion',
      description: 'Nominated and accepted as a security champion',
      category: 'LEADERSHIP',
      dateEarned: new Date(),
      points: 100,
    });
  }

  async assignMentee(championId: string, menteeId: string): Promise<void> {
    const champion = this.champions.get(championId);
    if (!champion) {
      throw new Error(`Champion not found: ${championId}`);
    }

    champion.menteeCount++;
    champion.lastActivityDate = new Date();

    await this.createMentoringPlan(championId, menteeId);
    
    this.logger.log(`Champion ${championId} assigned mentee ${menteeId}`);
  }

  async trackChampionActivity(championId: string, activity: string): Promise<void> {
    const champion = this.champions.get(championId);
    if (!champion) return;

    champion.lastActivityDate = new Date();

    // Award points based on activity type
    switch (activity) {
      case 'training-conducted':
        await this.awardAchievement(championId, {
          id: `training-${Date.now()}`,
          title: 'Training Conductor',
          description: 'Conducted security training session',
          category: 'TRAINING',
          dateEarned: new Date(),
          points: 50,
        });
        break;
      case 'mentoring-session':
        await this.awardAchievement(championId, {
          id: `mentoring-${Date.now()}`,
          title: 'Mentor',
          description: 'Completed mentoring session',
          category: 'MENTORING',
          dateEarned: new Date(),
          points: 25,
        });
        break;
      case 'security-contribution':
        await this.awardAchievement(championId, {
          id: `contribution-${Date.now()}`,
          title: 'Security Contributor',
          description: 'Made significant security contribution',
          category: 'CONTRIBUTION',
          dateEarned: new Date(),
          points: 75,
        });
        break;
    }

    await this.evaluateChampionLevelPromotion(championId);
  }

  async evaluateChampionLevelPromotion(championId: string): Promise<void> {
    const champion = this.champions.get(championId);
    if (!champion) return;

    const totalPoints = champion.achievements.reduce((sum, achievement) => sum + achievement.points, 0);
    const currentLevel = champion.level;
    let newLevel = currentLevel;

    // Define promotion criteria
    if (totalPoints >= 1000 && champion.menteeCount >= 5 && champion.certifications.length >= 3) {
      newLevel = 'PLATINUM';
    } else if (totalPoints >= 500 && champion.menteeCount >= 3 && champion.certifications.length >= 2) {
      newLevel = 'GOLD';
    } else if (totalPoints >= 200 && champion.menteeCount >= 1) {
      newLevel = 'SILVER';
    }

    if (newLevel !== currentLevel) {
      champion.level = newLevel;
      await this.awardAchievement(championId, {
        id: `promotion-${newLevel}`,
        title: `${newLevel} Champion`,
        description: `Promoted to ${newLevel} level champion`,
        category: 'LEADERSHIP',
        dateEarned: new Date(),
        points: 200,
      });
      
      this.logger.log(`Champion ${championId} promoted to ${newLevel} level`);
    }
  }

  async generateChampionReport(): Promise<any> {
    const allChampions = Array.from(this.champions.values());
    
    return {
      totalChampions: allChampions.length,
      levelDistribution: {
        BRONZE: allChampions.filter(c => c.level === 'BRONZE').length,
        SILVER: allChampions.filter(c => c.level === 'SILVER').length,
        GOLD: allChampions.filter(c => c.level === 'GOLD').length,
        PLATINUM: allChampions.filter(c => c.level === 'PLATINUM').length,
      },
      totalMentees: allChampions.reduce((sum, c) => sum + c.menteeCount, 0),
      averageTrainingsPerChampion: await this.calculateAverageTrainings(),
      topContributors: allChampions
        .sort((a, b) => this.getTotalPoints(b) - this.getTotalPoints(a))
        .slice(0, 5),
    };
  }

  private meetsChampionCriteria(userProgress: any): boolean {
    // Define criteria for becoming a security champion
    return userProgress.completedModules >= 3 && 
           userProgress.certifications >= 1 &&
           userProgress.averageScore >= 80;
  }

  private identifySkillAreas(userProgress: any): string[] {
    // Identify user's strong skill areas based on training progress
    return ['Web Security', 'Authentication', 'Input Validation'];
  }

  private async awardAchievement(championId: string, achievement: Achievement): Promise<void> {
    const champion = this.champions.get(championId);
    if (champion) {
      champion.achievements.push(achievement);
    }
  }

  private async createMentoringPlan(championId: string, menteeId: string): Promise<void> {
    // Create structured mentoring plan
    this.logger.log(`Created mentoring plan: ${championId} -> ${menteeId}`);
  }

  private getTotalPoints(champion: SecurityChampion): number {
    return champion.achievements.reduce((sum, achievement) => sum + achievement.points, 0);
  }

  private async getUserTrainingProgress(userId: string): Promise<any> {
