OWASP ZAP Integration​

// scripts/security-scan.ts
import { execSync } from 'child_process';
import * as fs from 'fs';

interface SecurityScanResult {
  alerts: Array<{
    alert: string;
    risk: 'High' | 'Medium' | 'Low' | 'Informational';
    confidence: string;
    url: string;
    description: string;
  }>;
}

export class SecurityScanner {
  private readonly zapApiKey: string;
  private readonly targetUrl: string;

  constructor(zapApiKey: string, targetUrl: string) {
    this.zapApiKey = zapApiKey;
    this.targetUrl = targetUrl;
  }

  async runBaselineScan(): Promise<SecurityScanResult> {
    const command = `
      docker run -t owasp/zap2docker-stable zap-baseline.py \
        -t ${this.targetUrl} \
        -f json \
        -J zap-report.json
    `;

    try {
      execSync(command, { stdio: 'inherit' });
      const report = JSON.parse(fs.readFileSync('zap-report.json', 'utf8'));
      return this.parseZapReport(report);
    } catch (error) {
      console.error('Security scan failed:', error);
      throw error;
    }
  }

  async runFullScan(): Promise<SecurityScanResult> {
    const command = `
      docker run -t owasp/zap2docker-stable zap-full-scan.py \
        -t ${this.targetUrl} \
        -f json \
        -J zap-full-report.json
    `;

    try {
      execSync(command, { stdio: 'inherit' });
      const report = JSON.parse(fs.readFileSync('zap-full-report.json', 'utf8'));
      return this.parseZapReport(report);
    } catch (error) {
      console.error('Full security scan failed:', error);
      throw error;
    }
  }

  private parseZapReport(report: any): SecurityScanResult {
    return {
      alerts: report.site?.[0]?.alerts?.map((alert: any) => ({
        alert: alert.name,
        risk: alert.riskdesc.split(' ')[0],
        confidence: alert.confidence,
        url: alert.url,
        description: alert.desc
      })) || []
    };
  }
}

Google Cloud Security Scanner Integration​

// src/security/cloud-security-scanner.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { SecurityScanner } from '@google-cloud/security-center';

@Injectable()
export class CloudSecurityScannerService {
  private readonly logger = new Logger(CloudSecurityScannerService.name);
  private readonly scanner: SecurityScanner;

  constructor(private configService: ConfigService) {
    this.scanner = new SecurityScanner({
      projectId: this.configService.get('GCP_PROJECT_ID'),
    });
  }

  async scanWebApplication(targetUrl: string): Promise<any> {
    try {
      const [operation] = await this.scanner.createScanConfig({
        parent: `projects/${this.configService.get('GCP_PROJECT_ID')}`,
        scanConfig: {
          displayName: 'Automated Security Scan',
          startingUrls: [targetUrl],
          targetPlatforms: ['APP_ENGINE', 'COMPUTE'],
          userAgent: 'CHROME_LINUX',
          maxQps: 5,
          authentication: {
            customAccount: {
              username: this.configService.get('SCAN_USERNAME'),
              password: this.configService.get('SCAN_PASSWORD'),
            },
          },
        },
      });

      this.logger.log(`Security scan started: ${operation.name}`);
      return operation;
    } catch (error) {
      this.logger.error('Failed to start security scan:', error);
      throw error;
    }
  }

  async getScanResults(scanName: string): Promise<any> {
    try {
      const [findings] = await this.scanner.listFindings({
        parent: scanName,
      });

      return findings.map(finding => ({
        name: finding.name,
        severity: finding.severity,
        category: finding.findingClass,
        description: finding.description,
        url: finding.httpMethod,
        reproductionUrl: finding.reproductionUrl,
      }));
    } catch (error) {
      this.logger.error('Failed to get scan results:', error);
      throw error;
    }
  }
}

Security Regression Testing​

NestJS Security Test Suite​

// test/security/auth.security.spec.ts
import { Test, TestingModule } from '@nestjs/testing';
import { INestApplication } from '@nestjs/common';
import * as request from 'supertest';
import { AppModule } from '../../src/app.module';
import { JwtService } from '@nestjs/jwt';

describe('Authentication Security Tests', () => {
  let app: INestApplication;
  let jwtService: JwtService;

  beforeEach(async () => {
    const moduleFixture: TestingModule = await Test.createTestingModule({
      imports: [AppModule],
    }).compile();

    app = moduleFixture.createNestApplication();
    jwtService = moduleFixture.get<JwtService>(JwtService);
    await app.init();
  });

  describe('JWT Security', () => {
    it('should reject invalid JWT tokens', async () => {
      const response = await request(app.getHttpServer())
        .get('/protected-route')
        .set('Authorization', 'Bearer invalid-token')
        .expect(401);

      expect(response.body.message).toContain('Unauthorized');
    });

    it('should reject expired JWT tokens', async () => {
      const expiredToken = jwtService.sign(
        { sub: 'test-user' },
        { expiresIn: '-1h' }
      );

      await request(app.getHttpServer())
        .get('/protected-route')
        .set('Authorization', `Bearer ${expiredToken}`)
        .expect(401);
    });

    it('should reject tokens with invalid signature', async () => {
      const token = jwtService.sign({ sub: 'test-user' });
      const tamperedToken = token.slice(0, -5) + 'xxxxx';

      await request(app.getHttpServer())
        .get('/protected-route')
        .set('Authorization', `Bearer ${tamperedToken}`)
        .expect(401);
    });
  });

  describe('SQL Injection Protection', () => {
    it('should prevent SQL injection in login endpoint', async () => {
      const maliciousPayload = {
        username: "admin' OR '1'='1",
        password: "password' OR '1'='1",
      };

      await request(app.getHttpServer())
        .post('/auth/login')
        .send(maliciousPayload)
        .expect(401);
    });

    it('should sanitize query parameters', async () => {
      const maliciousQuery = "1'; DROP TABLE users; --";

      await request(app.getHttpServer())
        .get(`/users/${maliciousQuery}`)
        .expect(400);
    });
  });

  describe('XSS Protection', () => {
    it('should sanitize HTML input', async () => {
      const xssPayload = {
        name: '<script>alert("XSS")</script>',
        email: 'test@example.com',
      };

      const response = await request(app.getHttpServer())
        .post('/users')
        .send(xssPayload)
        .expect(400);

      expect(response.body.message).toContain('validation failed');
    });

    it('should set proper security headers', async () => {
      const response = await request(app.getHttpServer())
        .get('/')
        .expect(200);

      expect(response.headers['x-content-type-options']).toBe('nosniff');
      expect(response.headers['x-frame-options']).toBe('DENY');
      expect(response.headers['x-xss-protection']).toBe('1; mode=block');
    });
  });

  describe('CSRF Protection', () => {
    it('should require CSRF token for state-changing operations', async () => {
      await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Test User' })
        .expect(403);
    });

    it('should accept valid CSRF tokens', async () => {
      // First get CSRF token
      const tokenResponse = await request(app.getHttpServer())
        .get('/csrf-token')
        .expect(200);

      const csrfToken = tokenResponse.body.token;

      await request(app.getHttpServer())
        .post('/users')
        .set('X-CSRF-Token', csrfToken)
        .send({ name: 'Test User' })
        .expect(201);
    });
  });
});

End-to-End Security Tests with Cypress​

// cypress/e2e/security.cy.ts
describe('Application Security Tests', () => {
  beforeEach(() => {
    cy.visit('/');
  });

  describe('Authentication Flow Security', () => {
    it('should redirect to login when accessing protected routes', () => {
      cy.visit('/dashboard');
      cy.url().should('include', '/login');
    });

    it('should prevent brute force attacks', () => {
      const attemptLogin = () => {
        cy.get('[data-cy=username]').type('admin');
        cy.get('[data-cy=password]').type('wrongpassword');
        cy.get('[data-cy=login-button]').click();
      };

      // Attempt multiple failed logins
      for (let i = 0; i < 5; i++) {
        attemptLogin();
      }

      // Should be rate limited
      cy.get('[data-cy=error-message]')
        .should('contain', 'Too many attempts');
    });

    it('should secure session management', () => {
      cy.login('testuser@example.com', 'password123');
      
      // Check session cookie is HttpOnly and Secure
      cy.getCookie('session').should('have.property', 'httpOnly', true);
      cy.getCookie('session').should('have.property', 'secure', true);
      
      // Check automatic logout after inactivity
      cy.wait(300000); // 5 minutes
      cy.visit('/dashboard');
      cy.url().should('include', '/login');
    });
  });

  describe('Input Validation Security', () => {
    it('should prevent XSS in user inputs', () => {
      cy.login('testuser@example.com', 'password123');
      cy.visit('/profile');

      const xssPayload = '<script>alert("XSS")</script>';
      
      cy.get('[data-cy=name-input]').type(xssPayload);
      cy.get('[data-cy=save-button]').click();

      // Should show validation error, not execute script
      cy.get('[data-cy=error-message]')
        .should('contain', 'Invalid characters');
      
      // Verify script didn't execute
      cy.window().its('alert').should('not.exist');
    });

    it('should validate file uploads', () => {
      cy.login('testuser@example.com', 'password123');
      cy.visit('/profile');

      // Try uploading executable file
      const fileName = 'malicious.exe';
      cy.get('[data-cy=file-input]').selectFile({
        contents: 'MZ', // PE header
        fileName,
        mimeType: 'application/octet-stream',
      });

      cy.get('[data-cy=upload-button]').click();
      cy.get('[data-cy=error-message]')
        .should('contain', 'File type not allowed');
    });
  });

  describe('Authorization Security', () => {
    it('should enforce role-based access control', () => {
      cy.login('user@example.com', 'password123'); // Regular user
      
      // Try accessing admin endpoint
      cy.request({
        method: 'GET',
        url: '/api/admin/users',
        failOnStatusCode: false,
      }).then((response) => {
        expect(response.status).to.eq(403);
      });
    });

    it('should prevent privilege escalation', () => {
      cy.login('user@example.com', 'password123');
      
      // Try modifying user role
      cy.request({
        method: 'PATCH',
        url: '/api/users/profile',
        body: { role: 'admin' },
        failOnStatusCode: false,
      }).then((response) => {
        expect(response.status).to.eq(400);
      });
    });
  });
});

GitHub Actions Security Workflow​

# .github/workflows/security.yml
name: Security Testing

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 2 * * *' # Daily at 2 AM

jobs:
  security-scan:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run security linting
        run: npm run security:lint
      
      - name: Audit dependencies
        run: npm audit --audit-level=moderate
      
      - name: Run Snyk security scan
        uses: snyk/actions/node@master
        env:
          SNYK_TOKEN: ${{ secrets.SNYK_TOKEN }}
        with:
          args: --severity-threshold=high
      
      - name: Run security tests
        run: npm run test:security
      
      - name: OWASP ZAP Baseline Scan
        uses: zaproxy/action-baseline@v0.7.0
        with:
          target: 'https://staging.example.com'
          rules_file_name: '.zap/rules.tsv'
          cmd_options: '-a'
      
      - name: Upload security reports
        uses: actions/upload-artifact@v3
        if: always()
        with:
          name: security-reports
          path: |
            zap-report.json
            snyk-report.json
            security-test-results.xml

Security Testing Utilities​

// src/testing/security-utils.ts
import * as crypto from 'crypto';
import { ValidationPipe } from '@nestjs/common';

export class SecurityTestUtils {
  static generateSqlInjectionPayloads(): string[] {
    return [
      "' OR '1'='1",
      "'; DROP TABLE users; --",
      "' UNION SELECT * FROM users --",
      "1'; EXEC sp_configure 'show advanced options', 1--",
      "' OR 1=1#",
      "admin'--",
      "admin' /*",
      "' OR 'x'='x",
    ];
  }

  static generateXssPayloads(): string[] {
    return [
      '<script>alert("XSS")</script>',
      '<img src=x onerror=alert("XSS")>',
      '<svg onload=alert("XSS")>',
      'javascript:alert("XSS")',
      '<iframe src="javascript:alert(`XSS`)">',
      '<input onfocus=alert("XSS") autofocus>',
      '<select onfocus=alert("XSS") autofocus>',
      '<textarea onfocus=alert("XSS") autofocus>',
    ];
  }

  static generateCommandInjectionPayloads(): string[] {
    return [
      '; ls -la',
      '| cat /etc/passwd',
      '&& whoami',
      '`whoami`',
      '$(whoami)',
      '; ping -c 1 google.com',
      '| nc -l 4444',
    ];
  }

  static validateSecurityHeaders(headers: Record<string, string>): boolean {
    const requiredHeaders = {
      'x-content-type-options': 'nosniff',
      'x-frame-options': ['DENY', 'SAMEORIGIN'],
      'x-xss-protection': '1; mode=block',
      'strict-transport-security': 'max-age=',
      'content-security-policy': 'default-src',
    };

    for (const [header, expectedValue] of Object.entries(requiredHeaders)) {
      const headerValue = headers[header.toLowerCase()];
      
      if (!headerValue) {
        console.error(`Missing security header: ${header}`);
        return false;
      }

      if (Array.isArray(expectedValue)) {
        if (!expectedValue.some(val => headerValue.includes(val))) {
          console.error(`Invalid ${header} header value: ${headerValue}`);
          return false;
        }
      } else if (!headerValue.includes(expectedValue)) {
        console.error(`Invalid ${header} header value: ${headerValue}`);
        return false;
      }
    }

    return true;
  }

  static createSecureValidationPipe(): ValidationPipe {
    return new ValidationPipe({
      whitelist: true,
      forbidNonWhitelisted: true,
      transform: true,
      disableErrorMessages: process.env.NODE_ENV === 'production',
      validateCustomDecorators: true,
    });
  }
}

Dockerfile Security Best Practices​

# Use official, minimal base image
FROM node:18-alpine AS base

# Create non-root user
RUN addgroup -g 1001 -S nodejs
RUN adduser -S nextjs -u 1001

# Set working directory
WORKDIR /app

# Install dependencies
COPY package*.json ./
RUN npm ci --only=production && npm cache clean --force

# Copy source code
COPY --chown=nextjs:nodejs . .

# Build application
FROM base AS build
RUN npm run build

# Production image
FROM node:18-alpine AS production

# Install security updates
RUN apk upgrade --no-cache

# Create app directory
WORKDIR /app

# Copy built application
COPY --from=build --chown=nextjs:nodejs /app/dist ./dist
COPY --from=build --chown=nextjs:nodejs /app/node_modules ./node_modules
COPY --from=build --chown=nextjs:nodejs /app/package.json ./package.json

# Use non-root user
USER nextjs

# Expose port
EXPOSE 3000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:3000/health || exit 1

# Start application
CMD ["node", "dist/main.js"]

Container Security Scanning Script​

#!/bin/bash
# scripts/container-security-scan.sh

set -e

IMAGE_NAME="$1"
if [ -z "$IMAGE_NAME" ]; then
    echo "Usage: $0 <image-name>"
    exit 1
fi

echo "Running container security scans for: $IMAGE_NAME"

# Trivy vulnerability scan
echo "Running Trivy vulnerability scan..."
trivy image --exit-code 1 --severity HIGH,CRITICAL "$IMAGE_NAME"

# Docker Bench Security
echo "Running Docker Bench Security..."
docker run --rm --net host --pid host --userns host --cap-add audit_control \
    -e DOCKER_CONTENT_TRUST=$DOCKER_CONTENT_TRUST \
    -v /etc:/etc:ro \
    -v /usr/bin/containerd:/usr/bin/containerd:ro \
    -v /usr/bin/runc:/usr/bin/runc:ro \
    -v /usr/lib/systemd:/usr/lib/systemd:ro \
    -v /var/lib:/var/lib:ro \
    -v /var/run/docker.sock:/var/run/docker.sock:ro \
    --label docker_bench_security \
    docker/docker-bench-security

# Hadolint for Dockerfile linting
echo "Running Hadolint..."
hadolint Dockerfile

# Dockle for container image linting
echo "Running Dockle..."
dockle --exit-code 1 "$IMAGE_NAME"

echo "Container security scan completed!"

Security Monitoring Dashboard​

// src/monitoring/security-metrics.service.ts
import { Injectable } from '@nestjs/common';
import { PrometheusService } from './prometheus.service';

@Injectable()
export class SecurityMetricsService {
  constructor(private prometheus: PrometheusService) {}

  recordFailedLogin(username: string, ip: string): void {
    this.prometheus.increment('failed_login_attempts_total', {
      username,
      ip,
    });
  }

  recordSecurityScanResult(severity: string, count: number): void {
    this.prometheus.set('security_vulnerabilities_total', count, {
      severity,
    });
  }

  recordSuspiciousActivity(type: string, ip: string): void {
    this.prometheus.increment('suspicious_activity_total', {
      type,
      ip,
    });
  }

  getSecurityMetrics(): any {
    return {
      failedLogins: this.prometheus.get('failed_login_attempts_total'),
      vulnerabilities: this.prometheus.get('security_vulnerabilities_total'),
      suspiciousActivity: this.prometheus.get('suspicious_activity_total'),
    };
  }
}
