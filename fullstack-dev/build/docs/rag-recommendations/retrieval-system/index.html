Retrieval System Implementation

This section focuses on building a robust retrieval system that efficiently finds relevant products based on user queries, preferences, and context. The retrieval system is the core component that bridges user intent with relevant products using vector similarity search.

Retrieval Architecture​

Our retrieval system uses a multi-stage approach to ensure both relevance and performance:

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│   User Query    │    │   Query         │    │   Retrieval     │
│   + Context     │───▶│   Processing    │───▶│   Engine        │
└─────────────────┘    └─────────────────┘    └─────────────────┘
│                      │                      │                 │
├─ Search Query        ├─ Query Expansion     ├─ Vector Search   │
├─ User Profile        ├─ Intent Detection    ├─ Hybrid Filtering│
├─ Session Context     ├─ Embedding Gen.     ├─ Re-ranking      │
├─ Behavioral Data     ├─ Filter Extraction  ├─ Diversity       │
└─ Product Context     └─ Query Enhancement  └─ Post-processing │

# retrieval/vector_db.py
from qdrant_client import QdrantClient
from qdrant_client.models import Distance, VectorParams, PointStruct
from qdrant_client.models import Filter, FieldCondition, Range, MatchValue
import numpy as np
from typing import List, Dict, Any, Optional
import logging

logger = logging.getLogger(__name__)

class QdrantVectorDB:
    """Qdrant vector database implementation for product retrieval"""
    
    def __init__(self, host: str = "localhost", port: int = 6333):
        self.client = QdrantClient(host=host, port=port)
        self.collections = {}
        
    async def setup_collection(self, 
                              collection_name: str, 
                              vector_size: int, 
                              distance: Distance = Distance.COSINE) -> bool:
        """Setup a new collection for vector storage"""
        try:
            # Check if collection exists
            collections = self.client.get_collections()
            existing_names = [c.name for c in collections.collections]
            
            if collection_name not in existing_names:
                self.client.create_collection(
                    collection_name=collection_name,
                    vectors_config=VectorParams(size=vector_size, distance=distance)
                )
                logger.info(f"Created collection: {collection_name}")
            else:
                logger.info(f"Collection already exists: {collection_name}")
            
            self.collections[collection_name] = {
                'vector_size': vector_size,
                'distance': distance
            }
            
            return True
            
        except Exception as e:
            logger.error(f"Failed to setup collection {collection_name}: {e}")
            return False
    
    async def upsert_vectors(self, 
                            collection_name: str, 
                            vectors_data: List[Dict[str, Any]]) -> bool:
        """Insert or update vectors in the collection"""
        try:
            points = []
            
            for data in vectors_data:
                point = PointStruct(
                    id=data['id'],
                    vector=data['vector'].tolist() if isinstance(data['vector'], np.ndarray) else data['vector'],
                    payload=data['payload']
                )
                points.append(point)
            
            # Batch upsert
            batch_size = 100
            for i in range(0, len(points), batch_size):
                batch = points[i:i + batch_size]
                self.client.upsert(
                    collection_name=collection_name,
                    points=batch
                )
            
            logger.info(f"Upserted {len(points)} vectors to {collection_name}")
            return True
            
        except Exception as e:
            logger.error(f"Failed to upsert vectors: {e}")
            return False
    
    async def search_vectors(self, 
                            collection_name: str,
                            query_vector: np.ndarray,
                            limit: int = 10,
                            filters: Optional[Filter] = None) -> List[Dict[str, Any]]:
        """Search for similar vectors"""
        try:
            results = self.client.search(
                collection_name=collection_name,
                query_vector=query_vector.tolist() if isinstance(query_vector, np.ndarray) else query_vector,
                query_filter=filters,
                limit=limit,
                with_payload=True,
                with_vectors=False
            )
            
            search_results = []
            for result in results:
                search_results.append({
                    'id': result.id,
                    'score': result.score,
                    'payload': result.payload
                })
            
            return search_results
            
        except Exception as e:
            logger.error(f"Search failed: {e}")
            return []
    
    def create_filter(self, filter_conditions: Dict[str, Any]) -> Filter:
        """Create Qdrant filter from conditions"""
        conditions = []
        
        for field, value in filter_conditions.items():
            if isinstance(value, list):
                # Multiple values - use should condition
                should_conditions = [
                    FieldCondition(key=field, match=MatchValue(value=v))
                    for v in value
                ]
                conditions.extend(should_conditions)
            elif isinstance(value, dict) and 'range' in value:
                # Range condition
                range_filter = value['range']
                conditions.append(
                    FieldCondition(
                        key=field,
                        range=Range(
                            gte=range_filter.get('gte'),
                            lte=range_filter.get('lte')
                        )
                    )
                )
            else:
                # Exact match
                conditions.append(
                    FieldCondition(key=field, match=MatchValue(value=value))
                )
        
        return Filter(must=conditions) if conditions else None

# retrieval/query_processor.py
import re
import spacy
from typing import List, Dict, Any, Optional, Set
import logging
from datetime import datetime

logger = logging.getLogger(__name__)

class QueryProcessor:
    """Advanced query processing for e-commerce search"""
    
    def __init__(self):
        # Load spaCy model for NLP processing
        try:
            self.nlp = spacy.load("en_core_web_sm")
        except OSError:
            logger.warning("spaCy model not found. Install with: python -m spacy download en_core_web_sm")
            self.nlp = None
        
        # Product categories and attributes
        self.categories = {
            'electronics': ['phone', 'laptop', 'computer', 'tablet', 'camera', 'headphones'],
            'clothing': ['shirt', 'pants', 'dress', 'shoes', 'jacket', 'jeans'],
            'home': ['furniture', 'decor', 'kitchen', 'bedroom', 'bathroom'],
            'books': ['novel', 'textbook', 'fiction', 'non-fiction', 'cookbook'],
            'sports': ['fitness', 'outdoor', 'gym', 'running', 'cycling']
        }
        
        # Price keywords
        self.price_keywords = {
            'cheap': {'range': {'lte': 50}},
            'affordable': {'range': {'lte': 100}},
            'budget': {'range': {'lte': 75}},
            'expensive': {'range': {'gte': 500}},
            'premium': {'range': {'gte': 300}},
            'luxury': {'range': {'gte': 1000}}
        }
        
        # Brand extraction patterns
        self.brand_patterns = [
            r'\b(apple|samsung|nike|adidas|sony|microsoft|google)\b',
            r'\b(amazon|dell|hp|lenovo|asus|lg)\b'
        ]
    
    def process_query(self, query: str, user_context: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
        """Process user query and extract structured information"""
        logger.info(f"Processing query: {query}")
        
        # Basic preprocessing
        query_lower = query.lower().strip()
        
        # Extract components
        processed_query = {
            'original_query': query,
            'cleaned_query': self._clean_query(query_lower),
            'intent': self._detect_intent(query_lower),
            'entities': self._extract_entities(query_lower),
            'filters': self._extract_filters(query_lower),
            'keywords': self._extract_keywords(query_lower),
            'embedding_text': self._create_embedding_text(query_lower, user_context),
            'query_expansion': self._expand_query(query_lower),
            'priority_terms': self._identify_priority_terms(query_lower)
        }
        
        logger.info(f"Processed query structure: {processed_query}")
        return processed_query
    
    def _clean_query(self, query: str) -> str:
        """Clean and normalize the query"""
        # Remove special characters but keep important ones
        query = re.sub(r'[^\w\s\-\$\.]', ' ', query)
        
        # Normalize whitespace
        query = re.sub(r'\s+', ' ', query)
        
        # Remove common stop words that don't add value
        stop_words = {'a', 'an', 'the', 'is', 'are', 'was', 'were', 'for', 'with'}
        words = query.split()
        words = [word for word in words if word not in stop_words]
        
        return ' '.join(words)
    
    def _detect_intent(self, query: str) -> str:
        """Detect user intent from query"""
        # Define intent patterns
        intent_patterns = {
            'specific_product': [r'\b(buy|purchase|order)\b', r'\b(specific|exact)\b'],
            'browse_category': [r'\b(show|browse|look|find)\b.*\b(category|type)\b'],
            'compare_products': [r'\b(compare|vs|versus|difference)\b'],
            'price_search': [r'\b(cheap|expensive|price|cost|budget)\b'],
            'brand_search': [r'\b(brand|manufacturer|make)\b'],
            'feature_search': [r'\b(with|having|feature|specification)\b'],
            'general_search': [r'.*']  # Default fallback
        }
        
        for intent, patterns in intent_patterns.items():
            if any(re.search(pattern, query) for pattern in patterns):
                return intent
        
        return 'general_search'
    
    def _extract_entities(self, query: str) -> Dict[str, List[str]]:
        """Extract named entities from the query"""
        entities = {
            'products': [],
            'brands': [],
            'categories': [],
            'attributes': [],
            'price_terms': []
        }
        
        # Extract brands
        for pattern in self.brand_patterns:
            matches = re.findall(pattern, query, re.IGNORECASE)
            entities['brands'].extend(matches)
        
        # Extract categories
        for category, keywords in self.categories.items():
            if any(keyword in query for keyword in keywords):
                entities['categories'].append(category)
                entities['products'].extend([kw for kw in keywords if kw in query])
        
        # Extract price terms
        for price_term in self.price_keywords.keys():
            if price_term in query:
                entities['price_terms'].append(price_term)
        
        # Use spaCy for additional entity extraction if available
        if self.nlp:
            doc = self.nlp(query)
            for ent in doc.ents:
                if ent.label_ in ['PRODUCT', 'ORG']:
                    entities['brands'].append(ent.text)
        
        return entities
    
    def _extract_filters(self, query: str) -> Dict[str, Any]:
        """Extract filter conditions from query"""
        filters = {}
        
        # Price filters
        for price_term, condition in self.price_keywords.items():
            if price_term in query:
                filters['price'] = condition
                break
        
        # Extract specific price mentions
        price_match = re.search(r'\$?(\d+(?:\.\d{2})?)', query)
        if price_match:
            price = float(price_match.group(1))
            if 'under' in query or 'below' in query:
                filters['price'] = {'range': {'lte': price}}
            elif 'over' in query or 'above' in query:
                filters['price'] = {'range': {'gte': price}}
            else:
                # Approximate range around the mentioned price
                filters['price'] = {'range': {'gte': price * 0.8, 'lte': price * 1.2}}
        
        # Category filters
        entities = self._extract_entities(query)
        if entities['categories']:
            filters['category'] = entities['categories']
        
        # Brand filters
        if entities['brands']:
            filters['brand'] = entities['brands']
        
        # Availability filters
        if any(term in query for term in ['available', 'in stock', 'ready']):
            filters['is_available'] = True
        
        # Rating filters
        if 'high rated' in query or 'top rated' in query:
            filters['rating'] = {'range': {'gte': 4.0}}
        elif 'highly rated' in query:
            filters['rating'] = {'range': {'gte': 4.5}}
        
        return filters
    
    def _extract_keywords(self, query: str) -> List[str]:
        """Extract important keywords for search"""
        # Basic keyword extraction
        words = query.split()
        
        # Remove common words
        stop_words = {'for', 'with', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'from'}
        keywords = [word for word in words if word not in stop_words and len(word) > 2]
        
        # Add entity-based keywords
        entities = self._extract_entities(query)
        keywords.extend(entities['products'])
        keywords.extend(entities['brands'])
        
        return list(set(keywords))  # Remove duplicates
    
    def _create_embedding_text(self, query: str, user_context: Optional[Dict[str, Any]] = None) -> str:
        """Create optimized text for embedding generation"""
        text_components = [query]
        
        # Add user context if available
        if user_context:
            if user_context.get('preferred_categories'):
                text_components.append(f"Categories: {', '.join(user_context['preferred_categories'])}")
            
            if user_context.get('preferred_brands'):
                text_components.append(f"Brands: {', '.join(user_context['preferred_brands'])}")
            
            if user_context.get('recent_searches'):
                recent = user_context['recent_searches'][:3]  # Last 3 searches
                text_components.append(f"Recent interests: {', '.join(recent)}")
        
        return ' '.join(text_components)
    
    def _expand_query(self, query: str) -> List[str]:
        """Expand query with synonyms and related terms"""
        # Simple synonym expansion
        synonyms = {
            'phone': ['smartphone', 'mobile', 'cellphone'],
            'laptop': ['notebook', 'computer', 'pc'],
            'cheap': ['affordable', 'budget', 'inexpensive'],
            'expensive': ['premium', 'luxury', 'high-end'],
            'good': ['quality', 'excellent', 'top', 'best']
        }
        
        expanded_terms = [query]
        words = query.split()
        
        for word in words:
            if word in synonyms:
                expanded_terms.extend(synonyms[word])
        
        return expanded_terms
    
    def _identify_priority_terms(self, query: str) -> List[str]:
        """Identify high-priority terms that should have higher weight"""
        priority_patterns = [
            r'\b(best|top|premium|quality)\b',
            r'\b(need|want|looking for)\b',
            r'\b(brand new|latest|newest)\b'
        ]
        
        priority_terms = []
        for pattern in priority_patterns:
            matches = re.findall(pattern, query)
            priority_terms.extend(matches)
        
        return priority_terms

Retrieval Engine​

# retrieval/retrieval_engine.py
import numpy as np
from typing import List, Dict, Any, Optional
import logging
from dataclasses import dataclass
import asyncio

logger = logging.getLogger(__name__)

@dataclass
class RetrievalResult:
    """Structure for retrieval results"""
    product_id: str
    score: float
    payload: Dict[str, Any]
    retrieval_method: str
    ranking_factors: Dict[str, float]

class RetrievalEngine:
    """Advanced retrieval engine with multiple search strategies"""
    
    def __init__(self, 
                 vector_db: QdrantVectorDB,
                 embedding_generator,
                 query_processor: QueryProcessor):
        self.vector_db = vector_db
        self.embedding_generator = embedding_generator
        self.query_processor = query_processor
        
        # Retrieval weights
        self.retrieval_weights = {
            'semantic_similarity': 0.6,
            'category_match': 0.2,
            'brand_match': 0.1,
            'price_relevance': 0.05,
            'popularity': 0.05
        }
    
    async def search(self, 
                    query: str,
                    user_context: Optional[Dict[str, Any]] = None,
                    limit: int = 20,
                    include_filters: bool = True) -> List[RetrievalResult]:
        """Main search interface"""
        logger.info(f"Searching for: {query}")
        
        # Process query
        processed_query = self.query_processor.process_query(query, user_context)
        
        # Multiple retrieval strategies
        results = await self._multi_stage_retrieval(processed_query, user_context, limit)
        
        # Re-rank results
        ranked_results = self._rerank_results(results, processed_query, user_context)
        
        # Apply diversity
        diverse_results = self._apply_diversity(ranked_results[:limit * 2])
        
        return diverse_results[:limit]
    
    async def _multi_stage_retrieval(self, 
                                   processed_query: Dict[str, Any],
                                   user_context: Optional[Dict[str, Any]],
                                   limit: int) -> List[RetrievalResult]:
        """Multi-stage retrieval with different strategies"""
        all_results = []
        
        # Stage 1: Semantic vector search
        semantic_results = await self._semantic_search(processed_query, limit)
        all_results.extend(semantic_results)
        
        # Stage 2: Filtered search (if filters exist)
        if processed_query['filters']:
            filtered_results = await self._filtered_search(processed_query, limit)
            all_results.extend(filtered_results)
        
        # Stage 3: User preference-based search
        if user_context:
            preference_results = await self._preference_search(processed_query, user_context, limit)
            all_results.extend(preference_results)
        
        # Deduplicate results
        seen_ids = set()
        unique_results = []
        for result in all_results:
            if result.product_id not in seen_ids:
                unique_results.append(result)
                seen_ids.add(result.product_id)
        
        return unique_results
    
    async def _semantic_search(self, processed_query: Dict[str, Any], limit: int) -> List[RetrievalResult]:
        """Semantic similarity search using vector embeddings"""
        try:
            # Generate query embedding
            query_text = processed_query['embedding_text']
            query_embedding = await self.embedding_generator.generate_embeddings([query_text])
            
            if len(query_embedding) == 0:
                return []
            
            # Search in vector database
            search_results = await self.vector_db.search_vectors(
                collection_name='products',
                query_vector=query_embedding[0],
                limit=limit * 2  # Get more results for re-ranking
            )
            
            # Convert to RetrievalResult objects
            results = []
            for result in search_results:
                retrieval_result = RetrievalResult(
                    product_id=result['id'],
                    score=result['score'],
                    payload=result['payload'],
                    retrieval_method='semantic',
                    ranking_factors={'semantic_similarity': result['score']}
                )
                results.append(retrieval_result)
            
            logger.info(f"Semantic search returned {len(results)} results")
            return results
            
        except Exception as e:
            logger.error(f"Semantic search failed: {e}")
            return []
    
    async def _filtered_search(self, processed_query: Dict[str, Any], limit: int) -> List[RetrievalResult]:
        """Search with specific filters"""
        try:
            filters = self.vector_db.create_filter(processed_query['filters'])
            
            # Generate query embedding
            query_text = processed_query['embedding_text']
            query_embedding = await self.embedding_generator.generate_embeddings([query_text])
            
            if len(query_embedding) == 0:
                return []
            
            # Search with filters
            search_results = await self.vector_db.search_vectors(
                collection_name='products',
                query_vector=query_embedding[0],
                limit=limit,
                filters=filters
            )
            
            # Convert to RetrievalResult objects
            results = []
            for result in search_results:
                ranking_factors = {
                    'semantic_similarity': result['score'],
                    'filter_match': 1.0  # Perfect filter match
                }
                
                retrieval_result = RetrievalResult(
                    product_id=result['id'],
                    score=result['score'],
                    payload=result['payload'],
                    retrieval_method='filtered',
                    ranking_factors=ranking_factors
                )
                results.append(retrieval_result)
            
            logger.info(f"Filtered search returned {len(results)} results")
            return results
            
        except Exception as e:
            logger.error(f"Filtered search failed: {e}")
            return []
    
    async def _preference_search(self, 
                               processed_query: Dict[str, Any],
                               user_context: Dict[str, Any],
                               limit: int) -> List[RetrievalResult]:
        """Search based on user preferences"""
        try:
            # Create preference-enhanced query
            preference_filters = {}
            
            if user_context.get('preferred_categories'):
                preference_filters['category'] = user_context['preferred_categories']
            
            if user_context.get('preferred_brands'):
                preference_filters['brand'] = user_context['preferred_brands']
            
            if user_context.get('price_range'):
                preference_filters['price'] = user_context['price_range']
            
            if not preference_filters:
                return []
            
            filters = self.vector_db.create_filter(preference_filters)
            
            # Generate query embedding
            query_text = processed_query['embedding_text']
            query_embedding = await self.embedding_generator.generate_embeddings([query_text])
            
            if len(query_embedding) == 0:
                return []
            
            # Search with preference filters
            search_results = await self.vector_db.search_vectors(
                collection_name='products',
                query_vector=query_embedding[0],
                limit=limit,
                filters=filters
            )
            
            # Convert to RetrievalResult objects
            results = []
            for result in search_results:
                ranking_factors = {
                    'semantic_similarity': result['score'],
                    'preference_match': 0.8  # High preference match
                }
                
                retrieval_result = RetrievalResult(
                    product_id=result['id'],
                    score=result['score'],
                    payload=result['payload'],
                    retrieval_method='preference',
                    ranking_factors=ranking_factors
                )
                results.append(retrieval_result)
            
            logger.info(f"Preference search returned {len(results)} results")
            return results
            
        except Exception as e:
            logger.error(f"Preference search failed: {e}")
            return []
    
    def _rerank_results(self, 
                       results: List[RetrievalResult],
                       processed_query: Dict[str, Any],
                       user_context: Optional[Dict[str, Any]]) -> List[RetrievalResult]:
        """Re-rank results using multiple factors"""
        logger.info(f"Re-ranking {len(results)} results")
        
        for result in results:
            # Calculate composite score
            composite_score = 0.0
            
            # Semantic similarity
            semantic_score = result.ranking_factors.get('semantic_similarity', 0.0)
            composite_score += semantic_score * self.retrieval_weights['semantic_similarity']
            
            # Category match
            category_score = self._calculate_category_score(result, processed_query)
            composite_score += category_score * self.retrieval_weights['category_match']
            result.ranking_factors['category_match'] = category_score
            
            # Brand match
            brand_score = self._calculate_brand_score(result, processed_query)
            composite_score += brand_score * self.retrieval_weights['brand_match']
            result.ranking_factors['brand_match'] = brand_score
            
            # Price relevance
            price_score = self._calculate_price_score(result, processed_query)
            composite_score += price_score * self.retrieval_weights['price_relevance']
            result.ranking_factors['price_relevance'] = price_score
            
            # Popularity score
            popularity_score = self._calculate_popularity_score(result)
            composite_score += popularity_score * self.retrieval_weights['popularity']
            result.ranking_factors['popularity'] = popularity_score
            
            # User context boost
            if user_context:
                context_score = self._calculate_context_score(result, user_context)
                composite_score += context_score * 0.1
                result.ranking_factors['context_match'] = context_score
            
            result.score = composite_score
        
        # Sort by composite score
        ranked_results = sorted(results, key=lambda x: x.score, reverse=True)
        
        logger.info("Re-ranking completed")
        return ranked_results
    
    def _calculate_category_score(self, result: RetrievalResult, processed_query: Dict[str, Any]) -> float:
        """Calculate category match score"""
        query_categories = processed_query['entities'].get('categories', [])
        if not query_categories:
            return 0.0
        
        product_category = result.payload.get('category', '')
        if product_category in query_categories:
            return 1.0
        
        # Partial match for subcategories
        product_subcategory = result.payload.get('subcategory', '')
        if any(cat in product_subcategory.lower() for cat in query_categories):
            return 0.5
        
        return 0.0
    
    def _calculate_brand_score(self, result: RetrievalResult, processed_query: Dict[str, Any]) -> float:
        """Calculate brand match score"""
        query_brands = processed_query['entities'].get('brands', [])
        if not query_brands:
            return 0.0
        
        product_brand = result.payload.get('brand', '').lower()
        for brand in query_brands:
            if brand.lower() in product_brand:
                return 1.0
        
        return 0.0
    
    def _calculate_price_score(self, result: RetrievalResult, processed_query: Dict[str, Any]) -> float:
        """Calculate price relevance score"""
        price_filters = processed_query['filters'].get('price')
        if not price_filters:
            return 0.5  # Neutral score
        
        product_price = result.payload.get('price', 0)
        
        if 'range' in price_filters:
            price_range = price_filters['range']
            min_price = price_range.get('gte', 0)
            max_price = price_range.get('lte', float('inf'))
            
            if min_price <= product_price <= max_price:
                return 1.0
            else:
                # Calculate how far outside the range
                if product_price < min_price:
                    ratio = product_price / min_price
                elif product_price > max_price:
                    ratio = max_price / product_price
                else:
                    ratio = 1.0
                
                return max(0.0, ratio)
        
        return 0.5
    
    def _calculate_popularity_score(self, result: RetrievalResult) -> float:
        """Calculate product popularity score"""
        # Normalize popularity metrics
        review_count = result.payload.get('review_count', 0)
        rating = result.payload.get('rating', 0.0)
        popularity = result.payload.get('popularity_score', 0.0)
        
        # Combine metrics
        review_score = min(1.0, review_count / 100)  # Normalize to 0-1
        rating_score = rating / 5.0 if rating > 0 else 0
        popularity_score = min(1.0, popularity / 100)  # Normalize to 0-1
        
        return (review_score + rating_score + popularity_score) / 3
    
    def _calculate_context_score(self, result: RetrievalResult, user_context: Dict[str, Any]) -> float:
        """Calculate user context match score"""
        score = 0.0
        
        # Preferred categories
        preferred_categories = user_context.get('preferred_categories', [])
        product_category = result.payload.get('category', '')
        if product_category in preferred_categories:
            score += 0.5
        
        # Preferred brands
        preferred_brands = user_context.get('preferred_brands', [])
        product_brand = result.payload.get('brand', '')
        if product_brand in preferred_brands:
            score += 0.3
        
        # Price range preference
        preferred_price_range = user_context.get('price_range')
        if preferred_price_range:
            product_price = result.payload.get('price', 0)
            min_price = preferred_price_range.get('min', 0)
            max_price = preferred_price_range.get('max', float('inf'))
            
            if min_price <= product_price <= max_price:
                score += 0.2
        
        return min(1.0, score)
    
    def _apply_diversity(self, results: List[RetrievalResult]) -> List[RetrievalResult]:
        """Apply diversity to results to avoid showing too similar products"""
        if len(results) <= 5:
            return results
        
        diverse_results = [results[0]]  # Always include top result
        
        for result in results[1:]:
            # Check if result is diverse enough
            is_diverse = True
            
            for existing_result in diverse_results:
                # Check category diversity
                if (result.payload.get('category') == existing_result.payload.get('category') and
                    result.payload.get('subcategory') == existing_result.payload.get('subcategory')):
                    
                    # Check if from same brand
                    if result.payload.get('brand') == existing_result.payload.get('brand'):
                        is_diverse = False
                        break
            
            if is_diverse:
                diverse_results.append(result)
            
            # Limit diversity checking to avoid performance issues
            if len(diverse_results) >= 10:
                break
        
        # Fill remaining slots with best remaining results
        remaining_slots = len(results) - len(diverse_results)
        if remaining_slots > 0:
            remaining_results = [r for r in results if r not in diverse_results]
            diverse_results.extend(remaining_results[:remaining_slots])
        
        logger.info(f"Applied diversity: {len(diverse_results)} results")
        return diverse_results

# retrieval/cache_manager.py
import redis
import json
import hashlib
from typing import Dict, Any, Optional, List
import logging
import pickle

logger = logging.getLogger(__name__)

class RetrievalCache:
    """Cache for retrieval results to improve performance"""
    
    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.cache_ttl = 3600  # 1 hour
        self.query_cache_ttl = 1800  # 30 minutes
    
    def _create_cache_key(self, query: str, filters: Dict[str, Any], user_id: Optional[str] = None) -> str:
        """Create a unique cache key for the query"""
        key_data = {
            'query': query,
            'filters': filters,
            'user_id': user_id
        }
        
        key_string = json.dumps(key_data, sort_keys=True)
        key_hash = hashlib.md5(key_string.encode()).hexdigest()
        
        return f"retrieval_cache:{key_hash}"
    
    async def get_cached_results(self, 
                               query: str, 
                               filters: Dict[str, Any],
                               user_id: Optional[str] = None) -> Optional[List[RetrievalResult]]:
        """Get cached retrieval results"""
        try:
            cache_key = self._create_cache_key(query, filters, user_id)
            cached_data = self.redis.get(cache_key)
            
            if cached_data:
                results_data = pickle.loads(cached_data)
                logger.info(f"Cache hit for query: {query}")
                return results_data
            
            return None
            
        except Exception as e:
            logger.error(f"Cache retrieval error: {e}")
            return None
    
    async def cache_results(self, 
                          query: str,
                          filters: Dict[str, Any],
                          results: List[RetrievalResult],
                          user_id: Optional[str] = None):
        """Cache retrieval results"""
        try:
            cache_key = self._create_cache_key(query, filters, user_id)
            results_data = pickle.dumps(results)
            
            self.redis.setex(cache_key, self.cache_ttl, results_data)
            logger.info(f"Cached results for query: {query}")
            
        except Exception as e:
            logger.error(f"Cache storage error: {e}")

async def search_products(
    request: SearchRequest,

        # Perform search

            query=request.query,
            results=formatted_results,
            total_results=len(formatted_results),
            search_time=search_time,
            filters_applied=request.filters or {}
