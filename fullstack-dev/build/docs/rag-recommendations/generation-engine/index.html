Generation and Personalization Engine

This section covers the LLM-powered generation component that creates personalized product recommendations, explanations, and content. The generation engine transforms retrieved products into compelling, contextual recommendations tailored to each user.

Generation Architecture​

The generation system combines retrieval results with user context to create personalized content:

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│   Retrieved     │    │   Generation    │    │   Personalized  │
│   Products      │───▶│   Engine        │───▶│   Content       │
└─────────────────┘    └─────────────────┘    └─────────────────┘
│                      │                      │                 │
├─ Product Data        ├─ Context Fusion     ├─ Recommendations │
├─ User Context        ├─ LLM Generation     ├─ Explanations    │
├─ Search Query        ├─ Template Engine    ├─ Product Summaries│
├─ Session History     ├─ Content Filtering  ├─ Comparison Text │
└─ Behavioral Data     └─ Quality Control    └─ Purchase Advice │

# generation/llm_client.py
import openai
import anthropic
from typing import Dict, Any, List, Optional, Union
import logging
import asyncio
from abc import ABC, abstractmethod
from dataclasses import dataclass

logger = logging.getLogger(__name__)

@dataclass
class GenerationRequest:
    """Request structure for content generation"""
    prompt: str
    max_tokens: int = 1000
    temperature: float = 0.7
    user_context: Optional[Dict[str, Any]] = None
    format_type: str = "text"  # text, json, markdown

@dataclass
class GenerationResponse:
    """Response structure for generated content"""
    content: str
    model_used: str
    tokens_used: int
    generation_time: float
    quality_score: Optional[float] = None

class BaseLLMProvider(ABC):
    """Abstract base class for LLM providers"""
    
    @abstractmethod
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        pass
    
    @abstractmethod
    def is_available(self) -> bool:
        pass

class OpenAIProvider(BaseLLMProvider):
    """OpenAI GPT provider"""
    
    def __init__(self, api_key: str, model: str = "gpt-4"):
        self.client = openai.AsyncOpenAI(api_key=api_key)
        self.model = model
        self.available = True
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate content using OpenAI API"""
        try:
            import time
            start_time = time.time()
            
            messages = [
                {"role": "system", "content": self._create_system_prompt(request)},
                {"role": "user", "content": request.prompt}
            ]
            
            response = await self.client.chat.completions.create(
                model=self.model,
                messages=messages,
                max_tokens=request.max_tokens,
                temperature=request.temperature
            )
            
            generation_time = time.time() - start_time
            
            return GenerationResponse(
                content=response.choices[0].message.content,
                model_used=self.model,
                tokens_used=response.usage.total_tokens,
                generation_time=generation_time
            )
            
        except Exception as e:
            logger.error(f"OpenAI generation failed: {e}")
            self.available = False
            raise
    
    def is_available(self) -> bool:
        return self.available
    
    def _create_system_prompt(self, request: GenerationRequest) -> str:
        """Create system prompt based on context"""
        base_prompt = """You are an expert e-commerce recommendation assistant. Your task is to create personalized, helpful product recommendations and explanations."""
        
        if request.user_context:
            context_info = []
            if request.user_context.get('preferred_categories'):
                context_info.append(f"User prefers: {', '.join(request.user_context['preferred_categories'])}")
            if request.user_context.get('budget_range'):
                context_info.append(f"Budget: ${request.user_context['budget_range']}")
            
            if context_info:
                base_prompt += f"\n\nUser Context: {'; '.join(context_info)}"
        
        if request.format_type == "json":
            base_prompt += "\n\nPlease respond in valid JSON format."
        elif request.format_type == "markdown":
            base_prompt += "\n\nPlease format your response using Markdown."
        
        return base_prompt

class AnthropicProvider(BaseLLMProvider):
    """Anthropic Claude provider"""
    
    def __init__(self, api_key: str, model: str = "claude-3-sonnet-20240229"):
        self.client = anthropic.AsyncAnthropic(api_key=api_key)
        self.model = model
        self.available = True
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate content using Anthropic API"""
        try:
            import time
            start_time = time.time()
            
            system_prompt = self._create_system_prompt(request)
            
            response = await self.client.messages.create(
                model=self.model,
                max_tokens=request.max_tokens,
                temperature=request.temperature,
                system=system_prompt,
                messages=[{"role": "user", "content": request.prompt}]
            )
            
            generation_time = time.time() - start_time
            
            return GenerationResponse(
                content=response.content[0].text,
                model_used=self.model,
                tokens_used=response.usage.input_tokens + response.usage.output_tokens,
                generation_time=generation_time
            )
            
        except Exception as e:
            logger.error(f"Anthropic generation failed: {e}")
            self.available = False
            raise
    
    def is_available(self) -> bool:
        return self.available
    
    def _create_system_prompt(self, request: GenerationRequest) -> str:
        """Create system prompt for Claude"""
        return """You are Claude, an AI assistant specialized in e-commerce recommendations. 
        You provide thoughtful, accurate product recommendations with clear explanations. 
        Always be helpful, honest, and consider the user's specific needs and preferences."""

class VertexAIProvider(BaseLLMProvider):
    """Google Vertex AI Gemini provider"""
    
    def __init__(self, project_id: str, location: str = "us-central1", model: str = "gemini-1.5-pro"):
        try:
            import vertexai
            from vertexai.generative_models import GenerativeModel, SafetySetting, HarmCategory
            
            # Initialize Vertex AI
            vertexai.init(project=project_id, location=location)
            
            # Safety settings for production use
            self.safety_settings = [
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_HARASSMENT,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_HATE_SPEECH,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_SEXUALLY_EXPLICIT,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_DANGEROUS_CONTENT,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
            ]
            
            # Generation config for consistent outputs
            self.generation_config = {
                "max_output_tokens": 2048,
                "temperature": 0.7,
                "top_p": 0.95,
                "top_k": 40,
            }
            
            self.model = GenerativeModel(model)
            self.project_id = project_id
            self.location = location
            self.model_name = model
            self.available = True
            
            logger.info(f"Initialized Vertex AI Gemini model: {model}")
            
        except ImportError:
            logger.error("Vertex AI SDK not installed. Install with: pip install google-cloud-aiplatform")
            self.available = False
            raise
        except Exception as e:
            logger.error(f"Failed to initialize Vertex AI: {e}")
            self.available = False
            raise
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate content using Vertex AI Gemini"""
        try:
            import time
            start_time = time.time()
            
            # Create system instruction
            system_instruction = self._create_system_instruction(request)
            
            # Update generation config with request parameters
            generation_config = self.generation_config.copy()
            generation_config.update({
                "max_output_tokens": min(request.max_tokens, 2048),
                "temperature": request.temperature,
            })
            
            # Create model instance with system instruction
            model_with_system = GenerativeModel(
                self.model_name,
                system_instruction=[system_instruction],
                generation_config=generation_config,
                safety_settings=self.safety_settings
            )
            
            # Generate content
            response = model_with_system.generate_content(
                request.prompt,
                stream=False
            )
            
            generation_time = time.time() - start_time
            
            # Extract text from response
            if response.text:
                content = response.text
            else:
                # Handle safety filter or other issues
                content = "I apologize, but I couldn't generate a response for that request. Please try rephrasing your query."
                logger.warning("Gemini response was filtered or empty")
            
            # Estimate token usage (Gemini doesn't provide exact counts)
            estimated_tokens = len(request.prompt.split()) + len(content.split())
            
            return GenerationResponse(
                content=content,
                model_used=self.model_name,
                tokens_used=estimated_tokens,
                generation_time=generation_time
            )
            
        except Exception as e:
            logger.error(f"Vertex AI Gemini generation failed: {e}")
            self.available = False
            raise
    
    def is_available(self) -> bool:
        return self.available
    
    def _create_system_instruction(self, request: GenerationRequest) -> str:
        """Create system instruction for Gemini"""
        base_instruction = """You are an expert e-commerce recommendation assistant powered by Google's Gemini AI. 
        Your role is to provide personalized, helpful, and accurate product recommendations with clear explanations.
        
        Key guidelines:
        - Always be helpful and user-focused
        - Provide specific, actionable recommendations
        - Explain your reasoning clearly
        - Consider user context and preferences
        - Be honest about product limitations
        - Use a friendly but professional tone"""
        
        if request.user_context:
            context_info = []
            if request.user_context.get('preferred_categories'):
                context_info.append(f"User typically shops for: {', '.join(request.user_context['preferred_categories'])}")
            if request.user_context.get('budget_range'):
                budget = request.user_context['budget_range']
                context_info.append(f"User's budget range: ${budget.get('min', 0)} - ${budget.get('max', 'unlimited')}")
            if request.user_context.get('experience_level'):
                context_info.append(f"User's experience level: {request.user_context['experience_level']}")
            
            if context_info:
                base_instruction += f"\n\nUser Context:\n{chr(10).join(context_info)}"
        
        if request.format_type == "json":
            base_instruction += "\n\nIMPORTANT: Format your response as valid JSON only. Do not include any text outside the JSON structure."
        elif request.format_type == "markdown":
            base_instruction += "\n\nFormat your response using proper Markdown syntax with headers, lists, and emphasis where appropriate."
        
        return base_instruction

class LLMClient:
    """Multi-provider LLM client with fallback support"""
    
    def __init__(self):
        self.providers: List[BaseLLMProvider] = []
        self.primary_provider: Optional[BaseLLMProvider] = None
    
    def add_provider(self, provider: BaseLLMProvider, is_primary: bool = False):
        """Add an LLM provider"""
        self.providers.append(provider)
        if is_primary or self.primary_provider is None:
            self.primary_provider = provider
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate content with fallback support"""
        # Try primary provider first
        if self.primary_provider and self.primary_provider.is_available():
            try:
                return await self.primary_provider.generate(request)
            except Exception as e:
                logger.warning(f"Primary provider failed: {e}")
        
        # Try fallback providers
        for provider in self.providers:
            if provider != self.primary_provider and provider.is_available():
                try:
                    logger.info(f"Using fallback provider: {provider.__class__.__name__}")
                    return await provider.generate(request)
                except Exception as e:
                    logger.warning(f"Fallback provider failed: {e}")
                    continue
        
        raise Exception("All LLM providers are unavailable")

from typing import Dict, Any, List
from dataclasses import dataclass
import json

@dataclass
class PromptTemplate:
    """Template for generating prompts"""
    template: str
    required_vars: List[str]
    optional_vars: List[str] = None
    format_type: str = "text"

class PromptManager:
    """Manages prompt templates for different recommendation scenarios"""
    
    def __init__(self):
        self.templates = {
            'product_recommendation': PromptTemplate(
                template="""Based on the user's search query "{query}" and the following product options, create personalized recommendations.

User Context:
- Preferred categories: {preferred_categories}
- Budget range: {budget_range}
- Previous purchases: {purchase_history}
- Search intent: {search_intent}

Available Products:
{product_list}

Please provide:
1. Top 3 product recommendations with explanations
2. Why each product matches the user's needs
3. Comparison between the recommended products
4. Any additional considerations or alternatives

Format as a helpful, conversational response.""",
                required_vars=['query', 'product_list'],
                optional_vars=['preferred_categories', 'budget_range', 'purchase_history', 'search_intent']
            ),
            
            'product_explanation': PromptTemplate(
                template="""Explain why the following product is a good match for the user's query "{query}".

Product Details:
{product_details}

User Context:
{user_context}

Provide a clear, compelling explanation that highlights:
1. How the product meets the user's specific needs
2. Key features and benefits relevant to the query
3. Value proposition compared to alternatives
4. Any potential considerations or limitations

Keep the explanation concise but informative (2-3 paragraphs).""",
                required_vars=['query', 'product_details'],
                optional_vars=['user_context']
            ),
            
            'comparison_analysis': PromptTemplate(
                template="""Compare the following products for a user searching for "{query}".

Products to Compare:
{products_data}

User Requirements:
{user_requirements}

Provide a detailed comparison including:
1. Feature comparison table
2. Pros and cons for each product
3. Best use cases for each option
4. Clear recommendation based on user needs
5. Price-value analysis

Format the response in clear sections with bullet points where appropriate.""",
                required_vars=['query', 'products_data'],
                optional_vars=['user_requirements'],
                format_type='markdown'
            ),
            
            'purchase_guide': PromptTemplate(
                template="""Create a comprehensive buying guide for "{product_category}" based on the user's search "{query}".

Available Products:
{product_options}

User Profile:
{user_profile}

Include:
1. Key factors to consider when choosing
2. Budget recommendations for different needs
3. Must-have vs nice-to-have features
4. Common mistakes to avoid
5. Specific product recommendations from the available options

Structure as an easy-to-follow guide with clear sections.""",
                required_vars=['query', 'product_category', 'product_options'],
                optional_vars=['user_profile'],
                format_type='markdown'
            ),
            
            'personalized_summary': PromptTemplate(
                template="""Create a personalized product summary for this user.

Product: {product_name}
Product Details: {product_details}

User Profile:
- Interests: {user_interests}
- Experience level: {experience_level}
- Usage scenario: {usage_scenario}
- Budget considerations: {budget_info}

Write a compelling 2-3 sentence summary that:
1. Highlights the most relevant features for this user
2. Connects to their specific interests and needs
3. Uses language appropriate for their experience level
4. Mentions value in their budget context

Keep it engaging and personal.""",
                required_vars=['product_name', 'product_details'],
                optional_vars=['user_interests', 'experience_level', 'usage_scenario', 'budget_info']
            )
        }
    
    def generate_prompt(self, template_name: str, variables: Dict[str, Any]) -> str:
        """Generate a prompt from template and variables"""
        if template_name not in self.templates:
            raise ValueError(f"Template '{template_name}' not found")
        
        template = self.templates[template_name]
        
        # Check required variables
        missing_vars = [var for var in template.required_vars if var not in variables]
        if missing_vars:
            raise ValueError(f"Missing required variables: {missing_vars}")
        
        # Set default values for optional variables
        prompt_vars = variables.copy()
        if template.optional_vars:
            for var in template.optional_vars:
                if var not in prompt_vars:
                    prompt_vars[var] = "Not specified"
        
        # Generate prompt
        try:
            prompt = template.template.format(**prompt_vars)
            return prompt
        except KeyError as e:
            raise ValueError(f"Variable substitution failed: {e}")
    
    def get_template_info(self, template_name: str) -> Dict[str, Any]:
        """Get information about a template"""
        if template_name not in self.templates:
            raise ValueError(f"Template '{template_name}' not found")
        
        template = self.templates[template_name]
        return {
            'required_vars': template.required_vars,
            'optional_vars': template.optional_vars or [],
            'format_type': template.format_type
        }

# vertex_ai_example.py
import asyncio
import os
import json
from typing import Dict, Any, List, Optional
import logging
from dataclasses import dataclass

# Set up logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

@dataclass
class ProductRecommendation:
    """Product recommendation with Vertex AI generated explanation"""
    product_id: str
    title: str
    price: float
    category: str
    description: str
    ai_explanation: str
    confidence_score: float
    recommendation_rank: int

class VertexAIRecommendationEngine:
    """Complete recommendation engine using Vertex AI Gemini"""
    
    def __init__(self, project_id: str, location: str = "us-central1"):
        self.project_id = project_id
        self.location = location
        self.model = None
        self.initialize_vertex_ai()
    
    def initialize_vertex_ai(self):
        """Initialize Vertex AI with proper configuration"""
        try:
            import vertexai
            from vertexai.generative_models import GenerativeModel, SafetySetting, HarmCategory
            
            # Initialize Vertex AI
            vertexai.init(project=self.project_id, location=self.location)
            
            # Configure safety settings for e-commerce content
            safety_settings = [
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_HARASSMENT,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_HATE_SPEECH,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_SEXUALLY_EXPLICIT,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
                SafetySetting(
                    category=HarmCategory.HARM_CATEGORY_DANGEROUS_CONTENT,
                    threshold=SafetySetting.HarmBlockThreshold.BLOCK_MEDIUM_AND_ABOVE,
                ),
            ]
            
            # Generation configuration optimized for recommendations
            generation_config = {
                "max_output_tokens": 1024,
                "temperature": 0.7,
                "top_p": 0.95,
                "top_k": 40,
            }
            
            # System instruction for e-commerce recommendations
            system_instruction = """You are an expert e-commerce recommendation assistant.
            Your role is to help customers find products that best match their needs and preferences.
            
            Guidelines:
            - Provide personalized recommendations based on user context
            - Explain why each product is suitable for the customer
            - Be honest about product limitations
            - Consider budget, preferences, and use cases
            - Use a helpful, friendly tone
            - Focus on value and relevance to the customer's needs"""
            
            self.model = GenerativeModel(
                "gemini-1.5-pro",
                system_instruction=[system_instruction],
                generation_config=generation_config,
                safety_settings=safety_settings
            )
            
            logger.info("Vertex AI Gemini model initialized successfully")
            
        except ImportError:
            logger.error("Vertex AI SDK not installed. Run: pip install google-cloud-aiplatform")
            raise
        except Exception as e:
            logger.error(f"Failed to initialize Vertex AI: {e}")
            raise
    
    async def generate_recommendations(self, 
                                     user_query: str,
                                     products: List[Dict[str, Any]],
                                     user_context: Optional[Dict[str, Any]] = None,
                                     max_recommendations: int = 5) -> List[ProductRecommendation]:
        """Generate personalized product recommendations using Vertex AI"""
        
        try:
            # Prepare the prompt with user context and products
            prompt = self._create_recommendation_prompt(user_query, products, user_context)
            
            # Generate recommendations using Gemini
            response = self.model.generate_content(prompt)
            
            # Parse the response
            recommendations = self._parse_gemini_response(response.text, products)
            
            # Limit to requested number of recommendations
            return recommendations[:max_recommendations]
            
        except Exception as e:
            logger.error(f"Failed to generate recommendations: {e}")
            raise
    
    def _create_recommendation_prompt(self, 
                                    user_query: str, 
                                    products: List[Dict[str, Any]], 
                                    user_context: Optional[Dict[str, Any]]) -> str:
        """Create a comprehensive prompt for Gemini"""
        
        # Format user context
        context_str = ""
        if user_context:
            context_parts = []
            if user_context.get('budget_range'):
                context_parts.append(f"Budget: ${user_context['budget_range'].get('min', 0)} - ${user_context['budget_range'].get('max', 'unlimited')}")
            if user_context.get('preferred_categories'):
                context_parts.append(f"Preferred categories: {', '.join(user_context['preferred_categories'])}")
            if user_context.get('previous_purchases'):
                context_parts.append(f"Previous purchases: {', '.join(user_context['previous_purchases'][:3])}")
            if user_context.get('experience_level'):
                context_parts.append(f"Experience level: {user_context['experience_level']}")
            
            if context_parts:
                context_str = f"User Context:\n{chr(10).join(context_parts)}\n\n"
        
        # Format products
        products_str = "Available Products:\n"
        for i, product in enumerate(products, 1):
            products_str += f"""
Product {i}:
- ID: {product.get('id', 'unknown')}
- Title: {product.get('title', 'No title')}
- Price: ${product.get('price', 0):.2f}
- Category: {product.get('category', 'Uncategorized')}
- Brand: {product.get('brand', 'Unknown')}
- Rating: {product.get('rating', 0):.1f}/5.0 ({product.get('review_count', 0)} reviews)
- Description: {product.get('description', 'No description')[:200]}{'...' if len(product.get('description', '')) > 200 else ''}
- Key Features: {', '.join(product.get('features', [])[:5])}
"""
        
        prompt = f"""User Query: "{user_query}"

{context_str}{products_str}

Task: Analyze the user's query and context, then recommend the top 3-5 products that best match their needs.

For each recommendation, provide:
1. Product ID and title
2. Why this product matches the user's needs (2-3 sentences)
3. Key benefits specific to their use case
4. Any considerations or limitations
5. Confidence score (0.0-1.0) for how well it matches

Format your response as JSON with this structure:
{{
  "recommendations": [
    {{
      "product_id": "product_id_here",
      "explanation": "Detailed explanation here...",
      "key_benefits": ["benefit1", "benefit2", "benefit3"],
      "considerations": ["consideration1", "consideration2"],
      "confidence_score": 0.95,
      "rank": 1
    }}
  ],
  "summary": "Overall recommendation summary...",
  "reasoning": "Why these products were selected..."
}}

Important: Respond only with valid JSON. Do not include any text outside the JSON structure."""
        
        return prompt
    
    def _parse_gemini_response(self, 
                             response_text: str, 
                             original_products: List[Dict[str, Any]]) -> List[ProductRecommendation]:
        """Parse Gemini's JSON response into ProductRecommendation objects"""
        
        try:
            # Clean the response text
            response_text = response_text.strip()
            
            # Remove markdown code blocks if present
            if response_text.startswith('```json'):
                response_text = response_text[7:]
            if response_text.endswith('```'):
                response_text = response_text[:-3]
            
            # Parse JSON
            data = json.loads(response_text)
            
            recommendations = []
            product_lookup = {p.get('id'): p for p in original_products}
            
            for rec_data in data.get('recommendations', []):
                product_id = rec_data.get('product_id')
                product = product_lookup.get(product_id, {})
                
                recommendation = ProductRecommendation(
                    product_id=product_id,
                    title=product.get('title', 'Unknown Product'),
                    price=product.get('price', 0.0),
                    category=product.get('category', 'Unknown'),
                    description=product.get('description', ''),
                    ai_explanation=rec_data.get('explanation', ''),
                    confidence_score=rec_data.get('confidence_score', 0.0),
                    recommendation_rank=rec_data.get('rank', 0)
                )
                recommendations.append(recommendation)
            
            # Sort by rank
            recommendations.sort(key=lambda x: x.recommendation_rank)
            
            logger.info(f"Successfully parsed {len(recommendations)} recommendations")
            return recommendations
            
        except json.JSONDecodeError as e:
            logger.error(f"Failed to parse JSON response: {e}")
            logger.error(f"Response text: {response_text}")
            
            # Fallback: create basic recommendations from original products
            return self._create_fallback_recommendations(original_products)
        
        except Exception as e:
            logger.error(f"Error parsing response: {e}")
            return self._create_fallback_recommendations(original_products)
    
    def _create_fallback_recommendations(self, products: List[Dict[str, Any]]) -> List[ProductRecommendation]:
        """Create basic recommendations when AI parsing fails"""
        recommendations = []
        
        for i, product in enumerate(products[:3]):
            recommendation = ProductRecommendation(
                product_id=product.get('id', f'fallback_{i}'),
                title=product.get('title', 'Product'),
                price=product.get('price', 0.0),
                category=product.get('category', 'Unknown'),
                description=product.get('description', ''),
                ai_explanation="This product matches your search criteria and is available in our catalog.",
                confidence_score=0.7,
                recommendation_rank=i + 1
            )
            recommendations.append(recommendation)
        
        return recommendations

@dataclass
class UsageMetrics:

class VertexAIMonitor:
    """Monitor Vertex AI usage and costs"""
    
    def __init__(self):
        self.metrics = UsageMetrics()
        self.request_history = []
        
        # Pricing (as of 2025 - check latest pricing)
        self.pricing = {
            "gemini-1.5-pro": {
                "input_tokens_per_1k": 0.0035,
                "output_tokens_per_1k": 0.0105
            }
        }
    
    def track_request(self, 
                     model_name: str,
                     input_tokens: int,
                     output_tokens: int,
                     response_time: float,
                     success: bool):
        """Track individual request metrics"""
        
        # Calculate cost
        pricing = self.pricing.get(model_name, self.pricing["gemini-1.5-pro"])
        input_cost = (input_tokens / 1000) * pricing["input_tokens_per_1k"]
        output_cost = (output_tokens / 1000) * pricing["output_tokens_per_1k"]
        total_cost = input_cost + output_cost
        
        # Update metrics
        self.metrics.total_requests += 1
        self.metrics.total_tokens += input_tokens + output_tokens
        self.metrics.total_cost += total_cost
        
        # Update average response time
        total_time = self.metrics.average_response_time * (self.metrics.total_requests - 1)
        self.metrics.average_response_time = (total_time + response_time) / self.metrics.total_requests
        
        if not success:
            self.metrics.error_count += 1
        
        # Store request history
        self.request_history.append({
            "timestamp": datetime.now(),
            "model": model_name,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "cost": total_cost,
            "response_time": response_time,
            "success": success
        })
        
        logging.info(f"Request tracked - Cost: ${total_cost:.4f}, Time: {response_time:.2f}s")
    
    def get_daily_summary(self) -> Dict[str, Any]:
        """Get daily usage summary"""
        today = datetime.now().date()
        today_requests = [
            req for req in self.request_history 
            if req["timestamp"].date() == today
        ]
        
        if not today_requests:
            return {"date": today, "no_requests": True}
        
        total_cost = sum(req["cost"] for req in today_requests)
        total_requests = len(today_requests)
        avg_response_time = sum(req["response_time"] for req in today_requests) / total_requests
        error_rate = sum(1 for req in today_requests if not req["success"]) / total_requests
        
        return {
            "date": today,
            "total_requests": total_requests,
            "total_cost": total_cost,
            "average_response_time": avg_response_time,
            "error_rate": error_rate,
            "cost_per_request": total_cost / total_requests if total_requests > 0 else 0
        }

import json
from typing import Dict, Any, List, Optional
import logging
from dataclasses import dataclass

logger = logging.getLogger(__name__)

@dataclass
class RecommendationItem:
    """Individual recommendation item"""
    product_id: str
    title: str
    explanation: str
    confidence_score: float
    ranking_factors: Dict[str, float]
    personalization_factors: Dict[str, Any]

@dataclass
class RecommendationSet:
    """Complete set of recommendations"""
    recommendations: List[RecommendationItem]
    summary: str
    comparison: Optional[str] = None
    buying_guide: Optional[str] = None
    metadata: Dict[str, Any] = None

class RecommendationGenerator:
    """Generates personalized recommendations using LLM"""
    
    def __init__(self, llm_client: LLMClient, prompt_manager: PromptManager):
        self.llm_client = llm_client
        self.prompt_manager = prompt_manager
        
        # Generation settings
        self.default_settings = {
            'max_tokens': 1500,
            'temperature': 0.7,
            'top_recommendations': 5
        }
    
    async def generate_recommendations(self,
                                     query: str,
                                     retrieved_products: List[Dict[str, Any]],
                                     user_context: Optional[Dict[str, Any]] = None,
                                     recommendation_type: str = 'standard') -> RecommendationSet:
        """Generate complete recommendation set"""
        logger.info(f"Generating recommendations for query: {query}")
        
        try:
            # Prepare product data
            product_list = self._format_products_for_prompt(retrieved_products)
            
            # Generate main recommendations
            recommendations = await self._generate_main_recommendations(
                query, product_list, user_context
            )
            
            # Generate summary
            summary = await self._generate_summary(query, recommendations, user_context)
            
            # Generate comparison if multiple products
            comparison = None
            if len(recommendations) > 1:
                comparison = await self._generate_comparison(query, recommendations, user_context)
            
            # Generate buying guide if requested
            buying_guide = None
            if recommendation_type == 'detailed':
                buying_guide = await self._generate_buying_guide(query, recommendations, user_context)
            
            return RecommendationSet(
                recommendations=recommendations,
                summary=summary,
                comparison=comparison,
                buying_guide=buying_guide,
                metadata={
                    'query': query,
                    'recommendation_type': recommendation_type,
                    'product_count': len(retrieved_products),
                    'user_context_available': user_context is not None
                }
            )
            
        except Exception as e:
            logger.error(f"Recommendation generation failed: {e}")
            raise
    
    async def _generate_main_recommendations(self,
                                           query: str,
                                           product_list: str,
                                           user_context: Optional[Dict[str, Any]]) -> List[RecommendationItem]:
        """Generate main product recommendations"""
        
        # Prepare variables for prompt
        variables = {
            'query': query,
            'product_list': product_list,
            'preferred_categories': self._format_user_preferences(user_context, 'categories'),
            'budget_range': self._format_user_preferences(user_context, 'budget'),
            'purchase_history': self._format_user_preferences(user_context, 'history'),
            'search_intent': self._infer_search_intent(query)
        }
        
        # Generate prompt
        prompt = self.prompt_manager.generate_prompt('product_recommendation', variables)
        
        # Create generation request
        request = GenerationRequest(
            prompt=prompt,
            max_tokens=self.default_settings['max_tokens'],
            temperature=self.default_settings['temperature'],
            user_context=user_context,
            format_type='json'
        )
        
        # Generate response
        response = await self.llm_client.generate(request)
        
        # Parse and structure recommendations
        return self._parse_recommendations(response.content, query)
    
    async def _generate_summary(self,
                              query: str,
                              recommendations: List[RecommendationItem],
                              user_context: Optional[Dict[str, Any]]) -> str:
        """Generate executive summary of recommendations"""
        
        # Create summary prompt
        rec_summaries = [f"- {rec.title}: {rec.explanation[:100]}..." for rec in recommendations]
        
        prompt = f"""Based on the search query "{query}" and the following recommendations, create a brief executive summary:

Recommendations:
{chr(10).join(rec_summaries)}

User Context: {self._format_user_context_summary(user_context)}

Provide a 2-3 sentence summary that:
1. Addresses the user's search intent
2. Highlights the best overall recommendation
3. Mentions key decision factors

Keep it concise and actionable."""
        
        request = GenerationRequest(
            prompt=prompt,
            max_tokens=200,
            temperature=0.6,
            user_context=user_context
        )
        
        response = await self.llm_client.generate(request)
        return response.content.strip()
    
    async def _generate_comparison(self,
                                 query: str,
                                 recommendations: List[RecommendationItem],
                                 user_context: Optional[Dict[str, Any]]) -> str:
        """Generate detailed comparison between recommendations"""
        
        # Prepare comparison data
        products_data = []
        for rec in recommendations:
            products_data.append(f"Product: {rec.title}\nFeatures: {rec.explanation}")
        
        variables = {
            'query': query,
            'products_data': '\n\n'.join(products_data),
            'user_requirements': self._format_user_requirements(user_context)
        }

        request = GenerationRequest(
            prompt=prompt,
            max_tokens=1000,
            temperature=0.6,
            user_context=user_context,
            format_type='markdown'
        )
        
        response = await self.llm_client.generate(request)
        return response.content
    
    async def _generate_buying_guide(self,
                                   query: str,
                                   recommendations: List[RecommendationItem],
                                   user_context: Optional[Dict[str, Any]]) -> str:
        """Generate comprehensive buying guide"""
        
        # Infer product category from query and recommendations
        product_category = self._infer_product_category(query, recommendations)
        
        # Format product options
        product_options = [f"- {rec.title}: {rec.explanation}" for rec in recommendations]
        
        variables = {
            'query': query,
            'product_category': product_category,
            'product_options': '\n'.join(product_options),
            'user_profile': self._format_user_profile(user_context)

        request = GenerationRequest(
            prompt=prompt,
            max_tokens=1500,
            temperature=0.7,
            user_context=user_context,
            format_type='markdown'
        )
        
        response = await self.llm_client.generate(request)
        return response.content
    
    def _format_products_for_prompt(self, products: List[Dict[str, Any]]) -> str:
        """Format product data for LLM prompt"""
        formatted_products = []
        
        for i, product in enumerate(products[:self.default_settings['top_recommendations']], 1):
            product_text = f"""Product {i}: {product.get('title', 'Unknown Product')}
- Category: {product.get('category', 'N/A')}
- Brand: {product.get('brand', 'N/A')}
- Price: ${product.get('price', 0):.2f}
- Rating: {product.get('rating', 0):.1f}/5.0 ({product.get('review_count', 0)} reviews)
- Description: {product.get('description', 'No description available')[:200]}...
- Key Features: {', '.join(product.get('keywords', [])[:5])}"""
            
            formatted_products.append(product_text)
        
        return '\n\n'.join(formatted_products)
    
    def _format_user_preferences(self, user_context: Optional[Dict[str, Any]], pref_type: str) -> str:
        """Format user preferences for prompts"""
        if not user_context:
            return "Not specified"
        
        if pref_type == 'categories':
            return ', '.join(user_context.get('preferred_categories', []))
        elif pref_type == 'budget':
            budget = user_context.get('budget_range', {})
            if budget:
                return f"${budget.get('min', 0)} - ${budget.get('max', 'unlimited')}"
        elif pref_type == 'history':
            history = user_context.get('purchase_history', [])
            return ', '.join(history[:3]) if history else "No previous purchases"
        
        return "Not specified"
    
    def _infer_search_intent(self, query: str) -> str:
        """Infer user's search intent from query"""
        query_lower = query.lower()
        
        if any(word in query_lower for word in ['buy', 'purchase', 'order']):
            return "Purchase intent"
        elif any(word in query_lower for word in ['compare', 'vs', 'difference']):
            return "Comparison intent"
        elif any(word in query_lower for word in ['best', 'top', 'recommend']):
            return "Recommendation seeking"
        elif any(word in query_lower for word in ['cheap', 'budget', 'affordable']):
            return "Budget-conscious"
        elif any(word in query_lower for word in ['premium', 'luxury', 'high-end']):
            return "Quality-focused"
        else:
            return "General exploration"
    
    def _parse_recommendations(self, llm_response: str, query: str) -> List[RecommendationItem]:
        """Parse LLM response into structured recommendations"""
        recommendations = []
        
        try:
            # Try to parse as JSON first
            if llm_response.strip().startswith('{') or llm_response.strip().startswith('['):
                data = json.loads(llm_response)
                if isinstance(data, dict) and 'recommendations' in data:
                    data = data['recommendations']
                
                for item in data:
                    rec = RecommendationItem(
                        product_id=item.get('product_id', ''),
                        title=item.get('title', ''),
                        explanation=item.get('explanation', ''),
                        confidence_score=item.get('confidence_score', 0.8),
                        ranking_factors=item.get('ranking_factors', {}),
                        personalization_factors=item.get('personalization_factors', {})
                    )
                    recommendations.append(rec)
            
        except json.JSONDecodeError:
            # Fallback: parse text format
            logger.warning("Failed to parse JSON, falling back to text parsing")
            recommendations = self._parse_text_recommendations(llm_response, query)
        
        return recommendations
    
    def _parse_text_recommendations(self, text: str, query: str) -> List[RecommendationItem]:
        """Parse text-based recommendations"""
        recommendations = []
        
        # Simple parsing logic - would be enhanced in production
        sections = text.split('\n\n')
        
        for i, section in enumerate(sections):
            if any(keyword in section.lower() for keyword in ['recommend', 'product', 'option']):
                rec = RecommendationItem(
                    product_id=f"parsed_{i}",
                    title=f"Recommendation {i+1}",
                    explanation=section.strip(),
                    confidence_score=0.7,
                    ranking_factors={'text_parsing': 1.0},
                    personalization_factors={}
                )
                recommendations.append(rec)
        
        return recommendations[:3]  # Limit to top 3

        # Required elements for recommendations
        self.required_elements = [
            'product name', 'price', 'key features', 'explanation'
        ]
    
    def validate_recommendation_set(self, recommendation_set: RecommendationSet) -> Dict[str, Any]:
        """Validate complete recommendation set quality"""
        validation_results = {
            'overall_score': 0.0,
            'issues': [],
            'recommendations': True,
            'detailed_scores': {}
        }
        
        try:
            # Validate individual recommendations
            rec_scores = []
            for i, rec in enumerate(recommendation_set.recommendations):
                rec_validation = self._validate_recommendation(rec)
                rec_scores.append(rec_validation['score'])
                validation_results['detailed_scores'][f'recommendation_{i}'] = rec_validation
                
                if rec_validation['score'] < 0.6:
                    validation_results['issues'].append(
                        f"Recommendation {i+1} quality below threshold: {rec_validation['issues']}"
                    )
            
            # Validate summary
            if recommendation_set.summary:
                summary_validation = self._validate_summary(recommendation_set.summary)
                validation_results['detailed_scores']['summary'] = summary_validation
                
                if summary_validation['score'] < 0.7:
                    validation_results['issues'].append(f"Summary quality issues: {summary_validation['issues']}")
            
            # Validate comparison (if exists)
            if recommendation_set.comparison:
                comparison_validation = self._validate_comparison(recommendation_set.comparison)
                validation_results['detailed_scores']['comparison'] = comparison_validation
            
            # Calculate overall score
            all_scores = rec_scores
            if recommendation_set.summary:
                all_scores.append(summary_validation['score'])
            
            validation_results['overall_score'] = sum(all_scores) / len(all_scores) if all_scores else 0
            
            # Determine if recommendations pass quality check
            validation_results['recommendations'] = (
                validation_results['overall_score'] >= 0.7 and
                len(validation_results['issues']) == 0
            )
            
        except Exception as e:
            logger.error(f"Validation failed: {e}")
            validation_results['recommendations'] = False
            validation_results['issues'].append(f"Validation error: {str(e)}")
        
        return validation_results
    
    def _validate_recommendation(self, recommendation: RecommendationItem) -> Dict[str, Any]:
        """Validate individual recommendation quality"""
        issues = []
        score_components = {}
        
        # Check explanation length
        explanation_length = len(recommendation.explanation)
        if explanation_length < self.min_explanation_length:
            issues.append(f"Explanation too short ({explanation_length} chars)")
            score_components['length'] = 0.3
        elif explanation_length > self.max_explanation_length:
            issues.append(f"Explanation too long ({explanation_length} chars)")
            score_components['length'] = 0.7
        else:
            score_components['length'] = 1.0
        
        # Check for prohibited phrases
        explanation_lower = recommendation.explanation.lower()
        prohibited_found = [phrase for phrase in self.prohibited_phrases if phrase in explanation_lower]
        if prohibited_found:
            issues.append(f"Contains prohibited phrases: {prohibited_found}")
            score_components['content'] = 0.2
        else:
            score_components['content'] = 1.0
        
        # Check for product-specific information
        has_product_info = any(keyword in explanation_lower for keyword in [
            'feature', 'price', 'quality', 'performance', 'benefit'
        ])
        score_components['relevance'] = 1.0 if has_product_info else 0.5
        
        # Check confidence score validity
        if 0 <= recommendation.confidence_score <= 1:
            score_components['confidence'] = 1.0
        else:
            issues.append(f"Invalid confidence score: {recommendation.confidence_score}")
            score_components['confidence'] = 0.0
        
        # Calculate overall score
        overall_score = sum(score_components.values()) / len(score_components)
        
        return {
            'score': overall_score,
            'issues': issues,
            'score_components': score_components
        }
    
    def _validate_summary(self, summary: str) -> Dict[str, Any]:
        """Validate summary quality"""
        issues = []
        score_components = {}
        
        # Check length
        summary_length = len(summary)
        if summary_length < self.min_summary_length:
            issues.append(f"Summary too short ({summary_length} chars)")
            score_components['length'] = 0.3
        elif summary_length > self.max_summary_length:
            issues.append(f"Summary too long ({summary_length} chars)")
            score_components['length'] = 0.7
        else:
            score_components['length'] = 1.0
        
        # Check for actionable content
        has_action = any(keyword in summary.lower() for keyword in [
            'recommend', 'suggest', 'best', 'choose', 'consider'
        ])
        score_components['actionable'] = 1.0 if has_action else 0.6
        
        # Check coherence (simple heuristic)
        sentences = summary.split('.')
        score_components['coherence'] = 1.0 if len(sentences) >= 2 else 0.7
        
        overall_score = sum(score_components.values()) / len(score_components)
        
        return {
            'score': overall_score,
            'issues': issues,
            'score_components': score_components
        }
    
    def _validate_comparison(self, comparison: str) -> Dict[str, Any]:
        """Validate comparison content quality"""
        issues = []
        score_components = {}
        
        # Check for comparison keywords
        comparison_keywords = ['vs', 'versus', 'compared', 'difference', 'better', 'advantage']
        has_comparison = any(keyword in comparison.lower() for keyword in comparison_keywords)
        score_components['comparison_content'] = 1.0 if has_comparison else 0.5
        
        # Check structure (looking for organized content)
        has_structure = any(marker in comparison for marker in ['1.', '2.', '-', '*', '##'])
        score_components['structure'] = 1.0 if has_structure else 0.7
        
        # Check length appropriateness
        comparison_length = len(comparison)
        if comparison_length < 100:
            score_components['depth'] = 0.5
        elif comparison_length > 2000:
            score_components['depth'] = 0.7
        else:
            score_components['depth'] = 1.0
        
        overall_score = sum(score_components.values()) / len(score_components)
        
        return {
            'score': overall_score,
            'issues': issues,
            'score_components': score_components
        }
    
    def improve_content(self, content: str, content_type: str) -> str:
        """Attempt to improve content quality automatically"""
        improved_content = content
        
        # Remove prohibited phrases
        for phrase in self.prohibited_phrases:
            improved_content = improved_content.replace(phrase, '')
        
        # Clean up formatting
        improved_content = re.sub(r'\n\s*\n', '\n\n', improved_content)  # Normalize line breaks
        improved_content = re.sub(r'\s+', ' ', improved_content)  # Normalize spaces
        improved_content = improved_content.strip()
        
        # Add structure if missing (for comparisons)
        if content_type == 'comparison' and not any(marker in improved_content for marker in ['1.', '-', '*']):
            # Simple structure addition
            sentences = improved_content.split('. ')
            if len(sentences) > 2:
                structured_sentences = [f"• {sentence.strip()}" for sentence in sentences if sentence.strip()]
                improved_content = '\n'.join(structured_sentences)
        
        return improved_content

from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel, Field
from typing import List, Dict, Any, Optional
import logging

logger = logging.getLogger(__name__)

app = FastAPI(title="RAG Recommendation Generation API")

class RecommendationRequest(BaseModel):
    query: str = Field(..., description="User search query")
    retrieved_products: List[Dict[str, Any]] = Field(..., description="Products from retrieval system")
    user_context: Optional[Dict[str, Any]] = Field(default=None, description="User context for personalization")
    recommendation_type: str = Field(default="standard", description="Type of recommendation (standard, detailed)")
    include_comparison: bool = Field(default=True, description="Include product comparison")
    include_buying_guide: bool = Field(default=False, description="Include buying guide")

class RecommendationResponse(BaseModel):
    recommendations: List[Dict[str, Any]]
    summary: str
    comparison: Optional[str] = None
    buying_guide: Optional[str] = None
    quality_score: float
    generation_metadata: Dict[str, Any]

# Global recommendation generator
recommendation_generator: Optional[RecommendationGenerator] = None

def get_recommendation_generator() -> RecommendationGenerator:
    """Dependency to get recommendation generator"""
    if recommendation_generator is None:
        raise HTTPException(status_code=500, detail="Recommendation generator not initialized")
    return recommendation_generator

@app.post("/generate-recommendations", response_model=RecommendationResponse)
async def generate_recommendations(
    request: RecommendationRequest,
    generator: RecommendationGenerator = Depends(get_recommendation_generator)
) -> RecommendationResponse:
    """Generate personalized product recommendations"""
    try:
        import time
        start_time = time.time()
        
        # Generate recommendations
        recommendation_set = await generator.generate_recommendations(
            query=request.query,
            retrieved_products=request.retrieved_products,
            user_context=request.user_context,
            recommendation_type=request.recommendation_type
        )
        
        # Validate quality
        quality_controller = ContentQualityController()
        quality_results = quality_controller.validate_recommendation_set(recommendation_set)
        
        generation_time = time.time() - start_time
        
        # Format response
        formatted_recommendations = []
        for rec in recommendation_set.recommendations:
            formatted_rec = {
                'product_id': rec.product_id,
                'title': rec.title,
                'explanation': rec.explanation,
                'confidence_score': rec.confidence_score,
                'ranking_factors': rec.ranking_factors,
                'personalization_factors': rec.personalization_factors
            }
            formatted_recommendations.append(formatted_rec)
        
        return RecommendationResponse(
            recommendations=formatted_recommendations,
            summary=recommendation_set.summary,
            comparison=recommendation_set.comparison if request.include_comparison else None,
            buying_guide=recommendation_set.buying_guide if request.include_buying_guide else None,
            quality_score=quality_results['overall_score'],
            generation_metadata={
                'generation_time': generation_time,
                'quality_validation': quality_results,
                'recommendation_count': len(formatted_recommendations),
                'user_personalization': request.user_context is not None
            }
        )
        
    except Exception as e:
        logger.error(f"Recommendation generation error: {e}")
        raise HTTPException(status_code=500, detail=str(e))

# main/rag_system.py
import asyncio
from typing import Dict, Any, List, Optional
import logging

logger = logging.getLogger(__name__)

class RAGRecommendationSystem:
    """Complete RAG-based recommendation system"""
    
    def __init__(self,
                 retrieval_engine: RetrievalEngine,
                 recommendation_generator: RecommendationGenerator,
                 cache_manager: Optional[RetrievalCache] = None):
        
        self.retrieval_engine = retrieval_engine
        self.recommendation_generator = recommendation_generator
        self.cache_manager = cache_manager
    
    async def get_recommendations(self,
                                query: str,
                                user_id: Optional[str] = None,
                                limit: int = 10,
                                include_explanations: bool = True) -> Dict[str, Any]:
        """Complete end-to-end recommendation pipeline"""
        
        try:
            logger.info(f"Processing recommendation request: {query}")
            
            # Get user context
            user_context = await self._get_user_context(user_id) if user_id else None
            
            # Check cache first
            cached_result = None
            if self.cache_manager:
                cached_result = await self.cache_manager.get_cached_results(
                    query, {}, user_id
                )
            
            if cached_result:
                logger.info("Returning cached results")
                return cached_result
            
            # Step 1: Retrieve relevant products
            retrieval_results = await self.retrieval_engine.search(
                query=query,
                user_context=user_context,
                limit=limit * 2  # Retrieve more for better generation
            )
            
            if not retrieval_results:
                return {
                    'recommendations': [],
                    'message': 'No products found matching your query',
                    'suggestions': await self._get_query_suggestions(query)
                }
            
            # Step 2: Convert retrieval results to product data
            product_data = [result.payload for result in retrieval_results]
            
            # Step 3: Generate personalized recommendations
            if include_explanations:
                recommendation_set = await self.recommendation_generator.generate_recommendations(
                    query=query,
                    retrieved_products=product_data,
                    user_context=user_context,
                    recommendation_type='detailed'
                )
                
                result = {
                    'query': query,
                    'recommendations': [
                        {
                            'product_id': rec.product_id,
                            'title': rec.title,
                            'explanation': rec.explanation,
                            'confidence_score': rec.confidence_score,
                            'product_data': next(
                                (p for p in product_data if p.get('id') == rec.product_id), 
                                {}
                            )
                        }
                        for rec in recommendation_set.recommendations
                    ],
                    'summary': recommendation_set.summary,
                    'comparison': recommendation_set.comparison,
                    'buying_guide': recommendation_set.buying_guide
                }
            else:
                # Simple recommendations without explanations
                result = {
                    'query': query,
                    'recommendations': [
                        {
                            'product_id': result.product_id,
                            'score': result.score,
                            'product_data': result.payload
                        }
                        for result in retrieval_results[:limit]
                    ]
                }
            
            # Cache the result
            if self.cache_manager:
                await self.cache_manager.cache_results(query, {}, result, user_id)
            
            logger.info(f"Generated {len(result['recommendations'])} recommendations")
            return result
            
        except Exception as e:
            logger.error(f"Recommendation pipeline failed: {e}")
            return {
                'error': str(e),
                'recommendations': [],
                'message': 'Sorry, we encountered an error processing your request'
            }

        # Placeholder implementation
        return {
