Data Pipeline and Vector Embeddings

This section covers building robust data pipelines and generating high-quality vector embeddings that capture the semantic essence of products and user behavior. Vector embeddings are the foundation that enables our recommendation system to understand relationships between products, users, and their preferences at a semantic level.

Data Pipeline Architecture​

Our data pipeline follows the Extract, Transform, Load (ETL) pattern, specifically designed for vector embedding generation:

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│   Raw Data      │    │  Processing     │    │   Vector        │
│   Sources       │───▶│  Pipeline       │───▶│   Storage       │
└─────────────────┘    └─────────────────┘    └─────────────────┘
│                      │                      │                 │
├─ Product Catalog     ├─ Data Cleaning       ├─ Product Vectors│
├─ User Profiles       ├─ Feature Extraction  ├─ User Vectors   │
├─ Interaction Logs    ├─ Embedding Generation├─ Query Vectors  │
├─ Reviews/Ratings     ├─ Quality Validation  ├─ Metadata Index │
└─ Search Queries      └─ Batch Processing    └─ Search Index   │

Product Data Processing​

# processors/product_processor.py
import pandas as pd
import numpy as np
from typing import List, Dict, Any
import re
import logging

logger = logging.getLogger(__name__)

class ProductDataProcessor:
    def __init__(self):
        self.stop_words = self._load_stop_words()
        self.category_keywords = self._load_category_keywords()
    
    def clean_product_data(self, products_df: pd.DataFrame) -> pd.DataFrame:
        """Clean and standardize product data"""
        logger.info(f"Cleaning {len(products_df)} products")
        
        # Remove duplicates
        products_df = products_df.drop_duplicates(subset=['id'])
        
        # Standardize text fields
        products_df['title'] = products_df['title'].apply(self._clean_text)
        products_df['description'] = products_df['description'].apply(self._clean_text)
        
        # Normalize prices
        products_df['price'] = pd.to_numeric(products_df['price'], errors='coerce')
        products_df = products_df.dropna(subset=['price'])
        
        # Fill missing values
        products_df['description'] = products_df['description'].fillna('')
        products_df['brand'] = products_df['brand'].fillna('Unknown')
        products_df['rating'] = products_df['rating'].fillna(0.0)
        products_df['review_count'] = products_df['review_count'].fillna(0)
        
        # Calculate derived features
        products_df['title_length'] = products_df['title'].str.len()
        products_df['description_length'] = products_df['description'].str.len()
        products_df['price_category'] = pd.cut(
            products_df['price'], 
            bins=[0, 25, 100, 500, np.inf], 
            labels=['budget', 'mid-range', 'premium', 'luxury']
        )
        
        logger.info(f"Cleaned data: {len(products_df)} products remaining")
        return products_df
    
    def extract_product_features(self, product: Product) -> Dict[str, Any]:
        """Extract meaningful features from product data"""
        features = {
            # Basic features
            'title': product.title,
            'description': product.description,
            'category': product.category.value,
            'subcategory': product.subcategory or '',
            'brand': product.brand or '',
            
            # Numeric features
            'price': product.price,
            'rating': product.rating or 0.0,
            'review_count': product.review_count,
            'popularity_score': product.popularity_score,
            
            # Text processing
            'combined_text': self._create_combined_text(product),
            'keywords': product.keywords,
            'search_terms': product.search_terms,
            
            # Categorical features
            'price_range': self._categorize_price(product.price),
            'rating_tier': self._categorize_rating(product.rating or 0.0),
            'popularity_tier': self._categorize_popularity(product.popularity_score),
            
            # Availability features
            'is_available': product.is_available,
            'stock_level': self._categorize_stock(product.stock_quantity),
        }
        
        # Add attribute-based features
        features.update(self._process_attributes(product.attributes))
        
        return features
    
    def _clean_text(self, text: str) -> str:
        """Clean and normalize text data"""
        if pd.isna(text) or not isinstance(text, str):
            return ""
        
        # Convert to lowercase
        text = text.lower()
        
        # Remove special characters but keep alphanumeric and spaces
        text = re.sub(r'[^a-z0-9\s]', ' ', text)
        
        # Remove extra whitespaces
        text = re.sub(r'\s+', ' ', text).strip()
        
        return text
    
    def _create_combined_text(self, product: Product) -> str:
        """Create combined text for embedding generation"""
        components = [
            product.title,
            product.description,
            product.brand or '',
            product.category.value,
            product.subcategory or '',
            ' '.join(product.keywords),
            ' '.join(product.search_terms)
        ]
        
        # Filter out empty components
        components = [comp for comp in components if comp and comp.strip()]
        
        return ' '.join(components)

import pandas as pd
import numpy as np
from typing import List, Dict
from datetime import datetime
from collections import defaultdict
import logging

logger = logging.getLogger(__name__)

class UserBehaviorProcessor:
    def __init__(self):
        self.interaction_weights = {
            InteractionType.VIEW: 1.0,
            InteractionType.CLICK: 2.0,
            InteractionType.ADD_TO_CART: 5.0,
            InteractionType.PURCHASE: 10.0,
            InteractionType.REVIEW: 8.0,
            InteractionType.SEARCH: 3.0,
            InteractionType.WISHLIST: 4.0,
        }
    
    def process_user_interactions(self, interactions_df: pd.DataFrame) -> pd.DataFrame:
        """Process and enrich user interaction data"""
        logger.info(f"Processing {len(interactions_df)} user interactions")
        
        # Convert timestamp to datetime
        interactions_df['timestamp'] = pd.to_datetime(interactions_df['timestamp'])
        
        # Add temporal features
        interactions_df['hour'] = interactions_df['timestamp'].dt.hour
        interactions_df['day_of_week'] = interactions_df['timestamp'].dt.dayofweek
        interactions_df['is_weekend'] = interactions_df['day_of_week'].isin([5, 6])
        
        # Calculate interaction weights
        interactions_df['interaction_weight'] = interactions_df['interaction_type'].map(
            lambda x: self.interaction_weights.get(InteractionType(x), 1.0)
        )
        
        # Add session-based features
        interactions_df = self._add_session_features(interactions_df)
        
        return interactions_df
    
    def create_user_profiles(self, users_df: pd.DataFrame, interactions_df: pd.DataFrame) -> Dict[str, Dict]:
        """Create comprehensive user profiles from interaction data"""
        user_profiles = {}
        
        for user_id in users_df['id'].unique():
            user_interactions = interactions_df[interactions_df['user_id'] == user_id]
            
            profile = {
                'user_id': user_id,
                'behavioral_features': self._extract_behavioral_features(user_interactions),
                'preference_features': self._extract_preference_features(user_interactions),
                'temporal_features': self._extract_temporal_features(user_interactions),
                'engagement_features': self._extract_engagement_features(user_interactions),
            }
            
            user_profiles[user_id] = profile
        
        logger.info(f"Created profiles for {len(user_profiles)} users")
        return user_profiles

Embedding Service Implementation​

# embeddings/embedding_generator.py
import numpy as np
import torch
from sentence_transformers import SentenceTransformer
from typing import List, Dict, Any, Optional
import logging
from abc import ABC, abstractmethod
import asyncio
import openai

logger = logging.getLogger(__name__)

class BaseEmbeddingGenerator(ABC):
    """Abstract base class for embedding generators"""
    
    @abstractmethod
    async def generate_embeddings(self, texts: List[str]) -> np.ndarray:
        pass
    
    @abstractmethod
    def get_embedding_dimension(self) -> int:
        pass

class SentenceTransformerEmbedding(BaseEmbeddingGenerator):
    """Local sentence transformer embedding generator"""
    
    def __init__(self, model_name: str = "all-MiniLM-L6-v2"):
        self.model_name = model_name
        self.model = SentenceTransformer(model_name)
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)
        
        logger.info(f"Loaded SentenceTransformer model: {model_name}")
    
    async def generate_embeddings(self, texts: List[str]) -> np.ndarray:
        """Generate embeddings for a list of texts"""
        if not texts:
            return np.array([])
        
        # Process in batches to manage memory
        batch_size = 32
        all_embeddings = []
        
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i + batch_size]
            embeddings = self.model.encode(
                batch,
                convert_to_numpy=True,
                show_progress_bar=False,
                batch_size=batch_size
            )
            all_embeddings.append(embeddings)
        
        return np.vstack(all_embeddings)
    
    def get_embedding_dimension(self) -> int:
        return self.model.get_sentence_embedding_dimension()

class OpenAIEmbedding(BaseEmbeddingGenerator):
    """OpenAI API embedding generator"""
    
    def __init__(self, model_name: str = "text-embedding-ada-002", api_key: Optional[str] = None):
        self.model_name = model_name
        openai.api_key = api_key or os.getenv("OPENAI_API_KEY")
        self.dimensions = {
            "text-embedding-ada-002": 1536,
            "text-embedding-3-small": 1536,
            "text-embedding-3-large": 3072,
        }
        
        logger.info(f"Initialized OpenAI embedding model: {model_name}")
    
    async def generate_embeddings(self, texts: List[str]) -> np.ndarray:
        """Generate embeddings using OpenAI API"""
        if not texts:
            return np.array([])
        
        batch_size = 100
        all_embeddings = []
        
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i + batch_size]
            
            try:
                response = await openai.Embedding.acreate(
                    model=self.model_name,
                    input=batch
                )
                
                batch_embeddings = [data.embedding for data in response.data]
                all_embeddings.extend(batch_embeddings)
                
                await asyncio.sleep(0.1)  # Rate limiting
                
            except Exception as e:
                logger.error(f"Error generating embeddings: {e}")
                zero_embedding = [0.0] * self.get_embedding_dimension()
                all_embeddings.extend([zero_embedding] * len(batch))
        
        return np.array(all_embeddings)
    
    def get_embedding_dimension(self) -> int:
        return self.dimensions.get(self.model_name, 1536)

Product Embedding Generation​

class ProductEmbeddingGenerator:
    """Specialized embedding generator for products"""
    
    def __init__(self, embedding_generator: BaseEmbeddingGenerator):
        self.embedding_generator = embedding_generator
        self.dimension = embedding_generator.get_embedding_dimension()
    
    async def generate_product_embeddings(self, products: List[Dict[str, Any]]) -> Dict[str, np.ndarray]:
        """Generate embeddings for products"""
        logger.info(f"Generating embeddings for {len(products)} products")
        
        # Prepare texts for embedding
        product_texts = []
        product_ids = []
        
        for product in products:
            text = self._create_product_text(product)
            product_texts.append(text)
            product_ids.append(product['id'])
        
        # Generate embeddings
        embeddings = await self.embedding_generator.generate_embeddings(product_texts)
        
        # Create mapping
        product_embeddings = {}
        for product_id, embedding in zip(product_ids, embeddings):
            product_embeddings[product_id] = embedding
        
        return product_embeddings
    
    def _create_product_text(self, product: Dict[str, Any]) -> str:
        """Create text representation of product for embedding"""
        components = []
        
        # Title (highest weight)
        if product.get('title'):
            components.append(f"Title: {product['title']}")
        
        # Category and subcategory
        if product.get('category'):
            components.append(f"Category: {product['category']}")
        if product.get('subcategory'):
            components.append(f"Subcategory: {product['subcategory']}")
        
        # Brand
        if product.get('brand'):
            components.append(f"Brand: {product['brand']}")
        
        # Description
        if product.get('description'):
            description = product['description'][:500]  # Truncate long descriptions
            components.append(f"Description: {description}")
        
        # Price range
        if product.get('price_range'):
            components.append(f"Price range: {product['price_range']}")
        
        # Keywords and tags
        if product.get('keywords'):
            components.append(f"Keywords: {', '.join(product['keywords'])}")
        
        return ' '.join(components)

# validation/embedding_validator.py
import numpy as np
from typing import List, Dict, Any
from sklearn.metrics.pairwise import cosine_similarity
from sklearn.decomposition import PCA
import logging

logger = logging.getLogger(__name__)

class EmbeddingQualityValidator:
    """Validate the quality of generated embeddings"""
    
    def __init__(self):
        self.similarity_threshold = 0.7
        self.diversity_threshold = 0.3
    
    def validate_product_embeddings(
        self, 
        product_embeddings: Dict[str, np.ndarray],
        products: List[Dict[str, Any]]
    ) -> Dict[str, Any]:
        """Comprehensive validation of product embeddings"""
        logger.info("Validating product embeddings...")
        
        results = {
            'total_products': len(product_embeddings),
            'embedding_dimension': len(next(iter(product_embeddings.values()))),
            'quality_metrics': {},
            'recommendations': []
        }
        
        # Create product lookup
        product_lookup = {p['id']: p for p in products}
        
        # Test semantic similarity
        similarity_results = self._test_semantic_similarity(product_embeddings, product_lookup)
        results['quality_metrics']['semantic_similarity'] = similarity_results
        
        # Test category clustering
        clustering_results = self._test_category_clustering(product_embeddings, product_lookup)
        results['quality_metrics']['category_clustering'] = clustering_results
        
        # Test embedding diversity
        diversity_results = self._test_embedding_diversity(product_embeddings)
        results['quality_metrics']['diversity'] = diversity_results
        
        # Generate recommendations
        results['recommendations'] = self._generate_quality_recommendations(results['quality_metrics'])
        
        overall_score = self._calculate_overall_score(results['quality_metrics'])
        logger.info(f"Validation complete. Overall quality score: {overall_score:.3f}")
        
        return results
    
    def _test_semantic_similarity(
        self, 
        embeddings: Dict[str, np.ndarray], 
        products: Dict[str, Dict[str, Any]]
    ) -> Dict[str, float]:
        """Test if semantically similar products have similar embeddings"""
        # Sample products for testing
        sample_size = min(100, len(embeddings))
        sampled_ids = list(embeddings.keys())[:sample_size]
        
        similarity_scores = []
        
        for i, product_id_1 in enumerate(sampled_ids):
            for product_id_2 in sampled_ids[i+1:]:
                # Calculate embedding similarity
                emb_sim = cosine_similarity(
                    embeddings[product_id_1].reshape(1, -1),
                    embeddings[product_id_2].reshape(1, -1)
                )[0, 0]
                
                # Calculate semantic similarity
                semantic_sim = self._calculate_semantic_similarity(
                    products[product_id_1], 
                    products[product_id_2]
                )
                
                similarity_scores.append({
                    'embedding_similarity': emb_sim,
                    'semantic_similarity': semantic_sim
                })
        
        # Calculate correlation
        emb_sims = [s['embedding_similarity'] for s in similarity_scores]
        sem_sims = [s['semantic_similarity'] for s in similarity_scores]
        
        correlation = np.corrcoef(emb_sims, sem_sims)[0, 1] if len(emb_sims) > 1 else 0
        
        return {
            'correlation': correlation,
            'avg_embedding_similarity': np.mean(emb_sims),
            'avg_semantic_similarity': np.mean(sem_sims),
            'sample_size': len(similarity_scores)
        }
    
    def _calculate_semantic_similarity(self, product1: Dict[str, Any], product2: Dict[str, Any]) -> float:
        """Calculate semantic similarity between two products"""
        similarity = 0.0
        
        # Category similarity (highest weight)
        if product1.get('category') == product2.get('category'):
            similarity += 0.4
            
            # Subcategory similarity
            if product1.get('subcategory') == product2.get('subcategory'):
                similarity += 0.2
        
        # Brand similarity
        if product1.get('brand') == product2.get('brand'):
            similarity += 0.2
        
        # Price range similarity
        if product1.get('price_range') == product2.get('price_range'):
            similarity += 0.1
        
        # Title similarity (simple word overlap)
        title1_words = set(product1.get('title', '').lower().split())
        title2_words = set(product2.get('title', '').lower().split())
        if title1_words and title2_words:
            word_overlap = len(title1_words & title2_words) / len(title1_words | title2_words)
            similarity += 0.1 * word_overlap
        
        return similarity

import numpy as np
from typing import Dict, List, Any
import logging

class VectorStorage:
    """Handles storage and retrieval of vector embeddings"""
    
    def __init__(self, vector_db_client):
        self.client = vector_db_client
    
    async def store_product_embeddings(
        self, 
        collection_name: str,
        product_embeddings: Dict[str, np.ndarray],
        product_metadata: Dict[str, Dict[str, Any]]
    ) -> bool:
        """Store product embeddings with metadata"""
        try:
            vectors_data = []
            
            for product_id, embedding in product_embeddings.items():
                vector_data = {
                    'id': product_id,
                    'vector': embedding,
                    'payload': product_metadata.get(product_id, {})
                }
                vectors_data.append(vector_data)
            
            # Batch insert
            success = await self.client.upsert_vectors(collection_name, vectors_data)
            
            if success:
                logger.info(f"Successfully stored {len(vectors_data)} product embeddings")
            else:
                logger.error("Failed to store product embeddings")
            
            return success
            
        except Exception as e:
            logger.error(f"Error storing embeddings: {e}")
            return False

        self,

class EmbeddingPipeline:

    async def run_product_pipeline(self, products_data: List[Dict[str, Any]]) -> bool:
        """Run complete product embedding pipeline"""
        try:
            logger.info("Starting product embedding pipeline")
            
            # Step 1: Clean and process product data
            logger.info("Processing product data...")
            processed_products = []
            for product_data in products_data:
                features = self.product_processor.extract_product_features(product_data)
                processed_products.append(features)
            
            # Step 2: Generate embeddings
            logger.info("Generating product embeddings...")
            product_embeddings = await self._generate_product_embeddings(processed_products)
            
            # Step 3: Validate embeddings
            logger.info("Validating embeddings...")
            validation_results = self._validate_embeddings(product_embeddings, processed_products)
            
            # Step 4: Store embeddings
            logger.info("Storing embeddings...")
            metadata = {p['id']: p for p in processed_products}
            success = await self.vector_storage.store_product_embeddings(
                'products', product_embeddings, metadata
            )
            
            if success:
                logger.info("Product pipeline completed successfully")
            
            return success
            
        except Exception as e:
            logger.error(f"Product pipeline failed: {e}")
            return False
    
    async def run_user_pipeline(self, user_profiles: Dict[str, Dict[str, Any]]) -> bool:
        """Run complete user embedding pipeline"""
        try:
            logger.info("Starting user embedding pipeline")
            
            # Generate user embeddings
            user_embeddings = await self._generate_user_embeddings(user_profiles)
            
            # Store embeddings
            success = await self.vector_storage.store_user_embeddings(
                'users', user_embeddings, user_profiles
            )
            
            return success
            
        except Exception as e:
            logger.error(f"User pipeline failed: {e}")
            return False
    
    async def _generate_product_embeddings(self, products: List[Dict[str, Any]]) -> Dict[str, np.ndarray]:
        """Generate embeddings for products"""
        product_texts = []
        product_ids = []
        
        for product in products:
            text = self._create_product_text(product)
            product_texts.append(text)
            product_ids.append(product['id'])
        
        embeddings = await self.embedding_generator.generate_embeddings(product_texts)
        
        return {
            product_id: embedding 
            for product_id, embedding in zip(product_ids, embeddings)
        }

Metrics Collection​

# monitoring/pipeline_metrics.py
import time
from typing import Dict, Any
import logging

logger = logging.getLogger(__name__)

class PipelineMetrics:
    """Collect and track pipeline performance metrics"""
    
    def __init__(self):
        self.metrics = {
            'processing_times': {},
            'embedding_quality': {},
            'throughput': {},
            'error_rates': {}
        }
    
    def track_processing_time(self, stage: str, duration: float):
        """Track processing time for pipeline stages"""
        if stage not in self.metrics['processing_times']:
            self.metrics['processing_times'][stage] = []
        
        self.metrics['processing_times'][stage].append(duration)
        
        logger.info(f"{stage} completed in {duration:.2f}s")
    
    def track_embedding_quality(self, quality_metrics: Dict[str, Any]):
        """Track embedding quality metrics"""
        self.metrics['embedding_quality'] = quality_metrics
        
        logger.info(f"Embedding quality: {quality_metrics}")
    
    def get_summary(self) -> Dict[str, Any]:
        """Get summary of all metrics"""
        summary = {}
        
        # Average processing times
        for stage, times in self.metrics['processing_times'].items():
            summary[f'{stage}_avg_time'] = sum(times) / len(times) if times else 0
        
        # Latest quality metrics
        summary['quality_metrics'] = self.metrics['embedding_quality']
        
        return summary
