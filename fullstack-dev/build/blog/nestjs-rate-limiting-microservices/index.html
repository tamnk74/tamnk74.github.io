Handling Rate Limiting in NestJS for Production Microservices

October 5, 2025 ¬∑ 26 min read
Tam Nguyen
Full Stack Developer, Next.js for Production Creator

Rate limiting is a critical security and performance mechanism in production microservice architectures. It protects your services from abuse, ensures fair resource allocation, and maintains system stability under high load. In this comprehensive guide, we'll explore how to implement robust, scalable rate limiting in NestJS applications designed for microservice environments.

Why Rate Limiting is Critical in Microservices‚Äã

üõ°Ô∏è Security Benefits‚Äã

  • DDoS Protection: Prevents overwhelming individual services with excessive requests
  • API Abuse Prevention: Stops malicious actors from exhausting system resources
  • Brute Force Attack Mitigation: Limits login attempts and sensitive operations
  • Resource Protection: Ensures no single client can monopolize service capacity

‚ö° Performance Benefits‚Äã

  • Fair Resource Allocation: Ensures all users get equitable access to services
  • Cascade Failure Prevention: Prevents one overloaded service from affecting others
  • Cost Control: Manages computational and bandwidth costs in cloud environments
  • SLA Compliance: Helps maintain service level agreements under varying loads

üèóÔ∏è Microservice-Specific Challenges‚Äã

  • Distributed State Management: Rate limits must work across multiple service instances
  • Service-to-Service Communication: Internal APIs need different limits than public ones
  • Dynamic Scaling: Rate limits must adapt to auto-scaling scenarios
  • Cross-Service Coordination: Shared limits across multiple related services

Popular Open Source Rate Limiting Libraries‚Äã

Before implementing custom rate limiting solutions, consider these battle-tested libraries that can significantly speed up development:

üöÄ @nestjs/throttler‚Äã

The official NestJS throttling library provides simple, decorator-based rate limiting.

// Installation
npm install @nestjs/throttler

// Basic setup
import { ThrottlerModule, ThrottlerGuard } from '@nestjs/throttler';
import { APP_GUARD } from '@nestjs/core';

@Module({
  imports: [
    ThrottlerModule.forRoot([
      {
        name: 'short',
        ttl: 1000,  // 1 second
        limit: 3,   // 3 requests per second
      },
      {
        name: 'medium',
        ttl: 10000, // 10 seconds
        limit: 20,  // 20 requests per 10 seconds
      },
      {
        name: 'long',
        ttl: 60000, // 1 minute
        limit: 100, // 100 requests per minute
      },
    ]),
  ],
  providers: [
    {
      provide: APP_GUARD,
      useClass: ThrottlerGuard,
    },
  ],
})
export class AppModule {}

// Usage in controllers
import { Throttle } from '@nestjs/throttler';

@Controller('api')
export class ApiController {
  @Get('public')
  @Throttle({ default: { limit: 10, ttl: 60000 } })
  getPublicData() {
    return { data: 'Limited to 10 requests per minute' };
  }

  @Post('auth/login')
  @Throttle({ default: { limit: 5, ttl: 300000 } }) // 5 attempts per 5 minutes
  login(@Body() credentials: any) {
    return { token: 'jwt-token' };
  }
}

üîÑ bottleneck‚Äã

Advanced rate limiting with queuing, clustering, and Redis support.

// Installation
npm install bottleneck

// Service implementation
import Bottleneck from 'bottleneck';
import { Injectable } from '@nestjs/common';

@Injectable()
export class AdvancedRateLimitService {
  private limiters = new Map<string, Bottleneck>();

  constructor() {
    // Create different limiters for different use cases
    this.setupLimiters();
  }

  private setupLimiters() {
    // API rate limiter with Redis clustering
    this.limiters.set('api', new Bottleneck({
      reservoir: 100,           // Initial capacity
      reservoirRefreshAmount: 100, // Tokens to add
      reservoirRefreshInterval: 60 * 1000, // Every minute
      maxConcurrent: 10,        // Max concurrent requests
      minTime: 100,             // Min time between requests (100ms)
      
      // Redis clustering for distributed rate limiting
      datastore: 'redis',
      clearDatastore: false,
      clientOptions: {
        host: 'localhost',
        port: 6379,
      },
      clusterNodes: [
        { host: 'redis-1', port: 6379 },
        { host: 'redis-2', port: 6379 },
      ],
    }));

    // Premium API with higher limits
    this.limiters.set('premium', new Bottleneck({
      reservoir: 1000,
      reservoirRefreshAmount: 1000,
      reservoirRefreshInterval: 60 * 1000,
      maxConcurrent: 50,
      minTime: 20,
    }));

    // Heavy processing limiter
    this.limiters.set('processing', new Bottleneck({
      maxConcurrent: 3,         // Only 3 heavy operations at once
      minTime: 2000,            // 2 seconds between operations
    }));
  }

  async executeWithRateLimit<T>(
    limiterType: string,
    operation: () => Promise<T>,
    priority?: number
  ): Promise<T> {
    const limiter = this.limiters.get(limiterType);
    if (!limiter) {
      throw new Error(`Unknown limiter type: ${limiterType}`);
    }

    return limiter.schedule({ priority }, operation);
  }

  async checkRateLimit(
    limiterType: string,
    weight: number = 1
  ): Promise<{ allowed: boolean; msBeforeNext: number }> {
    const limiter = this.limiters.get(limiterType);
    if (!limiter) {
      return { allowed: false, msBeforeNext: 0 };
    }

    try {
      const result = await limiter.check(weight);
      return {
        allowed: result >= 0,
        msBeforeNext: result < 0 ? Math.abs(result) : 0,
      };
    } catch (error) {
      return { allowed: false, msBeforeNext: 1000 };
    }
  }

  // Get limiter statistics
  getStats(limiterType: string) {
    const limiter = this.limiters.get(limiterType);
    if (!limiter) return null;

    return {
      running: limiter.running(),
      queued: limiter.queued(),
      reservoir: limiter.reservoir(),
      capacity: limiter.capacity,
    };
  }
}

// Usage in controllers
@Controller('api')
export class ApiController {
  constructor(
    private readonly rateLimitService: AdvancedRateLimitService
  ) {}

  @Get('data')
  async getData() {
    return this.rateLimitService.executeWithRateLimit(
      'api',
      async () => {
        // Your API logic here
        return { data: 'API response' };
      }
    );
  }

  @Post('heavy-process')
  async heavyProcess(@Body() data: any) {
    const check = await this.rateLimitService.checkRateLimit('processing');
    
    if (!check.allowed) {
      throw new HttpException(
        {
          message: 'Processing queue full',
          retryAfter: Math.ceil(check.msBeforeNext / 1000),
        },
        HttpStatus.TOO_MANY_REQUESTS
      );
    }

    return this.rateLimitService.executeWithRateLimit(
      'processing',
      async () => {
        // Heavy processing logic
        await this.performHeavyOperation(data);
        return { status: 'completed' };
      },
      5 // High priority
    );
  }

  @Get('stats')
  getSystemStats() {
    return {
      api: this.rateLimitService.getStats('api'),
      premium: this.rateLimitService.getStats('premium'),
      processing: this.rateLimitService.getStats('processing'),
    };
  }
}

// Service implementation
import RateLimiter from 'ioredis-rate-limiter';
import Redis from 'ioredis';
import { Injectable } from '@nestjs/common';

@Injectable()
export class RedisRateLimitService {
  private redis: Redis;
  private limiters = new Map<string, RateLimiter>();

  constructor() {
    this.redis = new Redis({
      host: 'localhost',
      port: 6379,
      retryDelayOnFailover: 100,
    });

    this.setupLimiters();
  }

  private setupLimiters() {
    // API rate limiter - 100 requests per minute
    this.limiters.set('api', new RateLimiter({
      redis: this.redis,
      key: (id: string) => `rate_limit:api:${id}`,
      window: 60000,        // 1 minute window
      limit: 100,           // 100 requests per window
    }));

    // Login rate limiter - 5 attempts per 15 minutes
    this.limiters.set('login', new RateLimiter({
      redis: this.redis,
      key: (id: string) => `rate_limit:login:${id}`,
      window: 15 * 60000,   // 15 minutes
      limit: 5,             // 5 attempts
    }));

    // Premium API - 1000 requests per minute
    this.limiters.set('premium', new RateLimiter({
      redis: this.redis,
      key: (id: string) => `rate_limit:premium:${id}`,
      window: 60000,
      limit: 1000,
    }));
  }

  async checkLimit(
    type: string,
    identifier: string,
    cost: number = 1
  ): Promise<{
    allowed: boolean;
    remaining: number;
    resetTime: Date;
    totalHits: number;
  }> {
    const limiter = this.limiters.get(type);
    if (!limiter) {
      throw new Error(`Unknown rate limiter type: ${type}`);
    }

    try {
      const result = await limiter.check(identifier, cost);
      
      return {
        allowed: result.allowed,
        remaining: result.remaining,
        resetTime: new Date(result.resetTime),
        totalHits: result.totalHits,
      };
    } catch (error) {
      // Fallback in case of Redis issues
      console.error('Rate limit check failed:', error);
      return {
        allowed: true, // Fail open for availability
        remaining: 0,
        resetTime: new Date(Date.now() + 60000),
        totalHits: 0,
      };
    }
  }

  async getRemainingQuota(type: string, identifier: string): Promise<number> {
    const result = await this.checkLimit(type, identifier, 0); // Cost 0 for check only
    return result.remaining;
  }
}

// Guard implementation
@Injectable()
export class RedisRateLimitGuard implements CanActivate {
  constructor(
    private readonly rateLimitService: RedisRateLimitService,
    private readonly reflector: Reflector
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest();
    const response = context.switchToHttp().getResponse();
    
    // Get rate limit configuration from decorator
    const config = this.reflector.get<{
      type: string;
      identifier?: (req: any) => string;
      cost?: number;
    }>('REDIS_RATE_LIMIT', context.getHandler());

    if (!config) {
      return true; // No rate limiting configured
    }

    // Generate identifier
    const identifier = config.identifier 
      ? config.identifier(request)
      : request.ip;

    // Check rate limit
    const result = await this.rateLimitService.checkLimit(
      config.type,
      identifier,
      config.cost || 1
    );

    // Set headers
    response.setHeader('X-RateLimit-Remaining', result.remaining);
    response.setHeader('X-RateLimit-Reset', result.resetTime.toISOString());
    response.setHeader('X-RateLimit-Total', result.totalHits);

    if (!result.allowed) {
      response.setHeader('Retry-After', Math.ceil((result.resetTime.getTime() - Date.now()) / 1000));
      
      throw new HttpException(
        {
          statusCode: HttpStatus.TOO_MANY_REQUESTS,
          message: 'Rate limit exceeded',
          retryAfter: result.resetTime.toISOString(),
        },
        HttpStatus.TOO_MANY_REQUESTS
      );
    }

    return true;
  }
}

// Decorator for easy usage
export const RedisRateLimit = (config: {
  type: string;
  identifier?: (req: any) => string;
  cost?: number;
}) => SetMetadata('REDIS_RATE_LIMIT', config);

// Usage example
@Controller('api')
@UseGuards(RedisRateLimitGuard)
export class ApiController {
  @Get('data')
  @RedisRateLimit({
    type: 'api',
    identifier: (req) => req.ip,
  })
  getData() {
    return { data: 'API response' };
  }

    type: 'login',
    identifier: (req) => `${req.ip}:${req.body.email}`,
  })
  login(@Body() credentials: any) {
    return { token: 'jwt-token' };
  }

  @Get('premium')
  @RedisRateLimit({
    type: 'premium',
    identifier: (req) => req.user?.id || req.ip,

Rate Limiting Strategies for Microservices‚Äã

1. Token Bucket Algorithm‚Äã

The token bucket algorithm is ideal for allowing bursts while maintaining average rate limits.

// rate-limiting/token-bucket.service.ts
import { Injectable } from '@nestjs/common';
import { RedisService } from '@nestjs-modules/ioredis';

interface TokenBucketConfig {
  capacity: number;      // Maximum tokens in bucket
  refillRate: number;    // Tokens added per second
  windowSize: number;    // Time window in seconds
}

@Injectable()
export class TokenBucketService {
  constructor(private readonly redis: RedisService) {}

  async checkRateLimit(
    key: string,
    config: TokenBucketConfig,
    tokensRequested: number = 1
  ): Promise<{
    allowed: boolean;
    remainingTokens: number;
    resetTime: number;
    retryAfter?: number;
  }> {
    const script = `
      local key = KEYS[1]
      local capacity = tonumber(ARGV[1])
      local refillRate = tonumber(ARGV[2])
      local windowSize = tonumber(ARGV[3])
      local tokensRequested = tonumber(ARGV[4])
      local now = tonumber(ARGV[5])
      
      -- Get current bucket state
      local bucket = redis.call('HMGET', key, 'tokens', 'lastRefill')
      local tokens = tonumber(bucket[1]) or capacity
      local lastRefill = tonumber(bucket[2]) or now
      
      -- Calculate tokens to add based on time elapsed
      local elapsed = math.max(0, now - lastRefill)
      local tokensToAdd = math.floor(elapsed * refillRate)
      tokens = math.min(capacity, tokens + tokensToAdd)
      
      -- Check if request can be fulfilled
      local allowed = tokens >= tokensRequested
      local remainingTokens = tokens
      local retryAfter = 0
      
      if allowed then
        remainingTokens = tokens - tokensRequested
        -- Update bucket state
        redis.call('HMSET', key, 'tokens', remainingTokens, 'lastRefill', now)
        redis.call('EXPIRE', key, windowSize * 2)
      else
        -- Calculate retry after time
        local tokensNeeded = tokensRequested - tokens
        retryAfter = math.ceil(tokensNeeded / refillRate)
      end
      
      return {
        allowed and 1 or 0,
        remainingTokens,
        now + (capacity - remainingTokens) / refillRate,
        retryAfter
      }
    `;

    const now = Math.floor(Date.now() / 1000);
    const result = await this.redis.eval(
      script,
      1,
      key,
      config.capacity,
      config.refillRate,
      config.windowSize,
      tokensRequested,
      now
    ) as [number, number, number, number];

    return {
      allowed: result[0] === 1,
      remainingTokens: result[1],
      resetTime: result[2],
      retryAfter: result[3] > 0 ? result[3] : undefined,
    };
  }
}

2. Sliding Window Algorithm‚Äã

Sliding window provides more accurate rate limiting by tracking requests in a rolling time window.

// rate-limiting/sliding-window.service.ts
import { Injectable } from '@nestjs/common';
import { RedisService } from '@nestjs-modules/ioredis';

interface SlidingWindowConfig {
  limit: number;         // Maximum requests per window
  windowMs: number;      // Window size in milliseconds
  precision: number;     // Sub-window precision (e.g., 10 for 10 sub-windows)
}

@Injectable()
export class SlidingWindowService {
  constructor(private readonly redis: RedisService) {}

  async checkRateLimit(
    key: string,
    config: SlidingWindowConfig
  ): Promise<{
    allowed: boolean;
    remaining: number;
    resetTime: number;
    currentUsage: number;
  }> {
    const script = `
      local key = KEYS[1]
      local limit = tonumber(ARGV[1])
      local windowMs = tonumber(ARGV[2])
      local precision = tonumber(ARGV[3])
      local now = tonumber(ARGV[4])
      
      -- Calculate sub-window size
      local subWindowMs = windowMs / precision
      local currentWindow = math.floor(now / subWindowMs)
      local windowStart = currentWindow - precision + 1
      
      -- Clean old sub-windows
      redis.call('ZREMRANGEBYSCORE', key, 0, windowStart - 1)
      
      -- Count current requests in the sliding window
      local currentCount = 0
      local subWindows = redis.call('ZRANGE', key, 0, -1, 'WITHSCORES')
      
      for i = 1, #subWindows, 2 do
        local subWindow = tonumber(subWindows[i + 1])
        local count = tonumber(subWindows[i])
        
        if subWindow >= windowStart then
          -- Calculate overlap percentage for partial windows
          local overlap = 1
          if subWindow == windowStart then
            local subWindowTime = subWindow * subWindowMs
            local windowTime = now - windowMs
            overlap = math.max(0, (subWindowTime + subWindowMs - windowTime) / subWindowMs)
          end
          currentCount = currentCount + (count * overlap)
        end
      end
      
      local allowed = currentCount < limit
      local remaining = math.max(0, limit - currentCount - 1)
      
      if allowed then
        -- Increment counter for current sub-window
        redis.call('ZINCRBY', key, 1, currentWindow)
        redis.call('EXPIRE', key, math.ceil(windowMs / 1000) * 2)
      end
      
      local resetTime = (currentWindow + 1) * subWindowMs
      
      return {
        allowed and 1 or 0,
        remaining,
        resetTime,
        currentCount + (allowed and 1 or 0)
      }
    `;

    const now = Date.now();
    const result = await this.redis.eval(
      script,
      1,
      key,
      config.limit,
      config.windowMs,
      config.precision,
      now
    ) as [number, number, number, number];

    return {
      allowed: result[0] === 1,
      remaining: result[1],
      resetTime: result[2],
      currentUsage: result[3],
    };
  }
}

Advanced Rate Limiting Guard‚Äã

Create a flexible, configurable rate limiting guard that supports multiple algorithms and strategies.

// guards/rate-limit.guard.ts
import {
  Injectable,
  CanActivate,
  ExecutionContext,
  HttpException,
  HttpStatus,
  Inject,
} from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { Request, Response } from 'express';
import { TokenBucketService } from '../rate-limiting/token-bucket.service';
import { SlidingWindowService } from '../rate-limiting/sliding-window.service';
import { RATE_LIMIT_OPTIONS } from '../decorators/rate-limit.decorator';

export interface RateLimitOptions {
  algorithm: 'token-bucket' | 'sliding-window' | 'fixed-window';
  keyGenerator?: (req: Request) => string;
  skipIf?: (req: Request) => boolean;
  onLimitReached?: (req: Request, res: Response) => void;
  
  // Token bucket specific
  capacity?: number;
  refillRate?: number;
  
  // Sliding/Fixed window specific
  limit?: number;
  windowMs?: number;
  precision?: number; // For sliding window
  
  // Common options
  message?: string;
  standardHeaders?: boolean;
  legacyHeaders?: boolean;
  skipSuccessfulRequests?: boolean;
  skipFailedRequests?: boolean;
}

@Injectable()
export class RateLimitGuard implements CanActivate {
  constructor(
    private readonly reflector: Reflector,
    private readonly tokenBucket: TokenBucketService,
    private readonly slidingWindow: SlidingWindowService,
    @Inject('CONFIG_SERVICE') private readonly config: any
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest<Request>();
    const response = context.switchToHttp().getResponse<Response>();
    
    // Get rate limit options from decorator or global config
    const options = this.getRateLimitOptions(context);
    
    if (!options) {
      return true; // No rate limiting configured
    }

    // Check if request should be skipped
    if (options.skipIf && options.skipIf(request)) {
      return true;
    }

    // Generate rate limit key
    const key = this.generateKey(request, options);
    
    // Apply rate limiting based on algorithm
    const result = await this.applyRateLimit(key, options);
    
    // Set response headers
    this.setHeaders(response, result, options);
    
    if (!result.allowed) {
      // Execute custom handler if provided
      if (options.onLimitReached) {
        options.onLimitReached(request, response);
      }
      
      // Throw rate limit exceeded exception
      throw new HttpException(
        {
          statusCode: HttpStatus.TOO_MANY_REQUESTS,
          message: options.message || 'Rate limit exceeded',
          error: 'Too Many Requests',
          retryAfter: result.retryAfter,
        },
        HttpStatus.TOO_MANY_REQUESTS
      );
    }

    return true;
  }

  private getRateLimitOptions(context: ExecutionContext): RateLimitOptions | null {
    // Check method-level decorator first
    const methodOptions = this.reflector.get<RateLimitOptions>(
      RATE_LIMIT_OPTIONS,
      context.getHandler()
    );
    
    if (methodOptions) {
      return methodOptions;
    }

    // Check class-level decorator
    const classOptions = this.reflector.get<RateLimitOptions>(
      RATE_LIMIT_OPTIONS,
      context.getClass()
    );
    
    if (classOptions) {
      return classOptions;
    }

    // Return global default options if configured
    return this.config.get('rateLimit.default');
  }

  private generateKey(request: Request, options: RateLimitOptions): string {
    if (options.keyGenerator) {
      return options.keyGenerator(request);
    }

    // Default key generation strategy
    const ip = request.ip || request.connection.remoteAddress;
    const userId = (request as any).user?.id || 'anonymous';
    const route = `${request.method}:${request.route?.path || request.path}`;
    
    return `rate_limit:${ip}:${userId}:${route}`;
  }

  private async applyRateLimit(
    key: string,
    options: RateLimitOptions
  ): Promise<{
    allowed: boolean;
    remaining?: number;
    resetTime?: number;
    retryAfter?: number;
    currentUsage?: number;
  }> {
    switch (options.algorithm) {
      case 'token-bucket':
        return this.tokenBucket.checkRateLimit(key, {
          capacity: options.capacity || 10,
          refillRate: options.refillRate || 1,
          windowSize: Math.floor((options.windowMs || 60000) / 1000),
        });

      case 'sliding-window':
        return this.slidingWindow.checkRateLimit(key, {
          limit: options.limit || 100,
          windowMs: options.windowMs || 60000,
          precision: options.precision || 10,
        });

      case 'fixed-window':
        return this.applyFixedWindow(key, {
          limit: options.limit || 100,
          windowMs: options.windowMs || 60000,
        });

      default:
        throw new Error(`Unknown rate limiting algorithm: ${options.algorithm}`);
    }
  }

  private async applyFixedWindow(
    key: string,
    config: { limit: number; windowMs: number }
  ): Promise<any> {
    // Simple fixed window implementation using Redis
    const window = Math.floor(Date.now() / config.windowMs);
    const windowKey = `${key}:${window}`;
    
    const current = await this.tokenBucket['redis'].incr(windowKey);
    await this.tokenBucket['redis'].expire(windowKey, Math.ceil(config.windowMs / 1000));
    
    const allowed = current <= config.limit;
    const remaining = Math.max(0, config.limit - current);
    const resetTime = (window + 1) * config.windowMs;
    
    return {
      allowed,
      remaining,
      resetTime,
      currentUsage: current,
    };
  }

  private setHeaders(response: Response, result: any, options: RateLimitOptions): void {
    if (options.standardHeaders !== false) {
      // Standard rate limit headers (draft RFC)
      response.setHeader('RateLimit-Limit', options.limit || options.capacity);
      response.setHeader('RateLimit-Remaining', result.remaining || 0);
      response.setHeader('RateLimit-Reset', result.resetTime);
      
      if (result.retryAfter) {
        response.setHeader('Retry-After', result.retryAfter);
      }
    }

    if (options.legacyHeaders) {
      // Legacy X-RateLimit headers for compatibility
      response.setHeader('X-RateLimit-Limit', options.limit || options.capacity);
      response.setHeader('X-RateLimit-Remaining', result.remaining || 0);
      response.setHeader('X-RateLimit-Reset', result.resetTime);
    }
  }
}

// services/distributed-rate-limit.service.ts
import { Injectable, Inject } from '@nestjs/common';
import { ClientProxy } from '@nestjs/microservices';
import { RedisService } from '@nestjs-modules/ioredis';

interface ServiceRateLimitConfig {
  serviceName: string;
  globalLimit: number;
  perInstanceLimit: number;
  windowMs: number;
  coordinationStrategy: 'redis' | 'gossip' | 'leader-election';
}

@Injectable()
export class DistributedRateLimitService {
  private instanceId: string;
  
  constructor(
    private readonly redis: RedisService,
    @Inject('MESSAGE_BROKER') private readonly messageBroker: ClientProxy
  ) {
    this.instanceId = `${process.env.POD_NAME || 'local'}-${Date.now()}`;
  }

  async checkServiceRateLimit(
    config: ServiceRateLimitConfig,
    clientKey: string
  ): Promise<{
    allowed: boolean;
    globalUsage: number;
    instanceUsage: number;
    recommendation: 'allow' | 'deny' | 'throttle';
  }> {
    const script = `
      local globalKey = KEYS[1]
      local instanceKey = KEYS[2]
      local globalLimit = tonumber(ARGV[1])
      local perInstanceLimit = tonumber(ARGV[2])
      local windowMs = tonumber(ARGV[3])
      local now = tonumber(ARGV[4])
      local instanceId = ARGV[5]
      
      -- Clean expired entries
      local cutoff = now - windowMs
      redis.call('ZREMRANGEBYSCORE', globalKey, 0, cutoff)
      redis.call('ZREMRANGEBYSCORE', instanceKey, 0, cutoff)
      
      -- Get current usage
      local globalUsage = redis.call('ZCARD', globalKey)
      local instanceUsage = redis.call('ZCARD', instanceKey)
      
      -- Calculate available capacity
      local globalAvailable = globalLimit - globalUsage
      local instanceAvailable = perInstanceLimit - instanceUsage
      
      -- Determine if request should be allowed
      local allowed = globalAvailable > 0 and instanceAvailable > 0
      local recommendation = 'allow'
      
      if not allowed then
        recommendation = 'deny'
      elseif globalAvailable < globalLimit * 0.1 then
        recommendation = 'throttle'
      end
      
      if allowed then
        -- Record the request
        local requestId = instanceId .. ':' .. now .. ':' .. math.random(1000000)
        redis.call('ZADD', globalKey, now, requestId)
        redis.call('ZADD', instanceKey, now, requestId)
        redis.call('EXPIRE', globalKey, math.ceil(windowMs / 1000) * 2)
        redis.call('EXPIRE', instanceKey, math.ceil(windowMs / 1000) * 2)
        
        globalUsage = globalUsage + 1
        instanceUsage = instanceUsage + 1
      end
      
      return {
        allowed and 1 or 0,
        globalUsage,
        instanceUsage,
        recommendation
      }
    `;

    const now = Date.now();
    const globalKey = `service_rate_limit:${config.serviceName}:global:${clientKey}`;
    const instanceKey = `service_rate_limit:${config.serviceName}:${this.instanceId}:${clientKey}`;

    const result = await this.redis.eval(
      script,
      2,
      globalKey,
      instanceKey,
      config.globalLimit,
      config.perInstanceLimit,
      config.windowMs,
      now,
      this.instanceId
    ) as [number, number, number, string];

    // Publish metrics for monitoring
    await this.publishMetrics(config.serviceName, {
      globalUsage: result[1],
      instanceUsage: result[2],
      recommendation: result[3],
      timestamp: now,
    });

    return {
      allowed: result[0] === 1,
      globalUsage: result[1],
      instanceUsage: result[2],
      recommendation: result[3] as any,
    };
  }

  private async publishMetrics(serviceName: string, metrics: any): Promise<void> {
    try {
      await this.messageBroker.emit('rate_limit.metrics', {
        serviceName,
        instanceId: this.instanceId,
        ...metrics,
      });
    } catch (error) {
      // Metrics publishing failure shouldn't affect rate limiting
      console.warn('Failed to publish rate limit metrics:', error);
    }
  }

  async getServiceHealth(serviceName: string): Promise<{
    totalInstances: number;
    healthyInstances: number;
    globalUsage: number;
    averageInstanceUsage: number;
  }> {
    const healthKey = `service_health:${serviceName}`;
    const instances = await this.redis.hgetall(healthKey);
    
    const now = Date.now();
    const healthyThreshold = 30000; // 30 seconds
    
    let healthyCount = 0;
    let totalUsage = 0;
    
    for (const [instanceId, lastSeen] of Object.entries(instances)) {
      if (now - parseInt(lastSeen) < healthyThreshold) {
        healthyCount++;
        
        // Get instance usage
        const usageKey = `service_rate_limit:${serviceName}:${instanceId}:*`;
        const usageKeys = await this.redis.keys(usageKey);
        for (const key of usageKeys) {
          const usage = await this.redis.zcard(key);
          totalUsage += usage;
        }
      }
    }
    
    return {
      totalInstances: Object.keys(instances).length,
      healthyInstances: healthyCount,
      globalUsage: totalUsage,
      averageInstanceUsage: healthyCount > 0 ? totalUsage / healthyCount : 0,
    };
  }
}

Middleware for Global Rate Limiting‚Äã

Implement middleware for application-wide rate limiting with exemptions.

// middleware/global-rate-limit.middleware.ts
import { Injectable, NestMiddleware, HttpException, HttpStatus } from '@nestjs/common';
import { Request, Response, NextFunction } from 'express';
import { TokenBucketService } from '../rate-limiting/token-bucket.service';

@Injectable()
export class GlobalRateLimitMiddleware implements NestMiddleware {
  constructor(private readonly tokenBucket: TokenBucketService) {}

  async use(req: Request, res: Response, next: NextFunction) {
    // Skip rate limiting for certain paths
    const exemptPaths = [
      '/health',
      '/metrics',
      '/api/v1/internal',
    ];

    if (exemptPaths.some(path => req.path.startsWith(path))) {
      return next();
    }

    // Skip for internal service calls
    if (req.headers['x-service-token'] === process.env.INTERNAL_SERVICE_TOKEN) {
      return next();
    }

    // Apply global rate limiting
    const clientKey = this.generateClientKey(req);
    const result = await this.tokenBucket.checkRateLimit(clientKey, {
      capacity: 10000,     // 10k requests
      refillRate: 100,     // 100 requests per second
      windowSize: 3600,    // 1 hour window
    });

    // Set rate limit headers
    res.setHeader('X-Global-RateLimit-Limit', '10000');
    res.setHeader('X-Global-RateLimit-Remaining', result.remainingTokens);
    res.setHeader('X-Global-RateLimit-Reset', result.resetTime);

    if (!result.allowed) {
      throw new HttpException(
        {
          statusCode: HttpStatus.TOO_MANY_REQUESTS,
          message: 'Global rate limit exceeded',
          error: 'Too Many Requests',
          retryAfter: result.retryAfter,
        },
        HttpStatus.TOO_MANY_REQUESTS
      );
    }

    next();
  }

  private generateClientKey(req: Request): string {
    // Prioritize API key, then user ID, then IP
    const apiKey = req.headers['x-api-key'] as string;
    const userId = (req as any).user?.id;
    const ip = req.ip || req.connection.remoteAddress;

    if (apiKey) {
      return `global:api:${apiKey}`;
    }
    if (userId) {
      return `global:user:${userId}`;
    }
    return `global:ip:${ip}`;
  }
}

    });

  }

    service: string, 
    algorithm: string, 
    keyType: string, 
    reason: string
  ): void {

  }

    service: string, 
    algorithm: string, 
    keyType: string, 
    ratio: number
  ): void {

  }

    this.rateLimitLatency.observe({ service, algorithm }, duration);
  }

  }
}
