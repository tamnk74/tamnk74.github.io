# generation/llm_client.py
import asyncio
import openai
import anthropic
from typing import List, Dict, Any, Optional, Union
import json
import logging
from abc import ABC, abstractmethod
from dataclasses import dataclass
from enum import Enum
import time

logger = logging.getLogger(__name__)

class ModelProvider(Enum):
    OPENAI = "openai"
    ANTHROPIC = "anthropic"
    LOCAL = "local"
    AZURE_OPENAI = "azure_openai"

@dataclass
class GenerationConfig:
    temperature: float = 0.7
    max_tokens: int = 1000
    top_p: float = 0.9
    frequency_penalty: float = 0.0
    presence_penalty: float = 0.0
    stop_sequences: Optional[List[str]] = None

@dataclass
class GenerationRequest:
    prompt: str
    config: GenerationConfig
    model: str
    system_prompt: Optional[str] = None
    context: Optional[Dict[str, Any]] = None

@dataclass
class GenerationResponse:
    content: str
    model: str
    provider: ModelProvider
    token_usage: Dict[str, int]
    latency: float
    cost: float

class BaseLLMClient(ABC):
    """Abstract base class for LLM clients"""
    
    @abstractmethod
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        pass
    
    @abstractmethod
    def calculate_cost(self, token_usage: Dict[str, int], model: str) -> float:
        pass

class OpenAIClient(BaseLLMClient):
    """OpenAI API client implementation"""
    
    def __init__(self, api_key: str, organization: Optional[str] = None):
        self.client = openai.AsyncOpenAI(
            api_key=api_key,
            organization=organization
        )
        self.pricing = {
            "gpt-4": {"input": 0.03, "output": 0.06},
            "gpt-4-turbo": {"input": 0.01, "output": 0.03},
            "gpt-3.5-turbo": {"input": 0.001, "output": 0.002},
            "gpt-3.5-turbo-16k": {"input": 0.003, "output": 0.004}
        }
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate response using OpenAI API"""
        start_time = time.time()
        
        try:
            messages = []
            
            # Add system prompt if provided
            if request.system_prompt:
                messages.append({
                    "role": "system",
                    "content": request.system_prompt
                })
            
            # Add main prompt
            messages.append({
                "role": "user",
                "content": request.prompt
            })
            
            # Make API call
            response = await self.client.chat.completions.create(
                model=request.model,
                messages=messages,
                temperature=request.config.temperature,
                max_tokens=request.config.max_tokens,
                top_p=request.config.top_p,
                frequency_penalty=request.config.frequency_penalty,
                presence_penalty=request.config.presence_penalty,
                stop=request.config.stop_sequences
            )
            
            latency = time.time() - start_time
            token_usage = {
                "prompt_tokens": response.usage.prompt_tokens,
                "completion_tokens": response.usage.completion_tokens,
                "total_tokens": response.usage.total_tokens
            }
            
            cost = self.calculate_cost(token_usage, request.model)
            
            return GenerationResponse(
                content=response.choices[0].message.content,
                model=request.model,
                provider=ModelProvider.OPENAI,
                token_usage=token_usage,
                latency=latency,
                cost=cost
            )
            
        except Exception as e:
            logger.error(f"OpenAI generation failed: {e}")
            raise
    
    def calculate_cost(self, token_usage: Dict[str, int], model: str) -> float:
        """Calculate cost based on token usage"""
        pricing = self.pricing.get(model, self.pricing["gpt-3.5-turbo"])
        
        input_cost = token_usage["prompt_tokens"] * pricing["input"] / 1000
        output_cost = token_usage["completion_tokens"] * pricing["output"] / 1000
        
        return input_cost + output_cost

class AnthropicClient(BaseLLMClient):
    """Anthropic Claude API client implementation"""
    
    def __init__(self, api_key: str):
        self.client = anthropic.AsyncAnthropic(api_key=api_key)
        self.pricing = {
            "claude-3-opus-20240229": {"input": 0.015, "output": 0.075},
            "claude-3-sonnet-20240229": {"input": 0.003, "output": 0.015},
            "claude-3-haiku-20240307": {"input": 0.00025, "output": 0.00125}
        }
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate response using Anthropic API"""
        start_time = time.time()
        
        try:
            # Prepare prompt for Claude
            full_prompt = request.prompt
            if request.system_prompt:
                full_prompt = f"{request.system_prompt}\n\n{request.prompt}"
            
            # Make API call
            response = await self.client.messages.create(
                model=request.model,
                max_tokens=request.config.max_tokens,
                temperature=request.config.temperature,
                top_p=request.config.top_p,
                stop_sequences=request.config.stop_sequences or [],
                messages=[
                    {"role": "user", "content": full_prompt}
                ]
            )
            
            latency = time.time() - start_time
            token_usage = {
                "prompt_tokens": response.usage.input_tokens,
                "completion_tokens": response.usage.output_tokens,
                "total_tokens": response.usage.input_tokens + response.usage.output_tokens
            }
            
            cost = self.calculate_cost(token_usage, request.model)
            
            return GenerationResponse(
                content=response.content[0].text,
                model=request.model,
                provider=ModelProvider.ANTHROPIC,
                token_usage=token_usage,
                latency=latency,
                cost=cost
            )
            
        except Exception as e:
            logger.error(f"Anthropic generation failed: {e}")
            raise
    
    def calculate_cost(self, token_usage: Dict[str, int], model: str) -> float:
        """Calculate cost based on token usage"""
        pricing = self.pricing.get(model, self.pricing["claude-3-haiku-20240307"])
        
        input_cost = token_usage["prompt_tokens"] * pricing["input"] / 1000
        output_cost = token_usage["completion_tokens"] * pricing["output"] / 1000
        
        return input_cost + output_cost

class LLMClientManager:
    """Manages multiple LLM clients with fallback and load balancing"""
    
    def __init__(self):
        self.clients: Dict[ModelProvider, BaseLLMClient] = {}
        self.model_mapping: Dict[str, ModelProvider] = {}
        self.fallback_order = [ModelProvider.OPENAI, ModelProvider.ANTHROPIC]
    
    def register_client(self, provider: ModelProvider, client: BaseLLMClient):
        """Register an LLM client"""
        self.clients[provider] = client
        logger.info(f"Registered {provider.value} client")
    
    def register_model(self, model_name: str, provider: ModelProvider):
        """Register a model with its provider"""
        self.model_mapping[model_name] = provider
    
    async def generate(self, request: GenerationRequest) -> GenerationResponse:
        """Generate response with fallback support"""
        provider = self.model_mapping.get(request.model)
        
        if not provider:
            raise ValueError(f"Unknown model: {request.model}")
        
        # Try primary provider
        if provider in self.clients:
            try:
                return await self.clients[provider].generate(request)
            except Exception as e:
                logger.warning(f"Primary provider {provider.value} failed: {e}")
        
        # Try fallback providers
        for fallback_provider in self.fallback_order:
            if fallback_provider != provider and fallback_provider in self.clients:
                try:
                    logger.info(f"Trying fallback provider: {fallback_provider.value}")
                    fallback_request = self._adapt_request_for_provider(request, fallback_provider)
                    return await self.clients[fallback_provider].generate(fallback_request)
                except Exception as e:
                    logger.warning(f"Fallback provider {fallback_provider.value} failed: {e}")
                    continue
        
        raise Exception("All LLM providers failed")
    
    def _adapt_request_for_provider(self, request: GenerationRequest, 
                                   provider: ModelProvider) -> GenerationRequest:
        """Adapt request for different provider"""
        # Map models between providers
        model_mappings = {
            ModelProvider.OPENAI: {
                "claude-3-opus-20240229": "gpt-4",
                "claude-3-sonnet-20240229": "gpt-3.5-turbo-16k",
                "claude-3-haiku-20240307": "gpt-3.5-turbo"
            },
            ModelProvider.ANTHROPIC: {
                "gpt-4": "claude-3-opus-20240229",
                "gpt-3.5-turbo": "claude-3-haiku-20240307",
                "gpt-3.5-turbo-16k": "claude-3-sonnet-20240229"
            }
        }
        
        adapted_model = model_mappings.get(provider, {}).get(request.model, request.model)
        
        return GenerationRequest(
            prompt=request.prompt,
            config=request.config,
            model=adapted_model,
            system_prompt=request.system_prompt,
            context=request.context
        )

# generation/prompt_engineer.py
from typing import Dict, Any, List, Optional
from dataclasses import dataclass
from enum import Enum
import json
import logging

logger = logging.getLogger(__name__)

class RecommendationType(Enum):
    PRODUCT_DISCOVERY = "product_discovery"
    SIMILAR_PRODUCTS = "similar_products"
    PERSONALIZED_SUGGESTIONS = "personalized_suggestions"
    TRENDING_ITEMS = "trending_items"
    BUNDLE_RECOMMENDATIONS = "bundle_recommendations"
    REORDER_SUGGESTIONS = "reorder_suggestions"

class PersonalityType(Enum):
    PROFESSIONAL = "professional"
    FRIENDLY = "friendly"
    ENTHUSIASTIC = "enthusiastic"
    MINIMALIST = "minimalist"
    EXPERT = "expert"

@dataclass
class PromptTemplate:
    template: str
    required_variables: List[str]
    optional_variables: List[str]
    output_format: str
    personality: PersonalityType

class PromptEngineer:
    """Advanced prompt engineering for e-commerce recommendations"""
    
    def __init__(self):
        self.templates = self._initialize_templates()
        self.personality_traits = self._initialize_personalities()
    
    def create_recommendation_prompt(self, 
                                   recommendation_type: RecommendationType,
                                   user_context: Dict[str, Any],
                                   products: List[Dict[str, Any]],
                                   personality: PersonalityType = PersonalityType.FRIENDLY,
                                   additional_context: Optional[Dict[str, Any]] = None) -> str:
        """Create a comprehensive recommendation prompt"""
        
        template = self.templates[recommendation_type][personality]
        personality_traits = self.personality_traits[personality]
        
        # Prepare product information
        product_info = self._format_products(products)
        
        # Prepare user context
        user_info = self._format_user_context(user_context)
        
        # Prepare additional context
        context_info = self._format_additional_context(additional_context or {})
        
        # Build the prompt
        prompt = template.template.format(
            personality_traits=personality_traits,
            user_context=user_info,
            products=product_info,
            additional_context=context_info,
            output_format=template.output_format
        )
        
        return prompt
    
    def create_explanation_prompt(self,
                                 recommended_product: Dict[str, Any],
                                 user_context: Dict[str, Any],
                                 reasoning_factors: List[str],
                                 personality: PersonalityType = PersonalityType.FRIENDLY) -> str:
        """Create a prompt for generating explanations"""
        
        personality_traits = self.personality_traits[personality]
        
        prompt = f"""
{personality_traits}

I need you to explain why this product is recommended for this specific user.

USER PROFILE:
{self._format_user_context(user_context)}

RECOMMENDED PRODUCT:
{self._format_single_product(recommended_product)}

REASONING FACTORS:
{self._format_reasoning_factors(reasoning_factors)}

Please provide a personalized explanation that:
1. Highlights why this product matches their needs
2. Mentions specific benefits relevant to their profile
3. Addresses any potential concerns
4. Keeps the tone {personality.value}

Format your response as a natural, conversational explanation (2-3 sentences).
"""
        
        return prompt
    
    def create_comparison_prompt(self,
                               products: List[Dict[str, Any]],
                               comparison_criteria: List[str],
                               user_context: Dict[str, Any],
                               personality: PersonalityType = PersonalityType.EXPERT) -> str:
        """Create a prompt for product comparisons"""
        
        personality_traits = self.personality_traits[personality]
        
        prompt = f"""
{personality_traits}

I need you to compare these products for a specific user.

USER PROFILE:
{self._format_user_context(user_context)}

PRODUCTS TO COMPARE:
{self._format_products_for_comparison(products)}

COMPARISON CRITERIA:
{', '.join(comparison_criteria)}

Please provide a detailed comparison that:
1. Compares each product across the specified criteria
2. Highlights strengths and weaknesses of each
3. Provides a recommendation based on the user's profile
4. Uses a {personality.value} tone

Format your response as structured comparison with clear recommendations.
"""
        
        return prompt
    
    def _initialize_templates(self) -> Dict[RecommendationType, Dict[PersonalityType, PromptTemplate]]:
        """Initialize prompt templates for different recommendation types and personalities"""
        templates = {}
        
        # Product Discovery Templates
        templates[RecommendationType.PRODUCT_DISCOVERY] = {
            PersonalityType.FRIENDLY: PromptTemplate(
                template="""
{personality_traits}

I'm helping a customer discover new products they might love. Here's what I know about them:

USER PROFILE:
{user_context}

AVAILABLE PRODUCTS:
{products}

ADDITIONAL CONTEXT:
{additional_context}

Please recommend 3-5 products from the list that would be perfect for this user. For each recommendation:

1. **Product Name & Brief Description**
2. **Why it's perfect for them** (personalized reasoning)
3. **Key benefits** they'll appreciate
4. **Price & value consideration**

{output_format}
""",
                required_variables=["user_context", "products"],
                optional_variables=["additional_context"],
                output_format="JSON format with product recommendations",
                personality=PersonalityType.FRIENDLY
            ),
            
            PersonalityType.PROFESSIONAL: PromptTemplate(
                template="""
{personality_traits}

Based on the user profile and available product catalog, I need to provide data-driven product recommendations.

USER ANALYSIS:
{user_context}

PRODUCT CATALOG:
{products}

CONTEXTUAL FACTORS:
{additional_context}

Please provide analytical product recommendations with:

1. **Product Selection Rationale**
2. **Compatibility Assessment**
3. **Value Proposition Analysis**
4. **Risk/Benefit Evaluation**

{output_format}
""",
                required_variables=["user_context", "products"],
                optional_variables=["additional_context"],
                output_format="Structured analytical format",
                personality=PersonalityType.PROFESSIONAL
            )
        }
        
        # Similar Products Templates
        templates[RecommendationType.SIMILAR_PRODUCTS] = {
            PersonalityType.FRIENDLY: PromptTemplate(
                template="""
{personality_traits}

A customer is interested in similar products to what they're currently viewing. Help me suggest alternatives!

USER PROFILE:
{user_context}

SIMILAR PRODUCTS:
{products}

Context: {additional_context}

Please recommend the best alternatives, explaining:
1. **How each product is similar** to their interest
2. **What makes each unique** or better
3. **Price comparison** and value
4. **Why they might prefer** each option

{output_format}
""",
                required_variables=["user_context", "products"],
                optional_variables=["additional_context"],
                output_format="Friendly recommendation list",
                personality=PersonalityType.FRIENDLY
            )
        }
        
        # Add more templates for other recommendation types...
        
        return templates
    
    def _initialize_personalities(self) -> Dict[PersonalityType, str]:
        """Initialize personality traits for different tones"""
        return {
            PersonalityType.FRIENDLY: """
You are a helpful, warm, and enthusiastic shopping assistant. You love helping people find products they'll genuinely enjoy. You speak conversationally, use occasional emojis, and always consider the human element in your recommendations. You're knowledgeable but not overwhelming, and you always keep the customer's best interests at heart.
""",
            
            PersonalityType.PROFESSIONAL: """
You are a professional product consultant with deep expertise in e-commerce and consumer behavior. You provide data-driven, analytical recommendations backed by clear reasoning. Your tone is authoritative but approachable, and you focus on providing maximum value through informed decision-making.
""",
            
            PersonalityType.ENTHUSIASTIC: """
You are an excited product expert who absolutely loves discovering amazing finds for customers! You're genuinely thrilled about every recommendation and can't wait to share why each product is fantastic. Your enthusiasm is contagious, and you help customers get excited about their potential purchases.
""",
            
            PersonalityType.MINIMALIST: """
You are a concise, no-nonsense product advisor. You provide clear, direct recommendations without unnecessary fluff. Every word serves a purpose, and you focus on the most important information customers need to make decisions quickly and confidently.
""",
            
            PersonalityType.EXPERT: """
You are a seasoned industry expert with deep knowledge across multiple product categories. You provide detailed, technical insights while remaining accessible. Your recommendations are backed by expertise, and you help customers understand the nuances that matter most.
"""
        }
    
    def _format_products(self, products: List[Dict[str, Any]]) -> str:
        """Format products for prompt inclusion"""
        formatted_products = []
        
        for i, product in enumerate(products, 1):
            payload = product.get('payload', product)
            
            product_text = f"""
Product {i}:
- ID: {payload.get('id', 'N/A')}
- Title: {payload.get('title', 'N/A')}
- Category: {payload.get('category', 'N/A')}
- Brand: {payload.get('brand', 'N/A')}
- Price: ${payload.get('price', 'N/A')}
- Rating: {payload.get('rating', 'N/A')}/5 ({payload.get('review_count', 0)} reviews)
- Description: {payload.get('description', 'N/A')[:200]}...
"""
            if product.get('score'):
                product_text += f"- Relevance Score: {product['score']:.3f}\n"
            
            formatted_products.append(product_text)
        
        return "\n".join(formatted_products)
    
    def _format_user_context(self, user_context: Dict[str, Any]) -> str:
        """Format user context for prompt inclusion"""
        context_parts = []
        
        # Demographics
        demographics = user_context.get('demographics', {})
        if demographics:
            context_parts.append(f"Demographics: {demographics}")
        
        # Preferences
        preferences = user_context.get('preferences', {})
        if preferences:
            pref_text = []
            if preferences.get('preferred_categories'):
                pref_text.append(f"Preferred categories: {', '.join(preferences['preferred_categories'])}")
            if preferences.get('preferred_brands'):
                pref_text.append(f"Preferred brands: {', '.join(preferences['preferred_brands'])}")
            if preferences.get('price_sensitivity'):
                price_level = "budget-conscious" if preferences['price_sensitivity'] < 0.3 else \
                             "premium-oriented" if preferences['price_sensitivity'] > 0.7 else "balanced"
                pref_text.append(f"Price preference: {price_level}")
            
            if pref_text:
                context_parts.append("Preferences: " + "; ".join(pref_text))
        
        # Behavioral patterns
        behavioral = user_context.get('behavioral_features', {})
        if behavioral:
            behavior_text = []
            purchase_freq = behavioral.get('purchase_frequency', 0)
            if purchase_freq > 10:
                behavior_text.append("frequent buyer")
            elif purchase_freq > 5:
                behavior_text.append("regular customer")
            else:
                behavior_text.append("occasional shopper")
            
            if behavioral.get('avg_session_duration', 0) > 300:
                behavior_text.append("detailed researcher")
            
            if behavior_text:
                context_parts.append("Shopping behavior: " + ", ".join(behavior_text))
        
        # Purchase history summary
        if user_context.get('recent_purchases'):
            recent = user_context['recent_purchases'][:3]  # Last 3 purchases
            context_parts.append(f"Recent purchases: {', '.join([p.get('title', 'Unknown') for p in recent])}")
        
        return "\n".join(context_parts) if context_parts else "New customer with limited profile data"
    
    def _format_additional_context(self, context: Dict[str, Any]) -> str:
        """Format additional context information"""
        context_parts = []
        
        if context.get('season'):
            context_parts.append(f"Season: {context['season']}")
        
        if context.get('current_trends'):
            context_parts.append(f"Current trends: {', '.join(context['current_trends'])}")
        
        if context.get('inventory_status'):
            context_parts.append(f"Inventory considerations: {context['inventory_status']}")
        
        if context.get('promotion_info'):
            context_parts.append(f"Active promotions: {context['promotion_info']}")
        
        return "\n".join(context_parts) if context_parts else "No additional context"
    
    def _format_single_product(self, product: Dict[str, Any]) -> str:
        """Format a single product for detailed analysis"""
        payload = product.get('payload', product)
        
        return f"""
Title: {payload.get('title', 'N/A')}
Category: {payload.get('category', 'N/A')}
Brand: {payload.get('brand', 'N/A')}
Price: ${payload.get('price', 'N/A')}
Rating: {payload.get('rating', 'N/A')}/5 ({payload.get('review_count', 0)} reviews)
Description: {payload.get('description', 'N/A')}
Key Features: {', '.join(payload.get('features', []))}
"""
    
    def _format_reasoning_factors(self, factors: List[str]) -> str:
        """Format reasoning factors for explanation"""
        return "\n".join([f"- {factor}" for factor in factors])
    
    def _format_products_for_comparison(self, products: List[Dict[str, Any]]) -> str:
        """Format products specifically for comparison"""
        formatted = []
        
        for i, product in enumerate(products, 1):
            payload = product.get('payload', product)
            formatted.append(f"""
Option {i}: {payload.get('title', 'N/A')}
- Brand: {payload.get('brand', 'N/A')}
- Price: ${payload.get('price', 'N/A')}
- Rating: {payload.get('rating', 'N/A')}/5
- Key Features: {', '.join(payload.get('features', [])[:3])}

import numpy as np
from typing import Dict, Any, List, Optional, Tuple
from dataclasses import dataclass
from enum import Enum
import logging
from datetime import datetime, timedelta
import json

logger = logging.getLogger(__name__)

class PersonalizationStrategy(Enum):
    COLLABORATIVE_FILTERING = "collaborative_filtering"
    CONTENT_BASED = "content_based"
    HYBRID = "hybrid"
    CONTEXTUAL = "contextual"
    DEEP_LEARNING = "deep_learning"

@dataclass
class PersonalizationConfig:
    strategy: PersonalizationStrategy
    weights: Dict[str, float]
    min_confidence: float = 0.5
    diversity_factor: float = 0.3
    novelty_factor: float = 0.2
    serendipity_factor: float = 0.1

@dataclass
class PersonalizedRecommendation:
    product_id: str
    score: float
    reasoning: List[str]
    confidence: float
    personalization_factors: Dict[str, float]
    explanation: str
    metadata: Dict[str, Any]

class PersonalizationEngine:
    """Advanced personalization engine for recommendation generation"""
    
    def __init__(self, config: PersonalizationConfig):
        self.config = config
        self.user_profiles = {}
        self.item_profiles = {}
        self.interaction_matrix = {}
        
    async def personalize_recommendations(self,
                                        user_id: str,
                                        candidate_products: List[Dict[str, Any]],
                                        user_context: Dict[str, Any],
                                        additional_context: Optional[Dict[str, Any]] = None) -> List[PersonalizedRecommendation]:
        """Generate personalized recommendations for a specific user"""
        
        # Get user profile
        user_profile = await self._get_user_profile(user_id, user_context)
        
        # Calculate personalization scores for each candidate
        personalized_products = []
        
        for product in candidate_products:
            personalization_score = await self._calculate_personalization_score(
                user_profile, product, additional_context or {}
            )
            
            if personalization_score.confidence >= self.config.min_confidence:
                personalized_products.append(personalization_score)
        
        # Apply diversity and novelty filters
        final_recommendations = self._apply_diversification(
            personalized_products, user_profile
        )
        
        # Sort by personalized score
        final_recommendations.sort(key=lambda x: x.score, reverse=True)
        
        return final_recommendations
    
    async def _get_user_profile(self, user_id: str, context: Dict[str, Any]) -> Dict[str, Any]:
        """Get or create comprehensive user profile"""
        
        if user_id in self.user_profiles:
            profile = self.user_profiles[user_id]
            # Update with latest context
            profile.update(context)
        else:
            # Create new profile
            profile = {
                'user_id': user_id,
                'preferences': context.get('preferences', {}),
                'behavioral_features': context.get('behavioral_features', {}),
                'demographic_features': context.get('demographics', {}),
                'interaction_history': context.get('interaction_history', []),
                'purchase_history': context.get('purchase_history', []),
                'created_at': datetime.utcnow(),
                'last_updated': datetime.utcnow()
            }
            self.user_profiles[user_id] = profile
        
        # Enrich profile with derived features
        profile = await self._enrich_user_profile(profile)
        
        return profile
    
    async def _enrich_user_profile(self, profile: Dict[str, Any]) -> Dict[str, Any]:
        """Enrich user profile with derived features"""
        
        # Calculate user preferences vector
        profile['preference_vector'] = self._calculate_preference_vector(profile)
        
        # Calculate user clusters/segments
        profile['user_segment'] = self._determine_user_segment(profile)
        
        # Calculate temporal patterns
        profile['temporal_patterns'] = self._analyze_temporal_patterns(profile)
        
        # Calculate price sensitivity
        profile['price_sensitivity'] = self._calculate_price_sensitivity(profile)
        
        # Calculate brand affinity
        profile['brand_affinity'] = self._calculate_brand_affinity(profile)
        
        return profile
    
    async def _calculate_personalization_score(self,
                                             user_profile: Dict[str, Any],
                                             product: Dict[str, Any],
                                             context: Dict[str, Any]) -> PersonalizedRecommendation:
        """Calculate comprehensive personalization score"""
        
        product_payload = product.get('payload', product)
        base_score = product.get('score', 0.5)
        
        # Calculate different personalization factors
        factors = {}
        reasoning = []
        
        # 1. Content-based similarity
        content_score = self._calculate_content_similarity(user_profile, product_payload)
        factors['content_similarity'] = content_score
        if content_score > 0.7:
            reasoning.append(f"Matches your interest in {product_payload.get('category', 'this category')}")
        
        # 2. Collaborative filtering score
        collaborative_score = self._calculate_collaborative_score(user_profile, product_payload)
        factors['collaborative_filtering'] = collaborative_score
        if collaborative_score > 0.6:
            reasoning.append("Popular among users with similar preferences")
        
        # 3. Behavioral alignment
        behavioral_score = self._calculate_behavioral_alignment(user_profile, product_payload)
        factors['behavioral_alignment'] = behavioral_score
        if behavioral_score > 0.6:
            reasoning.append("Matches your shopping behavior patterns")
        
        # 4. Contextual relevance
        contextual_score = self._calculate_contextual_relevance(user_profile, product_payload, context)
        factors['contextual_relevance'] = contextual_score
        if contextual_score > 0.6:
            reasoning.append("Relevant to your current context")
        
        # 5. Price alignment
        price_score = self._calculate_price_alignment(user_profile, product_payload)
        factors['price_alignment'] = price_score
        if price_score > 0.7:
            reasoning.append("Within your preferred price range")
        
        # 6. Brand preference
        brand_score = self._calculate_brand_preference(user_profile, product_payload)
        factors['brand_preference'] = brand_score
        if brand_score > 0.7:
            reasoning.append(f"From {product_payload.get('brand', 'a brand')} you've shown interest in")
        
        # 7. Temporal relevance
        temporal_score = self._calculate_temporal_relevance(user_profile, product_payload, context)
        factors['temporal_relevance'] = temporal_score
        if temporal_score > 0.6:
            reasoning.append("Timely for your current needs")
        
        # 8. Social proof
        social_score = self._calculate_social_proof(product_payload)
        factors['social_proof'] = social_score
        if social_score > 0.8:
            reasoning.append("Highly rated by other customers")
        
        # Calculate weighted personalization score
        personalization_weights = {
            'content_similarity': 0.20,
            'collaborative_filtering': 0.15,
            'behavioral_alignment': 0.15,
            'contextual_relevance': 0.10,
            'price_alignment': 0.15,
            'brand_preference': 0.10,
            'temporal_relevance': 0.10,
            'social_proof': 0.05
        }
        
        personalization_score = sum(
            factors[factor] * weight 
            for factor, weight in personalization_weights.items()
        )
        
        # Combine with base relevance score
        final_score = 0.6 * base_score + 0.4 * personalization_score
        
        # Calculate confidence based on data availability
        confidence = self._calculate_confidence(user_profile, factors)
        
        # Generate explanation
        explanation = self._generate_explanation(reasoning, product_payload, user_profile)
        
        return PersonalizedRecommendation(
            product_id=product_payload.get('id'),
            score=final_score,
            reasoning=reasoning,
            confidence=confidence,
            personalization_factors=factors,
            explanation=explanation,
            metadata={
                'base_score': base_score,
                'personalization_score': personalization_score,
                'weights_used': personalization_weights
            }
        )
    
    def _calculate_preference_vector(self, profile: Dict[str, Any]) -> np.ndarray:
        """Calculate user preference vector from interaction history"""
        # Simplified implementation - in production, use more sophisticated methods
        categories = ['electronics', 'clothing', 'books', 'home', 'sports', 'beauty']
        vector = np.zeros(len(categories))
        
        purchase_history = profile.get('purchase_history', [])
        for purchase in purchase_history:
            category = purchase.get('category', '').lower()
            if category in categories:
                idx = categories.index(category)
                vector[idx] += 1
        
        # Normalize
        if np.sum(vector) > 0:
            vector = vector / np.sum(vector)
        
        return vector
    
    def _determine_user_segment(self, profile: Dict[str, Any]) -> str:
        """Determine user segment based on behavior and preferences"""
        behavioral = profile.get('behavioral_features', {})
        
        purchase_freq = behavioral.get('purchase_frequency', 0)
        avg_order_value = behavioral.get('avg_order_value', 0)
        session_duration = behavioral.get('avg_session_duration', 0)
        
        if purchase_freq > 10 and avg_order_value > 100:
            return "premium_frequent"
        elif purchase_freq > 10:
            return "frequent_budget"
        elif avg_order_value > 100:
            return "premium_occasional"
        elif session_duration > 300:
            return "researcher"
        else:
            return "casual"
    
    def _analyze_temporal_patterns(self, profile: Dict[str, Any]) -> Dict[str, Any]:
        """Analyze user's temporal shopping patterns"""
        # Simplified implementation
        return {
            'preferred_shopping_time': 'evening',
            'seasonal_preferences': ['winter_clothing', 'summer_outdoor'],
            'frequency_pattern': 'monthly'
        }
    
    def _calculate_price_sensitivity(self, profile: Dict[str, Any]) -> float:
        """Calculate user's price sensitivity"""
        purchase_history = profile.get('purchase_history', [])
        if not purchase_history:
            return 0.5  # Default neutral sensitivity
        
        prices = [p.get('price', 0) for p in purchase_history]
        avg_price = np.mean(prices) if prices else 0
        
        # Normalize to 0-1 scale (simplified)
        return min(avg_price / 200, 1.0)  # Assuming 200 is high price threshold
    
    def _calculate_brand_affinity(self, profile: Dict[str, Any]) -> Dict[str, float]:
        """Calculate user's brand preferences"""
        purchase_history = profile.get('purchase_history', [])
        brand_counts = {}
        
        for purchase in purchase_history:
            brand = purchase.get('brand', 'unknown')
            brand_counts[brand] = brand_counts.get(brand, 0) + 1
        
        total_purchases = len(purchase_history)
        brand_affinity = {}
        
        for brand, count in brand_counts.items():
            brand_affinity[brand] = count / total_purchases if total_purchases > 0 else 0
        
        return brand_affinity
    
    def _calculate_content_similarity(self, user_profile: Dict[str, Any], 
                                    product: Dict[str, Any]) -> float:
        """Calculate content-based similarity"""
        user_preferences = user_profile.get('preferences', {})
        
        score = 0.0
        
        # Category preference
        preferred_categories = user_preferences.get('preferred_categories', [])
        product_category = product.get('category', '')
        if product_category in preferred_categories:
            score += 0.4
        
        # Brand preference
        brand_affinity = user_profile.get('brand_affinity', {})
        product_brand = product.get('brand', '')
        brand_score = brand_affinity.get(product_brand, 0)
        score += 0.3 * brand_score
        
        # Price alignment
        price_sensitivity = user_profile.get('price_sensitivity', 0.5)
        product_price = product.get('price', 0)
        # Simplified price scoring
        if product_price < 50 and price_sensitivity < 0.5:
            score += 0.3
        elif product_price > 100 and price_sensitivity > 0.7:
            score += 0.3
        elif 50 <= product_price <= 100:
            score += 0.2
        
        return min(score, 1.0)
    
    def _calculate_collaborative_score(self, user_profile: Dict[str, Any],
                                     product: Dict[str, Any]) -> float:
        """Calculate collaborative filtering score"""
        # Simplified implementation - in production, use matrix factorization
        user_segment = user_profile.get('user_segment', 'casual')
        
        # Mock collaborative scores based on user segment
        collaborative_scores = {
            'premium_frequent': 0.8,

            'premium_occasional': 0.7,
            'researcher': 0.65,
            'casual': 0.5
        }
        
        return collaborative_scores.get(user_segment, 0.5)
    
    def _calculate_behavioral_alignment(self, user_profile: Dict[str, Any],
                                      product: Dict[str, Any]) -> float:
        """Calculate behavioral pattern alignment"""
        behavioral = user_profile.get('behavioral_features', {})
        
        # Shopping frequency alignment
        purchase_freq = behavioral.get('purchase_frequency', 0)
        product_type = product.get('category', '').lower()
        
        if purchase_freq > 10 and product_type in ['consumables', 'electronics']:
            return 0.8
        elif purchase_freq < 5 and product_type in ['luxury', 'furniture']:
            return 0.7
        else:
            return 0.5
    
    def _calculate_contextual_relevance(self, user_profile: Dict[str, Any],
                                      product: Dict[str, Any],
                                      context: Dict[str, Any]) -> float:
        """Calculate contextual relevance score"""
        score = 0.5  # Base score
        
        # Seasonal relevance
        current_season = context.get('season', 'unknown')
        product_category = product.get('category', '').lower()
        
        seasonal_relevance = {
            'winter': ['coats', 'heating', 'warm_clothing'],
            'summer': ['swimwear', 'outdoor', 'cooling'],
            'spring': ['gardening', 'light_clothing'],
            'fall': ['jackets', 'school_supplies']
        }
        
        if current_season in seasonal_relevance:
            relevant_categories = seasonal_relevance[current_season]
            if any(cat in product_category for cat in relevant_categories):
                score += 0.3
        
        # Time-based relevance
        current_hour = datetime.now().hour
        if 9 <= current_hour <= 17 and 'office' in product_category:
            score += 0.2
        elif 18 <= current_hour <= 22 and 'entertainment' in product_category:
            score += 0.2
        
        return min(score, 1.0)
    
    def _calculate_price_alignment(self, user_profile: Dict[str, Any],
                                 product: Dict[str, Any]) -> float:
        """Calculate price alignment with user preferences"""
        user_price_sensitivity = user_profile.get('price_sensitivity', 0.5)
        product_price = product.get('price', 0)
        
        # Calculate price score based on user's typical spending
        purchase_history = user_profile.get('purchase_history', [])
        if purchase_history:
            avg_user_price = np.mean([p.get('price', 0) for p in purchase_history])
        else:
            avg_user_price = 50  # Default assumption
        
        price_ratio = product_price / avg_user_price if avg_user_price > 0 else 1
        
        if user_price_sensitivity < 0.3:  # Budget-conscious
            return max(0, 1 - (price_ratio - 0.8) / 0.5) if price_ratio > 0.8 else 1.0
        elif user_price_sensitivity > 0.7:  # Premium-oriented
            return min(1, price_ratio / 1.5) if price_ratio < 1.5 else 1.0
        else:  # Balanced
            return max(0, 1 - abs(price_ratio - 1) / 0.5)
    
    def _calculate_brand_preference(self, user_profile: Dict[str, Any],
                                  product: Dict[str, Any]) -> float:
        """Calculate brand preference alignment"""
        brand_affinity = user_profile.get('brand_affinity', {})
        product_brand = product.get('brand', '')
        
        return brand_affinity.get(product_brand, 0.3)  # Default slight preference
    
    def _calculate_temporal_relevance(self, user_profile: Dict[str, Any],
                                    product: Dict[str, Any],
                                    context: Dict[str, Any]) -> float:
        """Calculate temporal relevance"""
        temporal_patterns = user_profile.get('temporal_patterns', {})
        
        # Check if product aligns with user's shopping patterns
        current_time = datetime.now().hour
        preferred_time = temporal_patterns.get('preferred_shopping_time', 'evening')
        
        if preferred_time == 'morning' and 6 <= current_time <= 12:
            return 0.8
        elif preferred_time == 'evening' and 18 <= current_time <= 23:
            return 0.8
        else:
            return 0.5
    
    def _calculate_social_proof(self, product: Dict[str, Any]) -> float:
        """Calculate social proof score"""
        rating = product.get('rating', 0)
        review_count = product.get('review_count', 0)
        
        # Normalize rating (0-5 to 0-1)
        rating_score = rating / 5 if rating > 0 else 0
        
        # Review count bonus (logarithmic scale)
        review_bonus = min(np.log(review_count + 1) / np.log(1000), 0.3) if review_count > 0 else 0
        
        return min(rating_score + review_bonus, 1.0)
    
    def _calculate_confidence(self, user_profile: Dict[str, Any], 
                            factors: Dict[str, float]) -> float:
        """Calculate confidence in personalization"""
        # Base confidence on amount of user data available
        data_points = 0
        
        if user_profile.get('purchase_history'):
            data_points += len(user_profile['purchase_history'])
        
        if user_profile.get('interaction_history'):
            data_points += min(len(user_profile['interaction_history']), 50)
        
        if user_profile.get('preferences'):
            data_points += 10  # Explicit preferences are valuable
        
        # Normalize to 0-1 scale
        data_confidence = min(data_points / 100, 1.0)
        
        # Factor in score variance (more consistent scores = higher confidence)
        score_variance = np.var(list(factors.values()))
        variance_penalty = min(score_variance, 0.3)
        
        return max(data_confidence - variance_penalty, 0.1)
    
    def _generate_explanation(self, reasoning: List[str], 
                            product: Dict[str, Any],
                            user_profile: Dict[str, Any]) -> str:
        """Generate human-readable explanation"""
        if not reasoning:
            return f"This {product.get('category', 'product')} might interest you"
        
        primary_reason = reasoning[0]
        
        if len(reasoning) == 1:
            return primary_reason
        elif len(reasoning) == 2:
            return f"{primary_reason} and {reasoning[1].lower()}"
        else:
            return f"{primary_reason}, {reasoning[1].lower()}, and {len(reasoning)-2} other factors"
    
    def _apply_diversification(self, recommendations: List[PersonalizedRecommendation],
                             user_profile: Dict[str, Any]) -> List[PersonalizedRecommendation]:
        """Apply diversification to avoid echo chamber effect"""
        if len(recommendations) <= 3:
            return recommendations
        
        diversified = []
        category_counts = {}
        brand_counts = {}
        
        # Sort by score first
        recommendations.sort(key=lambda x: x.score, reverse=True)
        
        for rec in recommendations:
            # Get product info (simplified access)
            category = rec.metadata.get('category', 'unknown')
            brand = rec.metadata.get('brand', 'unknown')
            
            # Apply diversity constraints
            max_per_category = max(len(recommendations) // 4, 2)
            max_per_brand = max(len(recommendations) // 6, 1)
            
            if (category_counts.get(category, 0) < max_per_category and
                brand_counts.get(brand, 0) < max_per_brand):
                
                diversified.append(rec)
                category_counts[category] = category_counts.get(category, 0) + 1
                brand_counts[brand] = brand_counts.get(brand, 0) + 1
        
        return diversified
