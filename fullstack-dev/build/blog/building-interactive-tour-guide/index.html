Creating an effective interactive tour guide is crucial for helping users discover and adopt new features in your application. A well-designed tour can significantly improve user onboarding, feature adoption rates, and overall user satisfaction. In this comprehensive guide, we'll explore how to build a flexible, accessible, and engaging tour system that can highlight new features and guide users through complex workflows.

Tour System Architectureâ€‹

Let's build a tour system that's flexible, reusable, and easy to maintain. Our architecture will consist of several key components:

// Types for our tour system
interface TourStep {
  id: string;
  target: string; // CSS selector or element ID
  title: string;
  content: string;
  placement?: 'top' | 'bottom' | 'left' | 'right' | 'center';
  action?: 'click' | 'hover' | 'focus' | 'none';
  beforeShow?: () => Promise<void> | void;
  afterShow?: () => Promise<void> | void;
  beforeHide?: () => Promise<void> | void;
  skipable?: boolean;
  highlightPadding?: number;
  zIndex?: number;
}

interface Tour {
  id: string;
  name: string;
  description: string;
  steps: TourStep[];
  triggerConditions?: TriggerCondition[];
  userSegments?: string[];
  isActive: boolean;
  priority: number;
}

interface TourState {
  currentTour: Tour | null;
  currentStepIndex: number;
  isPlaying: boolean;
  completedTours: string[];
  skippedTours: string[];
}

Core Tour Engine Implementationâ€‹

Let's start with the main tour engine that will manage tour execution:

// hooks/useTourEngine.ts
import { useState, useEffect, useCallback, useRef } from 'react';
import { useLocalStorage } from './useLocalStorage';

export const useTourEngine = () => {
  const [tourState, setTourState] = useState<TourState>({
    currentTour: null,
    currentStepIndex: 0,
    isPlaying: false,
    completedTours: [],
    skippedTours: [],
  });

  const [completedTours, setCompletedTours] = useLocalStorage<string[]>('tour-completed', []);
  const [skippedTours, setSkippedTours] = useLocalStorage<string[]>('tour-skipped', []);
  
  const overlayRef = useRef<HTMLDivElement | null>(null);
  const highlightRef = useRef<HTMLDivElement | null>(null);

  // Start a tour
  const startTour = useCallback(async (tour: Tour) => {
    if (completedTours.includes(tour.id) || skippedTours.includes(tour.id)) {
      return;
    }

    setTourState({
      currentTour: tour,
      currentStepIndex: 0,
      isPlaying: true,
      completedTours,
      skippedTours,
    });

    // Execute beforeShow hook for first step
    if (tour.steps[0]?.beforeShow) {
      await tour.steps[0].beforeShow();
    }
  }, [completedTours, skippedTours]);

  // Navigate to next step
  const nextStep = useCallback(async () => {
    if (!tourState.currentTour) return;

    const currentStep = tourState.currentTour.steps[tourState.currentStepIndex];
    
    // Execute beforeHide hook
    if (currentStep?.beforeHide) {
      await currentStep.beforeHide();
    }

    const nextIndex = tourState.currentStepIndex + 1;
    
    if (nextIndex >= tourState.currentTour.steps.length) {
      // Tour completed
      completeTour();
      return;
    }

    setTourState(prev => ({
      ...prev,
      currentStepIndex: nextIndex,
    }));

    // Execute beforeShow hook for next step
    const nextStep = tourState.currentTour.steps[nextIndex];
    if (nextStep?.beforeShow) {
      await nextStep.beforeShow();
    }
  }, [tourState]);

  // Navigate to previous step
  const previousStep = useCallback(async () => {
    if (!tourState.currentTour || tourState.currentStepIndex === 0) return;

    const currentStep = tourState.currentTour.steps[tourState.currentStepIndex];
    
    if (currentStep?.beforeHide) {
      await currentStep.beforeHide();
    }

    const prevIndex = tourState.currentStepIndex - 1;
    setTourState(prev => ({
      ...prev,
      currentStepIndex: prevIndex,
    }));

    const prevStep = tourState.currentTour.steps[prevIndex];
    if (prevStep?.beforeShow) {
      await prevStep.beforeShow();
    }
  }, [tourState]);

  // Complete tour
  const completeTour = useCallback(() => {
    if (!tourState.currentTour) return;

    const newCompleted = [...completedTours, tourState.currentTour.id];
    setCompletedTours(newCompleted);
    
    setTourState(prev => ({
      ...prev,
      currentTour: null,
      currentStepIndex: 0,
      isPlaying: false,
      completedTours: newCompleted,
    }));
  }, [tourState.currentTour, completedTours, setCompletedTours]);

  // Skip tour
  const skipTour = useCallback(() => {
    if (!tourState.currentTour) return;

    const newSkipped = [...skippedTours, tourState.currentTour.id];
    setSkippedTours(newSkipped);
    
    setTourState(prev => ({
      ...prev,
      currentTour: null,
      currentStepIndex: 0,
      isPlaying: false,
      skippedTours: newSkipped,
    }));
  }, [tourState.currentTour, skippedTours, setSkippedTours]);

  // Get current step
  const getCurrentStep = useCallback((): TourStep | null => {
    if (!tourState.currentTour) return null;
    return tourState.currentTour.steps[tourState.currentStepIndex] || null;
  }, [tourState]);

  return {
    tourState,
    startTour,
    nextStep,
    previousStep,
    completeTour,
    skipTour,
    getCurrentStep,
  };
};

Tour Overlay Componentâ€‹

The overlay component creates the spotlight effect and manages the visual presentation:

// components/TourOverlay/TourOverlay.tsx
import React, { useEffect, useState, useRef } from 'react';
import { createPortal } from 'react-dom';
import { TourStep } from '../../types/tour.types';
import { TourTooltip } from './TourTooltip';

interface TourOverlayProps {
  step: TourStep;
  onNext: () => void;
  onPrevious: () => void;
  onSkip: () => void;
  onComplete: () => void;
  currentStepIndex: number;
  totalSteps: number;
  isFirstStep: boolean;
  isLastStep: boolean;
}

export const TourOverlay: React.FC<TourOverlayProps> = ({
  step,
  onNext,
  onPrevious,
  onSkip,
  onComplete,
  currentStepIndex,
  totalSteps,
  isFirstStep,
  isLastStep,
}) => {
  const [targetElement, setTargetElement] = useState<HTMLElement | null>(null);
  const [targetRect, setTargetRect] = useState<DOMRect | null>(null);
  const [tooltipPosition, setTooltipPosition] = useState({ x: 0, y: 0 });
  const overlayRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const element = document.querySelector(step.target) as HTMLElement;
    if (element) {
      setTargetElement(element);
      updatePositions(element);
      
      // Scroll element into view
      element.scrollIntoView({
        behavior: 'smooth',
        block: 'center',
        inline: 'center',
      });

      // Add highlight class to target element
      element.classList.add('tour-highlight');
      
      // Execute step action
      if (step.action === 'click') {
        element.click();
      } else if (step.action === 'focus') {
        element.focus();
      } else if (step.action === 'hover') {
        element.dispatchEvent(new MouseEvent('mouseenter', { bubbles: true }));
      }
    }

    return () => {
      if (element) {
        element.classList.remove('tour-highlight');
      }
    };
  }, [step]);

  useEffect(() => {
    const handleResize = () => {
      if (targetElement) {
        updatePositions(targetElement);
      }
    };

    const handleScroll = () => {
      if (targetElement) {
        updatePositions(targetElement);
      }
    };

    window.addEventListener('resize', handleResize);
    window.addEventListener('scroll', handleScroll, true);

    return () => {
      window.removeEventListener('resize', handleResize);
      window.removeEventListener('scroll', handleScroll, true);
    };
  }, [targetElement]);

  const updatePositions = (element: HTMLElement) => {
    const rect = element.getBoundingClientRect();
    setTargetRect(rect);

    // Calculate tooltip position based on placement preference
    const placement = step.placement || 'bottom';
    const padding = step.highlightPadding || 8;
    
    let tooltipX = 0;
    let tooltipY = 0;

    switch (placement) {
      case 'top':
        tooltipX = rect.left + rect.width / 2;
        tooltipY = rect.top - padding;
        break;
      case 'bottom':
        tooltipX = rect.left + rect.width / 2;
        tooltipY = rect.bottom + padding;
        break;
      case 'left':
        tooltipX = rect.left - padding;
        tooltipY = rect.top + rect.height / 2;
        break;
      case 'right':
        tooltipX = rect.right + padding;
        tooltipY = rect.top + rect.height / 2;
        break;
      case 'center':
        tooltipX = window.innerWidth / 2;
        tooltipY = window.innerHeight / 2;
        break;
    }

    setTooltipPosition({ x: tooltipX, y: tooltipY });
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    switch (e.key) {
      case 'Escape':
        onSkip();
        break;
      case 'ArrowRight':
      case 'Enter':
      case ' ':
        e.preventDefault();
        if (isLastStep) {
          onComplete();
        } else {
          onNext();
        }
        break;
      case 'ArrowLeft':
        e.preventDefault();
        if (!isFirstStep) {
          onPrevious();
        }
        break;
    }
  };

  if (!targetRect) return null;

  const overlayStyle: React.CSSProperties = {
    position: 'fixed',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0,
    backgroundColor: 'rgba(0, 0, 0, 0.5)',
    zIndex: step.zIndex || 1000,
    pointerEvents: 'auto',
  };

  const spotlightStyle: React.CSSProperties = {
    position: 'absolute',
    top: targetRect.top - (step.highlightPadding || 8),
    left: targetRect.left - (step.highlightPadding || 8),
    width: targetRect.width + (step.highlightPadding || 8) * 2,
    height: targetRect.height + (step.highlightPadding || 8) * 2,
    boxShadow: `0 0 0 9999px rgba(0, 0, 0, 0.5)`,
    borderRadius: '8px',
    pointerEvents: 'none',
  };

  return createPortal(
    <div
      ref={overlayRef}
      style={overlayStyle}
      onKeyDown={handleKeyDown}
      tabIndex={-1}
      role="dialog"
      aria-modal="true"
      aria-labelledby="tour-title"
      aria-describedby="tour-content"
    >
      {/* Spotlight effect */}
      <div style={spotlightStyle} />
      
      {/* Tooltip */}
      <TourTooltip
        step={step}
        position={tooltipPosition}
        onNext={onNext}
        onPrevious={onPrevious}
        onSkip={onSkip}
        onComplete={onComplete}
        currentStepIndex={currentStepIndex}
        totalSteps={totalSteps}
        isFirstStep={isFirstStep}
        isLastStep={isLastStep}
        placement={step.placement || 'bottom'}
      />
    </div>,
    document.body
  );
};

Tour Tooltip Componentâ€‹

The tooltip displays step content and navigation controls:

// components/TourOverlay/TourTooltip.tsx
import React from 'react';
import { TourStep } from '../../types/tour.types';

interface TourTooltipProps {
  step: TourStep;
  position: { x: number; y: number };
  placement: 'top' | 'bottom' | 'left' | 'right' | 'center';
  onNext: () => void;
  onPrevious: () => void;
  onSkip: () => void;
  onComplete: () => void;
  currentStepIndex: number;
  totalSteps: number;
  isFirstStep: boolean;
  isLastStep: boolean;
}

export const TourTooltip: React.FC<TourTooltipProps> = ({
  step,
  position,
  placement,
  onNext,
  onPrevious,
  onSkip,
  onComplete,
  currentStepIndex,
  totalSteps,
  isFirstStep,
  isLastStep,
}) => {
  const getTooltipStyle = (): React.CSSProperties => {
    const baseStyle: React.CSSProperties = {
      position: 'absolute',
      backgroundColor: 'white',
      borderRadius: '12px',
      padding: '20px',
      boxShadow: '0 10px 40px rgba(0, 0, 0, 0.2)',
      maxWidth: '320px',
      minWidth: '280px',
      zIndex: 1001,
      transform: getTransform(),
    };

    return {
      ...baseStyle,
      left: position.x,
      top: position.y,
    };
  };

  const getTransform = (): string => {
    switch (placement) {
      case 'top':
        return 'translate(-50%, -100%)';
      case 'bottom':
        return 'translate(-50%, 0)';
      case 'left':
        return 'translate(-100%, -50%)';
      case 'right':
        return 'translate(0, -50%)';
      case 'center':
        return 'translate(-50%, -50%)';
      default:
        return 'translate(-50%, 0)';
    }
  };

  const getArrowStyle = (): React.CSSProperties | null => {
    if (placement === 'center') return null;

    const baseArrow: React.CSSProperties = {
      position: 'absolute',
      width: 0,
      height: 0,
      borderStyle: 'solid',
    };

    switch (placement) {
      case 'top':
        return {
          ...baseArrow,
          bottom: '-8px',
          left: '50%',
          transform: 'translateX(-50%)',
          borderWidth: '8px 8px 0 8px',
          borderColor: 'white transparent transparent transparent',
        };
      case 'bottom':
        return {
          ...baseArrow,
          top: '-8px',
          left: '50%',
          transform: 'translateX(-50%)',
          borderWidth: '0 8px 8px 8px',
          borderColor: 'transparent transparent white transparent',
        };
      case 'left':
        return {
          ...baseArrow,
          right: '-8px',
          top: '50%',
          transform: 'translateY(-50%)',
          borderWidth: '8px 0 8px 8px',
          borderColor: 'transparent transparent transparent white',
        };
      case 'right':
        return {
          ...baseArrow,
          left: '-8px',
          top: '50%',
          transform: 'translateY(-50%)',
          borderWidth: '8px 8px 8px 0',
          borderColor: 'transparent white transparent transparent',
        };
      default:
        return null;
    }
  };

  return (
    <div style={getTooltipStyle()} className="tour-tooltip">
      {/* Arrow */}
      {getArrowStyle() && <div style={getArrowStyle()!} />}
      
      {/* Header */}
      <div className="tour-tooltip__header">
        <h3 id="tour-title" className="tour-tooltip__title">
          {step.title}
        </h3>
        <div className="tour-tooltip__progress">
          <span className="tour-tooltip__step-counter">
            {currentStepIndex + 1} of {totalSteps}
          </span>
          <div className="tour-tooltip__progress-bar">
            <div
              className="tour-tooltip__progress-fill"
              style={{
                width: `${((currentStepIndex + 1) / totalSteps) * 100}%`,
              }}
            />
          </div>
        </div>
      </div>

      {/* Content */}
      <div
        id="tour-content"
        className="tour-tooltip__content"
        dangerouslySetInnerHTML={{ __html: step.content }}
      />

      {/* Navigation */}
      <div className="tour-tooltip__actions">
        <div className="tour-tooltip__actions-left">
          {step.skipable !== false && (
            <button
              onClick={onSkip}
              className="tour-button tour-button--secondary"
              type="button"
            >
              Skip Tour
            </button>
          )}
        </div>
        
        <div className="tour-tooltip__actions-right">
          {!isFirstStep && (
            <button
              onClick={onPrevious}
              className="tour-button tour-button--secondary"
              type="button"
            >
              Previous
            </button>
          )}
          
          <button
            onClick={isLastStep ? onComplete : onNext}
            className="tour-button tour-button--primary"
            type="button"
            autoFocus
          >
            {isLastStep ? 'Finish' : 'Next'}
          </button>
        </div>
      </div>
    </div>
  );
};

Tour Manager for Multiple Toursâ€‹

A tour manager handles multiple tours and their triggering conditions:

// hooks/useTourManager.ts
import { useState, useEffect, useCallback } from 'react';
import { Tour, TriggerCondition } from '../types/tour.types';
import { useTourEngine } from './useTourEngine';

interface TourManagerConfig {
  tours: Tour[];
  autoStart?: boolean;
  debugMode?: boolean;
}

export const useTourManager = ({ tours, autoStart = true, debugMode = false }: TourManagerConfig) => {
  const [availableTours, setAvailableTours] = useState<Tour[]>([]);
  const tourEngine = useTourEngine();

  // Evaluate tour trigger conditions
  const evaluateTriggerConditions = useCallback((conditions: TriggerCondition[]): boolean => {
    return conditions.every(condition => {
      switch (condition.type) {
        case 'url_match':
          return window.location.pathname.includes(condition.value);
        
        case 'element_exists':
          return document.querySelector(condition.value) !== null;
        
        case 'user_segment':
          // This would typically check against user data from your auth system
          const userSegment = localStorage.getItem('user_segment');
          return userSegment === condition.value;
        
        case 'feature_flag':
          // Check if a feature flag is enabled
          const featureFlags = JSON.parse(localStorage.getItem('feature_flags') || '{}');
          return featureFlags[condition.value] === true;
        
        case 'page_visit_count':
          const visitCount = parseInt(localStorage.getItem('page_visit_count') || '0');
          return visitCount >= parseInt(condition.value);
        
        case 'time_on_page':
          // This would need to be implemented with a timer
          return true; // Simplified for example
        
        default:
          return false;
      }
    });
  }, []);

  // Find eligible tours
  const findEligibleTours = useCallback((): Tour[] => {
    return tours.filter(tour => {
      // Skip if tour is not active
      if (!tour.isActive) return false;
      
      // Skip if already completed or skipped
      if (tourEngine.tourState.completedTours.includes(tour.id) || 
          tourEngine.tourState.skippedTours.includes(tour.id)) {
        return false;
      }
      
      // Check trigger conditions
      if (tour.triggerConditions && tour.triggerConditions.length > 0) {
        return evaluateTriggerConditions(tour.triggerConditions);
      }
      
      return true;
    }).sort((a, b) => b.priority - a.priority); // Higher priority first
  }, [tours, tourEngine.tourState, evaluateTriggerConditions]);

  // Auto-start eligible tours
  useEffect(() => {
    if (!autoStart || tourEngine.tourState.isPlaying) return;

    const eligible = findEligibleTours();
    if (eligible.length > 0) {
      const nextTour = eligible[0];
      
      if (debugMode) {
        console.log('Starting tour:', nextTour.name);
      }
      
      // Small delay to ensure page is fully loaded
      setTimeout(() => {
        tourEngine.startTour(nextTour);
      }, 500);
    }
  }, [findEligibleTours, autoStart, tourEngine, debugMode]);

  // Manual tour starter
  const startTourById = useCallback((tourId: string) => {
    const tour = tours.find(t => t.id === tourId);
    if (tour) {
      tourEngine.startTour(tour);
    }
  }, [tours, tourEngine]);

  // Reset tour state (useful for testing)
  const resetTourState = useCallback(() => {
    localStorage.removeItem('tour-completed');
    localStorage.removeItem('tour-skipped');
    window.location.reload();
  }, []);

  return {
    ...tourEngine,
    availableTours: findEligibleTours(),
    startTourById,
    resetTourState,
  };
};

      if (element) {

        element.setAttribute('data-step', (index + 1).toString());
        element.setAttribute('data-title', step.title);
        if (step.position) {

        }
      }
    });

      exitOnOverlayClick: false,
      keyboardNavigation: true,
      disableInteraction: false,
    });

    intro.oncomplete(() => {
      onComplete?.();
    });

    intro.onexit(() => {
      onExit?.();
    });

    intro.start();

    return () => {
      intro.exit();
    };
  }, [steps, onComplete, onExit, autoStart]);

  return null; // This component doesn't render anything
};

  private tour: Shepherd.Tour;

      useModalOverlay: true,

        classes: 'shepherd-theme-custom',
        scrollTo: true,
        cancelIcon: {
          enabled: true,
        },
        when: {
          show: this.onStepShow.bind(this),
          hide: this.onStepHide.bind(this),
        },
      },
    });

    this.tour.on('complete', () => {
      this.onComplete?.();
    });
  }

  private onStepShow() {
    // Add custom behavior when step shows
    const currentStep = this.tour.getCurrentStep();
    if (currentStep) {
      // Track analytics
      this.trackStepView(currentStep.id || 'unknown');
    }
  }

  private onStepHide() {
    // Cleanup when step hides
  }

  private trackStepView(stepId: string) {
    if (typeof gtag !== 'undefined') {
      gtag('event', 'tour_step_view', {
        step_id: stepId,
        tour_name: 'dashboard_features',
      });
    }
  }

  addStep(step: {
    id: string;
    title: string;
    text: string;
    attachTo: {
      element: string;
      on: 'top' | 'bottom' | 'left' | 'right';
    };
    buttons?: Array<{
      text: string;
      action: () => void;
      classes?: string;
    }>;
    beforeShowPromise?: () => Promise<void>;
  }) {
    this.tour.addStep({
      id: step.id,
      title: step.title,
      text: step.text,
      attachTo: step.attachTo,
      buttons: step.buttons || [
        {
          text: 'Skip',
          classes: 'shepherd-button-secondary',
          action: () => this.tour.complete(),
        },
        {
          text: 'Next',
          classes: 'shepherd-button-primary',
          action: () => this.tour.next(),
        },
      ],
      beforeShowPromise: step.beforeShowPromise,
    });
  }

  start() {
    this.tour.start();
  }

  complete() {
    this.tour.complete();
  }
}

Choosing the Right Libraryâ€‹

Use Intro.js when:

  • You need maximum browser compatibility
  • Working with legacy codebases
  • Want extensive documentation and community support
  • Need proven stability in production

Use Reactour when:

  • Building React applications
  • Want modern hooks-based API
  • Need excellent TypeScript support
  • Prefer component-based architecture

Use Shepherd.js when:

  • Need maximum customization flexibility
  • Building complex, multi-step workflows
  • Working with multiple frameworks
  • Want advanced tour features (branching, conditionals)

Use Driver.js when:

  • Bundle size is critical
  • Need simple, straightforward tours
  • Want modern, clean API
  • Performance is a priority

Hybrid Approach: Combining Librariesâ€‹

You can also combine libraries for different use cases:

// tourStrategy.ts
import { TourLibrary, TourStep } from './types';

export class TourStrategy {
  private getOptimalLibrary(steps: TourStep[], context: any): TourLibrary {
    // Use lightweight library for simple tours
    if (steps.length <= 3 && !context.needsCustomization) {
      return 'driver';
    }
    
    // Use React-specific library for component-heavy tours
    if (context.isReactApp && steps.some(s => s.hasReactContent)) {
      return 'reactour';
    }
    
    // Use Shepherd for complex business workflows
    if (steps.length > 8 || context.needsBranching) {
      return 'shepherd';
    }
    
    // Default to Intro.js for compatibility
    return 'introjs';
  }

  async createTour(steps: TourStep[], context: any) {
    const library = this.getOptimalLibrary(steps, context);
    
    switch (library) {
      case 'driver':
        const { DriverTourManager } = await import('./drivers/driverTour');
        return new DriverTourManager();
      
      case 'reactour':
        const { ReactourManager } = await import('./drivers/reactourTour');
        return new ReactourManager();
      
      case 'shepherd':
        const { ShepherdTourManager } = await import('./drivers/shepherdTour');
        return new ShepherdTourManager();
      
      default:
        const { IntroJsManager } = await import('./drivers/introJsTour');
        return new IntroJsManager();
    }
  }
}

Analytics Integrationâ€‹

Track tour performance and user engagement:

// utils/tourAnalytics.ts
interface TourAnalyticsEvent {
  tourId: string;
  stepId: string;
  action: 'start' | 'next' | 'previous' | 'skip' | 'complete';
  timestamp: number;
  userSegment?: string;
}

export class TourAnalytics {
  private events: TourAnalyticsEvent[] = [];

  trackEvent(event: Omit<TourAnalyticsEvent, 'timestamp'>) {
    const fullEvent: TourAnalyticsEvent = {
      ...event,
      timestamp: Date.now(),
    };
    
    this.events.push(fullEvent);
    
    // Send to your analytics service
    this.sendToAnalytics(fullEvent);
  }

  private sendToAnalytics(event: TourAnalyticsEvent) {
    // Integration with your analytics platform
    if (typeof window !== 'undefined' && (window as any).gtag) {
      (window as any).gtag('event', 'tour_interaction', {
        tour_id: event.tourId,
        step_id: event.stepId,
        action: event.action,
        user_segment: event.userSegment,
      });
    }
    
    // Or send to your custom analytics endpoint
    fetch('/api/analytics/tour-events', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(event),
    }).catch(console.error);
  }

  getTourMetrics(tourId: string) {
    const tourEvents = this.events.filter(e => e.tourId === tourId);
    const startEvents = tourEvents.filter(e => e.action === 'start');
    const completeEvents = tourEvents.filter(e => e.action === 'complete');
    const skipEvents = tourEvents.filter(e => e.action === 'skip');
    
    return {
      starts: startEvents.length,
      completions: completeEvents.length,
      skips: skipEvents.length,
      completionRate: startEvents.length > 0 ? completeEvents.length / startEvents.length : 0,
      skipRate: startEvents.length > 0 ? skipEvents.length / startEvents.length : 0,
    };
  }
}

A/B Testing Supportâ€‹

Test different tour variations:

// utils/tourVariants.ts
interface TourVariant {
  id: string;
  name: string;
  tour: Tour;
  weight: number; // Percentage allocation
}

export class TourVariantManager {
  private variants: Map<string, TourVariant[]> = new Map();

  addVariant(baseTrackId: string, variant: TourVariant) {
    if (!this.variants.has(baseTrackId)) {
      this.variants.set(baseTrackId, []);
    }
    this.variants.get(baseTrackId)!.push(variant);
  }

  getVariantForUser(baseTrackId: string, userId: string): Tour | null {
    const variants = this.variants.get(baseTrackId);
    if (!variants || variants.length === 0) return null;

    // Use consistent hash of user ID for stable variant assignment
    const hash = this.hashString(userId + baseTrackId);
    const totalWeight = variants.reduce((sum, v) => sum + v.weight, 0);
    const target = (hash % 100) / 100 * totalWeight;
    
    let cumulative = 0;
    for (const variant of variants) {
      cumulative += variant.weight;
      if (target <= cumulative) {
        return variant.tour;
      }
    }
    
    return variants[0].tour; // Fallback
  }

  private hashString(str: string): number {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash; // Convert to 32-bit integer
    }
    return Math.abs(hash);
  }
}

Best Practices and Tipsâ€‹

ðŸŽ¯ Content Guidelinesâ€‹

  2. Use progressive disclosure: Start with overview, then dive into specifics
  3. Include visuals: Use emojis and bullet points for better readability
  4. Provide context: Explain the "why" behind features, not just the "how"

  1. Performance: Use lazy loading for tour components
  2. Accessibility: Ensure keyboard navigation and screen reader compatibility
  3. Mobile responsiveness: Test tours on different screen sizes
  4. Error handling: Gracefully handle missing target elements

ðŸ“Š Measurement and Optimizationâ€‹

  1. Track completion rates: Monitor which steps cause dropoffs
  2. Gather feedback: Add optional feedback collection at tour end
  3. A/B testing: Test different content and flows
  4. Timing: Experiment with when tours are triggered

ðŸŽ¨ Design Principlesâ€‹

  1. Non-intrusive: Allow users to continue working while touring
  2. Contextual: Show tours when users encounter relevant features
  3. Skippable: Always provide an easy exit option
  4. Memorable: Use consistent styling that matches your brand
