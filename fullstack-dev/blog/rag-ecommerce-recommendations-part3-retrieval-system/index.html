The retrieval system is the heart of our RAG architecture, responsible for efficiently finding relevant products from millions of items using semantic search, keyword matching, and hybrid approaches. We'll implement production-ready solutions using popular vector databases and explore advanced optimization techniques.

1. Qdrant Implementation​

# retrieval/qdrant_retriever.py
import asyncio
from typing import List, Dict, Any, Optional, Tuple
import numpy as np
from qdrant_client import QdrantClient
from qdrant_client.http import models
from qdrant_client.http.models import (
    Distance, VectorParams, CreateCollection, PointStruct,
    Filter, FieldCondition, PayloadSchemaType
)
import logging
from abc import ABC, abstractmethod

logger = logging.getLogger(__name__)

class BaseVectorRetriever(ABC):
    """Abstract base class for vector retrievers"""
    
    @abstractmethod
    async def create_collection(self, collection_name: str, vector_dimension: int) -> bool:
        pass
    
    @abstractmethod
    async def upsert_vectors(self, collection_name: str, vectors: List[Dict[str, Any]]) -> bool:
        pass
    
    @abstractmethod
    async def search(self, collection_name: str, query_vector: np.ndarray, 
                    limit: int = 10, filters: Optional[Dict] = None) -> List[Dict[str, Any]]:
        pass

class QdrantRetriever(BaseVectorRetriever):
    """Qdrant vector database retriever implementation"""
    
    def __init__(self, host: str = "localhost", port: int = 6333, api_key: Optional[str] = None):
        self.client = QdrantClient(
            host=host,
            port=port,
            api_key=api_key,
            timeout=30
        )
        self.collection_configs = {}
        logger.info(f"Connected to Qdrant at {host}:{port}")
    
    async def create_collection(self, collection_name: str, vector_dimension: int, 
                               distance_metric: Distance = Distance.COSINE) -> bool:
        """Create a new collection in Qdrant"""
        try:
            # Check if collection already exists
            collections = self.client.get_collections()
            existing_names = [col.name for col in collections.collections]
            
            if collection_name in existing_names:
                logger.info(f"Collection {collection_name} already exists")
                return True
            
            # Create collection
            self.client.create_collection(
                collection_name=collection_name,
                vectors_config=VectorParams(
                    size=vector_dimension,
                    distance=distance_metric
                ),
                optimizers_config=models.OptimizersConfig(
                    default_segment_number=2,
                    max_segment_size=20000,
                    memmap_threshold=20000,
                    indexing_threshold=20000,
                ),
                replication_factor=1,
                write_consistency_factor=1,
            )
            
            # Create payload schema for better performance
            self.client.create_payload_index(
                collection_name=collection_name,
                field_name="category",
                field_schema=PayloadSchemaType.KEYWORD
            )
            
            self.client.create_payload_index(
                collection_name=collection_name,
                field_name="brand",
                field_schema=PayloadSchemaType.KEYWORD
            )
            
            self.client.create_payload_index(
                collection_name=collection_name,
                field_name="price",
                field_schema=PayloadSchemaType.FLOAT
            )
            
            self.client.create_payload_index(
                collection_name=collection_name,
                field_name="rating",
                field_schema=PayloadSchemaType.FLOAT
            )
            
            logger.info(f"Created collection {collection_name} with dimension {vector_dimension}")
            return True
            
        except Exception as e:
            logger.error(f"Failed to create collection {collection_name}: {e}")
            return False
    
    async def upsert_vectors(self, collection_name: str, vectors: List[Dict[str, Any]]) -> bool:
        """Insert or update vectors in the collection"""
        try:
            points = []
            for i, vector_data in enumerate(vectors):
                point = PointStruct(
                    id=vector_data.get('id', i),
                    vector=vector_data['vector'].tolist() if isinstance(vector_data['vector'], np.ndarray) 
                           else vector_data['vector'],
                    payload=vector_data.get('payload', {})
                )
                points.append(point)
            
            # Batch upsert
            batch_size = 100
            for i in range(0, len(points), batch_size):
                batch = points[i:i + batch_size]
                operation_info = self.client.upsert(
                    collection_name=collection_name,
                    wait=True,
                    points=batch
                )
                logger.debug(f"Upserted batch {i//batch_size + 1}: {operation_info}")
            
            logger.info(f"Successfully upserted {len(vectors)} vectors to {collection_name}")
            return True
            
        except Exception as e:
            logger.error(f"Failed to upsert vectors to {collection_name}: {e}")
            return False
    
    async def search(self, collection_name: str, query_vector: np.ndarray, 
                    limit: int = 10, filters: Optional[Dict] = None,
                    score_threshold: Optional[float] = None) -> List[Dict[str, Any]]:
        """Search for similar vectors"""
        try:
            # Convert numpy array to list
            query_vector_list = query_vector.tolist() if isinstance(query_vector, np.ndarray) else query_vector
            
            # Build filters
            qdrant_filter = self._build_qdrant_filter(filters) if filters else None
            
            # Perform search
            search_result = self.client.search(
                collection_name=collection_name,
                query_vector=query_vector_list,
                limit=limit,
                query_filter=qdrant_filter,
                score_threshold=score_threshold,
                with_payload=True,
                with_vectors=False
            )
            
            # Format results
            results = []
            for scored_point in search_result:
                result = {
                    'id': scored_point.id,
                    'score': scored_point.score,
                    'payload': scored_point.payload
                }
                results.append(result)
            
            logger.debug(f"Found {len(results)} results for query in {collection_name}")
            return results
            
        except Exception as e:
            logger.error(f"Search failed in {collection_name}: {e}")
            return []
    
    async def hybrid_search(self, collection_name: str, query_vector: np.ndarray,
                           text_query: str, limit: int = 10, 
                           alpha: float = 0.7) -> List[Dict[str, Any]]:
        """Hybrid search combining vector and text search"""
        try:
            # Vector search
            vector_results = await self.search(
                collection_name=collection_name,
                query_vector=query_vector,
                limit=limit * 2  # Get more candidates
            )
            
            # Text-based filtering/scoring
            text_scored_results = self._score_text_relevance(vector_results, text_query)
            
            # Combine scores
            final_results = []
            for result in text_scored_results:
                vector_score = result['score']
                text_score = result.get('text_score', 0.0)
                
                # Weighted combination
                combined_score = alpha * vector_score + (1 - alpha) * text_score
                
                result['combined_score'] = combined_score
                result['vector_score'] = vector_score
                result['text_score'] = text_score
                
                final_results.append(result)
            
            # Sort by combined score and limit results
            final_results.sort(key=lambda x: x['combined_score'], reverse=True)
            return final_results[:limit]
            
        except Exception as e:
            logger.error(f"Hybrid search failed: {e}")
            return []
    
    def _build_qdrant_filter(self, filters: Dict[str, Any]) -> Filter:
        """Build Qdrant filter from dictionary"""
        conditions = []
        
        for field, value in filters.items():
            if isinstance(value, dict):
                # Range filter
                if 'gte' in value or 'lte' in value or 'gt' in value or 'lt' in value:
                    condition = FieldCondition(
                        key=field,
                        range=models.Range(**value)
                    )
                # Match any filter
                elif 'any' in value:
                    condition = FieldCondition(
                        key=field,
                        match=models.MatchAny(any=value['any'])
                    )
                else:
                    continue
            else:
                # Exact match
                condition = FieldCondition(
                    key=field,
                    match=models.MatchValue(value=value)
                )
            
            conditions.append(condition)
        
        return Filter(must=conditions) if conditions else None
    
    def _score_text_relevance(self, results: List[Dict[str, Any]], 
                             text_query: str) -> List[Dict[str, Any]]:
        """Score results based on text relevance"""
        query_terms = set(text_query.lower().split())
        
        for result in results:
            text_score = 0.0
            payload = result.get('payload', {})
            
            # Score based on title match
            title = payload.get('title', '').lower()
            title_words = set(title.split())
            title_overlap = len(query_terms & title_words) / max(len(query_terms), 1)
            text_score += title_overlap * 0.6
            
            # Score based on description match
            description = payload.get('description', '').lower()
            desc_words = set(description.split())
            desc_overlap = len(query_terms & desc_words) / max(len(query_terms), 1)
            text_score += desc_overlap * 0.3
            
            # Score based on category/brand match
            category = payload.get('category', '').lower()
            brand = payload.get('brand', '').lower()
            
            if any(term in category for term in query_terms):
                text_score += 0.1
            if any(term in brand for term in query_terms):
                text_score += 0.1
            
            result['text_score'] = text_score
        
        return results
    
    async def get_collection_info(self, collection_name: str) -> Dict[str, Any]:
        """Get information about a collection"""
        try:
            info = self.client.get_collection(collection_name)
            return {
                'status': info.status,
                'vectors_count': info.vectors_count,
                'points_count': info.points_count,
                'config': info.config
            }
        except Exception as e:
            logger.error(f"Failed to get collection info: {e}")
            return {}
    
    async def delete_collection(self, collection_name: str) -> bool:
        """Delete a collection"""
        try:
            self.client.delete_collection(collection_name)
            logger.info(f"Deleted collection {collection_name}")
            return True
        except Exception as e:
            logger.error(f"Failed to delete collection {collection_name}: {e}")
            return False

# retrieval/pinecone_retriever.py
import pinecone
import numpy as np
from typing import List, Dict, Any, Optional
import logging
import asyncio
from concurrent.futures import ThreadPoolExecutor

logger = logging.getLogger(__name__)

class PineconeRetriever(BaseVectorRetriever):
    """Pinecone vector database retriever implementation"""
    
    def __init__(self, api_key: str, environment: str):
        pinecone.init(api_key=api_key, environment=environment)
        self.api_key = api_key
        self.environment = environment
        self.indexes = {}
        self.executor = ThreadPoolExecutor(max_workers=10)
        logger.info(f"Initialized Pinecone in environment: {environment}")
    
    async def create_collection(self, collection_name: str, vector_dimension: int,
                               metric: str = "cosine", pods: int = 1,
                               pod_type: str = "p1.x1") -> bool:
        """Create a new index in Pinecone"""
        try:
            # Check if index exists
            if collection_name in pinecone.list_indexes():
                logger.info(f"Index {collection_name} already exists")
                self.indexes[collection_name] = pinecone.Index(collection_name)
                return True
            
            # Create index
            pinecone.create_index(
                name=collection_name,
                dimension=vector_dimension,
                metric=metric,
                pods=pods,
                pod_type=pod_type
            )
            
            # Wait for index to be ready
            await asyncio.sleep(10)  # Pinecone needs time to initialize
            
            self.indexes[collection_name] = pinecone.Index(collection_name)
            logger.info(f"Created Pinecone index {collection_name}")
            return True
            
        except Exception as e:
            logger.error(f"Failed to create Pinecone index {collection_name}: {e}")
            return False
    
    async def upsert_vectors(self, collection_name: str, vectors: List[Dict[str, Any]]) -> bool:
        """Insert or update vectors in Pinecone"""
        try:
            if collection_name not in self.indexes:
                self.indexes[collection_name] = pinecone.Index(collection_name)
            
            index = self.indexes[collection_name]
            
            # Prepare vectors for Pinecone format
            pinecone_vectors = []
            for vector_data in vectors:
                vector_id = str(vector_data.get('id'))
                vector_values = vector_data['vector'].tolist() if isinstance(vector_data['vector'], np.ndarray) else vector_data['vector']
                metadata = vector_data.get('payload', {})
                
                pinecone_vectors.append((vector_id, vector_values, metadata))
            
            # Batch upsert
            batch_size = 100
            for i in range(0, len(pinecone_vectors), batch_size):
                batch = pinecone_vectors[i:i + batch_size]
                
                # Run in executor to avoid blocking
                await asyncio.get_event_loop().run_in_executor(
                    self.executor, 
                    lambda: index.upsert(vectors=batch)
                )
            
            logger.info(f"Upserted {len(vectors)} vectors to Pinecone index {collection_name}")
            return True
            
        except Exception as e:
            logger.error(f"Failed to upsert vectors to Pinecone: {e}")
            return False
    
    async def search(self, collection_name: str, query_vector: np.ndarray,
                    limit: int = 10, filters: Optional[Dict] = None,
                    include_metadata: bool = True) -> List[Dict[str, Any]]:
        """Search for similar vectors in Pinecone"""
        try:
            if collection_name not in self.indexes:
                self.indexes[collection_name] = pinecone.Index(collection_name)
            
            index = self.indexes[collection_name]
            
            # Convert numpy array to list
            query_vector_list = query_vector.tolist() if isinstance(query_vector, np.ndarray) else query_vector
            
            # Perform search
            search_response = await asyncio.get_event_loop().run_in_executor(
                self.executor,
                lambda: index.query(
                    vector=query_vector_list,
                    top_k=limit,
                    filter=filters,
                    include_metadata=include_metadata
                )
            )
            
            # Format results
            results = []
            for match in search_response.matches:
                result = {
                    'id': match.id,
                    'score': match.score,
                    'payload': match.metadata if include_metadata else {}
                }
                results.append(result)
            
            logger.debug(f"Found {len(results)} results in Pinecone index {collection_name}")
            return results
            
        except Exception as e:
            logger.error(f"Pinecone search failed: {e}")
            return []
    
    async def delete_collection(self, collection_name: str) -> bool:
        """Delete a Pinecone index"""
        try:
            pinecone.delete_index(collection_name)
            if collection_name in self.indexes:
                del self.indexes[collection_name]
            logger.info(f"Deleted Pinecone index {collection_name}")
            return True
        except Exception as e:
            logger.error(f"Failed to delete Pinecone index {collection_name}: {e}")
            return False
    
    async def get_collection_info(self, collection_name: str) -> Dict[str, Any]:
        """Get information about a Pinecone index"""
        try:
            index_stats = await asyncio.get_event_loop().run_in_executor(
                self.executor,
                lambda: pinecone.Index(collection_name).describe_index_stats()
            )
            return {
                'total_vector_count': index_stats.total_vector_count,
                'dimension': index_stats.dimension,
                'index_fullness': index_stats.index_fullness
            }
        except Exception as e:
            logger.error(f"Failed to get Pinecone index info: {e}")
            return {}

# retrieval/hybrid_retriever.py
import asyncio
from typing import List, Dict, Any, Optional, Tuple
import numpy as np
from elasticsearch import AsyncElasticsearch
import logging
from collections import defaultdict

logger = logging.getLogger(__name__)

class HybridRetriever:
    """Combines vector search with traditional text search"""
    
    def __init__(self, vector_retriever: BaseVectorRetriever, 
                 elasticsearch_client: Optional[AsyncElasticsearch] = None):
        self.vector_retriever = vector_retriever
        self.elasticsearch_client = elasticsearch_client
        self.fusion_methods = {
            'rrf': self._reciprocal_rank_fusion,
            'weighted': self._weighted_fusion,
            'linear': self._linear_combination
        }
    
    async def hybrid_search(self, collection_name: str, query_vector: np.ndarray,
                           text_query: str, limit: int = 10,
                           vector_weight: float = 0.7,
                           fusion_method: str = 'rrf',
                           filters: Optional[Dict] = None) -> List[Dict[str, Any]]:
        """Perform hybrid search combining vector and text search"""
        
        # Run searches in parallel
        vector_task = self._vector_search(collection_name, query_vector, limit * 2, filters)
        text_task = self._text_search(collection_name, text_query, limit * 2, filters)
        
        vector_results, text_results = await asyncio.gather(vector_task, text_task)
        
        # Fuse results
        fusion_func = self.fusion_methods.get(fusion_method, self._reciprocal_rank_fusion)
        fused_results = fusion_func(vector_results, text_results, vector_weight)
        
        # Limit and return results
        return fused_results[:limit]
    
    async def _vector_search(self, collection_name: str, query_vector: np.ndarray,
                            limit: int, filters: Optional[Dict] = None) -> List[Dict[str, Any]]:
        """Perform vector similarity search"""
        try:
            results = await self.vector_retriever.search(
                collection_name=collection_name,
                query_vector=query_vector,
                limit=limit,
                filters=filters
            )
            
            # Add search type and normalize scores
            for i, result in enumerate(results):
                result['search_type'] = 'vector'
                result['rank'] = i + 1
                result['normalized_score'] = result['score']  # Assumes cosine similarity [0,1]
            
            return results
            
        except Exception as e:
            logger.error(f"Vector search failed: {e}")
            return []
    
    async def _text_search(self, collection_name: str, text_query: str,
                          limit: int, filters: Optional[Dict] = None) -> List[Dict[str, Any]]:
        """Perform full-text search using Elasticsearch"""
        if not self.elasticsearch_client:
            logger.warning("Elasticsearch client not configured, skipping text search")
            return []
        
        try:
            # Build Elasticsearch query
            es_query = {
                "query": {
                    "bool": {
                        "must": [
                            {
                                "multi_match": {
                                    "query": text_query,
                                    "fields": ["title^3", "description^2", "category", "brand"],
                                    "type": "best_fields",
                                    "fuzziness": "AUTO"
                                }
                            }
                        ]
                    }
                },
                "size": limit
            }
            
            # Add filters if provided
            if filters:
                es_filters = self._build_elasticsearch_filters(filters)
                es_query["query"]["bool"]["filter"] = es_filters
            
            # Execute search
            response = await self.elasticsearch_client.search(
                index=collection_name,
                body=es_query
            )
            
            # Format results
            results = []
            for i, hit in enumerate(response['hits']['hits']):
                result = {
                    'id': hit['_id'],
                    'score': hit['_score'],
                    'payload': hit['_source'],
                    'search_type': 'text',
                    'rank': i + 1,
                    'normalized_score': hit['_score'] / response['hits']['max_score'] if response['hits']['max_score'] > 0 else 0
                }
                results.append(result)
            
            return results
            
        except Exception as e:
            logger.error(f"Text search failed: {e}")
            return []
    
    def _reciprocal_rank_fusion(self, vector_results: List[Dict[str, Any]],
                               text_results: List[Dict[str, Any]],
                               vector_weight: float = 0.7,
                               k: int = 60) -> List[Dict[str, Any]]:
        """Combine results using Reciprocal Rank Fusion"""
        # Create score maps
        vector_scores = {result['id']: 1 / (k + result['rank']) for result in vector_results}
        text_scores = {result['id']: 1 / (k + result['rank']) for result in text_results}
        
        # Get all unique IDs
        all_ids = set(vector_scores.keys()) | set(text_scores.keys())
        
        # Calculate combined scores
        combined_results = []
        item_map = {}
        
        # Create item lookup
        for result in vector_results + text_results:
            if result['id'] not in item_map:
                item_map[result['id']] = result
        
        for item_id in all_ids:
            vector_score = vector_scores.get(item_id, 0)
            text_score = text_scores.get(item_id, 0)
            
            # Weighted RRF score
            combined_score = (vector_weight * vector_score + 
                            (1 - vector_weight) * text_score)
            
            if item_id in item_map:
                result = item_map[item_id].copy()
                result['rrf_score'] = combined_score
                result['vector_rrf'] = vector_score
                result['text_rrf'] = text_score
                combined_results.append(result)
        
        # Sort by combined score
        combined_results.sort(key=lambda x: x['rrf_score'], reverse=True)
        return combined_results
    
    def _weighted_fusion(self, vector_results: List[Dict[str, Any]],
                        text_results: List[Dict[str, Any]],
                        vector_weight: float = 0.7) -> List[Dict[str, Any]]:
        """Combine results using weighted score fusion"""
        # Create score maps
        vector_scores = {result['id']: result['normalized_score'] for result in vector_results}
        text_scores = {result['id']: result['normalized_score'] for result in text_results}
        
        # Get all unique IDs
        all_ids = set(vector_scores.keys()) | set(text_scores.keys())
        
        # Calculate combined scores
        combined_results = []
        item_map = {}
        
        # Create item lookup
        for result in vector_results + text_results:
            if result['id'] not in item_map:
                item_map[result['id']] = result
        
        for item_id in all_ids:
            vector_score = vector_scores.get(item_id, 0)
            text_score = text_scores.get(item_id, 0)
            
            # Weighted combination
            combined_score = (vector_weight * vector_score + 
                            (1 - vector_weight) * text_score)
            
            if item_id in item_map:
                result = item_map[item_id].copy()
                result['combined_score'] = combined_score
                result['vector_score'] = vector_score
                result['text_score'] = text_score
                combined_results.append(result)
        
        # Sort by combined score
        combined_results.sort(key=lambda x: x['combined_score'], reverse=True)
        return combined_results
    
    def _linear_combination(self, vector_results: List[Dict[str, Any]],
                           text_results: List[Dict[str, Any]],
                           vector_weight: float = 0.7) -> List[Dict[str, Any]]:
        """Simple linear combination of normalized scores"""
        return self._weighted_fusion(vector_results, text_results, vector_weight)
    
    def _build_elasticsearch_filters(self, filters: Dict[str, Any]) -> List[Dict[str, Any]]:
        """Convert filters to Elasticsearch format"""
        es_filters = []
        
        for field, value in filters.items():
            if isinstance(value, dict):
                if 'gte' in value or 'lte' in value:
                    es_filters.append({
                        "range": {
                            field: value
                        }
                    })
                elif 'any' in value:
                    es_filters.append({
                        "terms": {
                            field: value['any']
                        }
                    })
            else:
                es_filters.append({
                    "term": {
                        field: value
                    }
                })
        
        return es_filters

# retrieval/contextual_search.py
from typing import List, Dict, Any, Optional
import numpy as np
from datetime import datetime, timedelta
import logging

logger = logging.getLogger(__name__)

class ContextualSearchEngine:
    """Advanced search engine with context awareness"""
    
    def __init__(self, retriever: BaseVectorRetriever):
        self.retriever = retriever
        self.context_weights = {
            'seasonal': 0.1,
            'trending': 0.15,
            'personal': 0.3,
            'behavioral': 0.25,
            'collaborative': 0.2
        }
    
    async def contextual_search(self, collection_name: str, query_vector: np.ndarray,
                               user_context: Dict[str, Any], limit: int = 10) -> List[Dict[str, Any]]:
        """Perform context-aware search"""
        
        # Get base search results
        base_results = await self.retriever.search(
            collection_name=collection_name,
            query_vector=query_vector,
            limit=limit * 3  # Get more candidates for reranking
        )
        
        # Apply contextual reranking
        reranked_results = await self._apply_contextual_reranking(
            base_results, user_context
        )
        
        # Apply diversity filtering
        diverse_results = self._apply_diversity_filtering(reranked_results, limit)
        
        return diverse_results
    
    async def _apply_contextual_reranking(self, results: List[Dict[str, Any]], 
                                         context: Dict[str, Any]) -> List[Dict[str, Any]]:
        """Rerank results based on context"""
        
        for result in results:
            payload = result.get('payload', {})
            base_score = result['score']
            
            # Calculate context-based adjustments
            seasonal_boost = self._calculate_seasonal_boost(payload, context)
            trending_boost = self._calculate_trending_boost(payload, context)
            personal_boost = self._calculate_personal_boost(payload, context)
            behavioral_boost = self._calculate_behavioral_boost(payload, context)
            collaborative_boost = self._calculate_collaborative_boost(payload, context)
            
            # Apply weighted boosts
            context_score = (
                self.context_weights['seasonal'] * seasonal_boost +
                self.context_weights['trending'] * trending_boost +
                self.context_weights['personal'] * personal_boost +
                self.context_weights['behavioral'] * behavioral_boost +
                self.context_weights['collaborative'] * collaborative_boost
            )
            
            # Combine base score with context score
            result['contextual_score'] = base_score * (1 + context_score)
            result['context_breakdown'] = {
                'seasonal': seasonal_boost,
                'trending': trending_boost,
                'personal': personal_boost,
                'behavioral': behavioral_boost,
                'collaborative': collaborative_boost
            }
        
        # Sort by contextual score
        results.sort(key=lambda x: x['contextual_score'], reverse=True)
        return results
    
    def _calculate_seasonal_boost(self, product: Dict[str, Any], 
                                 context: Dict[str, Any]) -> float:
        """Calculate seasonal relevance boost"""
        current_month = datetime.now().month
        product_category = product.get('category', '').lower()
        
        # Define seasonal categories
        seasonal_categories = {
            'winter': ['coats', 'sweaters', 'boots', 'heating'],
            'spring': ['flowers', 'gardening', 'light_clothing'],
            'summer': ['swimwear', 'shorts', 'cooling', 'outdoor'],
            'fall': ['jackets', 'school_supplies', 'warm_drinks']
        }
        
        # Determine current season
        season_map = {
            12: 'winter', 1: 'winter', 2: 'winter',
            3: 'spring', 4: 'spring', 5: 'spring',
            6: 'summer', 7: 'summer', 8: 'summer',
            9: 'fall', 10: 'fall', 11: 'fall'
        }
        
        current_season = season_map.get(current_month, 'spring')
        seasonal_items = seasonal_categories.get(current_season, [])
        
        # Check if product matches current season
        for item in seasonal_items:
            if item in product_category or item in product.get('title', '').lower():
                return 0.2  # 20% boost for seasonal items
        
        return 0.0
    
    def _calculate_trending_boost(self, product: Dict[str, Any], 
                                 context: Dict[str, Any]) -> float:
        """Calculate trending boost based on recent popularity"""
        popularity_score = product.get('popularity_score', 0)
        recent_views = product.get('recent_views', 0)
        
        # Boost based on popularity and recent activity
        trending_score = 0.0
        
        if popularity_score > 0.8:
            trending_score += 0.15
        elif popularity_score > 0.6:
            trending_score += 0.1
        
        if recent_views > 1000:  # High recent activity
            trending_score += 0.1
        
        return trending_score
    
    def _calculate_personal_boost(self, product: Dict[str, Any], 
                                 context: Dict[str, Any]) -> float:
        """Calculate personal preference boost"""
        user_preferences = context.get('preferences', {})
        personal_score = 0.0
        
        # Category preference
        preferred_categories = user_preferences.get('preferred_categories', [])
        if product.get('category') in preferred_categories:
            personal_score += 0.3
        
        # Brand preference
        preferred_brands = user_preferences.get('preferred_brands', [])
        if product.get('brand') in preferred_brands:
            personal_score += 0.2
        
        # Price preference
        price_sensitivity = user_preferences.get('price_sensitivity', 0.5)
        product_price = product.get('price', 0)
        user_avg_price = context.get('avg_order_value', 100)
        
        if price_sensitivity < 0.3:  # Price-conscious user
            if product_price <= user_avg_price * 0.8:
                personal_score += 0.1
        elif price_sensitivity > 0.7:  # Premium-oriented user
            if product_price >= user_avg_price * 1.2:
                personal_score += 0.1
        
        return personal_score
    
    def _calculate_behavioral_boost(self, product: Dict[str, Any], 
                                   context: Dict[str, Any]) -> float:
        """Calculate behavioral pattern boost"""
        user_behavior = context.get('behavioral_features', {})
        behavioral_score = 0.0
        
        # Time-based patterns
        current_hour = datetime.now().hour
        user_peak_hour = context.get('peak_hour', 12)
        
        # Boost items typically purchased at current time
        if abs(current_hour - user_peak_hour) <= 2:
            behavioral_score += 0.1
        
        # Purchase frequency patterns
        purchase_frequency = user_behavior.get('purchase_frequency', 0)
        if purchase_frequency > 10:  # Frequent buyer
            # Boost consumable/replenishable items
            if any(keyword in product.get('title', '').lower() 
                   for keyword in ['refill', 'pack', 'bulk']):
                behavioral_score += 0.15
        
        return behavioral_score
    
    def _calculate_collaborative_boost(self, product: Dict[str, Any], 
                                      context: Dict[str, Any]) -> float:
        """Calculate collaborative filtering boost"""
        # This would typically use precomputed user similarity scores
        # For now, we'll use simplified logic
        
        similar_users_preferences = context.get('similar_users_bought', [])
        if product.get('id') in similar_users_preferences:
            return 0.2
        
        return 0.0
    
    def _apply_diversity_filtering(self, results: List[Dict[str, Any]], 
                                  limit: int, diversity_threshold: float = 0.3) -> List[Dict[str, Any]]:
        """Apply diversity filtering to avoid too similar items"""
        if not results:
            return results
        
        diverse_results = [results[0]]  # Always include top result
        
        for result in results[1:]:
            if len(diverse_results) >= limit:
                break
            
            # Check diversity against already selected items
            is_diverse = True
            result_category = result.get('payload', {}).get('category', '')
            result_brand = result.get('payload', {}).get('brand', '')
            
            category_count = sum(1 for r in diverse_results 
                               if r.get('payload', {}).get('category', '') == result_category)
            brand_count = sum(1 for r in diverse_results 
                            if r.get('payload', {}).get('brand', '') == result_brand)
            
            # Avoid too many items from same category/brand
            if (category_count >= limit * diversity_threshold or 
                brand_count >= limit * diversity_threshold):
                is_diverse = False
            
            if is_diverse:
                diverse_results.append(result)
        
        return diverse_results

# retrieval/performance_optimizer.py
import asyncio
from typing import List, Dict, Any, Optional
import numpy as np
import redis
import json
import hashlib
from datetime import datetime, timedelta
import logging

logger = logging.getLogger(__name__)

class SearchPerformanceOptimizer:
    """Optimize search performance through caching and query optimization"""
    
    def __init__(self, redis_client: Optional[redis.Redis] = None):
        self.redis_client = redis_client
        self.cache_ttl = {
            'search_results': 3600,    # 1 hour
            'user_embeddings': 86400,  # 24 hours
            'popular_queries': 7200,   # 2 hours
        }
        self.query_stats = {}
    
    async def optimized_search(self, search_func, cache_key: str, 
                              ttl: int = 3600, *args, **kwargs) -> List[Dict[str, Any]]:
        """Wrapper for optimized search with caching"""
        
        # Try cache first
        cached_result = await self._get_cached_result(cache_key)
        if cached_result is not None:
            logger.debug(f"Cache hit for key: {cache_key}")
            return cached_result
        
        # Execute search
        start_time = datetime.now()
        results = await search_func(*args, **kwargs)
        execution_time = (datetime.now() - start_time).total_seconds()
        
        # Cache results
        await self._cache_result(cache_key, results, ttl)
        
        # Update query statistics
        self._update_query_stats(cache_key, execution_time, len(results))
        
        logger.debug(f"Search executed in {execution_time:.3f}s, {len(results)} results")
        return results
    
    async def _get_cached_result(self, cache_key: str) -> Optional[List[Dict[str, Any]]]:
        """Retrieve cached search results"""
        if not self.redis_client:
            return None
        
        try:
            cached_data = self.redis_client.get(cache_key)
            if cached_data:
                return json.loads(cached_data)
        except Exception as e:
            logger.warning(f"Cache retrieval failed: {e}")
        
        return None
    
    async def _cache_result(self, cache_key: str, results: List[Dict[str, Any]], 
                           ttl: int) -> None:
        """Cache search results"""
        if not self.redis_client:
            return
        
        try:
            # Serialize results (handle numpy arrays)
            serializable_results = self._make_serializable(results)
            cached_data = json.dumps(serializable_results)
            
            self.redis_client.setex(cache_key, ttl, cached_data)
            logger.debug(f"Cached results for key: {cache_key}")
        except Exception as e:
            logger.warning(f"Caching failed: {e}")
    
    def _make_serializable(self, results: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """Convert results to JSON-serializable format"""
        serializable_results = []
        
        for result in results:
            serializable_result = {}
            for key, value in result.items():
                if isinstance(value, np.ndarray):
                    serializable_result[key] = value.tolist()
                elif isinstance(value, (np.int64, np.int32)):
                    serializable_result[key] = int(value)
                elif isinstance(value, (np.float64, np.float32)):
                    serializable_result[key] = float(value)
                else:
                    serializable_result[key] = value
            
            serializable_results.append(serializable_result)
        
        return serializable_results
    
    def _update_query_stats(self, cache_key: str, execution_time: float, 
                           result_count: int) -> None:
        """Update query performance statistics"""
        if cache_key not in self.query_stats:
            self.query_stats[cache_key] = {
                'count': 0,
                'total_time': 0.0,
                'total_results': 0,
                'avg_time': 0.0,
                'avg_results': 0.0
            }
        
        stats = self.query_stats[cache_key]
        stats['count'] += 1
        stats['total_time'] += execution_time
        stats['total_results'] += result_count
        stats['avg_time'] = stats['total_time'] / stats['count']
        stats['avg_results'] = stats['total_results'] / stats['count']
    
    def generate_cache_key(self, *args, **kwargs) -> str:
        """Generate cache key from search parameters"""
        # Create deterministic key from parameters
        key_data = {
            'args': args,
            'kwargs': kwargs
        }
        
        key_string = json.dumps(key_data, sort_keys=True, default=str)
        return hashlib.md5(key_string.encode()).hexdigest()
    
    async def warm_cache(self, collection_name: str, popular_queries: List[str],
                        retriever: BaseVectorRetriever, 
                        embedding_generator) -> None:
        """Pre-warm cache with popular queries"""
        logger.info(f"Warming cache with {len(popular_queries)} popular queries")
        
        for query in popular_queries:
            try:
                # Generate query embedding
                query_embedding = await embedding_generator.generate_query_embedding(query)
                
                # Generate cache key
                cache_key = self.generate_cache_key(
                    'search', collection_name, query, limit=20
                )
                
                # Execute search if not cached
                cached_result = await self._get_cached_result(cache_key)
                if cached_result is None:
                    results = await retriever.search(
                        collection_name=collection_name,
                        query_vector=query_embedding,
                        limit=20
                    )
                    await self._cache_result(cache_key, results, 
                                           self.cache_ttl['search_results'])
                
                # Rate limiting to avoid overwhelming the system
                await asyncio.sleep(0.1)
                
            except Exception as e:
                logger.warning(f"Failed to warm cache for query '{query}': {e}")
        
        logger.info("Cache warming completed")
    
    def get_performance_stats(self) -> Dict[str, Any]:
        """Get performance statistics"""
        total_queries = sum(stats['count'] for stats in self.query_stats.values())
        avg_execution_time = np.mean([stats['avg_time'] for stats in self.query_stats.values()]) if self.query_stats else 0
        
        return {
            'total_queries': total_queries,
            'unique_query_patterns': len(self.query_stats),
            'avg_execution_time': avg_execution_time,
            'query_breakdown': self.query_stats
        }
