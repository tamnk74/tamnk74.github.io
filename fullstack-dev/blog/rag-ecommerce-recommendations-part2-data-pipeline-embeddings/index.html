Building Intelligent E-commerce Recommendations with RAG: Part 2 - Data Pipeline and Vector Embeddings

October 8, 2025 · 21 min read
Tam Nguyen
Full Stack Developer, Next.js for Production Creator

Welcome back to our comprehensive series on building RAG-powered e-commerce recommendation systems! In Part 1, we explored the architecture and core concepts. Now, we'll get hands-on with the foundation of any RAG system: building robust data pipelines and generating high-quality vector embeddings that capture the semantic essence of products and user behavior.

Vector embeddings are the secret sauce that enables our recommendation system to understand the relationships between products, users, and their preferences at a semantic level. Let's dive into building a production-ready data processing pipeline that transforms raw e-commerce data into meaningful vector representations.

Data Pipeline Architecture Overview​

Our data pipeline follows the Extract, Transform, Load (ETL) pattern, specifically designed for vector embedding generation:

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│   Raw Data      │    │  Processing     │    │   Vector        │
│   Sources       │───▶│  Pipeline       │───▶│   Storage       │
└─────────────────┘    └─────────────────┘    └─────────────────┘
│                      │                      │                 │
├─ Product Catalog     ├─ Data Cleaning       ├─ Product Vectors│
├─ User Profiles       ├─ Feature Extraction  ├─ User Vectors   │
├─ Interaction Logs    ├─ Embedding Generation├─ Query Vectors  │
├─ Reviews/Ratings     ├─ Quality Validation  ├─ Metadata Index │
└─ Search Queries      └─ Batch Processing    └─ Search Index   │

Product Data Processor​

# processors/product_processor.py
import pandas as pd
import numpy as np
from typing import List, Dict, Tuple
import re
from collections import Counter
import logging
from models.product import Product, ProductReview

logger = logging.getLogger(__name__)

class ProductDataProcessor:
    def __init__(self):
        self.stop_words = self._load_stop_words()
        self.category_keywords = self._load_category_keywords()
    
    def clean_product_data(self, products_df: pd.DataFrame) -> pd.DataFrame:
        """Clean and standardize product data"""
        logger.info(f"Cleaning {len(products_df)} products")
        
        # Remove duplicates
        products_df = products_df.drop_duplicates(subset=['id'])
        
        # Standardize text fields
        products_df['title'] = products_df['title'].apply(self._clean_text)
        products_df['description'] = products_df['description'].apply(self._clean_text)
        
        # Normalize prices
        products_df['price'] = pd.to_numeric(products_df['price'], errors='coerce')
        products_df = products_df.dropna(subset=['price'])
        
        # Fill missing values
        products_df['description'] = products_df['description'].fillna('')
        products_df['brand'] = products_df['brand'].fillna('Unknown')
        products_df['rating'] = products_df['rating'].fillna(0.0)
        products_df['review_count'] = products_df['review_count'].fillna(0)
        
        # Calculate derived features
        products_df['title_length'] = products_df['title'].str.len()
        products_df['description_length'] = products_df['description'].str.len()
        products_df['price_category'] = pd.cut(
            products_df['price'], 
            bins=[0, 25, 100, 500, np.inf], 
            labels=['budget', 'mid-range', 'premium', 'luxury']
        )
        
        logger.info(f"Cleaned data: {len(products_df)} products remaining")
        return products_df
    
    def extract_product_features(self, product: Product) -> Dict[str, Any]:
        """Extract meaningful features from product data"""
        features = {
            # Basic features
            'title': product.title,
            'description': product.description,
            'category': product.category.value,
            'subcategory': product.subcategory or '',
            'brand': product.brand or '',
            
            # Numeric features
            'price': product.price,
            'rating': product.rating or 0.0,
            'review_count': product.review_count,
            'popularity_score': product.popularity_score,
            
            # Text processing
            'combined_text': self._create_combined_text(product),
            'keywords': product.keywords,
            'search_terms': product.search_terms,
            
            # Categorical features
            'price_range': self._categorize_price(product.price),
            'rating_tier': self._categorize_rating(product.rating or 0.0),
            'popularity_tier': self._categorize_popularity(product.popularity_score),
            
            # Availability features
            'is_available': product.is_available,
            'stock_level': self._categorize_stock(product.stock_quantity),
        }
        
        # Add attribute-based features
        features.update(self._process_attributes(product.attributes))
        
        return features
    
    def _clean_text(self, text: str) -> str:
        """Clean and normalize text data"""
        if pd.isna(text) or not isinstance(text, str):
            return ""
        
        # Convert to lowercase
        text = text.lower()
        
        # Remove special characters but keep alphanumeric and spaces
        text = re.sub(r'[^a-z0-9\s]', ' ', text)
        
        # Remove extra whitespaces
        text = re.sub(r'\s+', ' ', text).strip()
        
        return text
    
    def _create_combined_text(self, product: Product) -> str:
        """Create combined text for embedding generation"""
        components = [
            product.title,
            product.description,
            product.brand or '',
            product.category.value,
            product.subcategory or '',
            ' '.join(product.keywords),
            ' '.join(product.search_terms)
        ]
        
        # Filter out empty components
        components = [comp for comp in components if comp and comp.strip()]
        
        return ' '.join(components)
    
    def _categorize_price(self, price: float) -> str:
        """Categorize price into ranges"""
        if price < 25:
            return 'budget'
        elif price < 100:
            return 'mid-range'
        elif price < 500:
            return 'premium'
        else:
            return 'luxury'
    
    def _categorize_rating(self, rating: float) -> str:
        """Categorize rating into tiers"""
        if rating >= 4.5:
            return 'excellent'
        elif rating >= 4.0:
            return 'very_good'
        elif rating >= 3.5:
            return 'good'
        elif rating >= 3.0:
            return 'average'
        else:
            return 'poor'
    
    def _categorize_popularity(self, score: float) -> str:
        """Categorize popularity score"""
        if score >= 0.8:
            return 'viral'
        elif score >= 0.6:
            return 'popular'
        elif score >= 0.4:
            return 'moderate'
        elif score >= 0.2:
            return 'low'
        else:
            return 'niche'
    
    def _categorize_stock(self, quantity: int) -> str:
        """Categorize stock levels"""
        if quantity == 0:
            return 'out_of_stock'
        elif quantity < 10:
            return 'low_stock'
        elif quantity < 100:
            return 'medium_stock'
        else:
            return 'high_stock'
    
    def _process_attributes(self, attributes: Dict[str, Any]) -> Dict[str, str]:
        """Process product attributes into features"""
        processed = {}
        
        for key, value in attributes.items():
            # Normalize key
            clean_key = f"attr_{re.sub(r'[^a-z0-9]', '_', key.lower())}"
            
            # Convert value to string representation
            if isinstance(value, (list, tuple)):
                clean_value = ', '.join(str(v) for v in value)
            else:
                clean_value = str(value)
            
            processed[clean_key] = self._clean_text(clean_value)
        
        return processed
    
    def _load_stop_words(self) -> set:
        """Load stop words for text processing"""
        # Basic stop words - in production, use NLTK or spaCy
        return {
            'the', 'a', 'an', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for',
            'of', 'with', 'by', 'from', 'up', 'about', 'into', 'through', 'during'
        }
    
    def _load_category_keywords(self) -> Dict[str, List[str]]:
        """Load category-specific keywords"""
        return {
            'electronics': ['smartphone', 'laptop', 'tablet', 'tv', 'camera', 'headphones'],
            'clothing': ['shirt', 'pants', 'dress', 'shoes', 'jacket', 'accessories'],
            'home_garden': ['furniture', 'decor', 'kitchen', 'garden', 'tools', 'appliances'],
            'books': ['novel', 'textbook', 'magazine', 'ebook', 'audiobook', 'reference'],
            'sports': ['equipment', 'apparel', 'fitness', 'outdoor', 'team', 'individual'],
            'beauty': ['skincare', 'makeup', 'hair', 'fragrance', 'tools', 'accessories']
        }

# processors/user_processor.py
import pandas as pd
import numpy as np
from typing import List, Dict, Tuple
from datetime import datetime, timedelta
from collections import defaultdict, Counter
import logging
from models.user import User, UserInteraction, InteractionType

logger = logging.getLogger(__name__)

class UserBehaviorProcessor:
    def __init__(self):
        self.interaction_weights = {
            InteractionType.VIEW: 1.0,
            InteractionType.CLICK: 2.0,
            InteractionType.ADD_TO_CART: 5.0,
            InteractionType.PURCHASE: 10.0,
            InteractionType.REVIEW: 8.0,
            InteractionType.SEARCH: 3.0,
            InteractionType.WISHLIST: 4.0,
        }
    
    def process_user_interactions(self, interactions_df: pd.DataFrame) -> pd.DataFrame:
        """Process and enrich user interaction data"""
        logger.info(f"Processing {len(interactions_df)} user interactions")
        
        # Convert timestamp to datetime
        interactions_df['timestamp'] = pd.to_datetime(interactions_df['timestamp'])
        
        # Add temporal features
        interactions_df['hour'] = interactions_df['timestamp'].dt.hour
        interactions_df['day_of_week'] = interactions_df['timestamp'].dt.dayofweek
        interactions_df['is_weekend'] = interactions_df['day_of_week'].isin([5, 6])
        
        # Calculate interaction weights
        interactions_df['interaction_weight'] = interactions_df['interaction_type'].map(
            lambda x: self.interaction_weights.get(InteractionType(x), 1.0)
        )
        
        # Add session-based features
        interactions_df = self._add_session_features(interactions_df)
        
        return interactions_df
    
    def create_user_profiles(self, users_df: pd.DataFrame, interactions_df: pd.DataFrame) -> Dict[str, Dict]:
        """Create comprehensive user profiles from interaction data"""
        user_profiles = {}
        
        for user_id in users_df['id'].unique():
            user_interactions = interactions_df[interactions_df['user_id'] == user_id]
            
            profile = {
                'user_id': user_id,
                'behavioral_features': self._extract_behavioral_features(user_interactions),
                'preference_features': self._extract_preference_features(user_interactions),
                'temporal_features': self._extract_temporal_features(user_interactions),
                'engagement_features': self._extract_engagement_features(user_interactions),
            }
            
            user_profiles[user_id] = profile
        
        logger.info(f"Created profiles for {len(user_profiles)} users")
        return user_profiles
    
    def _add_session_features(self, interactions_df: pd.DataFrame) -> pd.DataFrame:
        """Add session-based features to interactions"""
        # Sort by user and timestamp
        interactions_df = interactions_df.sort_values(['user_id', 'timestamp'])
        
        # Calculate time since previous interaction
        interactions_df['time_since_prev'] = interactions_df.groupby('user_id')['timestamp'].diff()
        interactions_df['time_since_prev_seconds'] = interactions_df['time_since_prev'].dt.total_seconds()
        
        # Identify session boundaries (gap > 30 minutes)
        session_boundary = interactions_df['time_since_prev_seconds'] > 1800  # 30 minutes
        interactions_df['session_number'] = session_boundary.groupby(interactions_df['user_id']).cumsum()
        
        return interactions_df
    
    def _extract_behavioral_features(self, user_interactions: pd.DataFrame) -> Dict[str, float]:
        """Extract behavioral patterns from user interactions"""
        if len(user_interactions) == 0:
            return self._get_default_behavioral_features()
        
        features = {}
        
        # Interaction type distribution
        interaction_counts = user_interactions['interaction_type'].value_counts()
        total_interactions = len(user_interactions)
        
        for interaction_type in InteractionType:
            ratio = interaction_counts.get(interaction_type.value, 0) / total_interactions
            features[f'{interaction_type.value}_ratio'] = ratio
        
        # Purchase behavior
        purchases = user_interactions[user_interactions['interaction_type'] == InteractionType.PURCHASE.value]
        features['purchase_frequency'] = len(purchases)
        features['avg_time_to_purchase'] = self._calculate_avg_time_to_purchase(user_interactions)
        
        # Browsing behavior
        views = user_interactions[user_interactions['interaction_type'] == InteractionType.VIEW.value]
        features['avg_session_length'] = user_interactions.groupby('session_id').size().mean()
        features['avg_products_per_session'] = user_interactions.groupby('session_id')['product_id'].nunique().mean()
        
        # Engagement depth
        features['bounce_rate'] = self._calculate_bounce_rate(user_interactions)
        features['repeat_visit_rate'] = self._calculate_repeat_visit_rate(user_interactions)
        
        return features
    
    def _extract_preference_features(self, user_interactions: pd.DataFrame) -> Dict[str, Any]:
        """Extract user preferences from interaction patterns"""
        if len(user_interactions) == 0:
            return {}
        
        # This would typically join with product data to get categories, brands, etc.
        # For now, we'll create placeholder logic
        features = {
            'preferred_categories': [],
            'preferred_brands': [],
            'price_sensitivity': 0.5,
            'quality_preference': 0.5,
        }
        
        # Calculate preferences based on weighted interactions
        weighted_interactions = user_interactions.copy()
        weighted_interactions['weight'] = weighted_interactions['interaction_type'].map(
            lambda x: self.interaction_weights.get(InteractionType(x), 1.0)
        )
        
        # In a real implementation, you'd join with product data here
        # features['preferred_categories'] = self._get_top_categories(weighted_interactions)
        # features['preferred_brands'] = self._get_top_brands(weighted_interactions)
        
        return features
    
    def _extract_temporal_features(self, user_interactions: pd.DataFrame) -> Dict[str, float]:
        """Extract temporal behavior patterns"""
        if len(user_interactions) == 0:
            return {}
        
        features = {}
        
        # Time-based patterns
        hour_distribution = user_interactions['hour'].value_counts(normalize=True)
        features['peak_hour'] = hour_distribution.idxmax()
        features['activity_variance'] = hour_distribution.var()
        
        # Day-of-week patterns
        dow_distribution = user_interactions['day_of_week'].value_counts(normalize=True)
        features['weekend_activity_ratio'] = user_interactions['is_weekend'].mean()
        
        # Recency and frequency
        features['days_since_last_interaction'] = (
            datetime.utcnow() - user_interactions['timestamp'].max()
        ).days
        
        features['interaction_frequency'] = len(user_interactions) / max(
            (user_interactions['timestamp'].max() - user_interactions['timestamp'].min()).days, 1
        )
        
        return features
    
    def _extract_engagement_features(self, user_interactions: pd.DataFrame) -> Dict[str, float]:
        """Extract user engagement metrics"""
        if len(user_interactions) == 0:
            return {}
        
        features = {}
        
        # Session-based metrics
        session_stats = user_interactions.groupby('session_id').agg({
            'timestamp': ['min', 'max'],
            'product_id': 'nunique',
            'interaction_type': 'count'
        }).reset_index()
        
        # Calculate session durations
        session_stats['duration'] = (
            session_stats[('timestamp', 'max')] - session_stats[('timestamp', 'min')]
        ).dt.total_seconds()
        
        features['avg_session_duration'] = session_stats['duration'].mean()
        features['total_sessions'] = len(session_stats)
        features['avg_actions_per_session'] = session_stats[('interaction_type', 'count')].mean()
        
        # Conversion metrics
        sessions_with_purchase = user_interactions[
            user_interactions['interaction_type'] == InteractionType.PURCHASE.value
        ]['session_id'].nunique()
        
        features['session_conversion_rate'] = sessions_with_purchase / len(session_stats) if len(session_stats) > 0 else 0
        
        return features
    
    def _calculate_avg_time_to_purchase(self, user_interactions: pd.DataFrame) -> float:
        """Calculate average time from first interaction to purchase"""
        purchases = user_interactions[user_interactions['interaction_type'] == InteractionType.PURCHASE.value]
        
        if len(purchases) == 0:
            return 0.0
        
        times_to_purchase = []
        
        for _, purchase in purchases.iterrows():
            session_interactions = user_interactions[
                user_interactions['session_id'] == purchase['session_id']
            ].sort_values('timestamp')
            
            first_interaction = session_interactions.iloc[0]['timestamp']
            time_to_purchase = (purchase['timestamp'] - first_interaction).total_seconds()
            times_to_purchase.append(time_to_purchase)
        
        return np.mean(times_to_purchase) if times_to_purchase else 0.0
    
    def _calculate_bounce_rate(self, user_interactions: pd.DataFrame) -> float:
        """Calculate bounce rate (single-interaction sessions)"""
        session_sizes = user_interactions.groupby('session_id').size()
        single_interaction_sessions = (session_sizes == 1).sum()
        return single_interaction_sessions / len(session_sizes) if len(session_sizes) > 0 else 0
    
    def _calculate_repeat_visit_rate(self, user_interactions: pd.DataFrame) -> float:
        """Calculate repeat visit rate"""
        unique_days = user_interactions['timestamp'].dt.date.nunique()
        total_days = (
            user_interactions['timestamp'].max() - user_interactions['timestamp'].min()
        ).days + 1
        return unique_days / total_days if total_days > 0 else 0
    
    def _get_default_behavioral_features(self) -> Dict[str, float]:
        """Return default behavioral features for users with no interactions"""
        features = {}
        for interaction_type in InteractionType:
            features[f'{interaction_type.value}_ratio'] = 0.0
        
        features.update({
            'purchase_frequency': 0.0,
            'avg_time_to_purchase': 0.0,
            'avg_session_length': 0.0,
            'avg_products_per_session': 0.0,
            'bounce_rate': 1.0,
            'repeat_visit_rate': 0.0,
        })
        
        return features

import aiohttp
import openai
from tqdm import tqdm

logger = logging.getLogger(__name__)

class BaseEmbeddingGenerator(ABC):
    """Abstract base class for embedding generators"""
    
    @abstractmethod
    async def generate_embeddings(self, texts: List[str]) -> np.ndarray:
        pass
    
    @abstractmethod
    def get_embedding_dimension(self) -> int:
        pass

class SentenceTransformerEmbedding(BaseEmbeddingGenerator):
    """Local sentence transformer embedding generator"""
    
    def __init__(self, model_name: str = "all-MiniLM-L6-v2"):
        self.model_name = model_name
        self.model = SentenceTransformer(model_name)
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)
        
        logger.info(f"Loaded SentenceTransformer model: {model_name}")
    
    async def generate_embeddings(self, texts: List[str]) -> np.ndarray:
        """Generate embeddings for a list of texts"""
        if not texts:
            return np.array([])
        
        # Process in batches to manage memory
        batch_size = 32
        all_embeddings = []
        
        for i in tqdm(range(0, len(texts), batch_size), desc="Generating embeddings"):
            batch = texts[i:i + batch_size]
            embeddings = self.model.encode(
                batch,
                convert_to_numpy=True,
                show_progress_bar=False,
                batch_size=batch_size
            )
            all_embeddings.append(embeddings)
        
        return np.vstack(all_embeddings)
    
    def get_embedding_dimension(self) -> int:
        return self.model.get_sentence_embedding_dimension()

class OpenAIEmbedding(BaseEmbeddingGenerator):
    """OpenAI API embedding generator"""
    
    def __init__(self, model_name: str = "text-embedding-ada-002", api_key: Optional[str] = None):
        self.model_name = model_name
        openai.api_key = api_key or os.getenv("OPENAI_API_KEY")
        self.dimensions = {
            "text-embedding-ada-002": 1536,
            "text-embedding-3-small": 1536,
            "text-embedding-3-large": 3072,
        }
        
        logger.info(f"Initialized OpenAI embedding model: {model_name}")
    
    async def generate_embeddings(self, texts: List[str]) -> np.ndarray:
        """Generate embeddings using OpenAI API"""
        if not texts:
            return np.array([])
        
        # OpenAI API has rate limits, so we batch requests
        batch_size = 100  # OpenAI recommends up to 2048 texts per request
        all_embeddings = []
        
        for i in tqdm(range(0, len(texts), batch_size), desc="Generating OpenAI embeddings"):
            batch = texts[i:i + batch_size]
            
            try:
                response = await openai.Embedding.acreate(
                    model=self.model_name,
                    input=batch
                )
                
                batch_embeddings = [data.embedding for data in response.data]
                all_embeddings.extend(batch_embeddings)
                
                # Rate limiting
                await asyncio.sleep(0.1)
                
            except Exception as e:
                logger.error(f"Error generating embeddings for batch {i}: {e}")
                # Fill with zeros for failed batches
                zero_embedding = [0.0] * self.get_embedding_dimension()
                all_embeddings.extend([zero_embedding] * len(batch))
        
        return np.array(all_embeddings)
    
    def get_embedding_dimension(self) -> int:
        return self.dimensions.get(self.model_name, 1536)

class ProductEmbeddingGenerator:
    """Specialized embedding generator for products"""
    
    def __init__(self, embedding_generator: BaseEmbeddingGenerator):
        self.embedding_generator = embedding_generator
        self.dimension = embedding_generator.get_embedding_dimension()
    
    async def generate_product_embeddings(self, products: List[Dict[str, Any]]) -> Dict[str, np.ndarray]:
        """Generate embeddings for products"""
        logger.info(f"Generating embeddings for {len(products)} products")
        
        # Prepare texts for embedding
        product_texts = []
        product_ids = []
        
        for product in products:
            text = self._create_product_text(product)
            product_texts.append(text)
            product_ids.append(product['id'])
        
        # Generate embeddings
        embeddings = await self.embedding_generator.generate_embeddings(product_texts)
        
        # Create mapping
        product_embeddings = {}
        for product_id, embedding in zip(product_ids, embeddings):
            product_embeddings[product_id] = embedding
        
        logger.info(f"Generated embeddings for {len(product_embeddings)} products")
        return product_embeddings
    
    def _create_product_text(self, product: Dict[str, Any]) -> str:
        """Create text representation of product for embedding"""
        components = []
        
        # Title (highest weight)
        if product.get('title'):
            components.append(f"Title: {product['title']}")
        
        # Category and subcategory
        if product.get('category'):
            components.append(f"Category: {product['category']}")
        if product.get('subcategory'):
            components.append(f"Subcategory: {product['subcategory']}")
        
        # Brand
        if product.get('brand'):
            components.append(f"Brand: {product['brand']}")
        
        # Description
        if product.get('description'):
            # Truncate very long descriptions
            description = product['description'][:500]
            components.append(f"Description: {description}")
        
        # Price range
        if product.get('price_range'):
            components.append(f"Price range: {product['price_range']}")
        
        # Keywords and tags
        if product.get('keywords'):
            components.append(f"Keywords: {', '.join(product['keywords'])}")
        
        # Attributes
        for key, value in product.get('attributes', {}).items():
            if isinstance(value, (str, int, float)):
                components.append(f"{key}: {value}")
        
        return ' '.join(components)

class UserEmbeddingGenerator:
    """Specialized embedding generator for users"""
    
    def __init__(self, embedding_generator: BaseEmbeddingGenerator):
        self.embedding_generator = embedding_generator
        self.dimension = embedding_generator.get_embedding_dimension()
    
    async def generate_user_embeddings(self, user_profiles: Dict[str, Dict[str, Any]]) -> Dict[str, np.ndarray]:
        """Generate embeddings for user profiles"""
        logger.info(f"Generating embeddings for {len(user_profiles)} users")
        
        # Prepare texts for embedding
        user_texts = []
        user_ids = []
        
        for user_id, profile in user_profiles.items():
            text = self._create_user_text(profile)
            user_texts.append(text)
            user_ids.append(user_id)
        
        # Generate embeddings
        embeddings = await self.embedding_generator.generate_embeddings(user_texts)
        
        # Create mapping
        user_embeddings = {}
        for user_id, embedding in zip(user_ids, embeddings):
            user_embeddings[user_id] = embedding
        
        logger.info(f"Generated embeddings for {len(user_embeddings)} users")
        return user_embeddings
    
    def _create_user_text(self, profile: Dict[str, Any]) -> str:
        """Create text representation of user profile for embedding"""
        components = []
        
        # Behavioral features
        behavioral = profile.get('behavioral_features', {})
        if behavioral:
            high_interactions = [k.replace('_ratio', '') for k, v in behavioral.items() 
                               if k.endswith('_ratio') and v > 0.1]
            if high_interactions:
                components.append(f"Frequently: {', '.join(high_interactions)}")
        
        # Preferences
        preferences = profile.get('preference_features', {})
        if preferences.get('preferred_categories'):
            components.append(f"Prefers categories: {', '.join(preferences['preferred_categories'])}")
        if preferences.get('preferred_brands'):
            components.append(f"Prefers brands: {', '.join(preferences['preferred_brands'])}")
        
        # Engagement level
        engagement = profile.get('engagement_features', {})
        if engagement.get('session_conversion_rate', 0) > 0.1:
            components.append("High converter")
        elif engagement.get('avg_session_duration', 0) > 300:  # 5 minutes
            components.append("Engaged browser")
        
        # Temporal patterns
        temporal = profile.get('temporal_features', {})
        if temporal.get('weekend_activity_ratio', 0) > 0.6:
            components.append("Weekend shopper")
        
        if not components:
            components.append("New user")
        
        return ' '.join(components)

class QueryEmbeddingGenerator:
    """Generate embeddings for search queries"""
    
    def __init__(self, embedding_generator: BaseEmbeddingGenerator):
        self.embedding_generator = embedding_generator
    
    async def generate_query_embedding(self, query: str, context: Optional[Dict[str, Any]] = None) -> np.ndarray:
        """Generate embedding for a search query with optional context"""
        # Enhance query with context
        enhanced_query = self._enhance_query_with_context(query, context)
        
        # Generate embedding
        embeddings = await self.embedding_generator.generate_embeddings([enhanced_query])
        return embeddings[0]
    
    def _enhance_query_with_context(self, query: str, context: Optional[Dict[str, Any]] = None) -> str:
        """Enhance query with contextual information"""
        if not context:
            return query
        
        enhancements = [query]
        
        # Add user preferences
        if context.get('preferred_categories'):
            enhancements.append(f"Categories: {', '.join(context['preferred_categories'])}")
        
        # Add price context
        if context.get('price_range'):
            enhancements.append(f"Price range: {context['price_range']}")
        
        # Add temporal context
        if context.get('season'):
            enhancements.append(f"Season: {context['season']}")
        
        return ' '.join(enhancements)

# validation/embedding_validator.py
import numpy as np
from typing import List, Dict, Tuple, Any
from sklearn.metrics.pairwise import cosine_similarity
from sklearn.decomposition import PCA
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt
import seaborn as sns
from collections import defaultdict
import logging

logger = logging.getLogger(__name__)

class EmbeddingQualityValidator:
    """Validate the quality of generated embeddings"""
    
    def __init__(self):
        self.similarity_threshold = 0.7
        self.diversity_threshold = 0.3
    
    def validate_product_embeddings(
        self, 
        product_embeddings: Dict[str, np.ndarray],
        products: List[Dict[str, Any]]
    ) -> Dict[str, Any]:
        """Comprehensive validation of product embeddings"""
        logger.info("Validating product embeddings...")
        
        results = {
            'total_products': len(product_embeddings),
            'embedding_dimension': len(next(iter(product_embeddings.values()))),
            'quality_metrics': {},
            'recommendations': []
        }
        
        # Create product lookup
        product_lookup = {p['id']: p for p in products}
        
        # Test semantic similarity
        similarity_results = self._test_semantic_similarity(product_embeddings, product_lookup)
        results['quality_metrics']['semantic_similarity'] = similarity_results
        
        # Test category clustering
        clustering_results = self._test_category_clustering(product_embeddings, product_lookup)
        results['quality_metrics']['category_clustering'] = clustering_results
        
        # Test embedding diversity
        diversity_results = self._test_embedding_diversity(product_embeddings)
        results['quality_metrics']['diversity'] = diversity_results
        
        # Generate recommendations
        results['recommendations'] = self._generate_quality_recommendations(results['quality_metrics'])
        
        logger.info(f"Validation complete. Overall quality score: {self._calculate_overall_score(results['quality_metrics']):.3f}")
        
        return results
    
    def _test_semantic_similarity(
        self, 
        embeddings: Dict[str, np.ndarray], 
        products: Dict[str, Dict[str, Any]]
    ) -> Dict[str, float]:
        """Test if semantically similar products have similar embeddings"""
        # Sample products for testing (to avoid O(n²) computation)
        sample_size = min(100, len(embeddings))
        sampled_ids = list(embeddings.keys())[:sample_size]
        
        similarity_scores = []
        
        for i, product_id_1 in enumerate(sampled_ids):
            for product_id_2 in sampled_ids[i+1:]:
                # Calculate embedding similarity
                emb_sim = cosine_similarity(
                    embeddings[product_id_1].reshape(1, -1),
                    embeddings[product_id_2].reshape(1, -1)
                )[0, 0]
                
                # Calculate semantic similarity (based on category, brand, etc.)
                semantic_sim = self._calculate_semantic_similarity(
                    products[product_id_1], 
                    products[product_id_2]
                )
                
                similarity_scores.append({
                    'embedding_similarity': emb_sim,
                    'semantic_similarity': semantic_sim
                })
        
        # Calculate correlation
        emb_sims = [s['embedding_similarity'] for s in similarity_scores]
        sem_sims = [s['semantic_similarity'] for s in similarity_scores]
        
        correlation = np.corrcoef(emb_sims, sem_sims)[0, 1]
        
        return {
            'correlation': correlation,
            'avg_embedding_similarity': np.mean(emb_sims),
            'avg_semantic_similarity': np.mean(sem_sims),
            'sample_size': len(similarity_scores)
        }
    
    def _test_category_clustering(
        self, 
        embeddings: Dict[str, np.ndarray], 
        products: Dict[str, Dict[str, Any]]
    ) -> Dict[str, Any]:
        """Test if products from same category cluster together"""
        # Group by category
        category_embeddings = defaultdict(list)
        category_products = defaultdict(list)
        
        for product_id, embedding in embeddings.items():
            if product_id in products:
                category = products[product_id].get('category', 'unknown')
                category_embeddings[category].append(embedding)
                category_products[category].append(product_id)
        
        # Calculate intra-category similarity
        intra_similarities = {}
        for category, embs in category_embeddings.items():
            if len(embs) > 1:
                emb_matrix = np.array(embs)
                sim_matrix = cosine_similarity(emb_matrix)
                # Average similarity excluding diagonal
                mask = ~np.eye(sim_matrix.shape[0], dtype=bool)
                avg_similarity = sim_matrix[mask].mean()
                intra_similarities[category] = avg_similarity
        
        # Calculate inter-category similarity
        categories = list(category_embeddings.keys())
        inter_similarities = []
        
        for i, cat1 in enumerate(categories):
            for cat2 in categories[i+1:]:
                if len(category_embeddings[cat1]) > 0 and len(category_embeddings[cat2]) > 0:
                    # Calculate average similarity between categories
                    emb1 = np.mean(category_embeddings[cat1], axis=0)
                    emb2 = np.mean(category_embeddings[cat2], axis=0)
                    sim = cosine_similarity(emb1.reshape(1, -1), emb2.reshape(1, -1))[0, 0]
                    inter_similarities.append(sim)
        
        return {
            'intra_category_similarity': np.mean(list(intra_similarities.values())),
            'inter_category_similarity': np.mean(inter_similarities),
            'separation_score': np.mean(list(intra_similarities.values())) - np.mean(inter_similarities),
            'categories_analyzed': len(category_embeddings)
        }
    
    def _test_embedding_diversity(self, embeddings: Dict[str, np.ndarray]) -> Dict[str, float]:
        """Test diversity of embeddings to avoid mode collapse"""
        embedding_matrix = np.array(list(embeddings.values()))
        
        # Calculate pairwise similarities
        similarity_matrix = cosine_similarity(embedding_matrix)
        
        # Remove diagonal (self-similarity)
        mask = ~np.eye(similarity_matrix.shape[0], dtype=bool)
        similarities = similarity_matrix[mask]
        
        # Calculate diversity metrics
        avg_similarity = similarities.mean()
        similarity_std = similarities.std()
        
        # PCA to check dimensionality usage
        pca = PCA()
        pca.fit(embedding_matrix)
        explained_variance_ratio = pca.explained_variance_ratio_
        
        # Calculate effective dimensionality (95% variance)
        cumsum_variance = np.cumsum(explained_variance_ratio)
        effective_dims = np.argmax(cumsum_variance >= 0.95) + 1
        
        return {
            'average_similarity': avg_similarity,
            'similarity_std': similarity_std,
            'effective_dimensions': effective_dims,
            'total_dimensions': embedding_matrix.shape[1],
            'dimension_usage_ratio': effective_dims / embedding_matrix.shape[1]
        }
    
    def _calculate_semantic_similarity(self, product1: Dict[str, Any], product2: Dict[str, Any]) -> float:
        """Calculate semantic similarity between two products"""
        similarity = 0.0
        
        # Category similarity (highest weight)
        if product1.get('category') == product2.get('category'):
            similarity += 0.4
            
            # Subcategory similarity
            if product1.get('subcategory') == product2.get('subcategory'):
                similarity += 0.2
        
        # Brand similarity
        if product1.get('brand') == product2.get('brand'):
            similarity += 0.2
        
        # Price range similarity
        if product1.get('price_range') == product2.get('price_range'):
            similarity += 0.1
        
        # Title similarity (simple word overlap)
        title1_words = set(product1.get('title', '').lower().split())
        title2_words = set(product2.get('title', '').lower().split())
        if title1_words and title2_words:
            word_overlap = len(title1_words & title2_words) / len(title1_words | title2_words)
            similarity += 0.1 * word_overlap
        
        return similarity
    
    def _calculate_overall_score(self, metrics: Dict[str, Any]) -> float:
        """Calculate overall quality score"""
        scores = []
        
        # Semantic similarity correlation (should be positive)
        if 'semantic_similarity' in metrics:
            correlation = metrics['semantic_similarity'].get('correlation', 0)
            scores.append(max(0, correlation))
        
        # Category clustering separation
        if 'category_clustering' in metrics:
            separation = metrics['category_clustering'].get('separation_score', 0)
            scores.append(max(0, min(1, separation)))
        
        # Diversity (should not be too high or too low)
        if 'diversity' in metrics:
            avg_sim = metrics['diversity'].get('average_similarity', 0.5)
            # Optimal similarity around 0.3-0.5
            diversity_score = 1 - abs(avg_sim - 0.4) / 0.4
            scores.append(max(0, diversity_score))
        
        return np.mean(scores) if scores else 0.0
    
    def _generate_quality_recommendations(self, metrics: Dict[str, Any]) -> List[str]:
        """Generate recommendations based on quality metrics"""
        recommendations = []
        
        # Check semantic similarity
        if 'semantic_similarity' in metrics:
            correlation = metrics['semantic_similarity'].get('correlation', 0)
            if correlation < 0.3:
                recommendations.append(
                    "Low semantic similarity correlation. Consider improving text preprocessing "
                    "or using a more powerful embedding model."
                )
        
        # Check category clustering
        if 'category_clustering' in metrics:
            separation = metrics['category_clustering'].get('separation_score', 0)
            if separation < 0.1:
                recommendations.append(
                    "Poor category separation. Consider adding category information to "
                    "product text or using category-specific embeddings."
                )
        
        # Check diversity
        if 'diversity' in metrics:
            avg_sim = metrics['diversity'].get('average_similarity', 0.5)
            if avg_sim > 0.8:
                recommendations.append(
                    "Embeddings are too similar (mode collapse). Consider using a different "
                    "embedding model or improving text diversity."
                )
            elif avg_sim < 0.1:
                recommendations.append(
                    "Embeddings are too diverse. Consider improving text preprocessing "
                    "or using a more consistent embedding approach."
                )
        
        if not recommendations:
            recommendations.append("Embedding quality looks good!")
        
        return recommendations
