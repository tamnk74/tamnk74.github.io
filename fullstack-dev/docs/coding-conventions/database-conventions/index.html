Database Design Principles​

Normalization and Denormalization​

-- Good: Normalized design for transactional data
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE user_profiles (
    user_id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    bio TEXT,
    avatar_url VARCHAR(500),
    date_of_birth DATE,
    phone_number VARCHAR(20),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE addresses (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    type VARCHAR(20) NOT NULL CHECK (type IN ('billing', 'shipping', 'home', 'work')),
    street_line_1 VARCHAR(255) NOT NULL,
    street_line_2 VARCHAR(255),
    city VARCHAR(100) NOT NULL,
    state_province VARCHAR(100),
    postal_code VARCHAR(20),
    country_code CHAR(2) NOT NULL,
    is_default BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(user_id, type, is_default) WHERE is_default = TRUE
);

-- Strategic denormalization for read-heavy analytics
CREATE TABLE order_analytics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID NOT NULL REFERENCES orders(id),
    user_id UUID NOT NULL,
    user_email VARCHAR(255) NOT NULL, -- Denormalized for faster queries
    order_total DECIMAL(10,2) NOT NULL,
    order_date DATE NOT NULL,
    customer_segment VARCHAR(50),
    product_categories TEXT[], -- Array for faster filtering
    order_month DATE GENERATED ALWAYS AS (DATE_TRUNC('month', order_date)) STORED,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

Referential Integrity and Constraints​

-- Foreign key constraints with appropriate actions
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
    status VARCHAR(20) NOT NULL DEFAULT 'pending' 
        CHECK (status IN ('pending', 'confirmed', 'processing', 'shipped', 'delivered', 'cancelled')),
    total_amount DECIMAL(10,2) NOT NULL CHECK (total_amount >= 0),
    currency_code CHAR(3) NOT NULL DEFAULT 'USD',
    shipping_address_id UUID REFERENCES addresses(id) ON DELETE SET NULL,
    billing_address_id UUID REFERENCES addresses(id) ON DELETE SET NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    -- Ensure billing address belongs to the user
    CONSTRAINT fk_billing_address_user 
        FOREIGN KEY (billing_address_id, user_id) 
        REFERENCES addresses(id, user_id),
    
    -- Ensure shipping address belongs to the user
    CONSTRAINT fk_shipping_address_user 
        FOREIGN KEY (shipping_address_id, user_id) 
        REFERENCES addresses(id, user_id)
);

-- Check constraints for business rules
CREATE TABLE products (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    sku VARCHAR(50) UNIQUE NOT NULL,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL CHECK (price > 0),
    cost DECIMAL(10,2) CHECK (cost >= 0 AND cost <= price),
    weight_kg DECIMAL(8,3) CHECK (weight_kg > 0),
    stock_quantity INTEGER NOT NULL DEFAULT 0 CHECK (stock_quantity >= 0),
    min_stock_level INTEGER NOT NULL DEFAULT 0 CHECK (min_stock_level >= 0),
    max_stock_level INTEGER CHECK (max_stock_level IS NULL OR max_stock_level >= min_stock_level),
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    category_id UUID REFERENCES categories(id) ON DELETE SET NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    -- Ensure logical stock levels
    CONSTRAINT check_stock_levels 
        CHECK (max_stock_level IS NULL OR stock_quantity <= max_stock_level)
);

Naming Conventions​

Table and Column Naming​

-- Good: Clear, consistent naming conventions

-- Tables: plural nouns, snake_case
CREATE TABLE user_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    session_token VARCHAR(255) UNIQUE NOT NULL,
    expires_at TIMESTAMP WITH TIME ZONE NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    last_accessed_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    ip_address INET,
    user_agent TEXT
);

-- Junction tables: combine table names
CREATE TABLE user_roles (
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    assigned_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    assigned_by UUID REFERENCES users(id),
    
    PRIMARY KEY (user_id, role_id)
);

-- Audit tables: prefix with original table name
CREATE TABLE order_audit_log (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID NOT NULL REFERENCES orders(id) ON DELETE CASCADE,
    action VARCHAR(20) NOT NULL CHECK (action IN ('CREATE', 'UPDATE', 'DELETE')),
    old_values JSONB,
    new_values JSONB,
    changed_by UUID REFERENCES users(id),
    changed_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    change_reason TEXT
);

Index Naming​

-- Index naming convention: idx_[table]_[columns]_[type]
CREATE INDEX idx_orders_user_id_status ON orders(user_id, status);
CREATE INDEX idx_orders_created_at_btree ON orders(created_at);
CREATE INDEX idx_products_name_gin ON products USING gin(to_tsvector('english', name));

-- Unique index naming: uniq_[table]_[columns]
CREATE UNIQUE INDEX uniq_users_email_active ON users(email) WHERE is_active = TRUE;

-- Partial index naming: idx_[table]_[columns]_[condition]
CREATE INDEX idx_orders_total_amount_large ON orders(total_amount) 
WHERE total_amount > 1000;

-- Functional index naming: idx_[table]_[function]_[columns]
CREATE INDEX idx_users_email_lower ON users(lower(email));

Schema Design Patterns​

Soft Delete Pattern​

-- Soft delete implementation
CREATE TABLE categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    parent_category_id UUID REFERENCES categories(id),
    display_order INTEGER DEFAULT 0,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP WITH TIME ZONE,
    
    -- Unique constraint that ignores soft-deleted records
    UNIQUE(name, parent_category_id) WHERE deleted_at IS NULL
);

-- View for active records only
CREATE VIEW active_categories AS
SELECT id, name, description, parent_category_id, display_order, 
       created_at, updated_at
FROM categories
WHERE deleted_at IS NULL;

-- Function to soft delete with cascade
CREATE OR REPLACE FUNCTION soft_delete_category(category_id UUID)
RETURNS VOID AS $$
BEGIN
    -- Soft delete the category
    UPDATE categories 
    SET deleted_at = CURRENT_TIMESTAMP,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = category_id AND deleted_at IS NULL;
    
    -- Soft delete all child categories
    UPDATE categories 
    SET deleted_at = CURRENT_TIMESTAMP,
        updated_at = CURRENT_TIMESTAMP
    WHERE parent_category_id = category_id AND deleted_at IS NULL;
    
    -- Update products to remove category reference
    UPDATE products 
    SET category_id = NULL,
        updated_at = CURRENT_TIMESTAMP
    WHERE category_id = category_id;
END;
$$ LANGUAGE plpgsql;

Audit Trail Pattern​

-- Generic audit trail implementation
CREATE TABLE audit_log (
    id BIGSERIAL PRIMARY KEY,
    table_name VARCHAR(50) NOT NULL,
    record_id UUID NOT NULL,
    operation VARCHAR(10) NOT NULL CHECK (operation IN ('INSERT', 'UPDATE', 'DELETE')),
    old_values JSONB,
    new_values JSONB,
    changed_fields TEXT[],
    user_id UUID REFERENCES users(id),
    session_id UUID,
    ip_address INET,
    user_agent TEXT,
    timestamp TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    -- Index for common queries
    INDEX idx_audit_log_table_record (table_name, record_id),
    INDEX idx_audit_log_timestamp (timestamp),
    INDEX idx_audit_log_user_id (user_id)
);

-- Trigger function for automatic audit logging
CREATE OR REPLACE FUNCTION create_audit_log()
RETURNS TRIGGER AS $$
DECLARE
    old_data JSONB;
    new_data JSONB;
    changed_fields TEXT[] = '{}';
    field_name TEXT;
BEGIN
    -- Determine operation type
    IF TG_OP = 'DELETE' THEN
        old_data = to_jsonb(OLD);
        new_data = NULL;
    ELSIF TG_OP = 'INSERT' THEN
        old_data = NULL;
        new_data = to_jsonb(NEW);
    ELSIF TG_OP = 'UPDATE' THEN
        old_data = to_jsonb(OLD);
        new_data = to_jsonb(NEW);
        
        -- Identify changed fields
        FOR field_name IN SELECT key FROM jsonb_each(new_data) LOOP
            IF old_data->field_name IS DISTINCT FROM new_data->field_name THEN
                changed_fields = array_append(changed_fields, field_name);
            END IF;
        END LOOP;
    END IF;
    
    -- Insert audit record
    INSERT INTO audit_log (
        table_name, record_id, operation, old_values, new_values, 
        changed_fields, user_id, session_id, ip_address
    ) VALUES (
        TG_TABLE_NAME,
        COALESCE(NEW.id, OLD.id),
        TG_OP,
        old_data,
        new_data,
        changed_fields,
        current_setting('app.current_user_id', TRUE)::UUID,
        current_setting('app.current_session_id', TRUE)::UUID,
        current_setting('app.current_ip_address', TRUE)::INET
    );
    
    RETURN COALESCE(NEW, OLD);
END;
$$ LANGUAGE plpgsql;

-- Apply audit trigger to tables
CREATE TRIGGER audit_trigger_users
    AFTER INSERT OR UPDATE OR DELETE ON users
    FOR EACH ROW EXECUTE FUNCTION create_audit_log();

Versioning Pattern​

-- Document versioning system
CREATE TABLE documents (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    current_version_id UUID,
    title VARCHAR(255) NOT NULL,
    created_by UUID NOT NULL REFERENCES users(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE document_versions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id UUID NOT NULL REFERENCES documents(id) ON DELETE CASCADE,
    version_number INTEGER NOT NULL,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    content_type VARCHAR(50) DEFAULT 'text/plain',
    change_summary TEXT,
    created_by UUID NOT NULL REFERENCES users(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(document_id, version_number)
);

-- Add foreign key after both tables are created
ALTER TABLE documents 
ADD CONSTRAINT fk_documents_current_version 
FOREIGN KEY (current_version_id) REFERENCES document_versions(id);

-- Function to create new document version
CREATE OR REPLACE FUNCTION create_document_version(
    p_document_id UUID,
    p_title VARCHAR(255),
    p_content TEXT,
    p_change_summary TEXT,
    p_user_id UUID
) RETURNS UUID AS $$
DECLARE
    next_version INTEGER;
    version_id UUID;
BEGIN
    -- Get next version number
    SELECT COALESCE(MAX(version_number), 0) + 1 
    INTO next_version
    FROM document_versions 
    WHERE document_id = p_document_id;
    
    -- Create new version
    INSERT INTO document_versions (
        document_id, version_number, title, content, 
        change_summary, created_by
    ) VALUES (
        p_document_id, next_version, p_title, p_content,
        p_change_summary, p_user_id
    ) RETURNING id INTO version_id;
    
    -- Update document to point to new version
    UPDATE documents 
    SET current_version_id = version_id,
        title = p_title,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_document_id;
    
    RETURN version_id;
END;
$$ LANGUAGE plpgsql;

Indexing Strategies​

Composite Index Design​

-- Order of columns in composite indexes matters
-- Most selective column first, then by query patterns

-- Good: Status is selective, created_at for range queries
CREATE INDEX idx_orders_status_created_at ON orders(status, created_at);

-- Query this index supports efficiently:
SELECT * FROM orders 
WHERE status = 'pending' 
AND created_at >= '2024-01-01'
ORDER BY created_at DESC;

-- Multi-column index for complex queries
CREATE INDEX idx_products_category_active_price ON products(category_id, is_active, price)
WHERE is_active = TRUE;

-- Supports queries like:
SELECT * FROM products 
WHERE category_id = 'uuid-here' 
AND is_active = TRUE 
AND price BETWEEN 10.00 AND 100.00;

Partial and Functional Indexes​

-- Partial indexes for specific conditions
CREATE INDEX idx_orders_pending_created_at ON orders(created_at)
WHERE status = 'pending';

CREATE INDEX idx_users_active_email ON users(email)
WHERE is_active = TRUE AND deleted_at IS NULL;

-- Functional indexes for case-insensitive searches
CREATE INDEX idx_users_email_lower ON users(lower(email));
CREATE INDEX idx_products_name_lower ON products(lower(name));

-- GIN indexes for array and JSON columns
CREATE INDEX idx_order_analytics_categories_gin ON order_analytics 
USING gin(product_categories);

CREATE INDEX idx_user_preferences_gin ON user_profiles 
USING gin(preferences) WHERE preferences IS NOT NULL;

-- Full-text search indexes
CREATE INDEX idx_products_search_text ON products 
USING gin(to_tsvector('english', name || ' ' || COALESCE(description, '')));

-- Usage example for full-text search:
SELECT id, name, description,
       ts_rank(to_tsvector('english', name || ' ' || COALESCE(description, '')), 
               plainto_tsquery('english', 'search term')) as rank
FROM products
WHERE to_tsvector('english', name || ' ' || COALESCE(description, '')) 
      @@ plainto_tsquery('english', 'search term')
ORDER BY rank DESC;

Index Maintenance​

-- Monitor index usage
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_scan as index_scans,
    idx_tup_read as tuples_read,
    idx_tup_fetch as tuples_fetched
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;

-- Find unused indexes
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_scan
FROM pg_stat_user_indexes
WHERE idx_scan = 0
AND indexname NOT LIKE '%_pkey';

-- Index bloat monitoring
SELECT 
    schemaname,
    tablename,
    indexname,
    pg_size_pretty(pg_relation_size(indexrelid)) as index_size,
    idx_scan,
    round(100 * idx_scan::numeric / NULLIF(seq_scan + idx_scan, 0), 2) as index_usage_pct
FROM pg_stat_user_indexes
JOIN pg_stat_user_tables USING (schemaname, tablename)
ORDER BY pg_relation_size(indexrelid) DESC;

Query Optimization​

Efficient Join Patterns​

-- Good: Use appropriate join types and conditions
SELECT 
    u.id,
    u.email,
    u.first_name,
    u.last_name,
    p.bio,
    COUNT(o.id) as order_count,
    SUM(o.total_amount) as total_spent
FROM users u
LEFT JOIN user_profiles p ON u.id = p.user_id
LEFT JOIN orders o ON u.id = o.user_id AND o.status = 'delivered'
WHERE u.is_active = TRUE
GROUP BY u.id, u.email, u.first_name, u.last_name, p.bio
HAVING COUNT(o.id) > 0
ORDER BY total_spent DESC NULLS LAST
LIMIT 100;

-- Use EXISTS instead of IN for better performance with large datasets
SELECT p.id, p.name, p.price
FROM products p
WHERE EXISTS (
    SELECT 1 FROM order_items oi 
    WHERE oi.product_id = p.id 
    AND oi.created_at >= CURRENT_DATE - INTERVAL '30 days'
)
AND p.is_active = TRUE;

-- Window functions for analytics
SELECT 
    user_id,
    order_date,
    total_amount,
    ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY order_date) as order_sequence,
    LAG(total_amount) OVER (PARTITION BY user_id ORDER BY order_date) as previous_order_amount,
    SUM(total_amount) OVER (PARTITION BY user_id ORDER BY order_date ROWS UNBOUNDED PRECEDING) as running_total
FROM orders
WHERE status = 'delivered'
ORDER BY user_id, order_date;

Query Performance Optimization​

-- Use EXPLAIN ANALYZE to understand query performance
EXPLAIN (ANALYZE, BUFFERS, FORMAT JSON)
SELECT 
    u.email,
    COUNT(o.id) as order_count,
    AVG(o.total_amount) as avg_order_value
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE o.created_at >= CURRENT_DATE - INTERVAL '90 days'
GROUP BY u.id, u.email
HAVING COUNT(o.id) >= 5
ORDER BY avg_order_value DESC;

-- Optimize with CTEs for complex queries
WITH recent_orders AS (
    SELECT 
        user_id,
        COUNT(*) as order_count,
        AVG(total_amount) as avg_order_value,
        MAX(created_at) as last_order_date
    FROM orders 
    WHERE created_at >= CURRENT_DATE - INTERVAL '90 days'
    AND status = 'delivered'
    GROUP BY user_id
    HAVING COUNT(*) >= 5
),
user_segments AS (
    SELECT 
        user_id,
        order_count,
        avg_order_value,
        last_order_date,
        CASE 
            WHEN avg_order_value >= 500 THEN 'premium'
            WHEN avg_order_value >= 100 THEN 'standard'
            ELSE 'budget'
        END as customer_segment
    FROM recent_orders
)
SELECT 
    u.email,
    u.first_name,
    u.last_name,
    us.customer_segment,
    us.order_count,
    us.avg_order_value,
    us.last_order_date
FROM users u
JOIN user_segments us ON u.id = us.user_id
ORDER BY us.avg_order_value DESC;

Pagination Best Practices​

-- Cursor-based pagination for large datasets
-- Instead of OFFSET which becomes slow with large offsets
CREATE OR REPLACE FUNCTION get_orders_cursor(
    cursor_id UUID DEFAULT NULL,
    page_size INTEGER DEFAULT 20,
    order_direction TEXT DEFAULT 'ASC'
) RETURNS TABLE (
    id UUID,
    user_id UUID,
    total_amount DECIMAL(10,2),
    created_at TIMESTAMP WITH TIME ZONE,
    has_next_page BOOLEAN
) AS $$
DECLARE
    direction_operator TEXT;
    order_clause TEXT;
BEGIN
    -- Determine comparison operator based on direction
    direction_operator := CASE WHEN order_direction = 'DESC' THEN '<' ELSE '>' END;
    order_clause := 'ORDER BY created_at ' || order_direction || ', id ' || order_direction;
    
    RETURN QUERY EXECUTE format('
        WITH page_data AS (
            SELECT o.id, o.user_id, o.total_amount, o.created_at
            FROM orders o
            WHERE ($1 IS NULL OR o.created_at %s $2 OR (o.created_at = $2 AND o.id %s $1))
            %s
            LIMIT $3 + 1
        )
        SELECT 
            pd.id,
            pd.user_id,
            pd.total_amount,
            pd.created_at,
            (SELECT COUNT(*) FROM page_data) > $3 as has_next_page
        FROM page_data pd
        LIMIT $3',
        direction_operator,
        direction_operator,
        order_clause
    ) USING cursor_id, 
            CASE WHEN cursor_id IS NOT NULL THEN 
                (SELECT created_at FROM orders WHERE id = cursor_id) 
            END,
            page_size;
END;
$$ LANGUAGE plpgsql;

-- Migration: normalize_user_names
BEGIN;

-- Create backup table
CREATE TABLE users_backup_20240115 AS 
SELECT * FROM users;

-- Add temporary columns for normalized data
ALTER TABLE users 
ADD COLUMN first_name_normalized VARCHAR(100),

SET 
    first_name_normalized = TRIM(REGEXP_REPLACE(first_name, '\s+', ' ', 'g')),
    last_name_normalized = TRIM(REGEXP_REPLACE(last_name, '\s+', ' ', 'g'))

-- Validate data transformation
DO $$
DECLARE
    invalid_count INTEGER;
BEGIN
    SELECT COUNT(*) INTO invalid_count
    FROM users 
    WHERE (first_name IS NOT NULL AND first_name_normalized IS NULL)
       OR (last_name IS NOT NULL AND last_name_normalized IS NULL);
       
    IF invalid_count > 0 THEN
        RAISE EXCEPTION 'Data transformation validation failed: % invalid records', invalid_count;
    END IF;

ALTER TABLE users

Security Best Practices​

Row Level Security (RLS)​

-- Enable RLS on sensitive tables
ALTER TABLE orders ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE addresses ENABLE ROW LEVEL SECURITY;

-- Policy for users to see only their own orders
CREATE POLICY user_orders_policy ON orders
FOR ALL TO app_user
USING (user_id = current_setting('app.current_user_id')::UUID);

-- Policy for admins to see all orders
CREATE POLICY admin_orders_policy ON orders
FOR ALL TO app_admin
USING (true);

-- Policy for users to manage their own profile
CREATE POLICY user_profile_policy ON user_profiles
FOR ALL TO app_user
USING (user_id = current_setting('app.current_user_id')::UUID)
WITH CHECK (user_id = current_setting('app.current_user_id')::UUID);

-- Read-only policy for public product information
CREATE POLICY products_public_read ON products
FOR SELECT TO app_user, app_guest
USING (is_active = TRUE AND deleted_at IS NULL);

Data Encryption and Masking​

-- Store sensitive data encrypted
CREATE EXTENSION IF NOT EXISTS pgcrypto;

CREATE TABLE payment_methods (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    type VARCHAR(20) NOT NULL CHECK (type IN ('credit_card', 'debit_card', 'bank_account')),
    encrypted_number BYTEA NOT NULL, -- Encrypted card/account number
    last_four CHAR(4) NOT NULL, -- Last 4 digits for display
    encrypted_expiry BYTEA, -- Encrypted expiry date
    cardholder_name VARCHAR(255),
    is_default BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(user_id, is_default) WHERE is_default = TRUE
);

-- Function to encrypt sensitive data
CREATE OR REPLACE FUNCTION encrypt_payment_data(
    card_number TEXT,
    expiry_date TEXT,
    encryption_key TEXT
) RETURNS TABLE (
    encrypted_number BYTEA,
    encrypted_expiry BYTEA,
    last_four CHAR(4)
) AS $$
BEGIN
    RETURN QUERY SELECT 
        pgp_sym_encrypt(card_number, encryption_key)::BYTEA,
        pgp_sym_encrypt(expiry_date, encryption_key)::BYTEA,
        RIGHT(card_number, 4)::CHAR(4);
END;
$$ LANGUAGE plpgsql;

-- View for displaying masked payment methods
CREATE VIEW user_payment_methods_view AS

    CASE 
        WHEN type = 'credit_card' THEN '**** **** **** ' || last_four
        WHEN type = 'bank_account' THEN '****' || last_four
        ELSE '**** **** **** ****'
    END as masked_number,
    cardholder_name,
    is_default,
    created_at

-- Use parameterized queries and proper escaping
-- Example of secure dynamic query building
CREATE OR REPLACE FUNCTION search_products(
    search_term TEXT,
    category_filter UUID DEFAULT NULL,
    min_price DECIMAL DEFAULT NULL,
    max_price DECIMAL DEFAULT NULL,
    sort_by TEXT DEFAULT 'name',
    sort_order TEXT DEFAULT 'ASC'
) RETURNS TABLE (
    id UUID,
    name VARCHAR(255),
    price DECIMAL(10,2),
    category_name VARCHAR(100)
) AS $$
DECLARE
    query_text TEXT;
    sort_column TEXT;
BEGIN
    -- Validate and sanitize sort parameters
    CASE sort_by
        WHEN 'name' THEN sort_column := 'p.name';
        WHEN 'price' THEN sort_column := 'p.price';
        WHEN 'created_at' THEN sort_column := 'p.created_at';
        ELSE sort_column := 'p.name';
    END CASE;
    
    IF sort_order NOT IN ('ASC', 'DESC') THEN
        sort_order := 'ASC';
    END IF;
    
    -- Build query with proper parameterization
    query_text := '
        SELECT p.id, p.name, p.price, c.name as category_name
        FROM products p
        LEFT JOIN categories c ON p.category_id = c.id
        WHERE p.is_active = TRUE
        AND p.deleted_at IS NULL
        AND ($1 IS NULL OR to_tsvector(''english'', p.name || '' '' || COALESCE(p.description, '''')) @@ plainto_tsquery(''english'', $1))
        AND ($2 IS NULL OR p.category_id = $2)
        AND ($3 IS NULL OR p.price >= $3)
        AND ($4 IS NULL OR p.price <= $4)
        ORDER BY ' || sort_column || ' ' || sort_order;
    
    RETURN QUERY EXECUTE query_text 
    USING search_term, category_filter, min_price, max_price;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Grant execute permission to specific roles
GRANT EXECUTE ON FUNCTION search_products TO app_user;

Performance Monitoring​

Query Performance Tracking​

-- Enable query statistics collection
-- Add to postgresql.conf:
-- shared_preload_libraries = 'pg_stat_statements'
-- pg_stat_statements.track = all
-- pg_stat_statements.max = 10000

-- Query to find slow queries
SELECT 
    query,
    calls,
    total_time,
    total_time/calls as avg_time,
    stddev_time,
    rows,
    100.0 * shared_blks_hit / nullif(shared_blks_hit + shared_blks_read, 0) AS hit_percent
FROM pg_stat_statements
WHERE calls > 100
ORDER BY total_time DESC
LIMIT 20;

-- Find queries with high I/O
SELECT 
    query,
    calls,
    shared_blks_read,
    shared_blks_written,
    shared_blks_dirtied,
    shared_blks_hit,
    temp_blks_read,
    temp_blks_written
FROM pg_stat_statements
WHERE shared_blks_read > 1000 OR temp_blks_read > 100
ORDER BY shared_blks_read + temp_blks_read DESC;

Database Health Monitoring​

-- Create monitoring views for database health
CREATE VIEW database_health_summary AS
SELECT 
    'connections' as metric,
    current_setting('max_connections')::INTEGER as max_value,
    COUNT(*) as current_value,
    ROUND(100.0 * COUNT(*) / current_setting('max_connections')::INTEGER, 2) as usage_percent
FROM pg_stat_activity
WHERE state IS NOT NULL
UNION ALL
SELECT 
    'database_size' as metric,
    NULL as max_value,
    pg_database_size(current_database()) as current_value,
    NULL as usage_percent
UNION ALL
SELECT 
    'cache_hit_ratio' as metric,
    NULL as max_value,
    ROUND(100.0 * sum(blks_hit) / nullif(sum(blks_hit + blks_read), 0), 2)::BIGINT as current_value,
    NULL as usage_percent
FROM pg_stat_database
WHERE datname = current_database();

-- Monitor table bloat
CREATE VIEW table_bloat_summary AS
SELECT 
    schemaname,
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) as total_size,
    pg_size_pretty(pg_relation_size(schemaname||'.'||tablename)) as table_size,
    n_tup_ins as inserts,
    n_tup_upd as updates,
    n_tup_del as deletes,
    n_dead_tup as dead_tuples,
    CASE WHEN n_live_tup > 0 
         THEN ROUND(100.0 * n_dead_tup / n_live_tup, 2) 
         ELSE 0 
    END as dead_tuple_percent
FROM pg_stat_user_tables
WHERE n_live_tup > 1000
ORDER BY n_dead_tup DESC;

-- Automated maintenance tasks
CREATE OR REPLACE FUNCTION run_maintenance_tasks()
RETURNS TEXT AS $$
DECLARE
    result TEXT := '';
    table_rec RECORD;
BEGIN
    -- Auto-vacuum tables with high dead tuple percentage
    FOR table_rec IN 
        SELECT schemaname, tablename, dead_tuple_percent
        FROM table_bloat_summary
        WHERE dead_tuple_percent > 10
    LOOP
        EXECUTE 'VACUUM ANALYZE ' || quote_ident(table_rec.schemaname) || '.' || quote_ident(table_rec.tablename);
        result := result || 'Vacuumed ' || table_rec.schemaname || '.' || table_rec.tablename || E'\n';
    END LOOP;
    
    -- Update table statistics
    ANALYZE;
    result := result || 'Updated table statistics' || E'\n';
    
    RETURN result;
END;
$$ LANGUAGE plpgsql;
