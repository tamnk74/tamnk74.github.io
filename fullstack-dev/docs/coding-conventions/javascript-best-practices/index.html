JavaScript Best Practices

Modern JavaScript development guidelines for writing clean, maintainable, and performant code.

General Principles​

Use Strict Mode​

'use strict';

// All code should run in strict mode
function calculateTotal(items) {
  return items.reduce((sum, item) => sum + item.price, 0);
}

Meaningful Variable Names​

// ❌ Bad
const u = users.filter(u => u.a > 18);

// ✅ Good
const adultUsers = users.filter(user => user.age > 18);

Consistent Naming Conventions​

// Variables and functions: camelCase
const userProfile = {};
const calculateUserAge = () => {};

// Constants: UPPER_SNAKE_CASE
const MAX_RETRY_ATTEMPTS = 3;
const API_BASE_URL = 'https://api.example.com';

// Classes: PascalCase
class UserManager {}

ES6+ Features​

Use const and let​

// ❌ Bad
var userName = 'john';
var userAge = 25;

// ✅ Good
const userName = 'john';
let userAge = 25;

Arrow Functions​

// ❌ Bad
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(function(num) {
  return num * 2;
});

// ✅ Good
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(num => num * 2);

// ✅ For complex functions
const processUser = (user) => {
  const { name, email, age } = user;
  return {
    displayName: name.toUpperCase(),
    contact: email.toLowerCase(),
    isAdult: age >= 18,
  };
};

Destructuring​

// ❌ Bad
function processUser(user) {
  const name = user.name;
  const email = user.email;
  const age = user.age;
  
  return {
    name: name,
    email: email,
    isAdult: age >= 18
  };
}

// ✅ Good
function processUser(user) {
  const { name, email, age } = user;
  
  return {
    name,
    email,
    isAdult: age >= 18,
  };
}

// Array destructuring
const [first, second, ...rest] = items;

Template Literals​

// ❌ Bad
const message = 'Hello ' + user.name + ', you have ' + user.notifications.length + ' notifications.';

// ✅ Good
const message = `Hello ${user.name}, you have ${user.notifications.length} notifications.`;

// Multi-line strings
const htmlTemplate = `
  <div class="user-card">
    <h2>${user.name}</h2>
    <p>${user.email}</p>
  </div>
`;

Spread Operator​

// Array operations
const originalArray = [1, 2, 3];
const newArray = [...originalArray, 4, 5]; // [1, 2, 3, 4, 5]

// Object operations
const originalUser = { name: 'John', age: 25 };
const updatedUser = { ...originalUser, age: 26, city: 'New York' };

// Function parameters
const numbers = [1, 2, 3, 4, 5];
const max = Math.max(...numbers);

Default Parameters​

// ❌ Bad
function createUser(name, age, role) {
  name = name || 'Anonymous';
  age = age || 0;
  role = role || 'user';
  
  return { name, age, role };
}

// ✅ Good
function createUser(name = 'Anonymous', age = 0, role = 'user') {
  return { name, age, role };
}

Code Organization​

Module Structure​

// user.service.js
import { api } from './api.service.js';
import { logger } from './logger.service.js';

// Private functions at the top
function validateUserData(userData) {
  // Validation logic
}

function formatUserResponse(user) {
  // Formatting logic
}

// Main class or functions
export class UserService {
  async createUser(userData) {
    try {
      validateUserData(userData);
      const response = await api.post('/users', userData);
      return formatUserResponse(response.data);
    } catch (error) {
      logger.error('Failed to create user:', error);
      throw error;
    }
  }
  
  async getUserById(id) {
    try {
      const response = await api.get(`/users/${id}`);
      return formatUserResponse(response.data);
    } catch (error) {
      logger.error(`Failed to fetch user ${id}:`, error);
      throw error;
    }
  }
}

// Default export at the bottom
export default new UserService();

Barrel Exports​

// services/index.js
export { UserService } from './user.service.js';
export { AuthService } from './auth.service.js';
export { ApiService } from './api.service.js';
export { LoggerService } from './logger.service.js';

// Usage
import { UserService, AuthService } from './services';

Error Handling​

Try-Catch Blocks​

// ❌ Bad
function processData(data) {
  const result = JSON.parse(data);
  return result.items.map(item => item.name);
}

// ✅ Good
function processData(data) {
  try {
    const result = JSON.parse(data);
    
    if (!result.items || !Array.isArray(result.items)) {
      throw new Error('Invalid data structure: items array not found');
    }
    
    return result.items.map(item => {
      if (!item.name) {
        throw new Error(`Invalid item: missing name property`);
      }
      return item.name;
    });
  } catch (error) {
    logger.error('Failed to process data:', error);
    throw new Error(`Data processing failed: ${error.message}`);
  }
}

Promise Error Handling​

// ❌ Bad
fetchUserData()
  .then(data => processData(data))
  .then(result => updateUI(result));

// ✅ Good
async function handleUserData() {
  try {
    const data = await fetchUserData();
    const result = await processData(data);
    updateUI(result);
  } catch (error) {
    logger.error('User data handling failed:', error);
    showErrorMessage('Failed to load user data. Please try again.');
  }
}

// Alternative with .catch()
fetchUserData()
  .then(data => processData(data))
  .then(result => updateUI(result))
  .catch(error => {
    logger.error('User data handling failed:', error);
    showErrorMessage('Failed to load user data. Please try again.');
  });

Custom Error Classes​

class ValidationError extends Error {
  constructor(message, field) {
    super(message);
    this.name = 'ValidationError';
    this.field = field;
  }
}

class ApiError extends Error {
  constructor(message, status, endpoint) {
    super(message);
    this.name = 'ApiError';
    this.status = status;
    this.endpoint = endpoint;
  }
}

// Usage
function validateEmail(email) {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(email)) {
    throw new ValidationError('Invalid email format', 'email');
  }
}

Performance Considerations​

Avoid Global Variables​

// ❌ Bad
let globalCounter = 0;
let globalUserData = {};

function incrementCounter() {
  globalCounter++;
}

// ✅ Good
const CounterModule = (() => {
  let counter = 0;
  
  return {
    increment: () => ++counter,
    getCount: () => counter,
    reset: () => { counter = 0; }
  };
})();

Efficient Array Operations​

// ❌ Bad - Multiple iterations
const users = getUsers();
const activeUsers = users.filter(user => user.active);
const activeUserNames = activeUsers.map(user => user.name);
const sortedNames = activeUserNames.sort();

// ✅ Good - Single iteration with chaining
const sortedActiveUserNames = getUsers()
  .filter(user => user.active)
  .map(user => user.name)
  .sort();

// ✅ Better - Use reduce for complex operations
const userStats = users.reduce((stats, user) => {
  stats.total++;
  if (user.active) stats.active++;
  if (user.premium) stats.premium++;
  return stats;
}, { total: 0, active: 0, premium: 0 });

Debouncing and Throttling​

// Debounce function
function debounce(func, delay) {
  let timeoutId;
  return function (...args) {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(() => func.apply(this, args), delay);
  };
}

// Throttle function
function throttle(func, limit) {
  let inThrottle;
  return function (...args) {
    if (!inThrottle) {
      func.apply(this, args);
      inThrottle = true;
      setTimeout(() => inThrottle = false, limit);
    }
  };
}

// Usage
const debouncedSearch = debounce(searchFunction, 300);
const throttledScroll = throttle(scrollHandler, 100);

Testing Guidelines​

Unit Test Structure​

// user.service.test.js
import { UserService } from './user.service.js';
import { jest } from '@jest/globals';

describe('UserService', () => {
  let userService;
  
  beforeEach(() => {
    userService = new UserService();
  });
  
  describe('createUser', () => {
    it('should create a user with valid data', async () => {
      // Arrange
      const userData = {
        name: 'John Doe',
        email: 'john@example.com',
        age: 25
      };
      
      // Act
      const result = await userService.createUser(userData);
      
      // Assert
      expect(result).toHaveProperty('id');
      expect(result.name).toBe(userData.name);
      expect(result.email).toBe(userData.email);
    });
    
    it('should throw error for invalid email', async () => {
      // Arrange
      const userData = {
        name: 'John Doe',
        email: 'invalid-email',
        age: 25
      };
      
      // Act & Assert
      await expect(userService.createUser(userData))
        .rejects
        .toThrow('Invalid email format');
    });
  });
});

Mocking External Dependencies​

// Mock API calls
jest.mock('./api.service.js', () => ({
  api: {
    post: jest.fn(),
    get: jest.fn(),
  }
}));

import { api } from './api.service.js';

describe('UserService with mocked API', () => {
  beforeEach(() => {
    jest.clearAllMocks();
  });
  
  it('should handle API errors gracefully', async () => {
    // Arrange
    const mockError = new Error('Network error');
    api.post.mockRejectedValue(mockError);
    
    // Act & Assert
    await expect(userService.createUser({}))
      .rejects
      .toThrow('Data processing failed');
  });
});

Best Practices Summary​

  1. Use modern ES6+ features for cleaner, more readable code
  2. Write meaningful names for variables, functions, and classes
  3. Handle errors gracefully with proper try-catch blocks
  4. Optimize performance by avoiding unnecessary operations
  5. Write comprehensive tests for all functions and edge cases
  6. Use linting and formatting tools to maintain code quality
  7. Document complex logic with clear comments
  8. Follow consistent patterns throughout the codebase
