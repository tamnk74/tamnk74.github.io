Non-functional Requirements Validation

This guide establishes comprehensive validation frameworks for non-functional requirements in NestJS microservices and Next.js microfrontend applications, covering performance testing, security validation, scalability assessment, and compliance verification through automated testing and monitoring strategies.

Overview​

Non-functional requirements (NFRs) define system quality attributes such as performance, security, scalability, reliability, and usability. This validation framework ensures that applications meet these critical requirements through systematic testing, monitoring, and validation processes.

import http from 'k6/http';
import { check, sleep } from 'k6';

// Custom metrics
const errorRate = new Rate('errors');
const responseTimeThreshold = new Trend('response_time_threshold');

// Test configuration
export const options = {
  stages: [
    { duration: '5m', target: 100 }, // Ramp up to 100 users
    { duration: '10m', target: 100 }, // Stay at 100 users
    { duration: '5m', target: 200 }, // Ramp up to 200 users
    { duration: '10m', target: 200 }, // Stay at 200 users
    { duration: '5m', target: 0 }, // Ramp down
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'], // 95% of requests under 500ms
    http_req_failed: ['rate<0.05'], // Error rate under 5%
    errors: ['rate<0.05'],
    response_time_threshold: ['p(99)<1000'], // 99% under 1s
  },
  ext: {
    loadimpact: {
      distribution: {
        'amazon:us:ashburn': { loadZone: 'amazon:us:ashburn', percent: 50 },
        'amazon:ie:dublin': { loadZone: 'amazon:ie:dublin', percent: 50 },
      },

  },
};

const BASE_URL = __ENV.BASE_URL || 'http://localhost:3000';
let authToken;

export function setup() {
  // Authenticate and get token
  const loginResponse = http.post(`${BASE_URL}/auth/login`, {

export default function (data) {
  authToken = data.authToken;

  // Test user update

function testUsersList() {
  const response = http.get(`${BASE_URL}/api/users`, {
    headers: {
      Authorization: `Bearer ${authToken}`,
      'Content-Type': 'application/json',
    },
  });

  const success = check(response, {
    'users list status is 200': (r) => r.status === 200,
    'users list response time < 200ms': (r) => r.timings.duration < 200,
    'users list returns array': (r) => Array.isArray(r.json()),
  });

  errorRate.add(!success);
  responseTimeThreshold.add(response.timings.duration);
}

function testUserCreation() {
  const userData = {
    name: `Test User ${__VU}-${__ITER}`,
    email: `user-${__VU}-${__ITER}@example.com`,
    password: 'password123',
  };

  const response = http.post(`${BASE_URL}/api/users`, JSON.stringify(userData), {
    headers: {
      Authorization: `Bearer ${authToken}`,
      'Content-Type': 'application/json',
    },
  });

  const success = check(response, {
    'user creation status is 201': (r) => r.status === 201,
    'user creation response time < 300ms': (r) => r.timings.duration < 300,
    'user creation returns user id': (r) => r.json('id') !== undefined,
  });

  errorRate.add(!success);
  responseTimeThreshold.add(response.timings.duration);

  return response.json('id');
}

function testUserUpdate() {
  const userId = Math.floor(Math.random() * 100) + 1;
  const updateData = {
    name: `Updated User ${__VU}-${__ITER}`,
  };

  const response = http.patch(`${BASE_URL}/api/users/${userId}`, JSON.stringify(updateData), {
    headers: {
      Authorization: `Bearer ${authToken}`,
      'Content-Type': 'application/json',
    },
  });

  const success = check(response, {
    'user update status is 200 or 404': (r) => r.status === 200 || r.status === 404,
    'user update response time < 250ms': (r) => r.timings.duration < 250,
  });

  errorRate.add(!success && response.status !== 404);
  responseTimeThreshold.add(response.timings.duration);
}

Frontend Performance Testing​

// tests/performance/frontend/lighthouse-performance.ts
import lighthouse from 'lighthouse';
import * as chromeLauncher from 'chrome-launcher';
import * as fs from 'fs';
import * as path from 'path';

export interface PerformanceThresholds {
  performance: number;
  accessibility: number;
  bestPractices: number;
  seo: number;
  firstContentfulPaint: number;
  largestContentfulPaint: number;
  firstInputDelay: number;
  cumulativeLayoutShift: number;
}

export class FrontendPerformanceTester {
  private defaultThresholds: PerformanceThresholds = {
    performance: 90,
    accessibility: 95,
    bestPractices: 90,
    seo: 90,
    firstContentfulPaint: 1500,
    largestContentfulPaint: 2500,
    firstInputDelay: 100,
    cumulativeLayoutShift: 0.1,
  };

  async runPerformanceTests(
    urls: string[],
    thresholds: Partial<PerformanceThresholds> = {}
  ): Promise<boolean> {
    const finalThresholds = { ...this.defaultThresholds, ...thresholds };
    let allTestsPassed = true;

    for (const url of urls) {
      console.log(`Testing performance for: ${url}`);
      
      const chrome = await chromeLauncher.launch({ chromeFlags: ['--headless'] });
      
      try {
        const options = {
          logLevel: 'info' as const,
          output: 'json' as const,
          onlyCategories: ['performance', 'accessibility', 'best-practices', 'seo'],
          port: chrome.port,
        };

        const runnerResult = await lighthouse(url, options);
        
        if (!runnerResult || !runnerResult.report) {
          console.error(`Failed to run Lighthouse for ${url}`);
          allTestsPassed = false;
          continue;
        }

        const report = JSON.parse(runnerResult.report);
        const testResult = this.validatePerformanceResults(report, finalThresholds);
        
        if (!testResult.passed) {
          console.error(`Performance test failed for ${url}:`);
          testResult.failures.forEach(failure => console.error(`  - ${failure}`));
          allTestsPassed = false;
        } else {
          console.log(`✅ Performance test passed for ${url}`);
        }

        // Save detailed report
        await this.saveDetailedReport(url, report);
        
      } finally {
        await chrome.kill();
      }
    }

    return allTestsPassed;
  }

  private validatePerformanceResults(
    report: any,
    thresholds: PerformanceThresholds
  ): { passed: boolean; failures: string[] } {
    const failures: string[] = [];
    const categories = report.categories;
    const audits = report.audits;

    // Check category scores
    if (categories.performance.score * 100 < thresholds.performance) {
      failures.push(`Performance score: ${Math.round(categories.performance.score * 100)} < ${thresholds.performance}`);
    }

    if (categories.accessibility.score * 100 < thresholds.accessibility) {
      failures.push(`Accessibility score: ${Math.round(categories.accessibility.score * 100)} < ${thresholds.accessibility}`);
    }

    if (categories['best-practices'].score * 100 < thresholds.bestPractices) {
      failures.push(`Best practices score: ${Math.round(categories['best-practices'].score * 100)} < ${thresholds.bestPractices}`);
    }

    if (categories.seo.score * 100 < thresholds.seo) {
      failures.push(`SEO score: ${Math.round(categories.seo.score * 100)} < ${thresholds.seo}`);
    }

    // Check Core Web Vitals
    const fcp = audits['first-contentful-paint'].numericValue;
    if (fcp > thresholds.firstContentfulPaint) {
      failures.push(`First Contentful Paint: ${Math.round(fcp)}ms > ${thresholds.firstContentfulPaint}ms`);
    }

    const lcp = audits['largest-contentful-paint'].numericValue;
    if (lcp > thresholds.largestContentfulPaint) {
      failures.push(`Largest Contentful Paint: ${Math.round(lcp)}ms > ${thresholds.largestContentfulPaint}ms`);
    }

    const cls = audits['cumulative-layout-shift'].numericValue;
    if (cls > thresholds.cumulativeLayoutShift) {
      failures.push(`Cumulative Layout Shift: ${cls.toFixed(3)} > ${thresholds.cumulativeLayoutShift}`);
    }

    return {
      passed: failures.length === 0,
      failures,
    };
  }

  private async saveDetailedReport(url: string, report: any): Promise<void> {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const filename = `performance-${url.replace(/[^a-zA-Z0-9]/g, '_')}-${timestamp}.json`;
    const filepath = path.join('reports', 'performance', filename);

    // Ensure directory exists
    const dir = path.dirname(filepath);
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
    }

    fs.writeFileSync(filepath, JSON.stringify(report, null, 2));
    console.log(`Detailed performance report saved to: ${filepath}`);
  }
}

// Usage example
async function runTests() {
  const tester = new FrontendPerformanceTester();
  
  const urls = [
    'http://localhost:3000',
    'http://localhost:3000/users',
    'http://localhost:3000/dashboard',
  ];

  const customThresholds = {
    performance: 85, // Slightly lower threshold for complex pages
    largestContentfulPaint: 3000, // Allow 3s for LCP
  };

  const passed = await tester.runPerformanceTests(urls, customThresholds);
  
  if (!passed) {
    process.exit(1);
  }
}

if (require.main === module) {
  runTests().catch(console.error);
}

Automated Security Testing​

// tests/security/security-scanner.ts
import { ZAPClient } from 'zaproxy';
import axios from 'axios';
import * as fs from 'fs';
import * as path from 'path';

export interface SecurityTestConfig {
  targetUrl: string;
  apiEndpoints: string[];
  authToken?: string;
  excludeUrls: string[];
  scanPolicies: string[];
}

export interface SecurityVulnerability {
  name: string;
  risk: 'High' | 'Medium' | 'Low' | 'Informational';
  confidence: 'High' | 'Medium' | 'Low';
  url: string;
  param: string;
  attack: string;
  evidence: string;
  description: string;
  solution: string;
  cweid: string;
  wascid: string;
}

export class SecurityValidator {
  private zap: ZAPClient;

  constructor() {
    this.zap = new ZAPClient({
      proxy: 'http://localhost:8080',
    });
  }

  async runSecurityScan(config: SecurityTestConfig): Promise<SecurityVulnerability[]> {
    console.log('Starting security scan...');
    
    try {
      // Start ZAP session
      await this.zap.core.newSession();
      
      // Configure authentication if provided
      if (config.authToken) {
        await this.configureAuthentication(config);
      }

      // Spider the application
      console.log('Spidering application...');
      await this.spiderApplication(config.targetUrl, config.excludeUrls);

      // Perform active scan
      console.log('Performing active security scan...');
      await this.performActiveScan(config.targetUrl, config.scanPolicies);

      // Get results
      const vulnerabilities = await this.getVulnerabilities();
      
      // Generate report
      await this.generateSecurityReport(vulnerabilities);
      
      return vulnerabilities;
      
    } catch (error) {
      console.error('Security scan failed:', error);
      throw error;
    }
  }

  private async configureAuthentication(config: SecurityTestConfig): Promise<void> {
    // Configure session management
    await this.zap.authentication.setAuthenticationMethod(0, 'httpAuthentication', 
      `authUrl=${config.targetUrl}/auth/login&loginUrl=${config.targetUrl}/auth/login`
    );

    // Set authentication headers
    await this.zap.replacer.addRule('Authorization', `Bearer ${config.authToken}`, '', true, true);
  }

  private async spiderApplication(targetUrl: string, excludeUrls: string[]): Promise<void> {
    // Add target to scope
    await this.zap.core.includeInContext('Default Context', `${targetUrl}.*`);
    
    // Exclude URLs that shouldn't be tested
    for (const excludeUrl of excludeUrls) {
      await this.zap.core.excludeFromContext('Default Context', excludeUrl);
    }

    // Start spider
    const spiderId = await this.zap.spider.scan(targetUrl);
    
    // Wait for spider to complete
    let progress = 0;
    while (progress < 100) {
      await new Promise(resolve => setTimeout(resolve, 1000));
      const status = await this.zap.spider.status(spiderId);
      progress = parseInt(status);
      console.log(`Spider progress: ${progress}%`);
    }
  }

  private async performActiveScan(targetUrl: string, scanPolicies: string[]): Promise<void> {
    // Configure scan policies
    for (const policy of scanPolicies) {
      await this.zap.ascan.enableScanners(policy);
    }

    // Start active scan
    const scanId = await this.zap.ascan.scan(targetUrl);
    
    // Wait for scan to complete
    let progress = 0;
    while (progress < 100) {
      await new Promise(resolve => setTimeout(resolve, 5000));
      const status = await this.zap.ascan.status(scanId);
      progress = parseInt(status);
      console.log(`Active scan progress: ${progress}%`);
    }
  }

  private async getVulnerabilities(): Promise<SecurityVulnerability[]> {
    const alerts = await this.zap.core.alerts();
    
    return alerts.map((alert: any) => ({
      name: alert.name,
      risk: alert.risk,
      confidence: alert.confidence,
      url: alert.url,
      param: alert.param,
      attack: alert.attack,
      evidence: alert.evidence,
      description: alert.description,
      solution: alert.solution,
      cweid: alert.cweid,
      wascid: alert.wascid,
    }));
  }

  private async generateSecurityReport(vulnerabilities: SecurityVulnerability[]): Promise<void> {
    const report = {
      timestamp: new Date().toISOString(),
      totalVulnerabilities: vulnerabilities.length,
      riskDistribution: this.calculateRiskDistribution(vulnerabilities),
      vulnerabilities,
    };

    const reportPath = path.join('reports', 'security', `security-scan-${Date.now()}.json`);
    
    // Ensure directory exists
    const dir = path.dirname(reportPath);
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
    }

    fs.writeFileSync(reportPath, JSON.stringify(report, null, 2));
    console.log(`Security report saved to: ${reportPath}`);
  }

  private calculateRiskDistribution(vulnerabilities: SecurityVulnerability[]) {
    return vulnerabilities.reduce((acc, vuln) => {
      acc[vuln.risk] = (acc[vuln.risk] || 0) + 1;
      return acc;
    }, {} as Record<string, number>);
  }
}

// API Security Testing
export class APISecurityTester {
  async testAPIEndpoints(baseUrl: string, endpoints: string[], authToken?: string): Promise<void> {
    const headers = authToken ? { Authorization: `Bearer ${authToken}` } : {};

    for (const endpoint of endpoints) {
      await this.testSQLInjection(baseUrl + endpoint, headers);
      await this.testXSSVulnerabilities(baseUrl + endpoint, headers);
      await this.testAuthentication(baseUrl + endpoint);
      await this.testAuthorization(baseUrl + endpoint, headers);
      await this.testRateLimiting(baseUrl + endpoint, headers);
    }
  }

  private async testSQLInjection(url: string, headers: any): Promise<void> {
    const sqlPayloads = [
      "' OR '1'='1",
      "'; DROP TABLE users; --",
      "' UNION SELECT * FROM users --",
      "1' AND 1=1 --",
    ];

    for (const payload of sqlPayloads) {
      try {
        const response = await axios.get(url, {
          params: { q: payload },
          headers,
          timeout: 5000,
        });

        // Check for SQL error messages
        if (response.data.includes('SQL') || 
            response.data.includes('mysql') || 
            response.data.includes('postgres')) {
          console.warn(`⚠️  Potential SQL injection vulnerability in ${url}`);
        }
      } catch (error) {
        // Expected for malformed requests
      }
    }
  }

  private async testXSSVulnerabilities(url: string, headers: any): Promise<void> {
    const xssPayloads = [
      "<script>alert('XSS')</script>",
      "javascript:alert('XSS')",
      "<img src=x onerror=alert('XSS')>",
      "';alert('XSS');//",
    ];

    for (const payload of xssPayloads) {
      try {
        const response = await axios.post(url, { data: payload }, { headers });

        if (response.data.includes(payload)) {
          console.warn(`⚠️  Potential XSS vulnerability in ${url}`);
        }
      } catch (error) {
        // Expected for malformed requests
      }
    }
  }

  private async testAuthentication(url: string): Promise<void> {
    try {
      const response = await axios.get(url);
      
      if (response.status === 200) {
        console.warn(`⚠️  Endpoint ${url} accessible without authentication`);
      }
    } catch (error) {
      // Expected - endpoint should require authentication
    }
  }

  private async testAuthorization(url: string, headers: any): Promise<void> {
    // Test with modified/expired token
    const modifiedHeaders = {
      ...headers,
      Authorization: headers.Authorization?.replace(/.$/, 'X'), // Modify last character
    };

    try {
      const response = await axios.get(url, { headers: modifiedHeaders });
      
      if (response.status === 200) {
        console.warn(`⚠️  Endpoint ${url} accessible with invalid token`);
      }
    } catch (error) {
      // Expected - should reject invalid tokens
    }
  }

  private async testRateLimiting(url: string, headers: any): Promise<void> {
    const requests = Array.from({ length: 100 }, () => 
      axios.get(url, { headers, timeout: 1000 }).catch(() => null)
    );

    const responses = await Promise.all(requests);
    const successfulRequests = responses.filter(r => r && r.status === 200).length;

    if (successfulRequests > 50) {
      console.warn(`⚠️  Potential rate limiting issue in ${url}: ${successfulRequests}/100 requests succeeded`);
    }
  }
}

// tests/scalability/auto-scaling-test.ts
import { execSync } from 'child_process';
import axios from 'axios';
import * as k8s from '@kubernetes/client-node';

export interface ScalabilityTest {
  name: string;
  targetLoad: number;
  expectedScaling: {
    minReplicas: number;
    maxReplicas: number;
    targetCPU: number;
    targetMemory: number;
  };
  duration: number;
}

export class ScalabilityValidator {
  private k8sApi: k8s.AppsV1Api;
  private metricsApi: k8s.Metrics;

  constructor() {
    const kc = new k8s.KubeConfig();
    kc.loadFromDefault();
    
    this.k8sApi = kc.makeApiClient(k8s.AppsV1Api);
    this.metricsApi = new k8s.Metrics(kc);
  }

  async runScalabilityTest(test: ScalabilityTest): Promise<boolean> {
    console.log(`Starting scalability test: ${test.name}`);
    
    try {
      // Record initial state
      const initialMetrics = await this.getCurrentMetrics();
      
      // Generate load
      await this.generateLoad(test.targetLoad, test.duration);
      
      // Monitor scaling behavior
      const scalingResults = await this.monitorScaling(test);
      
      // Validate results
      const passed = this.validateScalingBehavior(scalingResults, test.expectedScaling);
      
      // Generate report
      await this.generateScalabilityReport(test, initialMetrics, scalingResults, passed);
      
      return passed;
      
    } catch (error) {
      console.error(`Scalability test failed: ${error}`);
      return false;
    }
  }

  private async generateLoad(targetRPS: number, duration: number): Promise<void> {
    const k6Script = `
      import http from 'k6/http';
      import { check } from 'k6';
      
      export let options = {
        stages: [
          { duration: '30s', target: ${targetRPS} },
          { duration: '${duration}s', target: ${targetRPS} },
          { duration: '30s', target: 0 },
        ],
      };
      
      export default function() {
        let response = http.get('${process.env.TARGET_URL || 'http://localhost:3000'}/api/health');
        check(response, {
          'status is 200': (r) => r.status === 200,
        });
      }
    `;

    // Write temporary K6 script
    require('fs').writeFileSync('/tmp/load-test.js', k6Script);
    
    // Run load test in background
    const loadTestProcess = execSync('k6 run /tmp/load-test.js', { 
      stdio: 'pipe',
      timeout: (duration + 120) * 1000,
    });

    console.log('Load test completed');
  }

  private async monitorScaling(test: ScalabilityTest): Promise<any> {
    const metrics: any[] = [];
    const startTime = Date.now();
    const monitorDuration = (test.duration + 120) * 1000; // Monitor for test duration + buffer

    while (Date.now() - startTime < monitorDuration) {
      try {
        const currentMetrics = await this.getCurrentMetrics();
        metrics.push({
          timestamp: new Date().toISOString(),
          ...currentMetrics,
        });

        console.log(`Replicas: ${currentMetrics.replicas}, CPU: ${currentMetrics.avgCPU}%, Memory: ${currentMetrics.avgMemory}%`);
        
        // Wait 30 seconds before next measurement
        await new Promise(resolve => setTimeout(resolve, 30000));
        
      } catch (error) {
        console.error('Error collecting metrics:', error);
      }
    }

    return metrics;
  }

  private async getCurrentMetrics(): Promise<any> {
    // Get deployment info
    const deployment = await this.k8sApi.readNamespacedDeployment('app-backend', 'default');
    const replicas = deployment.body.status?.replicas || 0;

    // Get pod metrics
    const podMetrics = await this.metricsApi.getPodMetrics('default');
    
    let totalCPU = 0;
    let totalMemory = 0;
    let podCount = 0;

    for (const pod of podMetrics.body.items) {
      if (pod.metadata?.name?.startsWith('app-backend')) {
        podCount++;
        
        for (const container of pod.containers) {
          // Parse CPU usage (convert from nano cores)
          const cpuUsage = parseInt(container.usage.cpu.replace('n', '')) / 1000000; // Convert to millicores
          totalCPU += cpuUsage;
          
          // Parse memory usage (convert from Ki to MB)
          const memoryUsage = parseInt(container.usage.memory.replace('Ki', '')) / 1024;
          totalMemory += memoryUsage;
        }
      }
    }

    return {
      replicas,
      avgCPU: podCount > 0 ? totalCPU / podCount : 0,
      avgMemory: podCount > 0 ? totalMemory / podCount : 0,
      podCount,
    };
  }

  private validateScalingBehavior(metrics: any[], expected: any): boolean {
    if (metrics.length === 0) {
      console.error('No metrics collected');
      return false;
    }

    // Find peak metrics
    const peakReplicas = Math.max(...metrics.map(m => m.replicas));
    const peakCPU = Math.max(...metrics.map(m => m.avgCPU));
    const peakMemory = Math.max(...metrics.map(m => m.avgMemory));

    console.log(`Peak metrics - Replicas: ${peakReplicas}, CPU: ${peakCPU}%, Memory: ${peakMemory}%`);

    // Validate scaling behavior
    const scalingValidation = {
      replicasInRange: peakReplicas >= expected.minReplicas && peakReplicas <= expected.maxReplicas,
      cpuUnderLimit: peakCPU <= expected.targetCPU * 1.2, // Allow 20% buffer
      memoryUnderLimit: peakMemory <= expected.targetMemory * 1.2,
      scalingOccurred: peakReplicas > expected.minReplicas,
    };

    const passed = Object.values(scalingValidation).every(v => v);

    if (!passed) {
      console.error('Scaling validation failed:', scalingValidation);
    }

    return passed;
  }

  private async generateScalabilityReport(
    test: ScalabilityTest,
    initialMetrics: any,
    scalingResults: any[],
    passed: boolean
  ): Promise<void> {
    const report = {
      testName: test.name,
      timestamp: new Date().toISOString(),
      passed,
      initialMetrics,
      scalingResults,
      summary: {
        maxReplicas: Math.max(...scalingResults.map(m => m.replicas)),
        maxCPU: Math.max(...scalingResults.map(m => m.avgCPU)),
        maxMemory: Math.max(...scalingResults.map(m => m.avgMemory)),
        averageReplicas: scalingResults.reduce((sum, m) => sum + m.replicas, 0) / scalingResults.length,
      },
    };

    const reportPath = `reports/scalability/scalability-${test.name}-${Date.now()}.json`;
    require('fs').writeFileSync(reportPath, JSON.stringify(report, null, 2));
    console.log(`Scalability report saved to: ${reportPath}`);
  }
}

Automated Accessibility Validation​

// tests/accessibility/accessibility-validator.ts
import { AxePuppeteer } from '@axe-core/puppeteer';
import puppeteer from 'puppeteer';
import * as fs from 'fs';
import * as path from 'path';

export interface AccessibilityTest {
  url: string;
  name: string;
  tags?: string[];
  rules?: string[];
  disableRules?: string[];
}

export interface AccessibilityViolation {
  id: string;
  impact: 'minor' | 'moderate' | 'serious' | 'critical';
  description: string;
  help: string;
  helpUrl: string;
  nodes: Array<{
    html: string;
    target: string[];
    failureSummary: string;
  }>;
}

export class AccessibilityValidator {
  async runAccessibilityTests(tests: AccessibilityTest[]): Promise<boolean> {
    console.log('Starting accessibility validation...');
    
    const browser = await puppeteer.launch({
      headless: true,
      args: ['--no-sandbox', '--disable-setuid-sandbox'],
    });

    let allTestsPassed = true;

    try {
      for (const test of tests) {
        console.log(`Testing accessibility for: ${test.name}`);
        
        const page = await browser.newPage();
        
        try {
          // Navigate to page
          await page.goto(test.url, { waitUntil: 'networkidle0' });
          
          // Wait for dynamic content
          await page.waitForTimeout(2000);
          
          // Configure axe
          const axe = new AxePuppeteer(page);
          
          if (test.tags) {
            axe.withTags(test.tags);
          }
          
          if (test.rules) {
            axe.withRules(test.rules);
          }
          
          if (test.disableRules) {
            axe.disableRules(test.disableRules);
          }

          // Run accessibility scan
          const results = await axe.analyze();
          
          // Process results
          const testPassed = this.processResults(test, results);
          
          if (!testPassed) {
            allTestsPassed = false;
          }
          
        } finally {
          await page.close();
        }
      }
    } finally {
      await browser.close();
    }

    return allTestsPassed;
  }

  private processResults(test: AccessibilityTest, results: any): boolean {
    const violations: AccessibilityViolation[] = results.violations.map((violation: any) => ({
      id: violation.id,
      impact: violation.impact,
      description: violation.description,
      help: violation.help,
      helpUrl: violation.helpUrl,
      nodes: violation.nodes.map((node: any) => ({
        html: node.html,
        target: node.target,
        failureSummary: node.failureSummary,
      })),
    }));

    // Log results
    if (violations.length === 0) {
      console.log(`✅ ${test.name}: No accessibility violations found`);
      return true;
    } else {
      console.error(`❌ ${test.name}: ${violations.length} accessibility violations found`);
      
      // Group violations by impact
      const violationsByImpact = violations.reduce((acc, violation) => {
        acc[violation.impact] = (acc[violation.impact] || 0) + 1;
        return acc;
      }, {} as Record<string, number>);

      console.error('Violations by impact:', violationsByImpact);
      
      // Save detailed report
      this.saveAccessibilityReport(test, violations);
      
      // Determine if test should fail based on impact
      const criticalViolations = violations.filter(v => v.impact === 'critical').length;
      const seriousViolations = violations.filter(v => v.impact === 'serious').length;
      
      // Fail test if there are critical or more than 5 serious violations
      return criticalViolations === 0 && seriousViolations <= 5;
    }
  }

  private saveAccessibilityReport(test: AccessibilityTest, violations: AccessibilityViolation[]): void {
    const report = {
      testName: test.name,
      url: test.url,
      timestamp: new Date().toISOString(),
      violationCount: violations.length,
      violations,
    };

    const reportPath = path.join('reports', 'accessibility', `a11y-${test.name.replace(/\s+/g, '-')}-${Date.now()}.json`);
    
    // Ensure directory exists
    const dir = path.dirname(reportPath);
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
    }

    fs.writeFileSync(reportPath, JSON.stringify(report, null, 2));
    console.log(`Accessibility report saved to: ${reportPath}`);
  }
}

// tests/compliance/gdpr-validator.ts
import axios from 'axios';
import * as puppeteer from 'puppeteer';

export interface GDPRComplianceCheck {
  name: string;
  endpoint?: string;
  pageUrl?: string;
  expected: {
    cookieConsent: boolean;
    privacyPolicy: boolean;
    dataPortability: boolean;
    rightToErasure: boolean;
    consentWithdrawal: boolean;
  };
}

export class GDPRComplianceValidator {
  async validateGDPRCompliance(baseUrl: string): Promise<boolean> {
    console.log('Starting GDPR compliance validation...');
    
    const checks: GDPRComplianceCheck[] = [
      {
        name: 'Cookie Consent Banner',
        pageUrl: `${baseUrl}/`,
        expected: {
          cookieConsent: true,
          privacyPolicy: false,
          dataPortability: false,
          rightToErasure: false,
          consentWithdrawal: false,
        },
      },
      {
        name: 'Privacy Policy Accessibility',
        pageUrl: `${baseUrl}/privacy`,
        expected: {
          cookieConsent: false,
          privacyPolicy: true,
          dataPortability: false,
          rightToErasure: false,
          consentWithdrawal: false,
        },
      },
      {
        name: 'Data Export Functionality',
        endpoint: `${baseUrl}/api/users/me/export`,
        expected: {
          cookieConsent: false,
          privacyPolicy: false,
          dataPortability: true,
          rightToErasure: false,
          consentWithdrawal: false,
        },
      },
      {
        name: 'Account Deletion',
        endpoint: `${baseUrl}/api/users/me`,
        expected: {
          cookieConsent: false,
          privacyPolicy: false,
          dataPortability: false,
          rightToErasure: true,
          consentWithdrawal: false,
        },
      },
      {
        name: 'Consent Management',
        pageUrl: `${baseUrl}/settings/privacy`,
        expected: {
          cookieConsent: false,
          privacyPolicy: false,
          dataPortability: false,
          rightToErasure: false,
          consentWithdrawal: true,
        },
      },
    ];

    let allChecksPassed = true;

    for (const check of checks) {
      try {
        const result = await this.performGDPRCheck(check);
        
        if (!result) {
          console.error(`❌ GDPR check failed: ${check.name}`);
          allChecksPassed = false;
        } else {
          console.log(`✅ GDPR check passed: ${check.name}`);
        }
      } catch (error) {
        console.error(`❌ GDPR check error for ${check.name}:`, error);
        allChecksPassed = false;
      }
    }

    return allChecksPassed;
  }

  private async performGDPRCheck(check: GDPRComplianceCheck): Promise<boolean> {
    if (check.pageUrl) {
      return this.checkPageCompliance(check);
    } else if (check.endpoint) {
      return this.checkAPICompliance(check);
    }
    
    return false;
  }

  private async checkPageCompliance(check: GDPRComplianceCheck): Promise<boolean> {
    const browser = await puppeteer.launch({ headless: true });
    
    try {
      const page = await browser.newPage();
      await page.goto(check.pageUrl!, { waitUntil: 'networkidle0' });

      let checksPassed = true;

      if (check.expected.cookieConsent) {
        const cookieBanner = await page.$('[data-testid="cookie-consent"], .cookie-banner, .cookie-notice');
        if (!cookieBanner) {
          console.error('Cookie consent banner not found');
          checksPassed = false;
        }

        // Check for reject option
        const rejectButton = await page.$('[data-testid="reject-cookies"], button:contains("Reject"), button:contains("Decline")');
        if (!rejectButton) {
          console.error('Cookie rejection option not found');
          checksPassed = false;
        }
      }

      if (check.expected.privacyPolicy) {
        const privacyContent = await page.$('[data-testid="privacy-policy"], .privacy-policy');
        if (!privacyContent) {
          console.error('Privacy policy content not found');
          checksPassed = false;
        }

        // Check for required GDPR elements
        const pageContent = await page.content();
        const requiredTerms = [
          'data controller',
          'legal basis',
          'data retention',
          'right to access',
          'right to rectification',
          'right to erasure',
          'right to portability',
        ];

        for (const term of requiredTerms) {
          if (!pageContent.toLowerCase().includes(term)) {
            console.error(`Privacy policy missing required term: ${term}`);
            checksPassed = false;
          }
        }
      }

      if (check.expected.consentWithdrawal) {
        const consentSettings = await page.$('[data-testid="consent-settings"], .consent-management');
        if (!consentSettings) {
          console.error('Consent management interface not found');
          checksPassed = false;
        }
      }

      return checksPassed;
      
    } finally {
      await browser.close();
    }
  }

  private async checkAPICompliance(check: GDPRComplianceCheck): Promise<boolean> {
    // This would require authentication in real scenarios
    const authToken = process.env.TEST_AUTH_TOKEN;
    
    if (!authToken) {
      console.warn('No auth token provided for API compliance check');
      return true; // Skip if no auth available
    }

    const headers = {
      Authorization: `Bearer ${authToken}`,
      'Content-Type': 'application/json',
    };

    if (check.expected.dataPortability) {
      try {
        const response = await axios.get(check.endpoint!, { headers });
        
        if (response.status !== 200) {
          console.error('Data export endpoint not accessible');
          return false;
        }

        // Check if response contains user data in structured format
        const data = response.data;
        if (!data || typeof data !== 'object') {
          console.error('Data export does not return structured data');
          return false;
        }

        // Check for personal data fields
        const expectedFields = ['email', 'name', 'created_at'];
        for (const field of expectedFields) {
          if (!(field in data)) {
            console.error(`Data export missing field: ${field}`);
            return false;
          }
        }

      } catch (error) {
        console.error('Data export endpoint failed:', error);
        return false;
      }
    }

    if (check.expected.rightToErasure) {
      try {
        // Check if DELETE method is supported
        const response = await axios.delete(check.endpoint!, { 
          headers,
          validateStatus: () => true, // Don't throw on any status
        });
        
        if (response.status !== 200 && response.status !== 204) {
          console.error('Account deletion endpoint not properly implemented');
          return false;
        }

      } catch (error) {
        console.error('Account deletion endpoint failed:', error);
        return false;
      }
    }

    return true;
  }
}

Real-time Quality Dashboard​

// tools/monitoring/nfr-monitor.ts
export interface NFRMetrics {
  performance: {
    responseTime: number;
    throughput: number;
    errorRate: number;
    availability: number;
  };
  security: {
    vulnerabilityCount: number;
    lastScanDate: Date;
    securityScore: number;
  };
  scalability: {
    currentReplicas: number;
    cpuUtilization: number;
    memoryUtilization: number;
    autoScalingEvents: number;
  };
  reliability: {
    uptime: number;
    mtbf: number; // Mean Time Between Failures
    mttr: number; // Mean Time To Recovery
  };
  usability: {
    accessibilityScore: number;
    performanceScore: number;
    lastAuditDate: Date;
  };
}

export class NFRMonitor {
  async collectMetrics(): Promise<NFRMetrics> {
    const [performance, security, scalability, reliability, usability] = await Promise.all([
      this.collectPerformanceMetrics(),
      this.collectSecurityMetrics(),
      this.collectScalabilityMetrics(),
      this.collectReliabilityMetrics(),
      this.collectUsabilityMetrics(),
    ]);

    return {
      performance,
      security,
      scalability,
      reliability,
      usability,
    };
  }

  private async collectPerformanceMetrics() {
    // Implementation would integrate with monitoring tools like Prometheus
    return {
      responseTime: 150, // ms
      throughput: 1200, // requests/second
      errorRate: 0.5, // percentage
      availability: 99.9, // percentage
    };
  }

  private async collectSecurityMetrics() {
    return {
      vulnerabilityCount: 2,
      lastScanDate: new Date(),
      securityScore: 85,
    };
  }

  private async collectScalabilityMetrics() {
    return {
      currentReplicas: 3,
      cpuUtilization: 65,
      memoryUtilization: 70,
      autoScalingEvents: 5,
    };
  }

  private async collectReliabilityMetrics() {
    return {
      uptime: 99.95,
      mtbf: 720, // hours
      mttr: 15, // minutes
    };
  }

  private async collectUsabilityMetrics() {
    return {
      accessibilityScore: 92,
      performanceScore: 88,
      lastAuditDate: new Date(),
    };
  }

  async validateNFRThresholds(metrics: NFRMetrics): Promise<boolean> {
    const thresholds = {
      performance: {
        responseTime: 500,
        errorRate: 1.0,
        availability: 99.5,
      },
      security: {
        maxVulnerabilities: 5,
        minSecurityScore: 80,
      },
      scalability: {
        maxCpuUtilization: 80,
        maxMemoryUtilization: 85,
      },
      reliability: {
        minUptime: 99.0,
        maxMttr: 30,
      },
      usability: {
        minAccessibilityScore: 90,
        minPerformanceScore: 85,
      },
    };

    const violations: string[] = [];

    // Check performance thresholds
    if (metrics.performance.responseTime > thresholds.performance.responseTime) {
      violations.push(`Response time: ${metrics.performance.responseTime}ms > ${thresholds.performance.responseTime}ms`);
    }

    if (metrics.performance.errorRate > thresholds.performance.errorRate) {
      violations.push(`Error rate: ${metrics.performance.errorRate}% > ${thresholds.performance.errorRate}%`);
    }

    if (metrics.performance.availability < thresholds.performance.availability) {
      violations.push(`Availability: ${metrics.performance.availability}% < ${thresholds.performance.availability}%`);
    }

    // Check security thresholds
    if (metrics.security.vulnerabilityCount > thresholds.security.maxVulnerabilities) {
      violations.push(`Vulnerabilities: ${metrics.security.vulnerabilityCount} > ${thresholds.security.maxVulnerabilities}`);
    }

    if (metrics.security.securityScore < thresholds.security.minSecurityScore) {
      violations.push(`Security score: ${metrics.security.securityScore} < ${thresholds.security.minSecurityScore}`);
    }

    // Check scalability thresholds
    if (metrics.scalability.cpuUtilization > thresholds.scalability.maxCpuUtilization) {
      violations.push(`CPU utilization: ${metrics.scalability.cpuUtilization}% > ${thresholds.scalability.maxCpuUtilization}%`);
    }

    if (metrics.scalability.memoryUtilization > thresholds.scalability.maxMemoryUtilization) {
      violations.push(`Memory utilization: ${metrics.scalability.memoryUtilization}% > ${thresholds.scalability.maxMemoryUtilization}%`);
    }

    // Check reliability thresholds
    if (metrics.reliability.uptime < thresholds.reliability.minUptime) {
      violations.push(`Uptime: ${metrics.reliability.uptime}% < ${thresholds.reliability.minUptime}%`);
    }

    if (metrics.reliability.mttr > thresholds.reliability.maxMttr) {
      violations.push(`MTTR: ${metrics.reliability.mttr}min > ${thresholds.reliability.maxMttr}min`);
    }

    // Check usability thresholds
    if (metrics.usability.accessibilityScore < thresholds.usability.minAccessibilityScore) {
      violations.push(`Accessibility score: ${metrics.usability.accessibilityScore} < ${thresholds.usability.minAccessibilityScore}`);
    }

    if (metrics.usability.performanceScore < thresholds.usability.minPerformanceScore) {
      violations.push(`Performance score: ${metrics.usability.performanceScore} < ${thresholds.usability.minPerformanceScore}`);
    }

    if (violations.length > 0) {
      console.error('NFR threshold violations:', violations);
      return false;
    }

    console.log('✅ All NFR thresholds met');
    return true;
  }
}
