Input Validation Guide

This guide provides comprehensive input validation strategies for NestJS microservices and Next.js microfrontend applications to ensure data integrity and security.

Overview​

Input validation is a critical security and data integrity practice that must be implemented at multiple layers of the application. This guide covers client-side validation for user experience, server-side validation for security, and database constraints for data integrity.

Validation Architecture​

Multi-Layer Validation Strategy​

graph TD
    A[User Input] --> B[Client-Side Validation]
    B --> C[API Gateway Validation]
    C --> D[Service-Level Validation]
    D --> E[Database Constraints]
    
    B --> F[User Feedback]
    C --> G[Request Rejection]
    D --> H[Business Rule Validation]
    E --> I[Data Integrity Protection]

// src/dtos/create-user.dto.ts
import {
  IsEmail,
  IsString,
  IsOptional,
  IsPhoneNumber,
  IsDate,
  MinLength,
  MaxLength,
  Matches,
  IsEnum,
  ValidateNested,
  ArrayMinSize,
  ArrayMaxSize,
} from 'class-validator';
import { Transform, Type } from 'class-transformer';

export enum UserRole {
  ADMIN = 'admin',
  USER = 'user',
  MODERATOR = 'moderator',
}

export class AddressDto {
  @IsString()
  @MinLength(1)
  @MaxLength(100)
  street: string;

  @IsString()
  @MinLength(1)
  @MaxLength(50)
  city: string;

  @IsString()
  @Matches(/^\d{5}(-\d{4})?$/, { message: 'Invalid postal code format' })
  postalCode: string;

  @IsString()
  @MinLength(2)
  @MaxLength(3)
  country: string;
}

export class CreateUserDto {
  @IsEmail({}, { message: 'Please provide a valid email address' })
  @MaxLength(255)
  @Transform(({ value }) => value?.toLowerCase()?.trim())
  email: string;

  @IsString()
  @MinLength(8, { message: 'Password must be at least 8 characters long' })
  @MaxLength(128)
  @Matches(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]/, {
    message: 'Password must contain uppercase, lowercase, number and special character',
  })
  password: string;

  @IsString()
  @MinLength(2)
  @MaxLength(50)
  @Matches(/^[a-zA-Z\s-']+$/, { message: 'First name can only contain letters, spaces, hyphens, and apostrophes' })
  firstName: string;

  @IsString()
  @MinLength(2)
  @MaxLength(50)
  @Matches(/^[a-zA-Z\s-']+$/, { message: 'Last name can only contain letters, spaces, hyphens, and apostrophes' })
  lastName: string;

  @IsOptional()
  @IsPhoneNumber(null, { message: 'Please provide a valid phone number' })
  phoneNumber?: string;

  @IsOptional()
  @IsDate()
  @Type(() => Date)
  @Transform(({ value }) => value && new Date(value))
  dateOfBirth?: Date;

  @IsEnum(UserRole, { message: 'Role must be one of: admin, user, moderator' })
  role: UserRole;

  @IsOptional()
  @ValidateNested()
  @Type(() => AddressDto)
  address?: AddressDto;

  @IsOptional()
  @IsString({ each: true })
  @ArrayMinSize(0)
  @ArrayMaxSize(10)
  @Transform(({ value }) => value?.map((tag: string) => tag.trim().toLowerCase()))
  tags?: string[];
}

// src/validators/custom-validators.ts
import { registerDecorator, ValidationOptions, ValidationArguments } from 'class-validator';

// Unique email validator
export function IsUniqueEmail(validationOptions?: ValidationOptions) {
  return function (object: Object, propertyName: string) {
    registerDecorator({
      name: 'isUniqueEmail',
      target: object.constructor,
      propertyName: propertyName,
      options: validationOptions,
      validator: {
        async validate(value: any, args: ValidationArguments) {
          const userService = Container.get(UserService);
          const existingUser = await userService.findByEmail(value);
          return !existingUser;
        },
        defaultMessage(args: ValidationArguments) {
          return `Email ${args.value} is already taken`;
        },
      },
    });
  };
}

// Strong password validator
export function IsStrongPassword(validationOptions?: ValidationOptions) {
  return function (object: Object, propertyName: string) {
    registerDecorator({
      name: 'isStrongPassword',
      target: object.constructor,
      propertyName: propertyName,
      options: validationOptions,
      validator: {
        validate(value: any, args: ValidationArguments) {
          if (typeof value !== 'string') return false;
          
          const hasMinLength = value.length >= 8;
          const hasUpperCase = /[A-Z]/.test(value);
          const hasLowerCase = /[a-z]/.test(value);
          const hasNumbers = /\d/.test(value);
          const hasSpecialChar = /[@$!%*?&]/.test(value);
          const hasNoSequence = !/(.)\1{2,}/.test(value); // No 3+ consecutive same chars
          
          return hasMinLength && hasUpperCase && hasLowerCase && 
                 hasNumbers && hasSpecialChar && hasNoSequence;
        },
        defaultMessage(args: ValidationArguments) {
          return 'Password must be at least 8 characters with uppercase, lowercase, number, special character, and no repetitive sequences';
        },
      },
    });
  };
}

// File type validator
export function IsValidFileType(allowedTypes: string[], validationOptions?: ValidationOptions) {
  return function (object: Object, propertyName: string) {
    registerDecorator({
      name: 'isValidFileType',
      target: object.constructor,
      propertyName: propertyName,
      constraints: [allowedTypes],
      options: validationOptions,
      validator: {
        validate(value: any, args: ValidationArguments) {
          if (!value || !value.mimetype) return false;
          const [allowedTypes] = args.constraints;
          return allowedTypes.includes(value.mimetype);
        },
        defaultMessage(args: ValidationArguments) {
          const [allowedTypes] = args.constraints;
          return `File type must be one of: ${allowedTypes.join(', ')}`;
        },
      },
    });
  };
}

Advanced Validation Techniques​

// src/validation/validation.service.ts
@Injectable()
export class ValidationService {
  private readonly logger = new Logger(ValidationService.name);

  // Sanitize input to prevent XSS
  sanitizeInput(input: string): string {
    return input
      .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '') // Remove script tags
      .replace(/javascript:/gi, '') // Remove javascript: protocol
      .replace(/on\w+="[^"]*"/gi, '') // Remove event handlers
      .trim();
  }

  // Validate business rules
  async validateBusinessRules(dto: any, context: string): Promise<ValidationResult> {
    const errors: string[] = [];

    switch (context) {
      case 'user-creation':
        await this.validateUserCreationRules(dto, errors);
        break;
      case 'order-creation':
        await this.validateOrderCreationRules(dto, errors);
        break;
      default:
        this.logger.warn(`Unknown validation context: ${context}`);
    }

    return {
      isValid: errors.length === 0,
      errors,
    };
  }

  private async validateUserCreationRules(dto: CreateUserDto, errors: string[]): Promise<void> {
    // Age validation
    if (dto.dateOfBirth) {
      const age = this.calculateAge(dto.dateOfBirth);
      if (age < 13) {
        errors.push('User must be at least 13 years old');
      }
    }

    // Email domain validation
    const emailDomain = dto.email.split('@')[1];
    const blockedDomains = await this.getBlockedEmailDomains();
    if (blockedDomains.includes(emailDomain)) {
      errors.push('Email domain is not allowed');
    }

    // Password history check
    if (dto.password) {
      const passwordHistory = await this.getUserPasswordHistory(dto.email);
      const isPasswordReused = await this.checkPasswordReuse(dto.password, passwordHistory);
      if (isPasswordReused) {
        errors.push('Password has been used recently, please choose a different password');
      }
    }
  }

  // Rate limiting validation
  async validateRateLimit(identifier: string, action: string): Promise<boolean> {
    const key = `rate_limit:${action}:${identifier}`;
    const current = await this.redisService.get(key);
    
    const limits = {
      'user-creation': { count: 5, window: 3600 }, // 5 attempts per hour
      'password-reset': { count: 3, window: 1800 }, // 3 attempts per 30 minutes
      'login': { count: 10, window: 900 }, // 10 attempts per 15 minutes
    };

    const limit = limits[action];
    if (!limit) return true;

    if (current >= limit.count) {
      return false;
    }

    await this.redisService.incr(key, limit.window);
    return true;
  }
}

Input Validation Middleware​

// src/middleware/validation.middleware.ts
@Injectable()
export class ValidationMiddleware implements NestMiddleware {
  constructor(private readonly validationService: ValidationService) {}

  use(req: Request, res: Response, next: NextFunction) {
    // Sanitize query parameters
    if (req.query) {
      for (const [key, value] of Object.entries(req.query)) {
        if (typeof value === 'string') {
          req.query[key] = this.validationService.sanitizeInput(value);
        }
      }
    }

    // Sanitize body parameters
    if (req.body && typeof req.body === 'object') {
      this.sanitizeObject(req.body);
    }

    // Validate content type for POST/PUT requests
    if (['POST', 'PUT', 'PATCH'].includes(req.method)) {
      const contentType = req.headers['content-type'];
      if (!contentType || !contentType.includes('application/json')) {
        return res.status(400).json({
          statusCode: 400,
          message: 'Content-Type must be application/json',
          error: 'Bad Request',
        });
      }
    }

    // Check payload size
    const contentLength = parseInt(req.headers['content-length'] || '0');
    if (contentLength > 10 * 1024 * 1024) { // 10MB limit
      return res.status(413).json({
        statusCode: 413,
        message: 'Payload too large',
        error: 'Payload Too Large',
      });
    }

    next();
  }

  private sanitizeObject(obj: any): void {
    for (const [key, value] of Object.entries(obj)) {
      if (typeof value === 'string') {
        obj[key] = this.validationService.sanitizeInput(value);
      } else if (typeof value === 'object' && value !== null) {
        this.sanitizeObject(value);
      }
    }
  }
}

// components/validation/password-strength-indicator.tsx
interface PasswordStrengthIndicatorProps {
  password: string;
}

export const PasswordStrengthIndicator: React.FC<PasswordStrengthIndicatorProps> = ({
  password,
}) => {
  const getPasswordStrength = (password: string): {
    score: number;
    label: string;
    color: string;
  } => {
    if (!password) return { score: 0, label: '', color: '' };

    let score = 0;
    const checks = {
      length: password.length >= 8,
      lowercase: /[a-z]/.test(password),
      uppercase: /[A-Z]/.test(password),
      numbers: /\d/.test(password),
      special: /[@$!%*?&]/.test(password),
      noSequence: !/(.)\1{2,}/.test(password),
    };

    score = Object.values(checks).filter(Boolean).length;

    const strengthLevels = [
      { score: 0, label: '', color: '' },
      { score: 1, label: 'Very Weak', color: 'red' },
      { score: 2, label: 'Weak', color: 'orange' },
      { score: 3, label: 'Fair', color: 'yellow' },
      { score: 4, label: 'Good', color: 'blue' },
      { score: 5, label: 'Strong', color: 'green' },
      { score: 6, label: 'Very Strong', color: 'green' },
    ];

    return strengthLevels[score] || strengthLevels[0];
  };

  const strength = getPasswordStrength(password);

  if (!password) return null;

  return (
    <div className="mt-2">
      <div className="flex items-center space-x-2">
        <div className="flex-1 bg-gray-200 rounded-full h-2">
          <div
            className={`h-2 rounded-full bg-${strength.color}-500 transition-all duration-300`}
            style={{ width: `${(strength.score / 6) * 100}%` }}
          />
        </div>
        <span className={`text-sm font-medium text-${strength.color}-600`}>
          {strength.label}
        </span>
      </div>
      
      <div className="mt-2 space-y-1">
        <PasswordRequirement
          met={password.length >= 8}
          text="At least 8 characters"
        />
        <PasswordRequirement
          met={/[a-z]/.test(password)}
          text="Contains lowercase letter"
        />
        <PasswordRequirement
          met={/[A-Z]/.test(password)}
          text="Contains uppercase letter"
        />
        <PasswordRequirement
          met={/\d/.test(password)}
          text="Contains number"
        />
        <PasswordRequirement
          met={/[@$!%*?&]/.test(password)}
          text="Contains special character"
        />
      </div>
    </div>
  );
};

    )}
    <span className={`text-xs ${met ? 'text-green-600' : 'text-gray-500'}`}>

    </span>
  </div>
);

        checks: [
          {
            name: 'check_email_format',

          },
          {
            name: 'check_name_length',
            expression: 'LENGTH(first_name) >= 2 AND LENGTH(last_name) >= 2',
          },
        ],
      }),
      true
    );

    await queryRunner.createIndex(
      'users',
      new Index('idx_users_email', ['email'])
    );

    await queryRunner.createIndex(
      'users',
      new Index('idx_users_active', ['is_active'])
    );

    await queryRunner.createIndex(
      'users',
      new Index('idx_users_role', ['role'])
    );
  }

Input Sanitization​

// src/security/sanitization.service.ts
@Injectable()
export class SanitizationService {
  // HTML sanitization
  sanitizeHtml(input: string): string {
    const allowedTags = ['p', 'br', 'strong', 'em', 'u', 'ol', 'ul', 'li'];
    const allowedAttributes = {
      'a': ['href'],
    };

    return sanitizeHtml(input, {
      allowedTags,
      allowedAttributes,
      disallowedTagsMode: 'discard',
    });
  }

  // SQL injection prevention
  sanitizeSqlInput(input: string): string {
    return input
      .replace(/['";\\]/g, '') // Remove dangerous characters
      .trim()
      .substring(0, 1000); // Limit length
  }

  // File name sanitization
  sanitizeFileName(fileName: string): string {
    return fileName
      .replace(/[^a-zA-Z0-9._-]/g, '') // Only allow safe characters
      .replace(/\.{2,}/g, '.') // Prevent directory traversal
      .substring(0, 255); // Limit length
  }

  // URL sanitization
  sanitizeUrl(url: string): string {
    try {
      const parsed = new URL(url);
      
      // Only allow specific protocols
      if (!['http:', 'https:'].includes(parsed.protocol)) {
        throw new Error('Invalid protocol');
      }
      
      return parsed.toString();
    } catch {
      throw new ValidationException('Invalid URL format');
    }
  }
}

Best Practices​

Validation Principles​

  1. Defense in Depth: Validate at multiple layers
  2. Fail Securely: Default to rejection when validation fails
  3. Clear Error Messages: Provide helpful feedback to users
  4. Performance: Validate efficiently without blocking
  5. Consistency: Use consistent validation patterns

Implementation Guidelines​

  1. Server-Side Priority: Never rely solely on client-side validation
  2. Sanitization: Clean inputs before validation
  3. Business Rules: Separate technical from business validation
  4. Rate Limiting: Prevent abuse through validation endpoints
  5. Logging: Log validation failures for security monitoring

        const result = await validate({ password });
        expect(result.isValid).toBe(true);
      }
    });

        'weak',
