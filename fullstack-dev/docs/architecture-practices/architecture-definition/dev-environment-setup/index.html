Development Scripts​

Setup Scripts​

scripts/setup-dev.sh​

#!/bin/bash

echo "🚀 Setting up development environment..."

# Check requirements
command -v node >/dev/null 2>&1 || { echo "Node.js is required but not installed. Aborting." >&2; exit 1; }
command -v docker >/dev/null 2>&1 || { echo "Docker is required but not installed. Aborting." >&2; exit 1; }
command -v pnpm >/dev/null 2>&1 || { echo "pnpm is required but not installed. Installing..." >&2; npm install -g pnpm; }

# Install dependencies
echo "📦 Installing dependencies..."
pnpm install

# Setup environment files
echo "⚙️ Setting up environment files..."
for app in apps/*/; do
  if [ ! -f "${app}.env" ]; then
    cp "${app}.env.example" "${app}.env"
    echo "Created ${app}.env"
  fi
done

# Start infrastructure
echo "🐳 Starting infrastructure services..."
docker-compose -f docker-compose.dev.yml up -d

# Wait for services
echo "⏳ Waiting for services to be ready..."
sleep 10

# Run database migrations
echo "🗄️ Running database migrations..."
pnpm --filter user-service typeorm:migration:run
pnpm --filter product-service typeorm:migration:run
pnpm --filter order-service typeorm:migration:run

echo "✅ Development environment setup complete!"
echo "Run 'pnpm dev' to start all services"

scripts/start-dev.sh​

#!/bin/bash

echo "🚀 Starting development environment..."

# Start infrastructure
docker-compose -f docker-compose.dev.yml up -d

# Start all services
pnpm dev

Quick Start Commands​

# Initial setup
chmod +x scripts/setup-dev.sh
./scripts/setup-dev.sh

# Daily development
pnpm dev

# Individual service development
pnpm --filter user-service dev
pnpm --filter admin-frontend dev

# Testing
pnpm test
pnpm --filter user-service test:watch

# Building
pnpm build
pnpm --filter customer-frontend build

Docker Development Environment​

Development Dockerfile​

apps/user-service/Dockerfile.dev​

FROM node:18-alpine

WORKDIR /app

# Install pnpm
RUN npm install -g pnpm

# Copy package files
COPY package.json pnpm-lock.yaml ./
COPY apps/user-service/package.json ./apps/user-service/

# Install dependencies
RUN pnpm install --frozen-lockfile

# Copy source code
COPY . .

# Expose port
EXPOSE 3001

# Start development server
CMD ["pnpm", "--filter", "user-service", "dev"]

Docker Compose for Development​

# docker-compose.dev.yml
version: '3.8'

services:
  # Infrastructure
  postgres:
    image: postgres:15
    environment:
      POSTGRES_MULTIPLE_DATABASES: user_service_dev,product_service_dev,order_service_dev
      POSTGRES_USER: dev
      POSTGRES_PASSWORD: devpass
    volumes:
      - ./docker/postgres-multiple-databases.sh:/docker-entrypoint-initdb.d/postgres-multiple-databases.sh
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"

  rabbitmq:
    image: rabbitmq:3-management
    environment:
      RABBITMQ_DEFAULT_USER: dev
      RABBITMQ_DEFAULT_PASS: devpass
    ports:
      - "5672:5672"
      - "15672:15672"

  # Optional: Development services
  user-service:
    build:
      context: .
      dockerfile: apps/user-service/Dockerfile.dev
    volumes:
      - .:/app
      - /app/node_modules
    ports:
      - "3001:3001"
    environment:
      - NODE_ENV=development
    depends_on:
      - postgres
      - redis

volumes:
  postgres_data:

Troubleshooting Guide​

Common Issues​

Port Conflicts​

# Check what's running on port
lsof -ti:3000

# Kill process
kill -9 $(lsof -ti:3000)

# Use different ports
PORT=3010 pnpm --filter user-service dev

Database Connection Issues​

# Check PostgreSQL status
docker-compose -f docker-compose.dev.yml ps postgres

# Reset database
docker-compose -f docker-compose.dev.yml down
docker volume rm ecommerce-platform_postgres_data
docker-compose -f docker-compose.dev.yml up -d

Node Modules Issues​

# Clean install
rm -rf node_modules
rm -rf apps/*/node_modules
rm pnpm-lock.yaml
pnpm install

Environment Health Checks​

Health Check Script​

#!/bin/bash
# scripts/health-check.sh

echo "🔍 Checking development environment health..."

# Check Node.js
node_version=$(node -v)
echo "Node.js: $node_version"

# Check services
services=("postgres:5432" "redis:6379" "rabbitmq:5672")
for service in "${services[@]}"; do
  if nc -z localhost ${service#*:}; then
    echo "✅ $service is running"
  else
    echo "❌ $service is not running"
  fi
done

# Check APIs
apis=("http://localhost:3001/health" "http://localhost:3002/health")
for api in "${apis[@]}"; do
  if curl -f -s "$api" > /dev/null; then
    echo "✅ $api is healthy"
  else
    echo "❌ $api is not responding"
  fi
done
