Architecture Foundation

This guide establishes the foundational architecture patterns and mechanisms for NestJS microservices and Next.js microfrontend applications.

Overview​

Architecture foundation defines the core layers, frameworks, patterns, and key mechanisms that form the backbone of a distributed system. This documentation ensures consistent architectural patterns across all services and applications.

Core Architecture Patterns​

Microservices Architecture (NestJS)​

Service Architecture Layers​

// src/architecture/layers.ts
export enum ArchitectureLayer {
  PRESENTATION = 'presentation',    // Controllers, DTOs, Validation
  APPLICATION = 'application',      // Use Cases, Business Logic
  DOMAIN = 'domain',               // Entities, Domain Services
  INFRASTRUCTURE = 'infrastructure' // Repositories, External Services
}

// Layered Architecture Implementation
@Injectable()
export class LayeredArchitectureService {
  // Presentation Layer - Controllers
  async handleRequest(dto: CreateUserDto): Promise<UserResponseDto> {
    // Input validation and transformation
    const command = this.mapToCommand(dto);
    
    // Delegate to application layer
    const result = await this.userApplicationService.createUser(command);
    
    // Transform to response DTO
    return this.mapToResponse(result);
  }
  
  // Application Layer - Use Cases
  async createUser(command: CreateUserCommand): Promise<User> {
    // Business logic orchestration
    const user = await this.userDomainService.createUser(command);
    await this.notificationService.sendWelcomeEmail(user);
    
    return user;
  }
  
  // Domain Layer - Business Logic
  createUser(userData: UserData): User {
    // Domain validation and business rules
    if (!this.isValidEmail(userData.email)) {
      throw new DomainException('Invalid email format');
    }
    
    return new User(userData);
  }
  
  // Infrastructure Layer - Data Access
  async saveUser(user: User): Promise<void> {
    const entity = this.userMapper.toEntity(user);
    await this.userRepository.save(entity);
  }
}

export class HttpClientService {
  private readonly httpService: HttpService;
  
  constructor(httpService: HttpService) {
    this.httpService = httpService;
  }
  
  async callUserService(userId: string): Promise<UserData> {
    try {
      const response = await this.httpService.axiosRef.get(
        `${process.env.USER_SERVICE_URL}/users/${userId}`,
        {
          timeout: 5000,
          headers: {
            'Authorization': `Bearer ${this.getServiceToken()}`,
            'X-Request-ID': this.generateRequestId(),
          },
        }
      );
      
      return response.data;
    } catch (error) {
      throw new ServiceCommunicationException(
        `Failed to call user service: ${error.message}`
      );
    }
  }
}

Microfrontend Architecture (Next.js)​

Application Architecture​

// lib/architecture/micro-frontend-shell.ts
export class MicrofrontendShell {
  private remotes: Map<string, RemoteModule> = new Map();
  private eventBus: EventBus = new EventBus();
  
  async loadRemote(name: string, url: string): Promise<RemoteModule> {
    try {
      const module = await import(/* webpackIgnore: true */ url);
      this.remotes.set(name, module);
      
      // Initialize remote with shared services
      if (module.initialize) {
        await module.initialize({
          eventBus: this.eventBus,
          authService: this.authService,
          themeProvider: this.themeProvider,
        });
      }
      
      return module;
    } catch (error) {
      throw new MicrofrontendLoadException(
        `Failed to load microfrontend ${name}: ${error.message}`
      );
    }
  }
  
  render(remoteName: string, props: any): React.ReactElement {
    const remote = this.remotes.get(remoteName);
    if (!remote) {
      return <MicrofrontendErrorBoundary remoteName={remoteName} />;
    }
    
    return (
      <ErrorBoundary>
        <remote.Component {...props} />
      </ErrorBoundary>
    );
  }
}

Component Architecture​

// components/architecture/base-component.tsx
export interface BaseComponentProps {
  className?: string;
  testId?: string;
  children?: React.ReactNode;
}

export abstract class BaseComponent<T extends BaseComponentProps> extends React.Component<T> {
  protected logger: Logger;
  
  constructor(props: T) {
    super(props);
    this.logger = new Logger(this.constructor.name);
  }
  
  componentDidCatch(error: Error, errorInfo: React.ErrorInfo): void {
    this.logger.error('Component error:', error, errorInfo);
    // Send to error tracking service
    ErrorTrackingService.captureException(error, {
      component: this.constructor.name,
      props: this.props,
      errorInfo,
    });
  }
  
  protected renderErrorState(error: Error): React.ReactElement {
    return (
      <div className="error-state" data-testid={`${this.props.testId}-error`}>
        <h3>Something went wrong</h3>
        {process.env.NODE_ENV === 'development' && (
          <pre>{error.message}</pre>
        )}
      </div>
    );
  }
}

Key Mechanisms​

Error Handling​

Global Exception Handling (NestJS)​

// src/common/filters/global-exception.filter.ts
@Catch()
export class GlobalExceptionFilter implements ExceptionFilter {
  private readonly logger = new Logger(GlobalExceptionFilter.name);
  
  catch(exception: unknown, host: ArgumentsHost): void {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();
    const request = ctx.getRequest<Request>();
    
    let status: number;
    let message: string;
    let code: string;
    
    if (exception instanceof HttpException) {
      status = exception.getStatus();
      const exceptionResponse = exception.getResponse();
      message = typeof exceptionResponse === 'string' 
        ? exceptionResponse 
        : (exceptionResponse as any).message;
      code = exception.constructor.name;
    } else if (exception instanceof DomainException) {
      status = 400;
      message = exception.message;
      code = exception.code;
    } else {
      status = 500;
      message = 'Internal server error';
      code = 'INTERNAL_ERROR';
    }
    
    const errorResponse = {
      statusCode: status,
      timestamp: new Date().toISOString(),
      path: request.url,
      method: request.method,
      message,
      code,
      requestId: request.headers['x-request-id'],
    };
    
    this.logger.error(
      `${request.method} ${request.url}`,
      exception instanceof Error ? exception.stack : exception,
      {
        ...errorResponse,
        userId: request.user?.id,
      }
    );
    
    response.status(status).json(errorResponse);
  }
}

Error Boundary (Next.js)​

// components/common/error-boundary.tsx
interface ErrorBoundaryState {
  hasError: boolean;
  error?: Error;
  errorInfo?: React.ErrorInfo;
}

export class ErrorBoundary extends React.Component<
  React.PropsWithChildren<{ fallback?: React.ComponentType<any> }>,
  ErrorBoundaryState
> {
  constructor(props: React.PropsWithChildren<{}>) {
    super(props);
    this.state = { hasError: false };
  }
  
  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { hasError: true, error };
  }
  
  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    this.setState({ error, errorInfo });
    
    // Log to monitoring service
    ErrorTrackingService.captureException(error, {
      tags: { boundary: 'global' },
      extra: { errorInfo },
    });
    
    // Report to analytics
    Analytics.track('error_boundary_triggered', {
      error: error.message,
      stack: error.stack,
      componentStack: errorInfo.componentStack,
    });
  }
  
  render() {
    if (this.state.hasError) {
      const FallbackComponent = this.props.fallback || DefaultErrorFallback;
      
      return (
        <FallbackComponent
          error={this.state.error}
          resetError={() => this.setState({ hasError: false })}
        />
      );
    }
    
    return this.props.children;
  }
}

Logging Architecture​

Structured Logging (NestJS)​

// src/common/logging/logger.service.ts
@Injectable()
export class LoggerService {
  private readonly winston: winston.Logger;
  
  constructor() {
    this.winston = winston.createLogger({
      level: process.env.LOG_LEVEL || 'info',
      format: winston.format.combine(
        winston.format.timestamp(),
        winston.format.errors({ stack: true }),
        winston.format.json(),
        winston.format.printf(({ timestamp, level, message, service, requestId, userId, ...meta }) => {
          return JSON.stringify({
            timestamp,
            level,
            message,
            service: process.env.SERVICE_NAME || 'unknown',
            requestId,
            userId,
            ...meta,
          });
        })
      ),
      transports: [
        new winston.transports.Console(),
        new winston.transports.File({ filename: 'logs/error.log', level: 'error' }),
        new winston.transports.File({ filename: 'logs/combined.log' }),
      ],
    });
  }
  
  log(message: string, context?: string, meta?: object): void {
    this.winston.info(message, { context, ...meta });
  }
  
  error(message: string, trace?: string, context?: string, meta?: object): void {
    this.winston.error(message, { trace, context, ...meta });
  }
  
  warn(message: string, context?: string, meta?: object): void {
    this.winston.warn(message, { context, ...meta });
  }
  
  debug(message: string, context?: string, meta?: object): void {
    this.winston.debug(message, { context, ...meta });
  }
}

Client-side Logging (Next.js)​

// lib/logging/client-logger.ts
export class ClientLogger {
  private static instance: ClientLogger;
  private buffer: LogEntry[] = [];
  private flushInterval: NodeJS.Timeout;
  
  private constructor() {
    this.flushInterval = setInterval(() => this.flush(), 5000);
  }
  
  static getInstance(): ClientLogger {
    if (!ClientLogger.instance) {
      ClientLogger.instance = new ClientLogger();
    }
    return ClientLogger.instance;
  }
  
  log(level: LogLevel, message: string, meta?: object): void {
    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      url: window.location.href,
      userAgent: navigator.userAgent,
      sessionId: this.getSessionId(),
      userId: this.getUserId(),
      ...meta,
    };
    
    this.buffer.push(entry);
    
    // Immediate flush for errors
    if (level === 'error') {
      this.flush();
    }
  }
  
  private async flush(): Promise<void> {
    if (this.buffer.length === 0) return;
    
    const logs = [...this.buffer];
    this.buffer = [];
    
    try {
      await fetch('/api/logs', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ logs }),
      });
    } catch (error) {
      // Re-add logs to buffer on failure
      this.buffer.unshift(...logs);
    }
  }
}

Transaction Control​

Database Transactions (NestJS)​

// src/common/decorators/transactional.decorator.ts
export function Transactional(): MethodDecorator {
  return function (target: any, propertyKey: string | symbol, descriptor: PropertyDescriptor) {
    const originalMethod = descriptor.value;
    
    descriptor.value = async function (...args: any[]) {
      const dataSource = getDataSource();
      const queryRunner = dataSource.createQueryRunner();
      
      await queryRunner.connect();
      await queryRunner.startTransaction();
      
      try {
        // Set the query runner in the context
        setQueryRunner(queryRunner);
        
        const result = await originalMethod.apply(this, args);
        
        await queryRunner.commitTransaction();
        return result;
      } catch (error) {
        await queryRunner.rollbackTransaction();
        throw error;
      } finally {
        await queryRunner.release();
        clearQueryRunner();
      }
    };
  };
}

// Usage example
@Injectable()
export class UserService {
  @Transactional()
  async createUserWithProfile(userData: CreateUserData): Promise<User> {
    const user = await this.userRepository.save(userData);
    await this.profileRepository.save({ userId: user.id, ...userData.profile });
    await this.auditRepository.save({ action: 'USER_CREATED', userId: user.id });
    
    return user;
  }
}

Authentication & Authorization​

// src/auth/jwt-auth.service.ts
@Injectable()
export class JwtAuthService {
  constructor(
    private readonly jwtService: JwtService,
    private readonly userService: UserService,
  ) {}
  
  async validateToken(token: string): Promise<User | null> {
    try {
      const payload = this.jwtService.verify(token);
      const user = await this.userService.findById(payload.sub);
      
      if (!user || !user.isActive) {
        return null;
      }
      
      return user;
    } catch (error) {
      return null;
    }
  }
  
  async generateTokens(user: User): Promise<TokenPair> {
    const payload = {
      sub: user.id,
      email: user.email,
      roles: user.roles,
    };
    
    const accessToken = this.jwtService.sign(payload, { expiresIn: '15m' });
    const refreshToken = this.jwtService.sign(
      { sub: user.id, type: 'refresh' },
      { expiresIn: '7d' }
    );
    
    return { accessToken, refreshToken };
  }
}

Data Protection​

// src/common/encryption/encryption.service.ts
@Injectable()
export class EncryptionService {
  private readonly algorithm = 'aes-256-gcm';
  private readonly keyLength = 32;
  
  encrypt(text: string, key?: string): EncryptedData {
    const encryptionKey = key ? Buffer.from(key, 'hex') : this.getDefaultKey();
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipher(this.algorithm, encryptionKey, iv);
    
    let encrypted = cipher.update(text, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    const authTag = cipher.getAuthTag();
    
    return {
      encrypted,
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex'),
    };
  }
  
  decrypt(encryptedData: EncryptedData, key?: string): string {
    const encryptionKey = key ? Buffer.from(key, 'hex') : this.getDefaultKey();
    const decipher = crypto.createDecipher(
      this.algorithm,
      encryptionKey,
      Buffer.from(encryptedData.iv, 'hex')
    );
    
    decipher.setAuthTag(Buffer.from(encryptedData.authTag, 'hex'));
    
    let decrypted = decipher.update(encryptedData.encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }
}

Caching Strategy​

// src/caching/redis-cache.service.ts
@Injectable()
export class RedisCacheService {
  constructor(
    @Inject('REDIS_CLIENT') private readonly redis: Redis,
  ) {}
  
  async get<T>(key: string): Promise<T | null> {
    try {
      const value = await this.redis.get(key);
      return value ? JSON.parse(value) : null;
    } catch (error) {
      this.logger.error(`Cache get error for key ${key}:`, error);
      return null;
    }
  }
  
  async set(key: string, value: any, ttl: number = 3600): Promise<void> {
    try {
      await this.redis.setex(key, ttl, JSON.stringify(value));
    } catch (error) {
      this.logger.error(`Cache set error for key ${key}:`, error);
    }
  }
  
  async del(key: string): Promise<void> {
    try {
      await this.redis.del(key);
    } catch (error) {
      this.logger.error(`Cache delete error for key ${key}:`, error);
    }
  }
  
  // Cache decorator
  @Cache('user:{id}', 300) // 5 minutes TTL
  async getUserById(id: string): Promise<User> {
    return this.userRepository.findById(id);
  }
}

Database Optimization​

// src/database/query-optimization.service.ts
@Injectable()
export class QueryOptimizationService {
  constructor(
    @InjectRepository(User) private readonly userRepository: Repository<User>,
  ) {}
  
  // Optimized query with proper joins and pagination
  async findUsersWithProfiles(
    filters: UserFilters,
    pagination: PaginationOptions,
  ): Promise<PaginatedResult<UserWithProfile>> {
    const queryBuilder = this.userRepository
      .createQueryBuilder('user')
      .leftJoinAndSelect('user.profile', 'profile')
      .leftJoinAndSelect('user.roles', 'roles');
    
    // Apply filters
    if (filters.email) {
      queryBuilder.andWhere('user.email ILIKE :email', { 
        email: `%${filters.email}%` 
      });
    }
    
    if (filters.status) {
      queryBuilder.andWhere('user.status = :status', { 
        status: filters.status 
      });
    }
    
    // Apply pagination
    queryBuilder
      .skip(pagination.offset)
      .take(pagination.limit)
      .orderBy('user.createdAt', 'DESC');
    
    const [users, total] = await queryBuilder.getManyAndCount();
    
    return {
      data: users,
      total,
      page: pagination.page,
      limit: pagination.limit,
    };
  }
}

Health Checks​

// src/health/health.service.ts
@Injectable()
export class HealthService {
  constructor(
    private readonly health: HealthCheckService,
    private readonly typeorm: TypeOrmHealthIndicator,
    private readonly redis: RedisHealthIndicator,
    private readonly microservice: MicroserviceHealthIndicator,
  ) {}
  
  @Get()
  @HealthCheck()
  check() {
    return this.health.check([
      // Database health
      () => this.typeorm.pingCheck('database'),
      
      // Redis health
      () => this.redis.pingCheck('redis'),
      
      // External service health
      () => this.microservice.pingCheck('user-service', {
        transport: Transport.HTTP,
        options: { url: process.env.USER_SERVICE_URL + '/health' },
      }),
      
      // Custom business logic health
      () => this.checkBusinessLogic(),
    ]);
  }
  
  private async checkBusinessLogic(): Promise<HealthIndicatorResult> {
    try {
      // Check critical business functionality
      const criticalCheck = await this.performCriticalCheck();
      
      if (criticalCheck.isHealthy) {
        return {
          'business-logic': {
            status: 'up',
            info: criticalCheck.info,
          },
        };
      } else {
        throw new Error(criticalCheck.error);
      }
    } catch (error) {
      return {
        'business-logic': {
          status: 'down',
          message: error.message,
        },
      };
    }
  }
}

Best Practices​

Architecture Guidelines​

  1. Separation of Concerns: Clear layer boundaries and responsibilities
  2. Dependency Inversion: Depend on abstractions, not concretions
  3. Single Responsibility: Each component has one reason to change
  4. Open/Closed Principle: Open for extension, closed for modification
  5. Interface Segregation: Clients shouldn't depend on unused interfaces

Implementation Standards​

  1. Error Handling: Comprehensive error handling at all layers
  2. Logging: Structured logging with correlation IDs
  3. Security: Defense in depth with multiple security layers
  4. Performance: Caching, optimization, and monitoring
  5. Testing: Unit, integration, and end-to-end test coverage

  1. API Documentation: OpenAPI/Swagger specifications
