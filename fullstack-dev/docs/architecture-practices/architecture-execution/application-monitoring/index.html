Application Performance Monitoring Guide

This guide provides comprehensive application performance monitoring (APM) strategies for NestJS microservices and Next.js microfrontend applications, covering performance optimization, real user monitoring, and automated performance testing.

Overview​

Application Performance Monitoring (APM) ensures optimal user experience by tracking application-level metrics, identifying performance bottlenecks, and enabling proactive optimization. This guide covers both server-side and client-side performance monitoring with actionable insights.

NestJS Performance Interceptor​

// src/interceptors/performance.interceptor.ts
import {
  Injectable,
  NestInterceptor,
  ExecutionContext,
  CallHandler,
  Logger,
} from '@nestjs/common';
import { Observable } from 'rxjs';
import { tap, catchError } from 'rxjs/operators';
import { Request, Response } from 'express';
import { PerformanceService } from '../monitoring/performance.service';

@Injectable()
export class PerformanceInterceptor implements NestInterceptor {
  private readonly logger = new Logger(PerformanceInterceptor.name);

  constructor(private readonly performanceService: PerformanceService) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const request = context.switchToHttp().getRequest<Request>();
    const response = context.switchToHttp().getResponse<Response>();
    
    const startTime = Date.now();
    const startMemory = process.memoryUsage();
    const startCpuUsage = process.cpuUsage();

    // Extract route information
    const method = request.method;
    const route = request.route?.path || request.url;
    const userAgent = request.headers['user-agent'];
    const requestId = request.headers['x-request-id'] as string;

    return next.handle().pipe(
      tap(() => {
        this.recordPerformanceMetrics(
          method,
          route,
          response.statusCode,
          startTime,
          startMemory,
          startCpuUsage,
          userAgent,
          requestId,
        );
      }),
      catchError((error) => {
        this.recordPerformanceMetrics(
          method,
          route,
          response.statusCode || 500,
          startTime,
          startMemory,
          startCpuUsage,
          userAgent,
          requestId,
          error,
        );
        throw error;
      }),
    );
  }

  private recordPerformanceMetrics(
    method: string,
    route: string,
    statusCode: number,
    startTime: number,
    startMemory: NodeJS.MemoryUsage,
    startCpuUsage: NodeJS.CpuUsage,
    userAgent: string,
    requestId: string,
    error?: Error,
  ): void {
    const endTime = Date.now();
    const duration = endTime - startTime;
    const endMemory = process.memoryUsage();
    const endCpuUsage = process.cpuUsage(startCpuUsage);

    const performanceData = {
      method,
      route,
      statusCode,
      duration,
      memoryUsage: {
        heapUsed: endMemory.heapUsed - startMemory.heapUsed,
        heapTotal: endMemory.heapTotal - startMemory.heapTotal,
        external: endMemory.external - startMemory.external,
        rss: endMemory.rss - startMemory.rss,
      },
      cpuUsage: {
        user: endCpuUsage.user,
        system: endCpuUsage.system,
      },
      userAgent,
      requestId,
      timestamp: new Date(startTime).toISOString(),
      error: error?.message,
    };

    // Record metrics
    this.performanceService.recordRequestPerformance(performanceData);

    // Log slow requests
    if (duration > 1000) {
      this.logger.warn(`Slow request detected: ${method} ${route} - ${duration}ms`, {
        requestId,
        performanceData,
      });
    }

    // Log high memory usage
    if (performanceData.memoryUsage.heapUsed > 50 * 1024 * 1024) { // 50MB
      this.logger.warn(`High memory usage detected: ${method} ${route}`, {
        requestId,
        memoryUsage: performanceData.memoryUsage,
      });
    }
  }
}

Database Query Performance Monitor​

// src/monitoring/database-performance.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { DataSource, QueryRunner } from 'typeorm';
import { Performance } from 'perf_hooks';

export interface QueryPerformanceData {
  query: string;
  parameters: any[];
  duration: number;
  rowCount?: number;
  timestamp: Date;
  executionPlan?: any;
  stackTrace?: string;
}

@Injectable()
export class DatabasePerformanceService {
  private readonly logger = new Logger(DatabasePerformanceService.name);
  private readonly slowQueryThreshold = 1000; // 1 second
  private readonly queryPerformanceData: QueryPerformanceData[] = [];

  constructor(private readonly dataSource: DataSource) {
    this.setupQueryLogging();
  }

  private setupQueryLogging(): void {
    // Override the query method to add performance tracking
    const originalQuery = this.dataSource.query.bind(this.dataSource);
    
    this.dataSource.query = async (query: string, parameters?: any[]): Promise<any> => {
      const startTime = performance.now();
      const stackTrace = new Error().stack;
      
      try {
        const result = await originalQuery(query, parameters);
        const duration = performance.now() - startTime;
        
        const performanceData: QueryPerformanceData = {
          query: this.sanitizeQuery(query),
          parameters: parameters || [],
          duration,
          rowCount: Array.isArray(result) ? result.length : undefined,
          timestamp: new Date(),
          stackTrace: this.extractRelevantStackTrace(stackTrace),
        };

        this.recordQueryPerformance(performanceData);
        
        return result;
      } catch (error) {
        const duration = performance.now() - startTime;
        
        this.logger.error(`Query failed after ${duration}ms: ${query}`, {
          error: error.message,
          parameters,
          duration,
        });
        
        throw error;
      }
    };
  }

  private recordQueryPerformance(data: QueryPerformanceData): void {
    // Store recent query performance data
    this.queryPerformanceData.push(data);
    
    // Keep only last 1000 queries to prevent memory leaks
    if (this.queryPerformanceData.length > 1000) {
      this.queryPerformanceData.shift();
    }

    // Log slow queries
    if (data.duration > this.slowQueryThreshold) {
      this.logger.warn(`Slow query detected: ${data.duration}ms`, {
        query: data.query,
        parameters: data.parameters,
        rowCount: data.rowCount,
        stackTrace: data.stackTrace,
      });

      // Get execution plan for slow queries
      this.getExecutionPlan(data.query, data.parameters).then(plan => {
        if (plan) {
          this.logger.warn(`Execution plan for slow query:`, plan);
        }
      });
    }
  }

  async getQueryPerformanceReport(): Promise<{
    slowQueries: QueryPerformanceData[];
    averageDuration: number;
    totalQueries: number;
    queryPatterns: Record<string, number>;
  }> {
    const slowQueries = this.queryPerformanceData.filter(q => q.duration > this.slowQueryThreshold);
    const averageDuration = this.queryPerformanceData.reduce((sum, q) => sum + q.duration, 0) / this.queryPerformanceData.length;
    
    const queryPatterns = this.queryPerformanceData.reduce((patterns, query) => {
      const pattern = this.extractQueryPattern(query.query);
      patterns[pattern] = (patterns[pattern] || 0) + 1;
      return patterns;
    }, {} as Record<string, number>);

    return {
      slowQueries: slowQueries.slice(-10), // Last 10 slow queries
      averageDuration,
      totalQueries: this.queryPerformanceData.length,
      queryPatterns,
    };
  }

  private async getExecutionPlan(query: string, parameters?: any[]): Promise<any> {
    try {
      const explainQuery = `EXPLAIN (ANALYZE, BUFFERS, FORMAT JSON) ${query}`;
      const result = await this.dataSource.query(explainQuery, parameters);
      return result[0]['QUERY PLAN'];
    } catch (error) {
      this.logger.error('Failed to get execution plan', error);
      return null;
    }
  }

  private sanitizeQuery(query: string): string {
    // Remove sensitive data from query logs
    return query.replace(/\$\d+/g, '?'); // Replace parameter placeholders
  }

  private extractQueryPattern(query: string): string {
    // Extract query pattern for grouping similar queries
    return query
      .replace(/\s+/g, ' ')
      .replace(/\d+/g, 'N')
      .replace(/'[^']*'/g, "'VALUE'")
      .trim()
      .substring(0, 100);
  }

  private extractRelevantStackTrace(stackTrace?: string): string {
    if (!stackTrace) return '';
    
    const lines = stackTrace.split('\n');
    const relevantLines = lines.filter(line => 
      line.includes('src/') && 
      !line.includes('node_modules') &&
      !line.includes('DatabasePerformanceService')
    );
    
    return relevantLines.slice(0, 3).join('\n');
  }
}

Memory and CPU Profiling​

// src/monitoring/profiling.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import * as v8 from 'v8';
import * as fs from 'fs/promises';
import * as path from 'path';

export interface ProfileData {
  timestamp: Date;
  memoryUsage: NodeJS.MemoryUsage;
  cpuUsage: NodeJS.CpuUsage;
  heapStatistics: v8.HeapStatistics;
  heapSpaceStatistics: v8.HeapSpaceStatistics[];
  eventLoopDelay?: number;
}

@Injectable()
export class ProfilingService {
  private readonly logger = new Logger(ProfilingService.name);
  private readonly profileDataHistory: ProfileData[] = [];
  private cpuUsageStart: NodeJS.CpuUsage;
  private eventLoopMonitor?: any;

  constructor() {
    this.cpuUsageStart = process.cpuUsage();
    this.initializeEventLoopMonitoring();
  }

  @Cron(CronExpression.EVERY_30_SECONDS)
  async collectProfileData(): Promise<void> {
    try {
      const profileData: ProfileData = {
        timestamp: new Date(),
        memoryUsage: process.memoryUsage(),
        cpuUsage: process.cpuUsage(this.cpuUsageStart),
        heapStatistics: v8.getHeapStatistics(),
        heapSpaceStatistics: v8.getHeapSpaceStatistics(),
        eventLoopDelay: this.eventLoopMonitor?.delay,
      };

      this.profileDataHistory.push(profileData);

      // Keep only last 100 records to prevent memory leaks
      if (this.profileDataHistory.length > 100) {
        this.profileDataHistory.shift();
      }

      // Check for memory leaks
      await this.checkForMemoryLeaks(profileData);

      // Check for high CPU usage
      this.checkForHighCpuUsage(profileData);

      // Reset CPU usage baseline
      this.cpuUsageStart = process.cpuUsage();
    } catch (error) {
      this.logger.error('Failed to collect profile data', error);
    }
  }

  private initializeEventLoopMonitoring(): void {
    if (typeof require !== 'undefined') {
      try {
        const { monitorEventLoopDelay } = require('perf_hooks');
        this.eventLoopMonitor = monitorEventLoopDelay({ resolution: 20 });
        this.eventLoopMonitor.enable();
      } catch (error) {
        this.logger.warn('Event loop monitoring not available', error);
      }
    }
  }

  private async checkForMemoryLeaks(currentProfile: ProfileData): Promise<void> {
    if (this.profileDataHistory.length < 10) return; // Need at least 10 data points

    const recentProfiles = this.profileDataHistory.slice(-10);
    const memoryGrowth = recentProfiles.map(profile => profile.memoryUsage.heapUsed);
    
    // Check if memory is consistently growing
    let consecutiveIncreases = 0;
    for (let i = 1; i < memoryGrowth.length; i++) {
      if (memoryGrowth[i] > memoryGrowth[i - 1]) {
        consecutiveIncreases++;
      } else {
        consecutiveIncreases = 0;
      }
    }

    if (consecutiveIncreases >= 8) { // 8 consecutive increases
      this.logger.warn('Potential memory leak detected', {
        currentHeapUsed: currentProfile.memoryUsage.heapUsed,
        memoryGrowthPattern: memoryGrowth,
      });

      // Trigger heap snapshot for analysis
      await this.createHeapSnapshot();
    }

    // Check for high memory usage
    const heapUsedMB = currentProfile.memoryUsage.heapUsed / 1024 / 1024;
    if (heapUsedMB > 512) { // Alert if heap usage exceeds 512MB
      this.logger.warn(`High memory usage detected: ${heapUsedMB.toFixed(2)}MB`, {
        memoryUsage: currentProfile.memoryUsage,
        heapStatistics: currentProfile.heapStatistics,
      });
    }
  }

  private checkForHighCpuUsage(currentProfile: ProfileData): void {
    const totalCpuTime = currentProfile.cpuUsage.user + currentProfile.cpuUsage.system;
    const cpuUsagePercent = (totalCpuTime / 30000000) * 100; // 30 seconds in microseconds

    if (cpuUsagePercent > 80) {
      this.logger.warn(`High CPU usage detected: ${cpuUsagePercent.toFixed(2)}%`, {
        cpuUsage: currentProfile.cpuUsage,
      });
    }

    // Check event loop delay
    if (currentProfile.eventLoopDelay && currentProfile.eventLoopDelay > 100) {
      this.logger.warn(`High event loop delay detected: ${currentProfile.eventLoopDelay}ms`);
    }
  }

  private async createHeapSnapshot(): Promise<void> {
    try {
      const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
      const filename = `heap-snapshot-${timestamp}.heapsnapshot`;
      const filepath = path.join(process.cwd(), 'logs', filename);

      // Ensure logs directory exists
      await fs.mkdir(path.dirname(filepath), { recursive: true });

      // Create heap snapshot
      const heapSnapshot = v8.getHeapSnapshot();
      const writeStream = require('fs').createWriteStream(filepath);
      
      heapSnapshot.pipe(writeStream);

      this.logger.log(`Heap snapshot created: ${filename}`);
    } catch (error) {
      this.logger.error('Failed to create heap snapshot', error);
    }
  }

  async getProfileReport(): Promise<{
    current: ProfileData;
    averages: {
      memoryUsage: number;
      cpuUsagePercent: number;
      eventLoopDelay: number;
    };
    trends: {
      memoryTrend: 'increasing' | 'decreasing' | 'stable';
      cpuTrend: 'increasing' | 'decreasing' | 'stable';
    };
  }> {
    const current = this.profileDataHistory[this.profileDataHistory.length - 1];
    
    if (!current) {
      throw new Error('No profile data available');
    }

    const recentProfiles = this.profileDataHistory.slice(-20); // Last 20 samples
    
    const averageMemory = recentProfiles.reduce((sum, p) => sum + p.memoryUsage.heapUsed, 0) / recentProfiles.length;
    const averageCpuPercent = recentProfiles.reduce((sum, p) => {
      const totalCpu = p.cpuUsage.user + p.cpuUsage.system;
      return sum + (totalCpu / 30000000) * 100;
    }, 0) / recentProfiles.length;
    const averageEventLoopDelay = recentProfiles.reduce((sum, p) => sum + (p.eventLoopDelay || 0), 0) / recentProfiles.length;

    // Calculate trends
    const memoryTrend = this.calculateTrend(recentProfiles.map(p => p.memoryUsage.heapUsed));
    const cpuTrend = this.calculateTrend(recentProfiles.map(p => p.cpuUsage.user + p.cpuUsage.system));

    return {
      current,
      averages: {
        memoryUsage: averageMemory,
        cpuUsagePercent: averageCpuPercent,
        eventLoopDelay: averageEventLoopDelay,
      },
      trends: {
        memoryTrend,
        cpuTrend,
      },
    };
  }

  private calculateTrend(values: number[]): 'increasing' | 'decreasing' | 'stable' {
    if (values.length < 2) return 'stable';

    const first = values.slice(0, Math.floor(values.length / 2));
    const second = values.slice(Math.floor(values.length / 2));

    const firstAvg = first.reduce((sum, val) => sum + val, 0) / first.length;
    const secondAvg = second.reduce((sum, val) => sum + val, 0) / second.length;

    const changePercent = ((secondAvg - firstAvg) / firstAvg) * 100;

    if (changePercent > 10) return 'increasing';
    if (changePercent < -10) return 'decreasing';
    return 'stable';
  }
}

Client-Side Performance Monitoring​

Next.js Performance Tracking​

// lib/performance/client-performance.ts
import { NextWebVitalsMetric } from 'next/app';

export interface ClientPerformanceMetric {
  id: string;
  name: string;
  value: number;
  delta?: number;
  rating: 'good' | 'needs-improvement' | 'poor';
  navigationType: string;
  url: string;
  timestamp: number;
  userAgent: string;
  connectionType?: string;
  deviceMemory?: number;
}

class ClientPerformanceTracker {
  private metrics: ClientPerformanceMetric[] = [];
  private sessionId: string;
  private userId?: string;

  constructor() {
    this.sessionId = this.generateSessionId();
    this.initializePerformanceObserver();
    this.trackPageVisibility();
    this.trackUserInteractions();
  }

  setUserId(userId: string): void {
    this.userId = userId;
  }

  // Track Core Web Vitals
  trackWebVital(metric: NextWebVitalsMetric): void {
    const performanceMetric: ClientPerformanceMetric = {
      id: metric.id,
      name: metric.name,
      value: metric.value,
      delta: metric.delta,
      rating: this.getMetricRating(metric.name, metric.value),
      navigationType: this.getNavigationType(),
      url: window.location.href,
      timestamp: Date.now(),
      userAgent: navigator.userAgent,
      connectionType: this.getConnectionType(),
      deviceMemory: this.getDeviceMemory(),
    };

    this.metrics.push(performanceMetric);
    this.sendMetric(performanceMetric);

    // Log poor performance
    if (performanceMetric.rating === 'poor') {
      console.warn(`Poor ${metric.name} performance:`, {
        value: metric.value,
        url: window.location.href,
      });
    }
  }

  // Track custom performance metrics
  trackCustomMetric(name: string, value: number, metadata?: Record<string, any>): void {
    const metric: ClientPerformanceMetric = {
      id: this.generateMetricId(),
      name,
      value,
      rating: 'good', // Custom metrics don't have predefined thresholds
      navigationType: this.getNavigationType(),
      url: window.location.href,
      timestamp: Date.now(),
      userAgent: navigator.userAgent,
      ...metadata,
    };

    this.metrics.push(metric);
    this.sendMetric(metric);
  }

  // Track resource loading performance
  private initializePerformanceObserver(): void {
    if ('PerformanceObserver' in window) {
      // Track resource timing
      const resourceObserver = new PerformanceObserver((list) => {
        for (const entry of list.getEntries()) {
          if (entry.entryType === 'resource') {
            this.trackResourceTiming(entry as PerformanceResourceTiming);
          }
        }
      });
      resourceObserver.observe({ entryTypes: ['resource'] });

      // Track navigation timing
      const navigationObserver = new PerformanceObserver((list) => {
        for (const entry of list.getEntries()) {
          if (entry.entryType === 'navigation') {
            this.trackNavigationTiming(entry as PerformanceNavigationTiming);
          }
        }
      });
      navigationObserver.observe({ entryTypes: ['navigation'] });

      // Track long tasks
      const longTaskObserver = new PerformanceObserver((list) => {
        for (const entry of list.getEntries()) {
          this.trackLongTask(entry);
        }
      });
      longTaskObserver.observe({ entryTypes: ['longtask'] });
    }
  }

  private trackResourceTiming(entry: PerformanceResourceTiming): void {
    const loadTime = entry.responseEnd - entry.requestStart;
    
    // Track slow resources
    if (loadTime > 1000) { // More than 1 second
      this.trackCustomMetric('slow_resource_load', loadTime, {
        resourceName: entry.name,
        resourceType: this.getResourceType(entry.name),
        transferSize: entry.transferSize,
        encodedBodySize: entry.encodedBodySize,
      });
    }

    // Track failed resources
    if (entry.responseStatus >= 400) {
      this.trackCustomMetric('resource_error', entry.responseStatus, {
        resourceName: entry.name,
        resourceType: this.getResourceType(entry.name),
      });
    }
  }

  private trackNavigationTiming(entry: PerformanceNavigationTiming): void {
    const metrics = {
      dns_lookup: entry.domainLookupEnd - entry.domainLookupStart,
      tcp_connection: entry.connectEnd - entry.connectStart,
      ssl_handshake: entry.connectEnd - entry.secureConnectionStart,
      ttfb: entry.responseStart - entry.requestStart,
      dom_content_loaded: entry.domContentLoadedEventEnd - entry.navigationStart,
      load_complete: entry.loadEventEnd - entry.navigationStart,
    };

    for (const [name, value] of Object.entries(metrics)) {
      if (value > 0) {
        this.trackCustomMetric(`navigation_${name}`, value);
      }
    }
  }

  private trackLongTask(entry: PerformanceEntry): void {
    this.trackCustomMetric('long_task', entry.duration, {
      taskType: (entry as any).name,
      startTime: entry.startTime,
    });
  }

  private trackPageVisibility(): void {
    let visibilityStart = Date.now();
    let isVisible = !document.hidden;

    const handleVisibilityChange = () => {
      const now = Date.now();
      
      if (document.hidden && isVisible) {
        // Page became hidden
        const visibleDuration = now - visibilityStart;
        this.trackCustomMetric('page_visible_duration', visibleDuration);
        isVisible = false;
      } else if (!document.hidden && !isVisible) {
        // Page became visible
        visibilityStart = now;
        isVisible = true;
      }
    };

    document.addEventListener('visibilitychange', handleVisibilityChange);

    // Track time on page when leaving
    window.addEventListener('beforeunload', () => {
      if (isVisible) {
        const visibleDuration = Date.now() - visibilityStart;
        this.trackCustomMetric('page_visible_duration', visibleDuration);
      }
    });
  }

  private trackUserInteractions(): void {
    const interactionStart = Date.now();

    // Track input delay
    ['click', 'keydown', 'touchstart'].forEach(eventType => {
      document.addEventListener(eventType, (event) => {
        const interactionDelay = Date.now() - interactionStart;
        
        if (interactionDelay > 100) { // More than 100ms
          this.trackCustomMetric('interaction_delay', interactionDelay, {
            eventType,
            targetElement: (event.target as Element)?.tagName,
          });
        }
      }, { passive: true });
    });

    // Track scroll performance
    let scrollStart = 0;
    let scrollCount = 0;
    
    window.addEventListener('scroll', () => {
      if (scrollStart === 0) {
        scrollStart = performance.now();
      }
      scrollCount++;
    }, { passive: true });

    // Debounced scroll end detection
    let scrollTimeout: NodeJS.Timeout;
    window.addEventListener('scroll', () => {
      clearTimeout(scrollTimeout);
      scrollTimeout = setTimeout(() => {
        if (scrollStart > 0) {
          const scrollDuration = performance.now() - scrollStart;
          this.trackCustomMetric('scroll_session', scrollDuration, {
            scrollCount,
          });
          scrollStart = 0;
          scrollCount = 0;
        }
      }, 150);
    }, { passive: true });
  }

  private async sendMetric(metric: ClientPerformanceMetric): Promise<void> {
    try {
      // Use sendBeacon for reliability, fallback to fetch
      const data = JSON.stringify({
        ...metric,
        sessionId: this.sessionId,
        userId: this.userId,
      });

      if (navigator.sendBeacon) {
        navigator.sendBeacon('/api/metrics/performance', data);
      } else {
        fetch('/api/metrics/performance', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: data,
        }).catch(error => {
          console.error('Failed to send performance metric:', error);
        });
      }
    } catch (error) {
      console.error('Error sending performance metric:', error);
    }
  }

  private getMetricRating(name: string, value: number): 'good' | 'needs-improvement' | 'poor' {
    const thresholds = {
      CLS: { good: 0.1, poor: 0.25 },
      FID: { good: 100, poor: 300 },
      FCP: { good: 1800, poor: 3000 },
      LCP: { good: 2500, poor: 4000 },
      TTFB: { good: 800, poor: 1800 },
    };

    const threshold = thresholds[name as keyof typeof thresholds];
    if (!threshold) return 'good';

    if (value <= threshold.good) return 'good';
    if (value <= threshold.poor) return 'needs-improvement';
    return 'poor';
  }

  private getNavigationType(): string {
    if ('navigation' in performance) {
      const navigation = performance.navigation;
      const types = ['navigate', 'reload', 'back_forward', 'prerender'];
      return types[navigation.type] || 'unknown';
    }
    return 'unknown';
  }

  private getConnectionType(): string | undefined {
    const connection = (navigator as any).connection || (navigator as any).mozConnection || (navigator as any).webkitConnection;
    return connection?.effectiveType;
  }

  private getDeviceMemory(): number | undefined {
    return (navigator as any).deviceMemory;
  }

  private getResourceType(url: string): string {
    const extension = url.split('.').pop()?.toLowerCase();
    
    if (['js', 'mjs'].includes(extension || '')) return 'script';
    if (['css'].includes(extension || '')) return 'stylesheet';
    if (['png', 'jpg', 'jpeg', 'gif', 'svg', 'webp'].includes(extension || '')) return 'image';
    if (['woff', 'woff2', 'ttf', 'eot'].includes(extension || '')) return 'font';
    if (url.includes('/api/')) return 'api';
    
    return 'other';
  }

  private generateSessionId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  private generateMetricId(): string {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  // Get performance summary
  getPerformanceSummary(): {
    totalMetrics: number;
    poorPerformanceCount: number;
    averageLoadTime: number;
    topIssues: Array<{ name: string; count: number; avgValue: number }>;
  } {
    const poorMetrics = this.metrics.filter(m => m.rating === 'poor');
    
    const issueGroups = this.metrics.reduce((groups, metric) => {
      if (metric.rating === 'poor') {
        const key = metric.name;
        if (!groups[key]) {
          groups[key] = { name: key, count: 0, totalValue: 0 };
        }
        groups[key].count++;
        groups[key].totalValue += metric.value;
      }
      return groups;
    }, {} as Record<string, { name: string; count: number; totalValue: number }>);

    const topIssues = Object.values(issueGroups)
      .map(group => ({
        name: group.name,
        count: group.count,
        avgValue: group.totalValue / group.count,
      }))
      .sort((a, b) => b.count - a.count)
      .slice(0, 5);

    const loadTimeMetrics = this.metrics.filter(m => m.name === 'LCP' || m.name === 'FCP');
    const averageLoadTime = loadTimeMetrics.length > 0 
      ? loadTimeMetrics.reduce((sum, m) => sum + m.value, 0) / loadTimeMetrics.length 
      : 0;

    return {
      totalMetrics: this.metrics.length,
      poorPerformanceCount: poorMetrics.length,
      averageLoadTime,
      topIssues,
    };
  }
}

// hooks/use-performance-monitoring.ts
import { useEffect } from 'react';
import { useRouter } from 'next/router';
import { clientPerformanceTracker } from '../lib/performance/client-performance';

export const usePerformanceMonitoring = (userId?: string) => {
  const router = useRouter();

  useEffect(() => {
    if (userId) {
      clientPerformanceTracker.setUserId(userId);
    }

    // Track route changes
    const handleRouteChangeStart = (url: string) => {
      const startTime = performance.now();
      
      const handleRouteChangeComplete = () => {
        const loadTime = performance.now() - startTime;
        clientPerformanceTracker.trackCustomMetric('route_change_duration', loadTime, {
          route: url,
        });
        router.events.off('routeChangeComplete', handleRouteChangeComplete);
      };
      
      router.events.on('routeChangeComplete', handleRouteChangeComplete);
    };

    router.events.on('routeChangeStart', handleRouteChangeStart);

    return () => {
      router.events.off('routeChangeStart', handleRouteChangeStart);
    };
  }, [router, userId]);

  // Manual performance tracking functions
  const trackCustomMetric = (name: string, value: number, metadata?: Record<string, any>) => {
    clientPerformanceTracker.trackCustomMetric(name, value, metadata);
  };

  const trackUserAction = (action: string, component: string, duration?: number) => {
    clientPerformanceTracker.trackCustomMetric('user_action', duration || 1, {
      action,
      component,
      url: window.location.pathname,
    });
  };

  const trackApiCall = (endpoint: string, method: string, duration: number, status: number) => {
    clientPerformanceTracker.trackCustomMetric('api_call_duration', duration, {
      endpoint,
      method,
      status,
    });
  };

  return {
    trackCustomMetric,
    trackUserAction,
    trackApiCall,
    getPerformanceSummary: () => clientPerformanceTracker.getPerformanceSummary(),
  };
};

Automated Performance Testing​

// scripts/performance-tests.ts
import puppeteer from 'puppeteer';
import lighthouse from 'lighthouse';

interface PerformanceTestResult {
  url: string;
  timestamp: Date;
  metrics: {
    firstContentfulPaint: number;
    largestContentfulPaint: number;
    firstInputDelay: number;
    cumulativeLayoutShift: number;
    speedIndex: number;
    timeToInteractive: number;
  };
  scores: {
    performance: number;
    accessibility: number;
    bestPractices: number;
    seo: number;
  };
  opportunities: Array<{
    id: string;
    title: string;
    description: string;
    score: number;
  }>;
}

class PerformanceTester {
  private browser?: puppeteer.Browser;

  async initialize(): Promise<void> {
    this.browser = await puppeteer.launch({
      headless: true,
      args: ['--no-sandbox', '--disable-setuid-sandbox'],
    });
  }

  async runPerformanceTest(url: string): Promise<PerformanceTestResult> {
    if (!this.browser) {
      await this.initialize();
    }

    const page = await this.browser!.newPage();
    
    try {
      // Configure page for performance testing
      await page.setCacheEnabled(false);
      await page.setViewport({ width: 1920, height: 1080 });

      // Run Lighthouse audit
      const { lhr } = await lighthouse(url, {
        port: new URL(this.browser!.wsEndpoint()).port,
        onlyCategories: ['performance', 'accessibility', 'best-practices', 'seo'],
        settings: {
          preset: 'desktop',
        },
      });

      const metrics = lhr.audits;
      
      return {
        url,
        timestamp: new Date(),
        metrics: {
          firstContentfulPaint: metrics['first-contentful-paint'].numericValue || 0,
          largestContentfulPaint: metrics['largest-contentful-paint'].numericValue || 0,
          firstInputDelay: metrics['max-potential-fid'].numericValue || 0,
          cumulativeLayoutShift: metrics['cumulative-layout-shift'].numericValue || 0,
          speedIndex: metrics['speed-index'].numericValue || 0,
          timeToInteractive: metrics['interactive'].numericValue || 0,
        },
        scores: {
          performance: (lhr.categories.performance?.score || 0) * 100,
          accessibility: (lhr.categories.accessibility?.score || 0) * 100,
          bestPractices: (lhr.categories['best-practices']?.score || 0) * 100,
          seo: (lhr.categories.seo?.score || 0) * 100,
        },
        opportunities: Object.values(lhr.audits)
          .filter(audit => audit.scoreDisplayMode === 'numeric' && audit.score !== null && audit.score < 0.9)
          .map(audit => ({
            id: audit.id,
            title: audit.title,
            description: audit.description,
            score: audit.score || 0,
          }))
          .sort((a, b) => a.score - b.score),
      };
    } finally {
      await page.close();
    }
  }

  async runLoadTest(url: string, concurrency: number = 10, duration: number = 60): Promise<{
    averageResponseTime: number;
    requestsPerSecond: number;
    errorRate: number;
    p95ResponseTime: number;
  }> {
    const results: Array<{ duration: number; success: boolean }> = [];
    const startTime = Date.now();
    const endTime = startTime + (duration * 1000);

    const runRequest = async (): Promise<void> => {
      while (Date.now() < endTime) {
        const requestStart = Date.now();
        
        try {
          const page = await this.browser!.newPage();
          await page.goto(url, { waitUntil: 'networkidle0' });
          await page.close();
          
          results.push({
            duration: Date.now() - requestStart,
            success: true,
          });
        } catch (error) {
          results.push({
            duration: Date.now() - requestStart,
            success: false,
          });
        }
      }
    };

    // Run concurrent requests
    const promises = Array(concurrency).fill(null).map(() => runRequest());
    await Promise.all(promises);

    // Calculate metrics
    const successfulResults = results.filter(r => r.success);
    const totalDuration = (Date.now() - startTime) / 1000;
    
    const averageResponseTime = successfulResults.length > 0 
      ? successfulResults.reduce((sum, r) => sum + r.duration, 0) / successfulResults.length 
      : 0;
    
    const requestsPerSecond = results.length / totalDuration;
    const errorRate = (results.length - successfulResults.length) / results.length;
    
    const sortedDurations = successfulResults.map(r => r.duration).sort((a, b) => a - b);
    const p95Index = Math.floor(sortedDurations.length * 0.95);
    const p95ResponseTime = sortedDurations[p95Index] || 0;

    return {
      averageResponseTime,
      requestsPerSecond,
      errorRate,
      p95ResponseTime,
    };
  }

  async close(): Promise<void> {
    if (this.browser) {
      await this.browser.close();
    }
  }
}

// Usage example
export async function runPerformanceTestSuite(): Promise<void> {
  const tester = new PerformanceTester();
  
  const urls = [
    'https://app.example.com',
    'https://app.example.com/dashboard',
    'https://app.example.com/profile',
  ];

  try {
    for (const url of urls) {
      console.log(`Testing ${url}...`);
      
      // Performance audit
      const performanceResult = await tester.runPerformanceTest(url);
      console.log(`Performance Score: ${performanceResult.scores.performance}`);
      console.log(`LCP: ${performanceResult.metrics.largestContentfulPaint}ms`);
      
      // Load test
      const loadResult = await tester.runLoadTest(url, 5, 30);
      console.log(`Average Response Time: ${loadResult.averageResponseTime}ms`);
      console.log(`Requests/sec: ${loadResult.requestsPerSecond}`);
      
      // Report issues
      if (performanceResult.opportunities.length > 0) {
        console.log('Performance Opportunities:');
        performanceResult.opportunities.slice(0, 3).forEach(opp => {
          console.log(`- ${opp.title} (Score: ${opp.score})`);
        });
      }
    }
  } finally {
    await tester.close();
  }
}
