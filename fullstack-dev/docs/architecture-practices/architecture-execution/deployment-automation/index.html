Deployment Automation Guide

This guide provides comprehensive deployment automation strategies for NestJS microservices and Next.js microfrontend applications, covering infrastructure as code, automated deployments, and production readiness.

Overview​

Deployment automation ensures consistent, reliable, and repeatable deployments across all environments. This guide covers Docker containerization, Kubernetes orchestration, Infrastructure as Code (IaC), and automated deployment pipelines for enterprise-grade applications.

Deployment Architecture​

Container Strategy​

Docker Configuration​

NestJS Microservice Dockerfile​

# Dockerfile for NestJS microservice
FROM node:18-alpine AS base

# Set working directory
WORKDIR /app

# Install dependencies for building native modules
RUN apk add --no-cache libc6-compat python3 make g++

# Copy package files
COPY package*.json ./
COPY tsconfig*.json ./

# Install dependencies
FROM base AS dependencies
RUN npm ci --only=production && npm cache clean --force

# Build stage
FROM base AS builder
COPY . .
RUN npm ci
RUN npm run build
RUN npm prune --production

# Production stage
FROM node:18-alpine AS production

# Create non-root user
RUN addgroup -g 1001 -S nodejs
RUN adduser -S nestjs -u 1001

# Set working directory
WORKDIR /app

# Copy built application
COPY --from=builder --chown=nestjs:nodejs /app/dist ./dist
COPY --from=builder --chown=nestjs:nodejs /app/node_modules ./node_modules
COPY --from=builder --chown=nestjs:nodejs /app/package.json ./package.json

# Health check script
COPY --chown=nestjs:nodejs scripts/health-check.sh ./health-check.sh
RUN chmod +x ./health-check.sh

# Switch to non-root user
USER nestjs

# Expose port
EXPOSE 3000

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD ./health-check.sh

# Start application
CMD ["node", "dist/main.js"]

Multi-Stage Build Optimization​

# Optimized multi-stage build with shared base
FROM node:18-alpine AS base
WORKDIR /app
RUN apk add --no-cache libc6-compat curl

# Dependencies cache layer
FROM base AS deps
COPY package*.json ./
RUN npm ci --only=production && npm cache clean --force

# Build dependencies layer  
FROM base AS build-deps
COPY package*.json ./
RUN npm ci

# Build layer
FROM build-deps AS builder
COPY . .
RUN npm run build
RUN npm run test:unit

# Production layer
FROM base AS runner
ENV NODE_ENV=production
RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 app

COPY --from=deps --chown=app:nodejs /app/node_modules ./node_modules
COPY --from=builder --chown=app:nodejs /app/dist ./dist
COPY --from=builder --chown=app:nodejs /app/package.json ./

USER app
EXPOSE 3000

CMD ["node", "dist/main.js"]

Deployment Orchestration​

// scripts/deploy.ts
import { execSync } from 'child_process';
import * as fs from 'fs';
import * as yaml from 'js-yaml';

interface DeploymentConfig {
  environment: string;
  namespace: string;
  imageTag: string;
  replicas: number;
  resources: {
    requests: { cpu: string; memory: string };
    limits: { cpu: string; memory: string };
  };
}

class DeploymentOrchestrator {
  private config: DeploymentConfig;

  constructor(environment: string) {
    this.config = this.loadConfig(environment);
  }

  async deploy(): Promise<void> {
    console.log(`Starting deployment to ${this.config.environment}`);

    try {
      // Pre-deployment checks
      await this.preDeploymentChecks();

      // Build and push image
      await this.buildAndPushImage();

      // Deploy to Kubernetes
      await this.deployToKubernetes();

      // Post-deployment verification
      await this.postDeploymentVerification();

      console.log('Deployment completed successfully');
    } catch (error) {
      console.error('Deployment failed:', error);
      await this.rollback();
      throw error;
    }
  }

  private loadConfig(environment: string): DeploymentConfig {
    const configPath = `./deploy/config/${environment}.json`;
    return JSON.parse(fs.readFileSync(configPath, 'utf8'));
  }

  private async preDeploymentChecks(): Promise<void> {
    console.log('Running pre-deployment checks...');

    // Check cluster connectivity
    try {
      execSync('kubectl cluster-info', { stdio: 'pipe' });
    } catch (error) {
      throw new Error('Cannot connect to Kubernetes cluster');
    }

    // Check namespace exists
    try {
      execSync(`kubectl get namespace ${this.config.namespace}`, { stdio: 'pipe' });
    } catch (error) {
      console.log(`Creating namespace ${this.config.namespace}`);
      execSync(`kubectl create namespace ${this.config.namespace}`);
    }

    // Validate configuration
    this.validateConfiguration();

    console.log('Pre-deployment checks completed');
  }

  private validateConfiguration(): void {
    const requiredFields = ['environment', 'namespace', 'imageTag', 'replicas'];
    for (const field of requiredFields) {
      if (!this.config[field]) {
        throw new Error(`Missing required configuration: ${field}`);
      }
    }

    if (this.config.replicas < 1) {
      throw new Error('Replicas must be at least 1');
    }
  }

  private async buildAndPushImage(): Promise<void> {
    console.log('Building and pushing Docker image...');

    const imageName = `myregistry/user-service:${this.config.imageTag}`;

    // Build image
    execSync(`docker build -t ${imageName} .`, { stdio: 'inherit' });

    // Security scan
    try {
      execSync(`docker run --rm -v /var/run/docker.sock:/var/run/docker.sock aquasec/trivy image ${imageName}`, { stdio: 'inherit' });
    } catch (error) {
      console.warn('Security scan failed, but continuing deployment');
    }

    // Push image
    execSync(`docker push ${imageName}`, { stdio: 'inherit' });

    console.log('Image built and pushed successfully');
  }

  private async deployToKubernetes(): Promise<void> {
    console.log('Deploying to Kubernetes...');

    // Generate deployment manifest
    const manifest = this.generateDeploymentManifest();
    fs.writeFileSync('./deploy/generated-manifest.yaml', manifest);

    // Apply manifest
    execSync(`kubectl apply -f ./deploy/generated-manifest.yaml`, { stdio: 'inherit' });

    // Wait for rollout to complete
    execSync(`kubectl rollout status deployment/user-service -n ${this.config.namespace} --timeout=600s`, { stdio: 'inherit' });

    console.log('Kubernetes deployment completed');
  }

  private generateDeploymentManifest(): string {
    const deployment = {
      apiVersion: 'apps/v1',
      kind: 'Deployment',
      metadata: {
        name: 'user-service',
        namespace: this.config.namespace,
        labels: {
          app: 'user-service',
          version: this.config.imageTag,
          environment: this.config.environment,
        },
      },
      spec: {
        replicas: this.config.replicas,
        strategy: {
          type: 'RollingUpdate',
          rollingUpdate: {
            maxSurge: 1,
            maxUnavailable: 0,
          },
        },
        selector: {
          matchLabels: {
            app: 'user-service',
          },
        },
        template: {
          metadata: {
            labels: {
              app: 'user-service',
              version: this.config.imageTag,
            },
          },
          spec: {
            containers: [
              {
                name: 'user-service',
                image: `myregistry/user-service:${this.config.imageTag}`,
                ports: [{ containerPort: 3000 }],
                resources: this.config.resources,
                livenessProbe: {
                  httpGet: {
                    path: '/health',
                    port: 3000,
                  },
                  initialDelaySeconds: 30,
                  periodSeconds: 30,
                },
                readinessProbe: {
                  httpGet: {
                    path: '/health/ready',
                    port: 3000,
                  },
                  initialDelaySeconds: 5,
                  periodSeconds: 10,
                },
              },
            ],
          },
        },
      },
    };

    return yaml.dump(deployment);
  }

  private async postDeploymentVerification(): Promise<void> {
    console.log('Running post-deployment verification...');

    // Wait for pods to be ready
    await this.waitForPodsReady();

    // Run health checks
    await this.runHealthChecks();

    // Run smoke tests
    await this.runSmokeTests();

    console.log('Post-deployment verification completed');
  }

  private async waitForPodsReady(): Promise<void> {
    console.log('Waiting for pods to be ready...');

    let retries = 30;
    while (retries > 0) {
      try {
        const output = execSync(
          `kubectl get pods -n ${this.config.namespace} -l app=user-service -o jsonpath='{.items[*].status.phase}'`,
          { encoding: 'utf8' }
        );

        const phases = output.trim().split(' ');
        if (phases.every(phase => phase === 'Running')) {
          console.log('All pods are running');
          return;
        }
      } catch (error) {
        console.warn('Error checking pod status:', error.message);
      }

      retries--;
      await new Promise(resolve => setTimeout(resolve, 10000)); // Wait 10 seconds
    }

    throw new Error('Pods did not become ready within timeout');
  }

  private async runHealthChecks(): Promise<void> {
    console.log('Running health checks...');

    try {
      // Port-forward to service for health check
      const portForwardProcess = execSync(
        `kubectl port-forward -n ${this.config.namespace} service/user-service 8080:80 &`,
        { stdio: 'pipe' }
      );

      // Wait a moment for port-forward to establish
      await new Promise(resolve => setTimeout(resolve, 5000));

      // Make health check request
      const response = await fetch('http://localhost:8080/health');
      if (!response.ok) {
        throw new Error(`Health check failed: ${response.status}`);
      }

      console.log('Health checks passed');
    } catch (error) {
      throw new Error(`Health check failed: ${error.message}`);
    }
  }

  private async runSmokeTests(): Promise<void> {
    console.log('Running smoke tests...');

    try {
      // Run smoke test suite
      execSync('npm run test:smoke', { stdio: 'inherit' });
      console.log('Smoke tests passed');
    } catch (error) {
      throw new Error(`Smoke tests failed: ${error.message}`);
    }
  }

  private async rollback(): Promise<void> {
    console.log('Rolling back deployment...');

    try {
      execSync(`kubectl rollout undo deployment/user-service -n ${this.config.namespace}`, { stdio: 'inherit' });
      console.log('Rollback completed');
    } catch (error) {
      console.error('Rollback failed:', error.message);
    }
  }
}

Blue-Green Deployment​

Blue-Green Strategy Implementation​

// scripts/blue-green-deploy.ts
interface BlueGreenConfig {
  blueNamespace: string;
  greenNamespace: string;
  serviceNamespace: string;
  healthCheckUrl: string;
  trafficShiftDelay: number;
}

class BlueGreenDeployment {
  private config: BlueGreenConfig;

  constructor(config: BlueGreenConfig) {
    this.config = config;
  }

  async deploy(newVersion: string): Promise<void> {
    console.log(`Starting blue-green deployment for version ${newVersion}`);

    try {
      // Determine current and target environments
      const currentEnv = await this.getCurrentEnvironment();
      const targetEnv = currentEnv === 'blue' ? 'green' : 'blue';
      const targetNamespace = targetEnv === 'blue' ? this.config.blueNamespace : this.config.greenNamespace;

      console.log(`Current environment: ${currentEnv}, Target environment: ${targetEnv}`);

      // Deploy to target environment
      await this.deployToEnvironment(targetNamespace, newVersion);

      // Run health checks
      await this.validateDeployment(targetNamespace);

      // Switch traffic
      await this.switchTraffic(targetEnv);

      // Verify traffic switch
      await this.verifyTrafficSwitch();

      // Cleanup old environment
      await this.cleanupOldEnvironment(currentEnv === 'blue' ? this.config.blueNamespace : this.config.greenNamespace);

      console.log('Blue-green deployment completed successfully');
    } catch (error) {
      console.error('Blue-green deployment failed:', error);
      throw error;
    }
  }

  private async getCurrentEnvironment(): Promise<'blue' | 'green'> {
    try {
      const serviceOutput = execSync(
        `kubectl get service user-service -n ${this.config.serviceNamespace} -o jsonpath='{.spec.selector.environment}'`,
        { encoding: 'utf8' }
      );
      return serviceOutput.trim() as 'blue' | 'green';
    } catch (error) {
      // Default to blue if service doesn't exist
      return 'blue';
    }
  }

  private async deployToEnvironment(namespace: string, version: string): Promise<void> {
    console.log(`Deploying version ${version} to ${namespace}`);

    const deployment = this.generateDeploymentManifest(namespace, version);
    fs.writeFileSync(`./deploy/${namespace}-manifest.yaml`, deployment);

    execSync(`kubectl apply -f ./deploy/${namespace}-manifest.yaml`, { stdio: 'inherit' });
    execSync(`kubectl rollout status deployment/user-service -n ${namespace} --timeout=600s`, { stdio: 'inherit' });
  }

  private async validateDeployment(namespace: string): Promise<void> {
    console.log(`Validating deployment in ${namespace}`);

    // Port-forward and run health checks
    const portForwardProcess = spawn('kubectl', [
      'port-forward',
      '-n', namespace,
      'service/user-service',
      '8080:80'
    ]);

    await new Promise(resolve => setTimeout(resolve, 5000));

    try {
      const response = await fetch('http://localhost:8080/health');
      if (!response.ok) {
        throw new Error(`Health check failed: ${response.status}`);
      }

      // Run additional validation tests
      await this.runValidationTests();

      console.log(`Deployment validation successful for ${namespace}`);
    } finally {
      portForwardProcess.kill();
    }
  }

  private async switchTraffic(targetEnv: 'blue' | 'green'): Promise<void> {
    console.log(`Switching traffic to ${targetEnv} environment`);

    const serviceManifest = this.generateServiceManifest(targetEnv);
    fs.writeFileSync('./deploy/service-switch.yaml', serviceManifest);

    execSync('kubectl apply -f ./deploy/service-switch.yaml', { stdio: 'inherit' });

    // Wait for DNS propagation
    await new Promise(resolve => setTimeout(resolve, this.config.trafficShiftDelay));
  }

  private async verifyTrafficSwitch(): Promise<void> {
    console.log('Verifying traffic switch...');

    let retries = 5;
    while (retries > 0) {
      try {
        const response = await fetch(this.config.healthCheckUrl);
        if (response.ok) {
          console.log('Traffic switch verified successfully');
          return;
        }
      } catch (error) {
        console.warn(`Traffic verification attempt failed: ${error.message}`);
      }

      retries--;
      await new Promise(resolve => setTimeout(resolve, 10000));
    }

    throw new Error('Traffic switch verification failed');
  }

  private generateServiceManifest(targetEnv: 'blue' | 'green'): string {
    const service = {
      apiVersion: 'v1',
      kind: 'Service',
      metadata: {
        name: 'user-service',
        namespace: this.config.serviceNamespace,
      },
      spec: {
        type: 'ClusterIP',
        ports: [
          {
            port: 80,
            targetPort: 3000,
            protocol: 'TCP',
          },
        ],
        selector: {
          app: 'user-service',
          environment: targetEnv,
        },
      },
    };

    return yaml.dump(service);
  }
}

Monitoring and Observability​

Deployment Metrics​

// src/deployment/deployment-metrics.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { Gauge, Counter, Histogram, register } from 'prom-client';

@Injectable()
export class DeploymentMetricsService {
  private readonly logger = new Logger(DeploymentMetricsService.name);

  private readonly deploymentCounter = new Counter({
    name: 'deployments_total',
    help: 'Total number of deployments',
    labelNames: ['environment', 'service', 'status'],
    registers: [register],
  });

  private readonly deploymentDuration = new Histogram({
    name: 'deployment_duration_seconds',
    help: 'Duration of deployments',
    labelNames: ['environment', 'service'],
    buckets: [30, 60, 120, 300, 600, 1200],
    registers: [register],
  });

  private readonly activeDeployments = new Gauge({
    name: 'active_deployments',
    help: 'Number of active deployments',
    labelNames: ['environment'],
    registers: [register],
  });

  recordDeploymentStart(environment: string, service: string): void {
    this.activeDeployments.inc({ environment });
    this.logger.log(`Deployment started: ${service} to ${environment}`);
  }

  recordDeploymentEnd(
    environment: string,
    service: string,
    status: 'success' | 'failure',
    duration: number,
  ): void {
    this.deploymentCounter.inc({ environment, service, status });
    this.deploymentDuration.observe({ environment, service }, duration);
    this.activeDeployments.dec({ environment });
    
    this.logger.log(
      `Deployment ${status}: ${service} to ${environment} (${duration}s)`
    );
  }
}

Deployment Guidelines​

  2. Progressive Delivery: Implement canary deployments for gradual rollouts
  3. Zero-Downtime: Use rolling updates and blue-green deployments
