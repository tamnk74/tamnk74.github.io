Configuration Management Guide

This guide provides comprehensive configuration management strategies for NestJS microservices and Next.js microfrontend applications, covering environment-specific configurations, secrets management, feature flags, and dynamic configuration updates.

Overview​

Configuration management ensures consistent, secure, and maintainable application settings across different environments. This guide covers centralized configuration, environment isolation, secrets handling, and configuration validation for microservices and microfrontend architectures.

Configuration Service with Hot Reload​

// src/config/dynamic-config.service.ts
import { Injectable, Logger, OnModuleInit } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { EventEmitter2 } from '@nestjs/event-emitter';
import { Interval } from '@nestjs/schedule';
import axios from 'axios';
import * as fs from 'fs/promises';
import * as path from 'path';

export interface ConfigChangeEvent {
  key: string;
  oldValue: any;
  newValue: any;
  timestamp: Date;
}

@Injectable()
export class DynamicConfigService implements OnModuleInit {
  private readonly logger = new Logger(DynamicConfigService.name);
  private configCache = new Map<string, any>();
  private lastUpdateTime = new Date();

  constructor(
    private readonly configService: ConfigService,
    private readonly eventEmitter: EventEmitter2,
  ) {}

  async onModuleInit(): Promise<void> {
    await this.loadInitialConfig();
    this.startConfigRefresh();
  }

  private async loadInitialConfig(): Promise<void> {
    try {
      // Load from environment variables
      this.loadEnvironmentConfig();

      // Load from remote config service if available
      const remoteConfigUrl = this.configService.get('REMOTE_CONFIG_URL');
      if (remoteConfigUrl) {
        await this.loadRemoteConfig(remoteConfigUrl);
      }

      // Load from local config files
      await this.loadFileConfig();

      this.logger.log('Initial configuration loaded successfully');
    } catch (error) {
      this.logger.error('Failed to load initial configuration', error);
    }
  }

  private loadEnvironmentConfig(): void {
    const envVars = process.env;
    for (const [key, value] of Object.entries(envVars)) {
      if (key.startsWith('APP_CONFIG_')) {
        const configKey = key.replace('APP_CONFIG_', '').toLowerCase();
        this.setConfig(configKey, this.parseConfigValue(value));
      }
    }
  }

  private async loadRemoteConfig(url: string): Promise<void> {
    try {
      const response = await axios.get(url, {
        timeout: 5000,
        headers: {
          'Authorization': `Bearer ${this.configService.get('REMOTE_CONFIG_TOKEN')}`,
        },
      });

      if (response.data && typeof response.data === 'object') {
        for (const [key, value] of Object.entries(response.data)) {
          this.setConfig(key, value);
        }
        this.logger.log('Remote configuration loaded successfully');
      }
    } catch (error) {
      this.logger.warn('Failed to load remote configuration', error.message);
    }
  }

  private async loadFileConfig(): Promise<void> {
    try {
      const configDir = path.join(process.cwd(), 'config');
      const environment = this.configService.get('NODE_ENV', 'development');
      const configFile = path.join(configDir, `${environment}.json`);

      try {
        const configContent = await fs.readFile(configFile, 'utf-8');
        const config = JSON.parse(configContent);
        
        for (const [key, value] of Object.entries(config)) {
          this.setConfig(key, value);
        }
        
        this.logger.log(`File configuration loaded from ${configFile}`);
      } catch (fileError) {
        this.logger.debug(`No configuration file found at ${configFile}`);
      }
    } catch (error) {
      this.logger.warn('Failed to load file configuration', error);
    }
  }

  @Interval(60000) // Refresh every minute
  private async startConfigRefresh(): Promise<void> {
    try {
      const remoteConfigUrl = this.configService.get('REMOTE_CONFIG_URL');
      if (remoteConfigUrl) {
        await this.refreshRemoteConfig(remoteConfigUrl);
      }
    } catch (error) {
      this.logger.error('Failed to refresh configuration', error);
    }
  }

  private async refreshRemoteConfig(url: string): Promise<void> {
    try {
      const response = await axios.get(url, {
        timeout: 5000,
        headers: {
          'Authorization': `Bearer ${this.configService.get('REMOTE_CONFIG_TOKEN')}`,
          'If-Modified-Since': this.lastUpdateTime.toISOString(),
        },
      });

      if (response.status === 200 && response.data) {
        const changes: ConfigChangeEvent[] = [];
        
        for (const [key, newValue] of Object.entries(response.data)) {
          const oldValue = this.configCache.get(key);
          if (JSON.stringify(oldValue) !== JSON.stringify(newValue)) {
            changes.push({
              key,
              oldValue,
              newValue,
              timestamp: new Date(),
            });
            this.setConfig(key, newValue);
          }
        }

        if (changes.length > 0) {
          this.logger.log(`Configuration updated: ${changes.length} changes`);
          for (const change of changes) {
            this.eventEmitter.emit('config.changed', change);
          }
        }

        this.lastUpdateTime = new Date();
      }
    } catch (error) {
      if (error.response?.status !== 304) { // 304 = Not Modified
        this.logger.warn('Failed to refresh remote configuration', error.message);
      }
    }
  }

  private setConfig(key: string, value: any): void {
    this.configCache.set(key, value);
  }

  get<T = any>(key: string, defaultValue?: T): T {
    return this.configCache.get(key) ?? defaultValue;
  }

  has(key: string): boolean {
    return this.configCache.has(key);
  }

  getAll(): Record<string, any> {
    return Object.fromEntries(this.configCache.entries());
  }

  private parseConfigValue(value: string | undefined): any {
    if (!value) return undefined;

    // Try to parse as JSON
    try {
      return JSON.parse(value);
    } catch {
      // Return as string if not valid JSON
      return value;
    }
  }

  async reloadConfig(): Promise<void> {
    this.logger.log('Manually reloading configuration...');
    await this.loadInitialConfig();
  }

  // Configuration validation
  validateConfig<T>(schema: any, config: T): { isValid: boolean; errors: string[] } {
    const { error } = schema.validate(config, { abortEarly: false });
    
    if (error) {
      return {
        isValid: false,
        errors: error.details.map((detail: any) => detail.message),
      };
    }

    return { isValid: true, errors: [] };
  }
}

// lib/config/environment.ts
import { z } from 'zod';

// Client-side configuration schema (publicly available)
const clientConfigSchema = z.object({
  APP_NAME: z.string().default('Microservice App'),
  APP_VERSION: z.string().default('1.0.0'),
  API_BASE_URL: z.string().url(),
  ENVIRONMENT: z.enum(['development', 'staging', 'production']).default('development'),
  FEATURE_FLAGS: z.object({
    enableNewUI: z.boolean().default(false),
    enableAnalytics: z.boolean().default(true),
    enableExperimentalFeatures: z.boolean().default(false),
  }),
  SENTRY_DSN: z.string().optional(),
  GOOGLE_ANALYTICS_ID: z.string().optional(),
});

// Server-side configuration schema (includes secrets)
const serverConfigSchema = clientConfigSchema.extend({
  DATABASE_URL: z.string(),
  REDIS_URL: z.string(),
  JWT_SECRET: z.string(),
  API_SECRET_KEY: z.string(),
  NEXTAUTH_SECRET: z.string(),
  NEXTAUTH_URL: z.string().url(),
});

export type ClientConfig = z.infer<typeof clientConfigSchema>;
export type ServerConfig = z.infer<typeof serverConfigSchema>;

class ConfigManager {
  private clientConfig: ClientConfig | null = null;
  private serverConfig: ServerConfig | null = null;

  getClientConfig(): ClientConfig {
    if (!this.clientConfig) {
      this.clientConfig = this.loadClientConfig();
    }
    return this.clientConfig;
  }

  getServerConfig(): ServerConfig {
    if (typeof window !== 'undefined') {
      throw new Error('Server configuration cannot be accessed on the client side');
    }

    if (!this.serverConfig) {
      this.serverConfig = this.loadServerConfig();
    }
    return this.serverConfig;
  }

  private loadClientConfig(): ClientConfig {
    const config = {
      APP_NAME: process.env.NEXT_PUBLIC_APP_NAME,
      APP_VERSION: process.env.NEXT_PUBLIC_APP_VERSION,
      API_BASE_URL: process.env.NEXT_PUBLIC_API_BASE_URL!,
      ENVIRONMENT: process.env.NEXT_PUBLIC_ENVIRONMENT,
      FEATURE_FLAGS: {
        enableNewUI: process.env.NEXT_PUBLIC_FEATURE_NEW_UI === 'true',
        enableAnalytics: process.env.NEXT_PUBLIC_FEATURE_ANALYTICS !== 'false',
        enableExperimentalFeatures: process.env.NEXT_PUBLIC_FEATURE_EXPERIMENTAL === 'true',
      },
      SENTRY_DSN: process.env.NEXT_PUBLIC_SENTRY_DSN,
      GOOGLE_ANALYTICS_ID: process.env.NEXT_PUBLIC_GA_ID,
    };

    const result = clientConfigSchema.safeParse(config);
    
    if (!result.success) {
      console.error('Client configuration validation failed:', result.error.format());
      throw new Error('Invalid client configuration');
    }

    return result.data;
  }

  private loadServerConfig(): ServerConfig {
    const config = {
      ...this.getClientConfig(),
      DATABASE_URL: process.env.DATABASE_URL!,
      REDIS_URL: process.env.REDIS_URL!,
      JWT_SECRET: process.env.JWT_SECRET!,
      API_SECRET_KEY: process.env.API_SECRET_KEY!,
      NEXTAUTH_SECRET: process.env.NEXTAUTH_SECRET!,
      NEXTAUTH_URL: process.env.NEXTAUTH_URL!,
    };

    const result = serverConfigSchema.safeParse(config);
    
    if (!result.success) {
      console.error('Server configuration validation failed:', result.error.format());
      throw new Error('Invalid server configuration');
    }

    return result.data;
  }

  // Reload configuration (useful for hot reloading in development)
  reload(): void {
    this.clientConfig = null;
    this.serverConfig = null;
  }

  // Get specific configuration value with type safety
  get<K extends keyof ClientConfig>(key: K): ClientConfig[K];
  get<K extends keyof ServerConfig>(key: K, serverOnly: true): ServerConfig[K];
  get<K extends keyof ClientConfig | keyof ServerConfig>(
    key: K,
    serverOnly = false
  ): any {
    if (serverOnly) {
      return this.getServerConfig()[key as keyof ServerConfig];
    }
    return this.getClientConfig()[key as keyof ClientConfig];
  }
}

export const configManager = new ConfigManager();

// Convenience functions
export const getClientConfig = () => configManager.getClientConfig();
export const getServerConfig = () => configManager.getServerConfig();

// Type-safe configuration getters
export const getConfig = <K extends keyof ClientConfig>(key: K) => 
  configManager.get(key);

export const getServerOnlyConfig = <K extends keyof ServerConfig>(key: K) =>
  configManager.get(key, true);

Runtime Configuration Provider​

// components/providers/config-provider.tsx
import React, { createContext, useContext, useEffect, useState } from 'react';
import { ClientConfig, getClientConfig } from '../../lib/config/environment';

interface ConfigContextType {
  config: ClientConfig;
  isLoading: boolean;
  error: Error | null;
  refresh: () => Promise<void>;
}

const ConfigContext = createContext<ConfigContextType | undefined>(undefined);

export const useConfig = () => {
  const context = useContext(ConfigContext);
  if (!context) {
    throw new Error('useConfig must be used within ConfigProvider');
  }
  return context;
};

interface ConfigProviderProps {
  children: React.ReactNode;
  initialConfig?: ClientConfig;
}

export const ConfigProvider: React.FC<ConfigProviderProps> = ({ 
  children, 
  initialConfig 
}) => {
  const [config, setConfig] = useState<ClientConfig>(() => 
    initialConfig || getClientConfig()
  );
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const refresh = async () => {
    setIsLoading(true);
    setError(null);

    try {
      // Fetch runtime configuration from API
      const response = await fetch('/api/config');
      if (!response.ok) {
        throw new Error(`Failed to fetch config: ${response.statusText}`);
      }

      const runtimeConfig = await response.json();
      
      // Merge with build-time configuration
      const mergedConfig = {
        ...config,
        ...runtimeConfig,
        FEATURE_FLAGS: {
          ...config.FEATURE_FLAGS,
          ...runtimeConfig.FEATURE_FLAGS,
        },
      };

      setConfig(mergedConfig);
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Unknown error'));
      console.error('Failed to refresh configuration:', err);
    } finally {
      setIsLoading(false);
    }
  };

  useEffect(() => {
    // Refresh configuration periodically
    const interval = setInterval(() => {
      refresh();
    }, 5 * 60 * 1000); // Every 5 minutes

    return () => clearInterval(interval);
  }, []);

  const value: ConfigContextType = {
    config,
    isLoading,
    error,
    refresh,
  };

  return (
    <ConfigContext.Provider value={value}>
      {children}
    </ConfigContext.Provider>
  );
};

Feature Flag Hook​

// hooks/use-feature-flag.ts
import { useConfig } from '../components/providers/config-provider';
import { useState, useEffect } from 'react';

export interface FeatureFlagOptions {
  userId?: string;
  userRole?: string;
  metadata?: Record<string, any>;
}

export const useFeatureFlag = (
  flagName: keyof ClientConfig['FEATURE_FLAGS'],
  options: FeatureFlagOptions = {}
) => {
  const { config } = useConfig();
  const [isEnabled, setIsEnabled] = useState(false);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const evaluateFlag = async () => {
      setIsLoading(true);

      try {
        // Get base flag value from configuration
        const baseValue = config.FEATURE_FLAGS[flagName];

        // For simple flags, just return the configured value
        if (typeof baseValue === 'boolean') {
          setIsEnabled(baseValue);
          return;
        }

        // For advanced feature flags, call evaluation service
        const response = await fetch('/api/feature-flags/evaluate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            flagName,
            context: {
              userId: options.userId,
              userRole: options.userRole,
              ...options.metadata,
            },
          }),
        });

        if (response.ok) {
          const result = await response.json();
          setIsEnabled(result.enabled);
        } else {
          // Fallback to configured value if service is unavailable
          setIsEnabled(Boolean(baseValue));
        }
      } catch (error) {
        console.error(`Failed to evaluate feature flag ${flagName}:`, error);
        // Fallback to configured value
        setIsEnabled(Boolean(config.FEATURE_FLAGS[flagName]));
      } finally {
        setIsLoading(false);
      }
    };

    evaluateFlag();
  }, [flagName, config.FEATURE_FLAGS, options.userId, options.userRole]);

  return { isEnabled, isLoading };
};

// HOC for feature flag-based rendering
export const withFeatureFlag = <P extends object>(
  Component: React.ComponentType<P>,
  flagName: keyof ClientConfig['FEATURE_FLAGS'],
  fallback?: React.ComponentType<P>
) => {
  const WrappedComponent: React.FC<P & FeatureFlagOptions> = (props) => {
    const { isEnabled, isLoading } = useFeatureFlag(flagName, props);

    if (isLoading) {
      return <div>Loading...</div>;
    }

    if (!isEnabled) {
      return fallback ? <fallback {...props} /> : null;
    }

    return <Component {...props} />;
  };

  WrappedComponent.displayName = `withFeatureFlag(${Component.displayName || Component.name})`;

  return WrappedComponent;
};

// src/config/vault.service.ts
import { Injectable, Logger, OnModuleInit } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import axios, { AxiosInstance } from 'axios';

export interface VaultSecret {
  data: Record<string, any>;
  metadata: {
    created_time: string;
    version: number;
  };
}

@Injectable()
export class VaultService implements OnModuleInit {
  private readonly logger = new Logger(VaultService.name);
  private client: AxiosInstance;
  private token: string | null = null;
  private secrets = new Map<string, any>();

  constructor(private readonly configService: ConfigService) {
    const vaultUrl = this.configService.get('VAULT_URL');
    
    this.client = axios.create({
      baseURL: vaultUrl,
      timeout: 10000,
    });
  }

  async onModuleInit(): Promise<void> {
    try {
      await this.authenticate();
      await this.loadSecrets();
    } catch (error) {
      this.logger.error('Failed to initialize Vault service', error);
    }
  }

  private async authenticate(): Promise<void> {
    const authMethod = this.configService.get('VAULT_AUTH_METHOD', 'token');

    switch (authMethod) {
      case 'token':
        await this.authenticateWithToken();
        break;
      case 'kubernetes':
        await this.authenticateWithKubernetes();
        break;
      case 'app-role':
        await this.authenticateWithAppRole();
        break;
      default:
        throw new Error(`Unsupported Vault auth method: ${authMethod}`);
    }
  }

  private async authenticateWithToken(): Promise<void> {
    this.token = this.configService.get('VAULT_TOKEN');
    if (!this.token) {
      throw new Error('Vault token not provided');
    }

    this.client.defaults.headers.common['X-Vault-Token'] = this.token;
    
    // Verify token
    const response = await this.client.get('/v1/auth/token/lookup-self');
    this.logger.log('Successfully authenticated with Vault using token');
  }

  private async authenticateWithKubernetes(): Promise<void> {
    const jwt = await this.getServiceAccountToken();
    const role = this.configService.get('VAULT_K8S_ROLE');

    const response = await this.client.post('/v1/auth/kubernetes/login', {
      role,
      jwt,
    });

    this.token = response.data.auth.client_token;
    this.client.defaults.headers.common['X-Vault-Token'] = this.token;
    
    this.logger.log('Successfully authenticated with Vault using Kubernetes auth');
  }

  private async authenticateWithAppRole(): Promise<void> {
    const roleId = this.configService.get('VAULT_ROLE_ID');
    const secretId = this.configService.get('VAULT_SECRET_ID');

    const response = await this.client.post('/v1/auth/approle/login', {
      role_id: roleId,
      secret_id: secretId,
    });

    this.token = response.data.auth.client_token;
    this.client.defaults.headers.common['X-Vault-Token'] = this.token;
    
    this.logger.log('Successfully authenticated with Vault using AppRole');
  }

  private async getServiceAccountToken(): Promise<string> {
    const fs = await import('fs/promises');
    const tokenPath = '/var/run/secrets/kubernetes.io/serviceaccount/token';
    return fs.readFile(tokenPath, 'utf8');
  }

  private async loadSecrets(): Promise<void> {
    const secretPaths = this.configService.get<string[]>('VAULT_SECRET_PATHS', []);
    
    for (const path of secretPaths) {
      try {
        const secret = await this.getSecret(path);
        this.secrets.set(path, secret);
        this.logger.log(`Loaded secret from ${path}`);
      } catch (error) {
        this.logger.error(`Failed to load secret from ${path}`, error);
      }
    }
  }

  async getSecret(path: string): Promise<VaultSecret> {
    try {
      const response = await this.client.get(`/v1/secret/data/${path}`);
      return response.data.data;
    } catch (error) {
      this.logger.error(`Failed to get secret from ${path}`, error);
      throw error;
    }
  }

  async setSecret(path: string, data: Record<string, any>): Promise<void> {
    try {
      await this.client.post(`/v1/secret/data/${path}`, { data });
      this.secrets.set(path, { data });
      this.logger.log(`Set secret at ${path}`);
    } catch (error) {
      this.logger.error(`Failed to set secret at ${path}`, error);
      throw error;
    }
  }

  getCachedSecret(path: string): any {
    return this.secrets.get(path)?.data;
  }

  async refreshSecrets(): Promise<void> {
    this.logger.log('Refreshing all secrets...');
    await this.loadSecrets();
  }

  // Periodic secret refresh
  async startSecretRefresh(intervalMs = 300000): Promise<void> { // 5 minutes
    setInterval(async () => {
      try {
        await this.refreshSecrets();
      } catch (error) {
        this.logger.error('Failed to refresh secrets', error);
      }
    }, intervalMs);
  }
}

// src/config/config.spec.ts
import { Test } from '@nestjs/testing';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { AppConfig } from './app.config';
import { DatabaseConfig } from './database.config';

describe('Configuration', () => {
  let configService: ConfigService;

  beforeEach(async () => {
    const module = await Test.createTestingModule({
      imports: [
        ConfigModule.forRoot({
          load: [AppConfig, DatabaseConfig],
          envFilePath: '.env.test',
        }),
      ],
    }).compile();

    configService = module.get<ConfigService>(ConfigService);
  });

  describe('App Configuration', () => {
    it('should load app configuration', () => {
      const appConfig = configService.get('app');
      expect(appConfig).toBeDefined();
      expect(appConfig.nodeEnv).toBe('test');
    });

    it('should have required configuration values', () => {
      expect(configService.get('app.port')).toBeDefined();
      expect(configService.get('app.appName')).toBeDefined();
    });

    it('should validate CORS configuration', () => {
      const corsConfig = configService.get('app.cors');
      expect(corsConfig.origins).toBeInstanceOf(Array);
      expect(corsConfig.credentials).toBeDefined();
    });
  });

  describe('Database Configuration', () => {
    it('should load database configuration', () => {
      const dbConfig = configService.get('database');
      expect(dbConfig).toBeDefined();
      expect(dbConfig.host).toBeDefined();
      expect(dbConfig.port).toBeGreaterThan(0);
    });

    it('should have proper pool configuration', () => {
      const poolConfig = configService.get('database.pool');
      expect(poolConfig.min).toBeGreaterThanOrEqual(1);
      expect(poolConfig.max).toBeGreaterThan(poolConfig.min);
    });
  });

  describe('Environment-specific settings', () => {
    it('should disable synchronize in production', () => {
      process.env.NODE_ENV = 'production';
      const dbConfig = configService.get('database');
      expect(dbConfig.synchronize).toBe(false);
    });

    it('should enable debug mode in development', () => {
      process.env.NODE_ENV = 'development';
      const appConfig = configService.get('app');
      expect(appConfig.debug).toBe(true);
    });
  });
});
