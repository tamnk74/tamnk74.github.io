Database Migration Management Guide

This guide provides comprehensive database migration strategies for NestJS microservices applications, covering schema versioning, zero-downtime migrations, rollback procedures, and data migration best practices for PostgreSQL and other databases.

Overview​

Database migration management ensures consistent, reliable, and safe evolution of database schemas across environments. This guide covers migration strategies, version control, automated migration pipelines, and data integrity practices for microservices architectures.

Migration Service​

// src/migration/migration.service.ts
import { Injectable, Logger, OnModuleInit } from '@nestjs/common';
import { DataSource } from 'typeorm';
import { InjectDataSource } from '@nestjs/typeorm';
import { ConfigService } from '@nestjs/config';
import * as fs from 'fs/promises';
import * as path from 'path';

export interface MigrationInfo {
  id: number;
  timestamp: number;
  name: string;
  executedAt: Date;
  executionTime: number;
}

export interface MigrationPlan {
  pending: MigrationInfo[];
  executed: MigrationInfo[];
  canRollback: boolean;
  rollbackTarget?: MigrationInfo;
}

@Injectable()
export class MigrationService implements OnModuleInit {
  private readonly logger = new Logger(MigrationService.name);

  constructor(
    @InjectDataSource()
    private readonly dataSource: DataSource,
    private readonly configService: ConfigService,
  ) {}

  async onModuleInit(): Promise<void> {
    if (this.configService.get('AUTO_RUN_MIGRATIONS') === 'true') {
      await this.runPendingMigrations();
    }
  }

  async getMigrationPlan(): Promise<MigrationPlan> {
    const queryRunner = this.dataSource.createQueryRunner();
    
    try {
      // Get executed migrations
      const executedMigrations = await queryRunner.query(`
        SELECT id, timestamp, name, "executedAt", "executionTime"
        FROM migrations_history 
        ORDER BY timestamp ASC
      `);

      // Get all available migrations
      const allMigrations = this.dataSource.migrations.map(migration => ({
        id: 0, // Will be set when executed
        timestamp: parseInt(migration.name.split('-')[0]),
        name: migration.name,
        executedAt: null,
        executionTime: 0,
      }));

      // Determine pending migrations
      const executedTimestamps = new Set(executedMigrations.map(m => m.timestamp));
      const pendingMigrations = allMigrations.filter(
        m => !executedTimestamps.has(m.timestamp)
      );

      return {
        pending: pendingMigrations,
        executed: executedMigrations,
        canRollback: executedMigrations.length > 0,
        rollbackTarget: executedMigrations[executedMigrations.length - 1],
      };
    } finally {
      await queryRunner.release();
    }
  }

  async runPendingMigrations(dryRun = false): Promise<{
    success: boolean;
    migrationsRun: string[];
    errors: string[];
    executionTime: number;
  }> {
    const startTime = Date.now();
    const migrationsRun: string[] = [];
    const errors: string[] = [];

    try {
      const plan = await this.getMigrationPlan();
      
      if (plan.pending.length === 0) {
        this.logger.log('No pending migrations to run');
        return {
          success: true,
          migrationsRun: [],
          errors: [],
          executionTime: Date.now() - startTime,
        };
      }

      this.logger.log(`Found ${plan.pending.length} pending migrations`);

      if (dryRun) {
        this.logger.log('DRY RUN: Would execute the following migrations:');
        plan.pending.forEach(migration => {
          this.logger.log(`- ${migration.name}`);
        });
        return {
          success: true,
          migrationsRun: plan.pending.map(m => m.name),
          errors: [],
          executionTime: Date.now() - startTime,
        };
      }

      // Create backup before migration
      await this.createBackup(`pre-migration-${Date.now()}`);

      // Run migrations
      await this.dataSource.runMigrations({
        transaction: 'each', // Run each migration in its own transaction
      });

      // Validate migrations
      const postPlan = await this.getMigrationPlan();
      if (postPlan.pending.length > 0) {
        throw new Error(`${postPlan.pending.length} migrations still pending after execution`);
      }

      plan.pending.forEach(migration => {
        migrationsRun.push(migration.name);
        this.logger.log(`Successfully executed migration: ${migration.name}`);
      });

      return {
        success: true,
        migrationsRun,
        errors,
        executionTime: Date.now() - startTime,
      };
    } catch (error) {
      this.logger.error('Migration failed', error);
      errors.push(error.message);

      // Attempt to rollback on failure
      try {
        await this.rollbackLastMigration();
        this.logger.log('Rollback completed successfully');
      } catch (rollbackError) {
        this.logger.error('Rollback failed', rollbackError);
        errors.push(`Rollback failed: ${rollbackError.message}`);
      }

      return {
        success: false,
        migrationsRun,
        errors,
        executionTime: Date.now() - startTime,
      };
    }
  }

  async rollbackLastMigration(): Promise<void> {
    try {
      await this.dataSource.undoLastMigration({
        transaction: 'each',
      });
      this.logger.log('Last migration rolled back successfully');
    } catch (error) {
      this.logger.error('Failed to rollback last migration', error);
      throw error;
    }
  }

  async rollbackToMigration(targetMigrationName: string): Promise<void> {
    const plan = await this.getMigrationPlan();
    const targetMigration = plan.executed.find(m => m.name === targetMigrationName);
    
    if (!targetMigration) {
      throw new Error(`Migration ${targetMigrationName} not found in executed migrations`);
    }

    // Create backup before rollback
    await this.createBackup(`pre-rollback-${Date.now()}`);

    // Rollback migrations one by one until target
    const migrationsToRollback = plan.executed
      .filter(m => m.timestamp > targetMigration.timestamp)
      .reverse(); // Rollback in reverse order

    for (const migration of migrationsToRollback) {
      try {
        await this.dataSource.undoLastMigration({
          transaction: 'each',
        });
        this.logger.log(`Rolled back migration: ${migration.name}`);
      } catch (error) {
        this.logger.error(`Failed to rollback migration: ${migration.name}`, error);
        throw error;
      }
    }
  }

  async createBackup(backupName: string): Promise<void> {
    const dbConfig = {
      host: this.configService.get('DATABASE_HOST'),
      port: this.configService.get('DATABASE_PORT'),
      database: this.configService.get('DATABASE_NAME'),
      username: this.configService.get('DATABASE_USERNAME'),
      password: this.configService.get('DATABASE_PASSWORD'),
    };

    const backupPath = path.join(process.cwd(), 'backups', `${backupName}.sql`);
    
    // Ensure backup directory exists
    await fs.mkdir(path.dirname(backupPath), { recursive: true });

    // Create pg_dump command
    const pgDumpCommand = [
      'pg_dump',
      `--host=${dbConfig.host}`,
      `--port=${dbConfig.port}`,
      `--username=${dbConfig.username}`,
      '--no-password',
      '--verbose',
      '--clean',
      '--if-exists',
      '--create',
      '--format=custom',
      `--file=${backupPath}`,
      dbConfig.database,
    ].join(' ');

    // Set PGPASSWORD environment variable
    process.env.PGPASSWORD = dbConfig.password;

    try {
      const { exec } = await import('child_process');
      const { promisify } = await import('util');
      const execAsync = promisify(exec);

      await execAsync(pgDumpCommand);
      this.logger.log(`Database backup created: ${backupPath}`);
    } catch (error) {
      this.logger.error('Failed to create database backup', error);
      throw error;
    } finally {
      delete process.env.PGPASSWORD;
    }
  }

  async validateMigration(migrationName: string): Promise<{
    isValid: boolean;
    issues: string[];
  }> {
    const issues: string[] = [];

    try {
      // Read migration file
      const migrationPath = path.join(process.cwd(), 'src/migrations', `${migrationName}.ts`);
      const migrationContent = await fs.readFile(migrationPath, 'utf-8');

      // Check for common migration issues
      const commonIssues = [
        {
          pattern: /DROP TABLE/gi,
          message: 'DROP TABLE operations can cause data loss',
        },
        {
          pattern: /ALTER TABLE.*DROP COLUMN/gi,
          message: 'DROP COLUMN operations can cause data loss',
        },
        {
          pattern: /ALTER TABLE.*RENAME COLUMN/gi,
          message: 'RENAME COLUMN may require application changes',
        },
        {
          pattern: /CREATE UNIQUE INDEX/gi,
          message: 'Creating unique indexes on existing data may fail',
        },
        {
          pattern: /ALTER TABLE.*ADD CONSTRAINT.*NOT NULL/gi,
          message: 'Adding NOT NULL constraints to existing columns may fail',
        },
      ];

      commonIssues.forEach(({ pattern, message }) => {
        if (pattern.test(migrationContent)) {
          issues.push(message);
        }
      });

      // Check for transaction handling
      if (!migrationContent.includes('transaction')) {
        issues.push('Migration should be wrapped in a transaction');
      }

      // Check for rollback method
      if (!migrationContent.includes('down(')) {
        issues.push('Migration should include a down() method for rollback');
      }

      return {
        isValid: issues.length === 0,
        issues,
      };
    } catch (error) {
      return {
        isValid: false,
        issues: [`Failed to validate migration: ${error.message}`],
      };
    }
  }

  async generateMigrationReport(): Promise<{
    totalMigrations: number;
    executedMigrations: number;
    pendingMigrations: number;
    lastMigration: MigrationInfo | null;
    migrationHistory: MigrationInfo[];
    averageExecutionTime: number;
  }> {
    const plan = await this.getMigrationPlan();
    
    const averageExecutionTime = plan.executed.length > 0
      ? plan.executed.reduce((sum, m) => sum + m.executionTime, 0) / plan.executed.length
      : 0;

    return {
      totalMigrations: plan.executed.length + plan.pending.length,
      executedMigrations: plan.executed.length,
      pendingMigrations: plan.pending.length,
      lastMigration: plan.executed[plan.executed.length - 1] || null,
      migrationHistory: plan.executed,
      averageExecutionTime,
    };
  }
}

import { MigrationInterface, QueryRunner, Table, Index, Column } from 'typeorm';

// Example: Safe column addition with default value
export class AddUserStatusColumn1640995200000 implements MigrationInterface {
  name = 'AddUserStatusColumn1640995200000';

  public async up(queryRunner: QueryRunner): Promise<void> {
    // Step 1: Add column with default value
    await queryRunner.addColumn('users', new Column({
      name: 'status',
      type: 'varchar',
      length: '20',
      default: "'active'",
      isNullable: false,
    }));

    // Step 2: Update existing records (if needed)
    await queryRunner.query(`
      UPDATE users 
      SET status = 'active' 
      WHERE status IS NULL OR status = ''
    `);

    // Step 3: Add index for performance
    await queryRunner.createIndex('users', new Index('IDX_users_status', ['status']));
  }

  public async down(queryRunner: QueryRunner): Promise<void> {
    await queryRunner.dropIndex('users', 'IDX_users_status');
    await queryRunner.dropColumn('users', 'status');
  }
}

// Example: Zero-downtime table restructuring
export class RestructureProductsTable1641081600000 implements MigrationInterface {
  name = 'RestructureProductsTable1641081600000';

  public async up(queryRunner: QueryRunner): Promise<void> {
    // Create new table structure
    await queryRunner.createTable(new Table({
      name: 'products_new',
      columns: [
        {
          name: 'id',
          type: 'uuid',
          isPrimary: true,
          generationStrategy: 'uuid',
          default: 'gen_random_uuid()',
        },
        {
          name: 'name',
          type: 'varchar',
          length: '255',
          isNullable: false,
        },
        {
          name: 'description',
          type: 'text',
          isNullable: true,
        },
        {
          name: 'price',
          type: 'decimal',
          precision: 10,
          scale: 2,
          isNullable: false,
        },
        {
          name: 'category_id',
          type: 'uuid',
          isNullable: false,
        },
        {
          name: 'created_at',
          type: 'timestamp',
          default: 'CURRENT_TIMESTAMP',
        },
        {
          name: 'updated_at',
          type: 'timestamp',
          default: 'CURRENT_TIMESTAMP',
        },
      ],
    }));

    // Copy data from old table to new table
    await queryRunner.query(`
      INSERT INTO products_new (id, name, description, price, category_id, created_at, updated_at)
      SELECT 
        id, 
        name, 
        description, 
        price::decimal(10,2),
        category_id,
        created_at,
        updated_at
      FROM products
    `);

    // Create indexes on new table
    await queryRunner.createIndex('products_new', new Index('IDX_products_new_category', ['category_id']));
    await queryRunner.createIndex('products_new', new Index('IDX_products_new_name', ['name']));

    // Atomic table swap (requires coordination with application deployment)
    await queryRunner.query('BEGIN');
    try {
      await queryRunner.renameTable('products', 'products_old');
      await queryRunner.renameTable('products_new', 'products');
      await queryRunner.query('COMMIT');
    } catch (error) {
      await queryRunner.query('ROLLBACK');
      throw error;
    }
  }

  public async down(queryRunner: QueryRunner): Promise<void> {
    // Rollback table swap
    await queryRunner.query('BEGIN');
    try {
      await queryRunner.renameTable('products', 'products_new');
      await queryRunner.renameTable('products_old', 'products');
      await queryRunner.query('COMMIT');
    } catch (error) {
      await queryRunner.query('ROLLBACK');
      throw error;
    }

    // Clean up
    await queryRunner.dropTable('products_new');
  }
}

// Example: Data migration with batch processing
export class MigrateUserPreferences1641168000000 implements MigrationInterface {
  name = 'MigrateUserPreferences1641168000000';

  public async up(queryRunner: QueryRunner): Promise<void> {
    // Create new preferences table
    await queryRunner.createTable(new Table({
      name: 'user_preferences',
      columns: [
        {
          name: 'id',
          type: 'uuid',
          isPrimary: true,
          generationStrategy: 'uuid',
          default: 'gen_random_uuid()',
        },
        {
          name: 'user_id',
          type: 'uuid',
          isNullable: false,
        },
        {
          name: 'preference_key',
          type: 'varchar',
          length: '100',
          isNullable: false,
        },
        {
          name: 'preference_value',
          type: 'jsonb',
          isNullable: true,
        },
        {
          name: 'created_at',
          type: 'timestamp',
          default: 'CURRENT_TIMESTAMP',
        },
      ],
    }));

    // Migrate data in batches to avoid memory issues
    const batchSize = 1000;
    let offset = 0;
    let hasMoreData = true;

    while (hasMoreData) {
      const users = await queryRunner.query(`
        SELECT id, settings 
        FROM users 
        WHERE settings IS NOT NULL 
        ORDER BY id
        LIMIT $1 OFFSET $2
      `, [batchSize, offset]);

      if (users.length === 0) {
        hasMoreData = false;
        break;
      }

      // Transform and insert preferences
      for (const user of users) {
        const settings = JSON.parse(user.settings || '{}');
        
        for (const [key, value] of Object.entries(settings)) {
          await queryRunner.query(`
            INSERT INTO user_preferences (user_id, preference_key, preference_value)
            VALUES ($1, $2, $3)
          `, [user.id, key, JSON.stringify(value)]);
        }
      }

      offset += batchSize;
      
      // Log progress
      console.log(`Migrated preferences for ${offset} users`);
    }

    // Create indexes
    await queryRunner.createIndex('user_preferences', 
      new Index('IDX_user_preferences_user_key', ['user_id', 'preference_key'], { isUnique: true })
    );
  }

  public async down(queryRunner: QueryRunner): Promise<void> {
    await queryRunner.dropTable('user_preferences');
  }
}

Migration CI/CD Script​

#!/bin/bash
# scripts/migrate-database.sh

set -euo pipefail

ENVIRONMENT=${1:-development}
DRY_RUN=${2:-false}
BACKUP_ENABLED=${3:-true}

echo "Starting database migration for environment: $ENVIRONMENT"

# Load environment-specific configuration
case $ENVIRONMENT in
  "development")
    DB_HOST=$DEV_DB_HOST
    DB_NAME=$DEV_DB_NAME
    DB_USER=$DEV_DB_USER
    DB_PASS=$DEV_DB_PASS
    ;;
  "staging")
    DB_HOST=$STAGING_DB_HOST
    DB_NAME=$STAGING_DB_NAME
    DB_USER=$STAGING_DB_USER
    DB_PASS=$STAGING_DB_PASS
    ;;
  "production")
    DB_HOST=$PROD_DB_HOST
    DB_NAME=$PROD_DB_NAME
    DB_USER=$PROD_DB_USER
    DB_PASS=$PROD_DB_PASS
    ;;
  *)
    echo "Unknown environment: $ENVIRONMENT"
    exit 1
    ;;
esac

# Set database connection
export DATABASE_HOST=$DB_HOST
export DATABASE_NAME=$DB_NAME
export DATABASE_USERNAME=$DB_USER
export DATABASE_PASSWORD=$DB_PASS

# Function to run migration with monitoring
run_migration() {
  local start_time=$(date +%s)
  
  echo "Checking migration plan..."
  npm run migration:plan
  
  if [ "$DRY_RUN" = "true" ]; then
    echo "Running dry-run migration..."
    npm run migration:run -- --dry-run
    return 0
  fi
  
  if [ "$BACKUP_ENABLED" = "true" ]; then
    echo "Creating pre-migration backup..."
    npm run migration:backup
  fi
  
  echo "Running database migrations..."
  npm run migration:run
  
  local end_time=$(date +%s)
  local duration=$((end_time - start_time))
  
  echo "Migration completed in ${duration} seconds"
  
  # Validate migration
  echo "Validating migration..."
  npm run migration:validate
  
  # Generate migration report
  npm run migration:report > "migration-report-${ENVIRONMENT}-$(date +%Y%m%d-%H%M%S).json"
}

# Function to handle migration failure
handle_failure() {
  echo "Migration failed. Checking if rollback is needed..."
  
  if [ "$ENVIRONMENT" = "production" ]; then
    echo "Production migration failed. Manual intervention required."
    echo "Please check logs and consider manual rollback."
    exit 1
  else
    echo "Attempting automatic rollback..."
    npm run migration:rollback
    exit 1
  fi
}

# Trap errors and handle failure
trap handle_failure ERR

# Pre-migration checks
echo "Running pre-migration checks..."

# Check database connectivity
npm run migration:check-connection || {
  echo "Database connection failed"
  exit 1
}

# Check migration files
npm run migration:validate-files || {
  echo "Migration file validation failed"
  exit 1
}

# Check for pending migrations
PENDING_MIGRATIONS=$(npm run migration:pending --silent | wc -l)
if [ "$PENDING_MIGRATIONS" -eq 0 ]; then
  echo "No pending migrations found"
  exit 0
fi

echo "Found $PENDING_MIGRATIONS pending migrations"

# Production-specific checks
if [ "$ENVIRONMENT" = "production" ]; then
  echo "Production environment detected. Running additional checks..."
  
  # Check for destructive operations
  npm run migration:check-destructive || {
    echo "Destructive operations detected in migrations"
    echo "Please review migrations manually"
    exit 1
  }
  
  # Require explicit confirmation for production
  if [ -z "${MIGRATION_CONFIRMED:-}" ]; then
    echo "Production migration requires explicit confirmation"
    echo "Set MIGRATION_CONFIRMED=true to proceed"
    exit 1
  fi
fi

# Run migration
run_migration

echo "Migration pipeline completed successfully"

# .github/workflows/migrate.yml
name: Database Migration

on:
  workflow_dispatch:
    inputs:
      environment:
        description: 'Target environment'
        required: true
        default: 'staging'
        type: choice
        options:
        - development
        - staging
        - production
      dry_run:
        description: 'Dry run only'
        required: false
        default: true
        type: boolean

jobs:
  migrate:
    runs-on: ubuntu-latest
    environment: ${{ github.event.inputs.environment }}
    
    steps:
    - name: Checkout
      uses: actions/checkout@v4

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'

    - name: Install dependencies
      run: npm ci

    - name: Setup environment variables
      run: |
        case "${{ github.event.inputs.environment }}" in
          "development")
            echo "DATABASE_HOST=${{ secrets.DEV_DB_HOST }}" >> $GITHUB_ENV
            echo "DATABASE_NAME=${{ secrets.DEV_DB_NAME }}" >> $GITHUB_ENV
            echo "DATABASE_USERNAME=${{ secrets.DEV_DB_USER }}" >> $GITHUB_ENV
            echo "DATABASE_PASSWORD=${{ secrets.DEV_DB_PASS }}" >> $GITHUB_ENV
            ;;
          "staging")
            echo "DATABASE_HOST=${{ secrets.STAGING_DB_HOST }}" >> $GITHUB_ENV
            echo "DATABASE_NAME=${{ secrets.STAGING_DB_NAME }}" >> $GITHUB_ENV
            echo "DATABASE_USERNAME=${{ secrets.STAGING_DB_USER }}" >> $GITHUB_ENV
            echo "DATABASE_PASSWORD=${{ secrets.STAGING_DB_PASS }}" >> $GITHUB_ENV
            ;;
          "production")
            echo "DATABASE_HOST=${{ secrets.PROD_DB_HOST }}" >> $GITHUB_ENV
            echo "DATABASE_NAME=${{ secrets.PROD_DB_NAME }}" >> $GITHUB_ENV
            echo "DATABASE_USERNAME=${{ secrets.PROD_DB_USER }}" >> $GITHUB_ENV
            echo "DATABASE_PASSWORD=${{ secrets.PROD_DB_PASS }}" >> $GITHUB_ENV
            ;;
        esac

    - name: Run migration checks
      run: |
        npm run migration:check-connection
        npm run migration:validate-files

    - name: Run migration plan
      run: npm run migration:plan

    - name: Run migration (dry run)
      if: ${{ github.event.inputs.dry_run == 'true' }}
      run: npm run migration:run -- --dry-run

    - name: Create backup
      if: ${{ github.event.inputs.dry_run == 'false' && github.event.inputs.environment == 'production' }}
      run: npm run migration:backup

    - name: Run migration
      if: ${{ github.event.inputs.dry_run == 'false' }}
      run: npm run migration:run

    - name: Validate migration
      if: ${{ github.event.inputs.dry_run == 'false' }}
      run: npm run migration:validate

    - name: Generate migration report
      run: npm run migration:report > migration-report.json

    - name: Upload migration report
      uses: actions/upload-artifact@v3
      with:
        name: migration-report-${{ github.event.inputs.environment }}
        path: migration-report.json

    - name: Notify on success
      if: success()
      uses: 8398a7/action-slack@v3
      with:
        status: success
        text: |
          ✅ Database migration completed successfully
          Environment: ${{ github.event.inputs.environment }}
          Dry Run: ${{ github.event.inputs.dry_run }}
      env:
        SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK }}

    - name: Notify on failure
      if: failure()
      uses: 8398a7/action-slack@v3
      with:
        status: failure
        text: |
          ❌ Database migration failed
          Environment: ${{ github.event.inputs.environment }}
          Please check logs and consider rollback
      env:
        SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK }}

Migration Metrics Collection​

// src/migration/migration-metrics.service.ts
import { Injectable } from '@nestjs/common';
import { PrometheusService } from '../monitoring/prometheus.service';
import { Counter, Histogram, Gauge } from 'prom-client';

@Injectable()
export class MigrationMetricsService {
  private readonly migrationCounter: Counter<string>;
  private readonly migrationDuration: Histogram<string>;
  private readonly pendingMigrations: Gauge<string>;
  private readonly migrationErrors: Counter<string>;

  constructor(private readonly prometheusService: PrometheusService) {
    this.migrationCounter = new Counter({
      name: 'database_migrations_total',
      help: 'Total number of database migrations executed',
      labelNames: ['status', 'migration_name'],
      registers: [this.prometheusService.getRegistry()],
    });

    this.migrationDuration = new Histogram({
      name: 'database_migration_duration_seconds',
      help: 'Duration of database migration execution',
      labelNames: ['migration_name'],
      buckets: [1, 5, 10, 30, 60, 300, 600],
      registers: [this.prometheusService.getRegistry()],
    });

    this.pendingMigrations = new Gauge({
      name: 'database_pending_migrations',
      help: 'Number of pending database migrations',
      registers: [this.prometheusService.getRegistry()],
    });

    this.migrationErrors = new Counter({
      name: 'database_migration_errors_total',
      help: 'Total number of database migration errors',
      labelNames: ['migration_name', 'error_type'],
      registers: [this.prometheusService.getRegistry()],
    });
  }

  recordMigrationSuccess(migrationName: string, duration: number): void {
    this.migrationCounter.inc({ status: 'success', migration_name: migrationName });
    this.migrationDuration.observe({ migration_name: migrationName }, duration);
  }

  recordMigrationFailure(migrationName: string, errorType: string): void {
    this.migrationCounter.inc({ status: 'failure', migration_name: migrationName });
    this.migrationErrors.inc({ migration_name: migrationName, error_type: errorType });
  }

  updatePendingMigrations(count: number): void {
    this.pendingMigrations.set(count);
  }
}
