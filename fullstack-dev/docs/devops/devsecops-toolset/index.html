<!-- dependency-check-maven.xml -->
<plugin>
    <groupId>org.owasp</groupId>
    <artifactId>dependency-check-maven</artifactId>
    <version>8.4.2</version>

</plugin>

Node.js implementation:

// dependency-check-runner.js
const { execSync } = require('child_process');
const fs = require('fs');

function runDependencyCheck() {
  try {
    // Download dependency-check CLI
    execSync(
      'wget https://github.com/jeremylong/DependencyCheck/releases/download/v8.4.2/dependency-check-8.4.2-release.zip',
    );
    execSync('unzip -q dependency-check-8.4.2-release.zip');

    // Run dependency check
    execSync(
      './dependency-check/bin/dependency-check.sh --project "ecommerce-app" --scan . --format JSON --format HTML --failOnCVSS 7',
    );

    // Parse results
    const results = JSON.parse(fs.readFileSync('dependency-check-report.json', 'utf8'));
    console.log(`Found ${results.dependencies.length} dependencies`);

    // Check for vulnerabilities
    const vulnerabilities = results.dependencies.filter((dep) => dep.vulnerabilities && dep.vulnerabilities.length > 0);
    if (vulnerabilities.length > 0) {
      console.error(`Found ${vulnerabilities.length} dependencies with vulnerabilities`);
      process.exit(1);
    }
  } catch (error) {
    console.error('Dependency check failed:', error.message);
    process.exit(1);
  }
}

runDependencyCheck();

Multi-Tool Security Pipeline​

# .github/workflows/comprehensive-security.yml
name: Comprehensive Security Scan

on:
  push:
    branches: [main]
  schedule:
    - cron: '0 2 * * 1' # Weekly scan on Monday at 2 AM

jobs:
  comprehensive-scan:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        scanner: [trivy, osv, grype]

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run Trivy
        if: matrix.scanner == 'trivy'
        uses: aquasecurity/trivy-action@master
        with:
          scan-type: 'fs'
          scan-ref: '.'
          format: 'json'
          output: 'trivy-results.json'

      - name: Run OSV Scanner
        if: matrix.scanner == 'osv'
        run: |
          go install github.com/google/osv-scanner/cmd/osv-scanner@v1
          osv-scanner --format json --output osv-results.json .

      - name: Run Grype
        if: matrix.scanner == 'grype'
        run: |
          curl -sSfL https://raw.githubusercontent.com/anchore/grype/main/install.sh | sh -s -- -b /usr/local/bin
          grype dir:. -o json > grype-results.json

      - name: Compare Results
        run: |
          echo "Scanner: ${{ matrix.scanner }}"
          if [ -f "${{ matrix.scanner }}-results.json" ]; then
            echo "Results found for ${{ matrix.scanner }}"
            # Add result comparison logic here
          fi

      - name: Upload artifacts
        uses: actions/upload-artifact@v3
        with:
          name: security-results-${{ matrix.scanner }}
          path: '${{ matrix.scanner }}-results.json'

Cost-Effective Recommendation​

For your ecommerce application, I recommend this hybrid approach:

Free Tier Solution:

# Primary scanning (free)
npm run security:audit     # Built-in npm audit
npm run security:trivy     # Trivy for comprehensive scanning
npm run security:osv       # OSV Scanner for Google's database

Cost-Effective Enterprise Solution:

  • Primary: Trivy + OSV Scanner (Free)
  • Secondary: Snyk (for critical environments only)

# .github/workflows/incident-response.yml
name: Security Incident Response

on:
  repository_dispatch:
    types: [security-incident]

    runs-on: ubuntu-latest

    steps:
      - name: Parse Incident Data
        id: parse
        run: |
          echo "severity=${{ github.event.client_payload.severity }}" >> $GITHUB_OUTPUT
          echo "type=${{ github.event.client_payload.type }}" >> $GITHUB_OUTPUT
          echo "source=${{ github.event.client_payload.source }}" >> $GITHUB_OUTPUT

      - name: Critical Incident Response
        if: steps.parse.outputs.severity == 'critical'
        run: |
          # Auto-scale down compromised services
          kubectl scale deployment compromised-service --replicas=0 -n ecommerce

          # Trigger circuit breaker
          kubectl patch destinationrule ecommerce-api -n ecommerce \
            --type='merge' -p='{"spec":{"trafficPolicy":{"outlierDetection":{"consecutiveErrors":1}}}}'

          # Alert security team
          curl -X POST "${{ secrets.SLACK_SECURITY_WEBHOOK }}" \
            -H 'Content-type: application/json' \
            --data '{"text":"🚨 CRITICAL Security Incident Detected - Auto-response activated"}'

      - name: Create Security Incident Issue
        uses: actions/github-script@v6
        with:
          script: |
            github.rest.issues.create({
              owner: context.repo.owner,
              repo: context.repo.repo,
              title: `Security Incident: ${{ steps.parse.outputs.type }}`,
              body: `
                **Severity:** ${{ steps.parse.outputs.severity }}
                **Type:** ${{ steps.parse.outputs.type }}
                **Source:** ${{ steps.parse.outputs.source }}
                **Time:** ${new Date().toISOString()}
                
                **Auto-Response Actions Taken:**
                - [ ] Service isolation
                - [ ] Traffic redirection

2. GDPR Compliance Implementation​

// src/compliance/gdpr-compliance.service.ts
import { Injectable } from '@nestjs/common';
import { trace } from '@opentelemetry/api';

@Injectable()
export class GDPRComplianceService {
  private readonly tracer = trace.getTracer('gdpr-compliance');

  async processDataRequest(requestType: 'access' | 'deletion' | 'portability', userId: string) {
    return this.tracer.startActiveSpan(`gdpr.${requestType}_request`, async (span) => {
      span.setAttributes({
        'gdpr.request_type': requestType,
        'gdpr.user_id': userId,
        'gdpr.timestamp': new Date().toISOString(),
      });

      try {
        switch (requestType) {
          case 'access':
            return await this.generateDataExport(userId);
          case 'deletion':
            return await this.deleteUserData(userId);
          case 'portability':
            return await this.exportPortableData(userId);
        }
      } catch (error) {
        span.recordException(error);
        throw error;
      } finally {
        span.end();
      }
    });
  }

  private async generateDataExport(userId: string) {
    // Implementation for data export
    const userData = await this.collectUserData(userId);

    // Audit trail
    await this.auditService.log({
      action: 'gdpr_data_access',
      userId,
      timestamp: new Date(),
      details: { dataTypes: Object.keys(userData) },
    });

    return userData;
  }

  private async deleteUserData(userId: string) {
    // Implementation for data deletion with audit trail
    const deletedData = await this.performDataDeletion(userId);

    // Audit trail
    await this.auditService.log({
      action: 'gdpr_data_deletion',
      userId,
      timestamp: new Date(),
      details: { deletedRecords: deletedData.length },
    });

    return { deleted: true, recordCount: deletedData.length };
  }
}
