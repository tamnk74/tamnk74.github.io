2. Kafka Service Base Classes​

// shared/kafka/src/services/kafka-producer.service.ts
import { Injectable, Logger, OnModuleDestroy, OnModuleInit } from '@nestjs/common';
import { ClientKafka } from '@nestjs/microservices';
import { BaseEvent } from '@ecommerce/types';

@Injectable()
export abstract class KafkaProducerService implements OnModuleInit, OnModuleDestroy {
  protected readonly logger = new Logger(this.constructor.name);
  protected abstract kafkaClient: ClientKafka;

  async onModuleInit() {
    await this.kafkaClient.connect();
    this.logger.log('Kafka producer connected');
  }

  async onModuleDestroy() {
    await this.kafkaClient.close();
    this.logger.log('Kafka producer disconnected');
  }

  async publishEvent<T extends BaseEvent>(topic: string, event: T): Promise<void> {
    try {
      await this.kafkaClient.emit(topic, {
        key: event.aggregateId,
        value: JSON.stringify(event),
        headers: {
          eventType: event.type,
          version: event.version.toString(),
          timestamp: event.timestamp.toISOString(),
        },
      });

      this.logger.debug(`Event published to ${topic}:`, event.type);
    } catch (error) {
      this.logger.error(`Failed to publish event to ${topic}:`, error);
      throw error;
    }
  }
}
// shared/kafka/src/services/kafka-consumer.service.ts
import { Logger } from '@nestjs/common';
import { KafkaContext } from '@nestjs/microservices';

export abstract class KafkaConsumerService {
  protected readonly logger = new Logger(this.constructor.name);

  protected async handleMessage<T>(
    payload: any,
    context: KafkaContext,
    processor: (event: T) => Promise<void>,
  ): Promise<void> {
    const message = context.getMessage();
    const { partition, offset } = message;

    try {
      const event = this.parseEvent<T>(payload);
      await processor(event);

      this.logger.debug(`Processed message from partition ${partition}, offset ${offset}`);
    } catch (error) {
      this.logger.error(`Failed to process message:`, error);
      throw error;
    }
  }

  private parseEvent<T>(payload: any): T {
    try {
      return typeof payload === 'string' ? JSON.parse(payload) : payload;
    } catch (error) {
      throw new Error(`Invalid event format: ${error.message}`);
    }
  }
}

import { Injectable } from '@nestjs/common';
import { OrderProducerService } from '../kafka/order-producer.service';
import { CreateOrderDto } from './dto/create-order.dto';

@Injectable()
export class OrdersService {

    private readonly orderProducer: OrderProducerService, // ... other dependencies
  ) {}

  async createOrder(createOrderDto: CreateOrderDto, userId: string): Promise<Order> {
    // 1. Validate order data
    // 2. Check inventory availability
    // 3. Calculate totals

    const order = await this.orderRepository.save({
      ...createOrderDto,
      userId,
      status: 'PENDING',
      createdAt: new Date(),
    });

    // Publish order created event
    await this.orderProducer.publishOrderCreated({
      orderId: order.id,
      userId: order.userId,
      items: order.items,
      totalAmount: order.totalAmount,
      shippingAddress: order.shippingAddress,
    });

    return order;
  }

    const order = await this.orderRepository.findOne({ where: { id: orderId } });
    if (!order) {
      throw new NotFoundException('Order not found');
    }

    order.status = 'CONFIRMED';
    order.estimatedDelivery = this.calculateDeliveryDate();
    await this.orderRepository.save(order);

    await this.orderProducer.publishOrderConfirmed({
      orderId: order.id,
      userId: order.userId,
      estimatedDelivery: order.estimatedDelivery,
    });
  }
}

3. Notification Service Consumer​

// services/notification-service/src/kafka/notification-consumer.controller.ts
import { Controller } from '@nestjs/common';
import { EventPattern, Payload, Ctx, KafkaContext } from '@nestjs/microservices';
import { KafkaConsumerService, kafkaTopics } from '@ecommerce/kafka';
import { OrderEvents, PaymentEvents, UserEvents } from '@ecommerce/types';
import { NotificationService } from '../services/notification.service';

@Controller()
export class NotificationConsumerController extends KafkaConsumerService {
  constructor(private readonly notificationService: NotificationService) {
    super();
  }

  @EventPattern(kafkaTopics.ORDER_EVENTS)
  async handleOrderEvent(@Payload() payload: any, @Ctx() context: KafkaContext): Promise<void> {
    await this.handleMessage(payload, context, async (event: any) => {
      switch (event.type) {
        case 'ORDER_CREATED':
          await this.notificationService.sendOrderConfirmation(event);
          break;
        case 'ORDER_SHIPPED':
          await this.notificationService.sendShippingNotification(event);
          break;
        case 'ORDER_DELIVERED':
          await this.notificationService.sendDeliveryNotification(event);
          break;
      }
    });
  }

  @EventPattern(kafkaTopics.PAYMENT_EVENTS)
  async handlePaymentEvent(@Payload() payload: any, @Ctx() context: KafkaContext): Promise<void> {
    await this.handleMessage(payload, context, async (event: any) => {
      switch (event.type) {
        case 'PAYMENT_COMPLETED':
          await this.notificationService.sendPaymentConfirmation(event);
          break;
        case 'PAYMENT_FAILED':
          await this.notificationService.sendPaymentFailure(event);
          break;
      }
    });
  }
}

4. Product Service with Inventory Management​

// services/product-service/src/kafka/inventory-consumer.controller.ts
import { Controller } from '@nestjs/common';
import { EventPattern, Payload, Ctx, KafkaContext } from '@nestjs/microservices';
import { KafkaConsumerService, kafkaTopics } from '@ecommerce/kafka';
import { OrderEvents } from '@ecommerce/types';
import { InventoryService } from '../services/inventory.service';
import { ProductProducerService } from './product-producer.service';

@Controller()
export class InventoryConsumerController extends KafkaConsumerService {
  constructor(
    private readonly inventoryService: InventoryService,
    private readonly productProducer: ProductProducerService,
  ) {
    super();
  }

  @EventPattern(kafkaTopics.ORDER_EVENTS)
  async handleOrderEvent(@Payload() payload: any, @Ctx() context: KafkaContext): Promise<void> {
    await this.handleMessage(payload, context, async (event: any) => {
      switch (event.type) {
        case 'ORDER_CREATED':
          await this.reserveInventory(event);
          break;
        case 'ORDER_CANCELLED':
          await this.releaseInventory(event);
          break;
      }
    });
  }

  private async reserveInventory(event: OrderEvents['ORDER_CREATED']): Promise<void> {
    for (const item of event.items) {
      const updated = await this.inventoryService.reserveQuantity(item.productId, item.quantity);

      if (updated) {
        await this.productProducer.publishInventoryUpdated({
          productId: item.productId,
          quantity: -item.quantity,
          operation: 'subtract',
        });
      }
    }
  }

  private async releaseInventory(event: OrderEvents['ORDER_CANCELLED']): Promise<void> {
    // Implementation for releasing reserved inventory
    for (const item of event.items) {
      await this.inventoryService.releaseReservedQuantity(item.productId, item.quantity);

      await this.productProducer.publishInventoryUpdated({
        productId: item.productId,
        quantity: item.quantity,
        operation: 'add',
      });
    }
  }
}

// shared/monitoring/src/kafka-metrics.service.ts
import { Injectable } from '@nestjs/common';
import { Counter, Histogram, Gauge, register } from 'prom-client';

@Injectable()
export class KafkaMetricsService {
  private readonly messageCounter = new Counter({
    name: 'kafka_messages_total',
    help: 'Total number of Kafka messages processed',
    labelNames: ['topic', 'status', 'service'],
  });

  private readonly messageProcessingDuration = new Histogram({
    name: 'kafka_message_processing_duration_seconds',
    help: 'Duration of message processing in seconds',
    labelNames: ['topic', 'service'],
    buckets: [0.01, 0.05, 0.1, 0.5, 1, 5, 10],
  });

  private readonly messageSize = new Histogram({
    name: 'kafka_message_size_bytes',
    help: 'Size of Kafka messages in bytes',
    labelNames: ['topic', 'service'],
    buckets: [100, 1000, 10000, 100000, 1000000],
  });

  private readonly connectionGauge = new Gauge({
    name: 'kafka_connections_active',
    help: 'Number of active Kafka connections',
    labelNames: ['service', 'type'], // type: producer or consumer
  });

  constructor() {
    register.registerMetric(this.messageCounter);
    register.registerMetric(this.messageProcessingDuration);
    register.registerMetric(this.messageSize);
    register.registerMetric(this.connectionGauge);
  }

  recordMessageProcessed(topic: string, service: string, status: 'success' | 'error'): void {
    this.messageCounter.inc({ topic, status, service });
  }

  recordProcessingDuration(topic: string, service: string, duration: number): void {
    this.messageProcessingDuration.observe({ topic, service }, duration);
  }

  recordMessageSize(topic: string, service: string, size: number): void {
    this.messageSize.observe({ topic, service }, size);
  }

  setActiveConnections(service: string, type: 'producer' | 'consumer', count: number): void {
    this.connectionGauge.set({ service, type }, count);
  }
}

// shared/kafka/src/decorators/retry.decorator.ts
import { Logger } from '@nestjs/common';

export function KafkaRetry(maxRetries: number = 3, backoffMs: number = 1000) {
  return function (target: any, propertyName: string, descriptor: PropertyDescriptor) {
    const method = descriptor.value;
    const logger = new Logger(`${target.constructor.name}:${propertyName}`);

    descriptor.value = async function (...args: any[]) {
      let lastError: Error;

      for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
          return await method.apply(this, args);
        } catch (error) {
          lastError = error;
          logger.warn(`Attempt ${attempt}/${maxRetries} failed:`, error.message);

          if (attempt < maxRetries) {
            const delay = backoffMs * Math.pow(2, attempt - 1); // Exponential backoff
            await new Promise((resolve) => setTimeout(resolve, delay));
          }
        }
      }

      logger.error(`All ${maxRetries} attempts failed. Last error:`, lastError);
      throw lastError;
    };
  };
}

// shared/kafka/src/services/dead-letter-queue.service.ts
import { Injectable, Logger } from '@nestjs/common';
import { ClientKafka } from '@nestjs/microservices';

@Injectable()
export class DeadLetterQueueService {
  private readonly logger = new Logger(DeadLetterQueueService.name);

  constructor(private readonly kafkaClient: ClientKafka) {}

  async sendToDeadLetterQueue(
    originalTopic: string,
    message: any,
    error: Error,
    retryCount: number = 0,
  ): Promise<void> {
    const dlqTopic = `${originalTopic}.dlq`;

    try {
      await this.kafkaClient.emit(dlqTopic, {
        ...message,
        metadata: {
          ...message.metadata,
          originalTopic,
          error: error.message,
          retryCount,
          failedAt: new Date().toISOString(),
        },
      });

      this.logger.error(`Message sent to DLQ: ${dlqTopic}`, {
        originalTopic,
        error: error.message,
        retryCount,
      });
    } catch (dlqError) {
      this.logger.error(`Failed to send message to DLQ: ${dlqTopic}`, dlqError);
    }
  }
}

import { Logger } from '@nestjs/common';
import { KafkaContext } from '@nestjs/microservices';
import { KafkaMetricsService } from '@ecommerce/monitoring';
import { DeadLetterQueueService } from './dead-letter-queue.service';

export abstract class EnhancedKafkaConsumerService {
  protected readonly logger = new Logger(this.constructor.name);

  constructor(
    protected readonly metricsService: KafkaMetricsService,
    protected readonly dlqService: DeadLetterQueueService,
  ) {}

  protected async handleMessageWithRetry<T>(
    payload: any,
    context: KafkaContext,
    processor: (event: T) => Promise<void>,
    maxRetries: number = 3,
  ): Promise<void> {
    const message = context.getMessage();
    const topic = context.getTopic();
    const { partition, offset } = message;
    const startTime = Date.now();

    try {
      const event = this.parseEvent<T>(payload);
      await processor(event);

      const processingTime = Date.now() - startTime;
      this.metricsService.recordMessageProcessed(topic, this.constructor.name, 'success');
      this.metricsService.recordProcessingDuration(topic, this.constructor.name, processingTime / 1000);

      this.logger.debug(`Successfully processed message from ${topic}, partition ${partition}, offset ${offset}`);
    } catch (error) {
      const processingTime = Date.now() - startTime;
      this.metricsService.recordMessageProcessed(topic, this.constructor.name, 'error');
      this.metricsService.recordProcessingDuration(topic, this.constructor.name, processingTime / 1000);

      const retryCount = this.getRetryCount(message);

      if (retryCount < maxRetries) {
        this.logger.warn(`Retrying message processing. Attempt ${retryCount + 1}/${maxRetries}`, error);
        // Implement retry logic here (could be immediate retry or scheduled)
        throw error; // Let Kafka handle the retry
      } else {
        this.logger.error(`Max retries exceeded. Sending to DLQ`, error);
        await this.dlqService.sendToDeadLetterQueue(topic, payload, error, retryCount);
      }
    }
  }

  private parseEvent<T>(payload: any): T {
    try {
      return typeof payload === 'string' ? JSON.parse(payload) : payload;
    } catch (error) {
      throw new Error(`Invalid event format: ${error.message}`);
    }
  }

  private getRetryCount(message: any): number {
    return message.headers?.retryCount ? parseInt(message.headers.retryCount) : 0;
  }
}

#!/bin/bash
# scripts/manage-topics.sh

KAFKA_BROKER="kafka-service.kafka.svc.cluster.local:9092"
KUBECTL_KAFKA_POD=$(kubectl get pod -n kafka -l app=kafka-broker -o jsonpath='{.items[0].metadata.name}')

function list_topics() {
    echo "📋 Listing Kafka topics..."
    kubectl exec -n kafka $KUBECTL_KAFKA_POD -- kafka-topics --bootstrap-server $KAFKA_BROKER --list
}

function create_topic() {
    local topic_name=$1
    local partitions=${2:-3}
    local replication_factor=${3:-1}

    echo "➕ Creating topic: $topic_name (partitions: $partitions, replication: $replication_factor)"
    kubectl exec -n kafka $KUBECTL_KAFKA_POD -- kafka-topics \
        --bootstrap-server $KAFKA_BROKER \
        --create \
        --topic $topic_name \
        --partitions $partitions \
        --replication-factor $replication_factor \
        --if-not-exists
}

function delete_topic() {
    local topic_name=$1
    echo "🗑️  Deleting topic: $topic_name"
    kubectl exec -n kafka $KUBECTL_KAFKA_POD -- kafka-topics \
        --bootstrap-server $KAFKA_BROKER \
        --delete \
        --topic $topic_name
}

function describe_topic() {
    local topic_name=$1
    echo "🔍 Describing topic: $topic_name"
    kubectl exec -n kafka $KUBECTL_KAFKA_POD -- kafka-topics \
        --bootstrap-server $KAFKA_BROKER \
        --describe \
        --topic $topic_name
}

function consumer_groups() {
    echo "👥 Listing consumer groups..."
    kubectl exec -n kafka $KUBECTL_KAFKA_POD -- kafka-consumer-groups \
        --bootstrap-server $KAFKA_BROKER \
        --list
}

function group_info() {
    local group_name=$1
    echo "📊 Consumer group info: $group_name"
    kubectl exec -n kafka $KUBECTL_KAFKA_POD -- kafka-consumer-groups \
        --bootstrap-server $KAFKA_BROKER \
        --describe \
        --group $group_name
}

case "$1" in
    list)
        list_topics
        ;;
    create)
        create_topic $2 $3 $4
        ;;
    delete)
        delete_topic $2
        ;;
    describe)
        describe_topic $2
        ;;
    groups)
        consumer_groups
        ;;
    group-info)
        group_info $2
        ;;
    *)
        echo "Usage: $0 {list|create|delete|describe|groups|group-info}"
        echo ""
        echo "Examples:"
        echo "  $0 list                           # List all topics"
        echo "  $0 create my-topic 6 1           # Create topic with 6 partitions, replication factor 1"
        echo "  $0 delete my-topic               # Delete topic"
        echo "  $0 describe my-topic             # Describe topic"
        echo "  $0 groups                        # List consumer groups"
        echo "  $0 group-info my-group           # Show consumer group info"
        exit 1
        ;;
esac

#!/bin/bash
# scripts/backup-kafka-topics.sh

KAFKA_BROKER="kafka-service:9092"
BACKUP_DIR="/tmp/kafka-backup-$(date +%Y%m%d-%H%M%S)"
KAFKA_POD=$(kubectl get pod -n kafka -l app=kafka-broker -o jsonpath='{.items[0].metadata.name}')

mkdir -p $BACKUP_DIR

echo "📦 Starting Kafka backup process..."

# Get list of topics
TOPICS=$(kubectl exec -n kafka $KAFKA_POD -- kafka-topics --bootstrap-server $KAFKA_BROKER --list)

for topic in $TOPICS; do
    echo "💾 Backing up topic: $topic"

    # Export topic data
    kubectl exec -n kafka $KAFKA_POD -- kafka-console-consumer \
        --bootstrap-server $KAFKA_BROKER \
        --topic $topic \
        --from-beginning \
        --timeout-ms 10000 > "$BACKUP_DIR/$topic.json" 2>/dev/null

    # Export topic configuration
    kubectl exec -n kafka $KAFKA_POD -- kafka-configs \
        --bootstrap-server $KAFKA_BROKER \
        --describe \
        --entity-type topics \
        --entity-name $topic > "$BACKUP_DIR/$topic.config"

    echo "✅ Backed up topic: $topic"
done

# Create archive
tar -czf "kafka-backup-$(date +%Y%m%d-%H%M%S).tar.gz" -C $(dirname $BACKUP_DIR) $(basename $BACKUP_DIR)

echo "🎉 Backup completed: kafka-backup-$(date +%Y%m%d-%H%M%S).tar.gz"

#!/bin/bash

function debug_consumer_groups() {
    echo "🔍 Consumer Groups Debug Information"
    echo "=================================="

    # List all consumer groups
    echo "📋 Consumer Groups:"
    kubectl exec -n kafka $KAFKA_POD -- kafka-consumer-groups \

    echo ""

    # Show detailed info for each group
    GROUPS=$(kubectl exec -n kafka $KAFKA_POD -- kafka-consumer-groups \
        --bootstrap-server $KAFKA_BROKER --list)

    for group in $GROUPS; do
        echo "👥 Group: $group"
        kubectl exec -n kafka $KAFKA_POD -- kafka-consumer-groups \
            --bootstrap-server $KAFKA_BROKER \
            --describe --group $group
        echo ""
    done
}

function debug_topic_details() {
    local topic=$1
    echo "🔍 Topic Debug Information: $topic"
    echo "=================================="

    # Topic configuration
    echo "⚙️  Topic Configuration:"
    kubectl exec -n kafka $KAFKA_POD -- kafka-configs \
        --bootstrap-server $KAFKA_BROKER \
        --describe --entity-type topics --entity-name $topic

    echo ""

    # Topic details
    echo "📊 Topic Details:"
    kubectl exec -n kafka $KAFKA_POD -- kafka-topics \
        --bootstrap-server $KAFKA_BROKER \
        --describe --topic $topic

    echo ""

    # Log segments
    echo "📁 Log Segments:"
    kubectl exec -n kafka $KAFKA_POD -- kafka-log-dirs \
        --bootstrap-server $KAFKA_BROKER \
        --describe --json | jq ".brokers[].logDirs[].partitions | to_entries[] | select(.key | contains(\"$topic\"))"
}
