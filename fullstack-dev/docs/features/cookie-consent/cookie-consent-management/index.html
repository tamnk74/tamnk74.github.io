European Union (GDPR + ePrivacy Directive)​

Applies to: All EU/EEA countries (27 EU countries + Iceland, Liechtenstein, Norway)

Key Requirements:

  • Explicit Consent: Users must actively opt-in for non-essential cookies
  • Granular Control: Users must be able to choose specific cookie categories
  • Withdrawal: Users must be able to withdraw consent at any time
  • Clear Information: Cookie purposes must be clearly explained in plain language
  • Legitimate Interest: Must be properly documented and justified
  • Cookie Wall Prohibition: Cannot block access to services for refusing consent
  • Pre-checked Boxes Prohibited: Consent boxes cannot be pre-ticked

// GDPR-compliant consent recording
function recordGDPRConsent(consent: GDPRConsent): void {
  const record = {
    ...consent,
    ipAddress: anonymizeIP(getUserIP()), // Last octet anonymized
    userAgent: navigator.userAgent,
    referrer: document.referrer,
    consentMethod: 'banner_interaction'
  };
  
  // Store locally and send to server
  localStorage.setItem('gdpr_consent', JSON.stringify(record));
  sendConsentToServer(record);
}

Multi-Jurisdictional Compliance Strategy​

interface GlobalComplianceManager {
  detectUserJurisdiction(): string;
  getApplicableLaws(jurisdiction: string): PrivacyLaw[];
  generateComplianceConfig(laws: PrivacyLaw[]): ConsentConfig;
  validateCompliance(config: ConsentConfig): ComplianceResult;
}

class GlobalPrivacyCompliance implements GlobalComplianceManager {
  private jurisdictionMap = new Map<string, PrivacyLaw[]>([
    ['EU', [PrivacyLaw.GDPR, PrivacyLaw.EPRIVACY]],
    ['UK', [PrivacyLaw.UK_GDPR, PrivacyLaw.PECR]],
    ['US-CA', [PrivacyLaw.CCPA, PrivacyLaw.CPRA]],
    ['US-VA', [PrivacyLaw.VCDPA]],
    ['US-CO', [PrivacyLaw.CPA]],
    ['CA', [PrivacyLaw.PIPEDA]],
    ['CA-QC', [PrivacyLaw.PIPEDA, PrivacyLaw.QUEBEC_LAW25]],
    ['BR', [PrivacyLaw.LGPD]],
    ['AU', [PrivacyLaw.PRIVACY_ACT]],
    ['CN', [PrivacyLaw.PIPL]],
    ['IN', [PrivacyLaw.DPDP]],
    ['JP', [PrivacyLaw.JAPAN_PIPA]],
    ['KR', [PrivacyLaw.KOREA_PIPA]],
    ['SG', [PrivacyLaw.SINGAPORE_PDPA]]
  ]);

  detectUserJurisdiction(): string {
    // Use geo-IP, browser locale, or explicit user selection
    const geoIP = this.getGeoIPCountry();
    const browserLocale = navigator.language;
    const userSelection = localStorage.getItem('user_jurisdiction');
    
    return userSelection || this.mapToJurisdiction(geoIP) || 'US';
  }

  getApplicableLaws(jurisdiction: string): PrivacyLaw[] {
    return this.jurisdictionMap.get(jurisdiction) || [PrivacyLaw.DEFAULT];
  }

  generateComplianceConfig(laws: PrivacyLaw[]): ConsentConfig {
    const config: ConsentConfig = {
      consentType: 'opt-in',
      granularControl: false,
      cookieWallAllowed: false,
      gpcSupport: false,
      dataSubjectRights: [],
      retentionRequirements: [],
      breachNotification: false
    };

    for (const law of laws) {
      switch (law) {
        case PrivacyLaw.GDPR:
          config.consentType = 'opt-in';
          config.granularControl = true;
          config.cookieWallAllowed = false;
          config.dataSubjectRights.push('access', 'deletion', 'portability', 'rectification');
          break;
          
        case PrivacyLaw.CCPA:
          config.gpcSupport = true;
          config.dataSubjectRights.push('access', 'deletion', 'opt-out-sale');
          break;
          
        case PrivacyLaw.LGPD:
          config.consentType = 'opt-in';
          config.dataSubjectRights.push('access', 'correction', 'deletion', 'portability', 'opposition');
          break;
          
        // Add more laws as needed
      }
    }

    return config;
  }

  validateCompliance(config: ConsentConfig): ComplianceResult {
    const issues: string[] = [];
    
    // Validate configuration against requirements
    if (config.consentType === 'opt-out' && config.granularControl === false) {
      issues.push('Granular control required for opt-out consent');
    }
    
    if (config.gpcSupport === false && this.requiresGPC(config)) {
      issues.push('Global Privacy Control support required');
    }
    
    return {
      isCompliant: issues.length === 0,
      issues,
      recommendations: this.generateRecommendations(issues)
    };
  }

  private getGeoIPCountry(): string {
    // Implementation to get country from geo-IP
    return 'US'; // Placeholder
  }

  private mapToJurisdiction(country: string): string {
    const countryMap: Record<string, string> = {
      'US': 'US',
      'CA': 'CA', 'GB': 'UK', 'DE': 'EU', 'FR': 'EU', 'IT': 'EU',
      'BR': 'BR', 'AU': 'AU', 'CN': 'CN', 'IN': 'IN',
      'JP': 'JP', 'KR': 'KR', 'SG': 'SG'
    };
    
    return countryMap[country] || 'US';
  }

  private requiresGPC(config: ConsentConfig): boolean {
    // Check if jurisdiction requires GPC support
    return config.dataSubjectRights.includes('opt-out-sale');
  }

  private generateRecommendations(issues: string[]): string[] {
    // Generate actionable recommendations based on issues
    return issues.map(issue => `Recommendation: ${issue}`);
  }
}

1. Cookie Consent Provider​

// lib/cookie-consent.tsx
'use client';

import React, { createContext, useContext, useState, useEffect } from 'react';

export interface CookieConsent {
  necessary: boolean;
  analytics: boolean;
  functional: boolean;
  marketing: boolean;
}

interface CookieConsentContextType {
  consent: CookieConsent;
  updateConsent: (newConsent: Partial<CookieConsent>) => void;
  showBanner: boolean;
  acceptAll: () => void;
  rejectAll: () => void;
  isLoaded: boolean;
}

const CookieConsentContext = createContext<CookieConsentContextType | undefined>(undefined);

const CONSENT_COOKIE_NAME = 'cookie-consent';
const CONSENT_VERSION = '1.0';

export function CookieConsentProvider({ children }: { children: React.ReactNode }) {
  const [consent, setConsent] = useState<CookieConsent>({
    necessary: true, // Always true
    analytics: false,
    functional: false,
    marketing: false,
  });
  const [showBanner, setShowBanner] = useState(false);
  const [isLoaded, setIsLoaded] = useState(false);

  useEffect(() => {
    // Load consent from localStorage or cookie
    const savedConsent = loadConsentFromStorage();
    if (savedConsent) {
      setConsent(savedConsent);
      setShowBanner(false);
    } else {
      setShowBanner(true);
    }
    setIsLoaded(true);
  }, []);

  const updateConsent = (newConsent: Partial<CookieConsent>) => {
    const updatedConsent = { ...consent, ...newConsent, necessary: true };
    setConsent(updatedConsent);
    saveConsentToStorage(updatedConsent);
    setShowBanner(false);
    
    // Trigger consent change event
    window.dispatchEvent(new CustomEvent('cookieConsentChange', {
      detail: updatedConsent
    }));
  };

  const acceptAll = () => {
    updateConsent({
      analytics: true,
      functional: true,
      marketing: true,
    });
  };

  const rejectAll = () => {
    updateConsent({
      analytics: false,
      functional: false,
      marketing: false,
    });
  };

  return (
    <CookieConsentContext.Provider 
      value={{ 
        consent, 
        updateConsent, 
        showBanner, 
        acceptAll, 
        rejectAll, 
        isLoaded 
      }}
    >
      {children}
    </CookieConsentContext.Provider>
  );
}

export const useCookieConsent = () => {
  const context = useContext(CookieConsentContext);
  if (!context) {
    throw new Error('useCookieConsent must be used within CookieConsentProvider');
  }
  return context;
};

function loadConsentFromStorage(): CookieConsent | null {
  try {
    const stored = localStorage.getItem(CONSENT_COOKIE_NAME);
    if (stored) {
      const parsed = JSON.parse(stored);
      if (parsed.version === CONSENT_VERSION) {
        return parsed.consent;
      }
    }
  } catch (error) {
    console.error('Error loading consent from storage:', error);
  }
  return null;
}

function saveConsentToStorage(consent: CookieConsent) {
  try {
    const toStore = {
      consent,
      version: CONSENT_VERSION,
      timestamp: new Date().toISOString(),
    };
    localStorage.setItem(CONSENT_COOKIE_NAME, JSON.stringify(toStore));
    
    // Also set a cookie for server-side access
    document.cookie = `${CONSENT_COOKIE_NAME}=${JSON.stringify(toStore)}; path=/; max-age=31536000; SameSite=Lax`;
  } catch (error) {
    console.error('Error saving consent to storage:', error);
  }
}

// hooks/useCookieManager.ts
import { useEffect } from 'react';
import { useCookieConsent } from '@/lib/cookie-consent';

export function useCookieManager() {
  const { consent, isLoaded } = useCookieConsent();

  useEffect(() => {
    if (!isLoaded) return;

    // Google Analytics
    if (consent.analytics) {
      loadGoogleAnalytics();
    } else {
      removeGoogleAnalytics();
    }

    // Facebook Pixel
    if (consent.marketing) {
      loadFacebookPixel();
    } else {
      removeFacebookPixel();
    }

    // Hotjar
    if (consent.analytics) {
      loadHotjar();
    } else {
      removeHotjar();
    }

  }, [consent, isLoaded]);

  return { consent, isLoaded };
}

function loadGoogleAnalytics() {
  if (typeof window === 'undefined') return;
  
  // Load Google Analytics script
  const script = document.createElement('script');
  script.src = `https://www.googletagmanager.com/gtag/js?id=${process.env.NEXT_PUBLIC_GA_ID}`;
  script.async = true;
  document.head.appendChild(script);

  // Initialize gtag
  window.dataLayer = window.dataLayer || [];
  function gtag(...args: any[]) {
    window.dataLayer.push(arguments);
  }
  window.gtag = gtag;
  
  gtag('js', new Date());
  gtag('config', process.env.NEXT_PUBLIC_GA_ID, {
    anonymize_ip: true,
    cookie_flags: 'SameSite=Lax;Secure'
  });
}

function removeGoogleAnalytics() {
  // Remove GA cookies
  const gaCookies = ['_ga', '_gid', '_gat_gtag_' + process.env.NEXT_PUBLIC_GA_ID];
  gaCookies.forEach(cookie => {
    document.cookie = `${cookie}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
  });
}

function loadFacebookPixel() {
  if (typeof window === 'undefined') return;
  
  // Facebook Pixel code
  window.fbq = window.fbq || function() {
    window.fbq.callMethod ? window.fbq.callMethod.apply(window.fbq, arguments) : 
    window.fbq.queue.push(arguments);
  };
  
  if (!window._fbq) window._fbq = window.fbq;
  window.fbq.push = window.fbq;
  window.fbq.loaded = true;
  window.fbq.version = '2.0';
  window.fbq.queue = [];
  
  const script = document.createElement('script');
  script.src = 'https://connect.facebook.net/en_US/fbevents.js';
  script.async = true;
  document.head.appendChild(script);
  
  window.fbq('init', process.env.NEXT_PUBLIC_FACEBOOK_PIXEL_ID);
  window.fbq('track', 'PageView');
}

function removeFacebookPixel() {
  // Remove Facebook Pixel cookies
  const fbCookies = ['_fbp', '_fbc'];
  fbCookies.forEach(cookie => {
    document.cookie = `${cookie}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
  });
}

function loadHotjar() {
  if (typeof window === 'undefined') return;
  
  window.hj = window.hj || function() {
    (window.hj.q = window.hj.q || []).push(arguments);
  };
  window._hjSettings = { hjid: process.env.NEXT_PUBLIC_HOTJAR_ID, hjsv: 6 };
  
  const script = document.createElement('script');
  script.src = `https://static.hotjar.com/c/hotjar-${process.env.NEXT_PUBLIC_HOTJAR_ID}.js?sv=6`;
  script.async = true;
  document.head.appendChild(script);
}

function removeHotjar() {
  // Remove Hotjar cookies
  const hjCookies = Object.keys(document.cookie).filter(key => key.startsWith('_hj'));
  hjCookies.forEach(cookie => {
    document.cookie = `${cookie}=; expires=Thu, 01 Jan 1970 00:00:00 UTC; path=/;`;
  });
}

@Injectable()
export class CookieConsentMiddleware implements NestMiddleware {
  use(req: Request, res: Response, next: NextFunction) {
    const consentCookie = req.cookies['cookie-consent'];
    
    if (consentCookie) {
      try {
        const consent = JSON.parse(consentCookie);
        req.cookieConsent = consent;
      } catch (error) {
        console.error('Error parsing cookie consent:', error);
      }
    }
    
    next();
  }
}

function checkNoCookiesSetBeforeConsent() {
  // Check that no non-essential cookies are set before consent
  const cookies = document.cookie.split(';');
  const nonEssentialCookies = cookies.filter(cookie => {
    const name = cookie.split('=')[0].trim();
    return !['session_id', 'csrf_token', 'load_balancer'].includes(name);
  });
  return nonEssentialCookies.length === 0;
}

function checkConsentBannerExists() {
  return document.querySelector('[data-testid="cookie-banner"]') !== null;
}

function checkGranularConsentOptions() {
  const categories = ['analytics', 'marketing', 'functional'];
  return categories.every(category => 
    document.querySelector(`[data-consent-category="${category}"]`) !== null
  );
}

function checkConsentWithdrawal() {
  return document.querySelector('[data-testid="cookie-settings"]') !== null;
}

function checkCookiePolicyAccessible() {
  const policyLink = document.querySelector('a[href*="cookie-policy"]');
  return policyLink !== null && policyLink.href.length > 0;
}

function checkConsentRecording() {
  // Verify consent is recorded with timestamp and version
  const consent = localStorage.getItem('cookie_consent');
  if (!consent) return false;
  
  try {
    const consentData = JSON.parse(consent);
    return consentData.timestamp && consentData.version && consentData.choices;
  } catch {
    return false;
  }
}

// CCPA Compliance Verification
function verifyCCPACompliance() {
  const tests = {
    doNotSellLink: checkDoNotSellLink(),
    gpcSupport: checkGPCSupport(),
    optOutMechanism: checkOptOutMechanism(),
    privacyRights: checkPrivacyRights(),
    dataInventory: checkDataInventory()
  };
  
  console.log('CCPA Compliance Test Results:', tests);
  return Object.values(tests).every(test => test === true);
}

function checkDoNotSellLink() {
  const doNotSellLink = document.querySelector('a[href*="do-not-sell"]') ||
                       document.querySelector('a:contains("Do Not Sell")');
  return doNotSellLink !== null;
}

function checkGPCSupport() {
  // Check if site respects Global Privacy Control
  return navigator.globalPrivacyControl === undefined || 
         localStorage.getItem('gpc_respected') === 'true';
}

function checkOptOutMechanism() {
  // Verify there's a clear opt-out mechanism
  const optOutButton = document.querySelector('[data-testid="opt-out-sale"]') ||
                      document.querySelector('button:contains("Opt Out")');
  return optOutButton !== null;
}

function checkPrivacyRights() {
  // Check privacy rights page exists and is accessible
  const privacyRightsLink = document.querySelector('a[href*="privacy-rights"]') ||
                           document.querySelector('a[href*="consumer-rights"]');
  return privacyRightsLink !== null;
}

function checkDataInventory() {
  // Check if privacy policy includes data inventory
  return fetch('/privacy-policy')
    .then(response => response.text())
    .then(text => text.includes('categories of personal information'))
    .catch(() => false);
}

function checkMeaningfulConsent() {
  // PIPEDA requires "meaningful consent"
  const consentForm = document.querySelector('[data-testid="consent-form"]');
  if (!consentForm) return false;
  
  // Check for clear, understandable language
  const consentText = consentForm.textContent || '';
  return !consentText.includes('legal jargon') && 
         consentText.length > 100 && // Sufficient detail
         consentText.includes('purpose');
}
