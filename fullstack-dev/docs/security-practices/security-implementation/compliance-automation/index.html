interface PersonalDataField {

  retentionPeriod: number; // in days

@Injectable()
export class DataDiscoveryService {
  private readonly logger = new Logger(DataDiscoveryService.name);
  
  private readonly personalDataPatterns = [
    { pattern: /email/i, type: 'PII', classification: 'Contact Information' },
    { pattern: /phone|mobile/i, type: 'PII', classification: 'Contact Information' },
    { pattern: /address|street|city|postal|zip/i, type: 'PII', classification: 'Address Information' },
    { pattern: /birth|dob|age/i, type: 'PII', classification: 'Demographic Information' },
    { pattern: /ssn|social_security|national_id/i, type: 'SENSITIVE_PII', classification: 'Identity Numbers' },
    { pattern: /passport|driver_license|id_number/i, type: 'SENSITIVE_PII', classification: 'Identity Documents' },
    { pattern: /medical|health|diagnosis/i, type: 'SPECIAL_CATEGORY', classification: 'Health Data' },
    { pattern: /religion|belief|political/i, type: 'SPECIAL_CATEGORY', classification: 'Belief/Opinion Data' },
    { pattern: /biometric|fingerprint|facial/i, type: 'SPECIAL_CATEGORY', classification: 'Biometric Data' }
  ];

  async discoverPersonalData(): Promise<PersonalDataField[]> {
    const discoveredFields: PersonalDataField[] = [];
    
    // Get database schema information
    const tables = await this.getDatabaseTables();
    
    for (const table of tables) {
      const columns = await this.getTableColumns(table);
      
      for (const column of columns) {
        const classification = this.classifyColumn(column.name, column.type);
        
        if (classification) {
          discoveredFields.push({
            table: table.name,
            column: column.name,
            dataType: classification.type,
            classification: classification.classification,
            retentionPeriod: this.getRetentionPeriod(classification.type),
            lawfulBasis: this.determineLawfulBasis(classification.type),
            processingPurpose: this.determineProcessingPurpose(classification.classification)
          });
        }
      }
    }

    // Store discovered data in compliance database
    await this.storeDataInventory(discoveredFields);
    
    this.logger.log(`Discovered ${discoveredFields.length} personal data fields`);
    return discoveredFields;
  }

  private classifyColumn(columnName: string, columnType: string) {
    for (const pattern of this.personalDataPatterns) {
      if (pattern.pattern.test(columnName)) {
        return {
          type: pattern.type,
          classification: pattern.classification
        };
      }
    }
    return null;
  }

  private getRetentionPeriod(dataType: string): number {
    switch (dataType) {
      case 'SPECIAL_CATEGORY': return 365; // 1 year
      case 'SENSITIVE_PII': return 2555; // 7 years
      case 'PII': return 1095; // 3 years
      default: return 365;
    }
  }

  private determineLawfulBasis(dataType: string): string {
    switch (dataType) {
      case 'SPECIAL_CATEGORY': return 'Explicit consent';
      case 'SENSITIVE_PII': return 'Legal obligation';
      case 'PII': return 'Legitimate interest';
      default: return 'Consent';
    }
  }

  private determineProcessingPurpose(classification: string): string {
    switch (classification) {
      case 'Contact Information': return 'Customer communication';
      case 'Address Information': return 'Service delivery';
      case 'Identity Numbers': return 'Legal compliance';

import { Injectable, Logger } from '@nestjs/common';

interface DataSubjectRequest {
  requestId: string;
  userId: string;
  requestType: 'ACCESS' | 'RECTIFICATION' | 'ERASURE' | 'PORTABILITY' | 'RESTRICTION' | 'OBJECTION';
  requestDate: Date;
  verificationStatus: 'PENDING' | 'VERIFIED' | 'REJECTED';
  fulfillmentStatus: 'PENDING' | 'IN_PROGRESS' | 'COMPLETED' | 'REJECTED';
  completionDate?: Date;
  rejectionReason?: string;
  requestDetails: any;
}

@Injectable()
export class DataSubjectRightsService {
  private readonly logger = new Logger(DataSubjectRightsService.name);

  async submitDataSubjectRequest(
    userId: string,
    requestType: DataSubjectRequest['requestType'],
    requestDetails: any
  ): Promise<DataSubjectRequest> {
    const request: DataSubjectRequest = {
      requestId: this.generateRequestId(),
      userId,
      requestType,
      requestDate: new Date(),
      verificationStatus: 'PENDING',
      fulfillmentStatus: 'PENDING',
      requestDetails
    };

    await this.storeRequest(request);
    
    // Trigger verification process
    await this.initiateVerificationProcess(request);

    this.logger.log(`Data subject request submitted: ${request.requestId}`);
    return request;
  }

  async fulfillAccessRequest(requestId: string): Promise<any> {
    const request = await this.getRequest(requestId);
    
    if (request.verificationStatus !== 'VERIFIED') {
      throw new Error('Request not verified');
    }

    // Collect all personal data for the user
    const personalData = await this.collectPersonalData(request.userId);
    
    // Generate data export
    const exportData = await this.generateDataExport(personalData);
    
    // Update request status
    await this.updateRequestStatus(requestId, 'COMPLETED');
    
    this.logger.log(`Access request fulfilled: ${requestId}`);
    return exportData;
  }

  async fulfillErasureRequest(requestId: string): Promise<void> {
    const request = await this.getRequest(requestId);
    
    if (request.verificationStatus !== 'VERIFIED') {
      throw new Error('Request not verified');
    }

    // Check if erasure is legally possible
    const canErase = await this.checkErasureLegality(request.userId);
    
    if (!canErase.allowed) {
      await this.rejectRequest(requestId, canErase.reason);
      return;
    }

    // Perform data erasure
    await this.performDataErasure(request.userId, request.requestDetails);
    
    // Update request status
    await this.updateRequestStatus(requestId, 'COMPLETED');
    
    this.logger.log(`Erasure request fulfilled: ${requestId}`);
  }

  async fulfillPortabilityRequest(requestId: string): Promise<any> {
    const request = await this.getRequest(requestId);
    
    if (request.verificationStatus !== 'VERIFIED') {
      throw new Error('Request not verified');
    }

    // Collect portable data (data provided by the data subject or generated by their use of the service)
    const portableData = await this.collectPortableData(request.userId);
    
    // Generate structured data export (JSON, CSV, XML)
    const exportFormat = request.requestDetails.format || 'JSON';
    const structuredExport = await this.generateStructuredExport(portableData, exportFormat);
    
    // Update request status
    await this.updateRequestStatus(requestId, 'COMPLETED');
    
    this.logger.log(`Portability request fulfilled: ${requestId}`);
    return structuredExport;
  }

  async generateComplianceReport(): Promise<any> {
    const thirtyDaysAgo = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000);
    const requests = await this.getRequestsSince(thirtyDaysAgo);
    
    return {

// src/compliance/pci/card-data-protection.service.ts
import { Injectable, Logger } from '@nestjs/common';

@Injectable()
export class CardDataProtectionService {
  private readonly logger = new Logger(CardDataProtectionService.name);
  
  private readonly cardDataPatterns = [
    /\b4[0-9]{12}(?:[0-9]{3})?\b/, // Visa
    /\b5[1-5][0-9]{14}\b/, // MasterCard
    /\b3[47][0-9]{13}\b/, // American Express
    /\b3[0-9]{4}[\s\-]?[0-9]{6}[\s\-]?[0-9]{5}\b/, // Diners Club
    /\b6(?:011|5[0-9]{2})[0-9]{12}\b/ // Discover
  ];

  async scanForCardData(text: string): Promise<{ found: boolean; violations: string[] }> {
    const violations: string[] = [];
    
    for (const pattern of this.cardDataPatterns) {
      const matches = text.match(pattern);
      if (matches) {
        violations.push(`Potential card data found: ${this.maskCardNumber(matches[0])}`);
      }
    }

    if (violations.length > 0) {
      await this.reportPCIViolation('CARD_DATA_EXPOSURE', violations);
    }

    return {
      found: violations.length > 0,
      violations
    };
  }

  async enforceDataRetentionPolicies(): Promise<void> {
    // PCI DSS Requirement 3.1: Keep cardholder data storage to a minimum
    const retentionPolicies = [
      { dataType: 'CARDHOLDER_DATA', maxRetentionDays: 365 },
      { dataType: 'SENSITIVE_AUTH_DATA', maxRetentionDays: 0 }, // Should not be stored
      { dataType: 'TRANSACTION_LOGS', maxRetentionDays: 365 * 3 } // 3 years
    ];

    for (const policy of retentionPolicies) {
      await this.enforceRetentionPolicy(policy);
    }
  }

  async validateEncryptionCompliance(): Promise<{ compliant: boolean; issues: string[] }> {
    const issues: string[] = [];
    
    // Check encryption at rest
    const encryptionAtRest = await this.checkEncryptionAtRest();
    if (!encryptionAtRest.compliant) {

    // Check key management

    if (!keyManagement.compliant) {
      issues.push(...keyManagement.issues);
    }

    return {
      compliant: issues.length === 0,
      issues
    };
  }

  async monitorCardholderDataAccess(userId: string, resource: string, action: string) {

    const accessDecision = await this.evaluateAccess(userId, resource, action);
    
    // Log all access attempts
    await this.logAccessAttempt({
      userId,
      resource,
      action,
      allowed: accessDecision.allowed,
      reason: accessDecision.reason,
      timestamp: new Date(),
      riskLevel: this.calculateRiskLevel(userId, resource, action)
    });

    if (!accessDecision.allowed) {
      await this.reportAccessViolation(userId, resource, action, accessDecision.reason);
    }

    return accessDecision;
  }

  async enforcePasswordPolicy(userId: string, password: string): Promise<{ compliant: boolean; violations: string[] }> {
    // PCI DSS Requirement 8.2: Password requirements
    const violations: string[] = [];
    
    if (password.length < 8) {
      violations.push('Password must be at least 8 characters long');
    }
    
    if (!/[A-Z]/.test(password)) {
      violations.push('Password must contain uppercase letters');
    }
    
    if (!/[a-z]/.test(password)) {
      violations.push('Password must contain lowercase letters');
    }
    
    if (!/[0-9]/.test(password)) {
      violations.push('Password must contain numbers');
    }
    
    if (!/[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(password)) {
      violations.push('Password must contain special characters');
    }

    // Check against password history
    const isReused = await this.checkPasswordHistory(userId, password);
    if (isReused) {
      violations.push('Password cannot be the same as any of the last 4 passwords');
    }

    return {
      compliant: violations.length === 0,
      violations
    };
  }

  async monitorPrivilegedAccess(): Promise<void> {
    // PCI DSS Requirement 8.5: Monitor privileged access
    const privilegedUsers = await this.getPrivilegedUsers();
    
    for (const user of privilegedUsers) {

      // Check for unusual activity patterns

  async evaluatePolicy(policyId: string, context: any): Promise<{ allowed: boolean; violations: string[] }> {
    const policy = this.policies.find(p => p.id === policyId);
    
    if (!policy || !policy.enabled) {
      return { allowed: true, violations: [] };
    }

    const violations: string[] = [];
    const allowed = await this.evaluateCondition(policy.condition, context);

    if (!allowed) {
      violations.push(`Policy violation: ${policy.name}`);
      
      // Execute policy action
      await this.executePolicyAction(policy, context);
    }

    return { allowed, violations };
  }

  async evaluateAllPolicies(context: any): Promise<{ allowed: boolean; violations: string[] }> {
    const allViolations: string[] = [];
    let overallAllowed = true;

    for (const policy of this.policies.filter(p => p.enabled)) {
      const result = await this.evaluatePolicy(policy.id, context);
      
      if (!result.allowed) {
        allViolations.push(...result.violations);
        
        if (policy.action === 'DENY') {
          overallAllowed = false;
        }
      }
    }

    return {
      allowed: overallAllowed,
      violations: allViolations
    };
  }

  async generatePolicyReport(): Promise<any> {
    const now = new Date();
    const last30Days = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);

    return {

Best Practices​

1. Implementation Guidelines​

  • Risk-Based Approach: Prioritize compliance controls based on risk assessment
  • Automation First: Automate compliance checks and reporting wherever possible
  • Continuous Monitoring: Implement real-time compliance monitoring and alerting
  • Evidence Collection: Automatically collect and preserve compliance evidence

2. Data Management​

  • Data Classification: Implement automated data discovery and classification
  • Retention Policies: Enforce automated data retention and deletion policies
  • Access Controls: Implement least-privilege access with automated reviews
  • Audit Trails: Maintain comprehensive, tamper-proof audit trails

3. Privacy by Design​

  • Default Privacy: Implement privacy-friendly defaults in all systems
  • Data Minimization: Collect and process only necessary data
  • Consent Management: Implement granular, auditable consent management
  • Transparency: Provide clear, accessible privacy information to users

4. Security Integration​

  • Defense in Depth: Layer security controls throughout the system
  • Incident Response: Integrate compliance requirements into incident response
  • Vulnerability Management: Regular security assessments and remediation
  • Encryption: Implement strong encryption for data at rest and in transit
